71pub fn build_authorize_url(pkce: &PkceState) -> String {
72    let mut params = BTreeMap::new();
73    params.insert("response_type", "code");
74    params.insert("client_id", OPENAI_OAUTH_CLIENT_ID);
75    params.insert("redirect_uri", OPENAI_OAUTH_REDIRECT_URI);
76    params.insert("scope", "openid profile email offline_access");
77    params.insert("code_challenge", pkce.code_challenge.as_str());
78    params.insert("code_challenge_method", "S256");
79    params.insert("state", pkce.state.as_str());
80    params.insert("codex_cli_simplified_flow", "true");
81    params.insert("id_token_add_organizations", "true");
82
83    let mut encoded: Vec<String> = Vec::with_capacity(params.len());
84    for (k, v) in params {
85        encoded.push(format!("{}={}", url_encode(k), url_encode(v)));
86    }
87
88    format!("{OPENAI_OAUTH_AUTHORIZE_URL}?{}", encoded.join("&"))
89}

91pub async fn exchange_code_for_tokens(
92    client: &Client,
93    code: &str,
94    pkce: &PkceState,
95) -> Result<TokenSet> {
96    let form = [
97        ("grant_type", "authorization_code"),
98        ("code", code),
99        ("client_id", OPENAI_OAUTH_CLIENT_ID),
100        ("redirect_uri", OPENAI_OAUTH_REDIRECT_URI),
101        ("code_verifier", pkce.code_verifier.as_str()),
102    ];
103
104    let response = client
105        .post(OPENAI_OAUTH_TOKEN_URL)
106        .form(&form)
107        .send()
108        .await
109        .context("Failed to exchange OpenAI OAuth authorization code")?;
110
111    parse_token_response(response).await
112}

131pub async fn start_device_code_flow(client: &Client) -> Result<DeviceCodeStart> {
132    let form = [
133        ("client_id", OPENAI_OAUTH_CLIENT_ID),
134        ("scope", "openid profile email offline_access"),
135    ];
136
137    let response = client
138        .post(OPENAI_OAUTH_DEVICE_CODE_URL)
139        .form(&form)
140        .send()
141        .await
142        .context("Failed to start OpenAI OAuth device-code flow")?;
143
144    if !response.status().is_success() {
145        let status = response.status();
146        let body = response.text().await.unwrap_or_default();
147        anyhow::bail!("OpenAI device-code start failed ({status}): {body}");
148    }
149
150    let parsed: DeviceCodeResponse = response
151        .json()
152        .await
153        .context("Failed to parse OpenAI device-code response")?;
154
155    Ok(DeviceCodeStart {
156        device_code: parsed.device_code,
157        user_code: parsed.user_code,
158        verification_uri: parsed.verification_uri,
159        verification_uri_complete: parsed.verification_uri_complete,
160        expires_in: parsed.expires_in,
161        interval: parsed.interval.unwrap_or(5).max(1),
162        message: parsed.message,
163    })
164}

166pub async fn poll_device_code_tokens(
167    client: &Client,
168    device: &DeviceCodeStart,
169) -> Result<TokenSet> {
170    let started = Instant::now();
171    let mut interval_secs = device.interval.max(1);
172
173    loop {
174        if started.elapsed() > Duration::from_secs(device.expires_in) {
175            anyhow::bail!("Device-code flow timed out before authorization completed");
176        }
177
178        tokio::time::sleep(Duration::from_secs(interval_secs)).await;
179
180        let form = [
181            ("grant_type", "urn:ietf:params:oauth:grant-type:device_code"),
182            ("device_code", device.device_code.as_str()),
183            ("client_id", OPENAI_OAUTH_CLIENT_ID),
184        ];
185
186        let response = client
187            .post(OPENAI_OAUTH_TOKEN_URL)
188            .form(&form)
189            .send()
190            .await
191            .context("Failed polling OpenAI device-code token endpoint")?;
192
193        if response.status().is_success() {
194            return parse_token_response(response).await;
195        }
196
197        let status = response.status();
198        let text = response.text().await.unwrap_or_default();
199
200        if let Ok(err) = serde_json::from_str::<OAuthErrorResponse>(&text) {
201            match err.error.as_str() {
202                "authorization_pending" => {
203                    continue;
204                }
205                "slow_down" => {
206                    interval_secs = interval_secs.saturating_add(5);
207                    continue;
208                }
209                "access_denied" => {
210                    anyhow::bail!("OpenAI device-code authorization was denied")
211                }
212                "expired_token" => {
213                    anyhow::bail!("OpenAI device-code expired")
214                }
215                _ => {
216                    anyhow::bail!(
217                        "OpenAI device-code polling failed ({status}): {}",
218                        err.error_description.unwrap_or(err.error)
219                    )
220                }
221            }
222        }
223
224        anyhow::bail!("OpenAI device-code polling failed ({status}): {text}");
225    }
226}

228pub async fn receive_loopback_code(expected_state: &str, timeout: Duration) -> Result<String> {
229    let listener = TcpListener::bind("127.0.0.1:1455")
230        .await
231        .context("Failed to bind callback listener at 127.0.0.1:1455")?;
232
233    let accepted = tokio::time::timeout(timeout, listener.accept())
234        .await
235        .context("Timed out waiting for browser callback")?
236        .context("Failed to accept callback connection")?;
237
238    let (mut stream, _) = accepted;
239    let mut buffer = vec![0_u8; 8192];
240    let bytes_read = stream
241        .read(&mut buffer)
242        .await
243        .context("Failed to read callback request")?;
244
245    let request = String::from_utf8_lossy(&buffer[..bytes_read]);
246    let first_line = request
247        .lines()
248        .next()
249        .ok_or_else(|| anyhow::anyhow!("Malformed callback request"))?;
250
251    let path = first_line
252        .split_whitespace()
253        .nth(1)
254        .ok_or_else(|| anyhow::anyhow!("Callback request missing path"))?;
255
256    let code = parse_code_from_redirect(path, Some(expected_state))?;
257
258    let body =
259        "<html><body><h2>ZeroClaw login complete</h2><p>You can close this tab.</p></body></html>";
260    let response = format!(
261        "HTTP/1.1 200 OK\r\nContent-Type: text/html; charset=utf-8\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
262        body.len(),
263        body
264    );
265    let _ = stream.write_all(response.as_bytes()).await;
266
267    Ok(code)
268}

270pub fn parse_code_from_redirect(input: &str, expected_state: Option<&str>) -> Result<String> {
271    let trimmed = input.trim();
272    if trimmed.is_empty() {
273        anyhow::bail!("No OAuth code provided");
274    }
275
276    let query = if let Some((_, right)) = trimmed.split_once('?') {
277        right
278    } else {
279        trimmed
280    };
281
282    let params = parse_query_params(query);
283    let is_callback_payload = trimmed.contains('?')
284        || params.contains_key("code")
285        || params.contains_key("state")
286        || params.contains_key("error");
287
288    if let Some(err) = params.get("error") {
289        let desc = params
290            .get("error_description")
291            .cloned()
292            .unwrap_or_else(|| "OAuth authorization failed".to_string());
293        anyhow::bail!("OpenAI OAuth error: {err} ({desc})");
294    }
295
296    if let Some(expected_state) = expected_state {
297        if let Some(got) = params.get("state") {
298            if got != expected_state {
299                anyhow::bail!("OAuth state mismatch");
300            }
301        } else if is_callback_payload {
302            anyhow::bail!("Missing OAuth state in callback");
303        }
304    }
305
306    if let Some(code) = params.get("code").cloned() {
307        return Ok(code);
308    }
309
310    if !is_callback_payload {
311        return Ok(trimmed.to_string());
312    }
313
314    anyhow::bail!("Missing OAuth code in callback")
315}

317pub fn extract_account_id_from_jwt(token: &str) -> Option<String> {
318    let payload = token.split('.').nth(1)?;
319    let decoded = base64::engine::general_purpose::URL_SAFE_NO_PAD
320        .decode(payload)
321        .ok()?;
322    let claims: serde_json::Value = serde_json::from_slice(&decoded).ok()?;
323
324    for key in [
325        "account_id",
326        "accountId",
327        "acct",
328        "sub",
329        "https://api.openai.com/account_id",
330    ] {
331        if let Some(value) = claims.get(key).and_then(|v| v.as_str())
332            && !value.trim().is_empty()
333        {
334            return Some(value.to_string());
335        }
336    }
337
338    None
339}

341pub fn extract_expiry_from_jwt(token: &str) -> Option<chrono::DateTime<Utc>> {
342    let payload = token.split('.').nth(1)?;
343    let decoded = base64::engine::general_purpose::URL_SAFE_NO_PAD
344        .decode(payload)
345        .ok()?;
346    let claims: serde_json::Value = serde_json::from_slice(&decoded).ok()?;
347    let exp = claims.get("exp").and_then(|v| v.as_i64())?;
348    chrono::DateTime::<Utc>::from_timestamp(exp, 0)
349}

351async fn parse_token_response(response: reqwest::Response) -> Result<TokenSet> {
352    if !response.status().is_success() {
353        let status = response.status();
354        let body = response.text().await.unwrap_or_default();
355        anyhow::bail!("OpenAI OAuth token request failed ({status}): {body}");
356    }
357
358    let token: TokenResponse = response
359        .json()
360        .await
361        .context("Failed to parse OpenAI token response")?;
362
363    let expires_at = token.expires_in.and_then(|seconds| {
364        if seconds <= 0 {
365            None
366        } else {
367            Some(Utc::now() + chrono::Duration::seconds(seconds))
368        }
369    });
370
371    Ok(TokenSet {
372        access_token: token.access_token,
373        refresh_token: token.refresh_token,
374        id_token: token.id_token,
375        expires_at,
376        token_type: token.token_type,
377        scope: token.scope,
378    })
379}
