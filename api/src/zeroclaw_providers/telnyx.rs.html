1//! Telnyx AI inference provider.
2//!
3//! Telnyx provides AI inference through an OpenAI-compatible API at
4//! <https://api.telnyx.com/v2/ai> with access to 53+ models including
5//! GPT-4o, Claude, Llama, Mistral, and more.
6//!
7//! # Configuration
8//!
9//! Set the `TELNYX_API_KEY` environment variable or configure in `config.toml`:
10//!
11//! ```toml
12//! default_provider = "telnyx"
13//! default_model = "openai/gpt-4o"
14//! ```

21/// Telnyx Inference Engine public endpoint.
22const BASE_URL: &str = "https://api.telnyx.com/v2/ai";

24/// Telnyx AI inference provider.
25///
26/// Uses the OpenAI-compatible chat completions API at `/v2/ai/chat/completions`.
27/// Supports 53+ models including OpenAI, Anthropic (via API), Meta Llama,
28/// Mistral, and more.
29///
30/// # Example
31///
32/// ```rust,ignore
33/// use zeroclaw::providers::telnyx::TelnyxProvider;
34/// use zeroclaw::providers::Provider;
35///
36/// let provider = TelnyxProvider::new(Some("your-api-key"));
37/// let response = provider.chat("Hello!", "openai/gpt-4o", 0.7).await?;
38/// ```
39pub struct TelnyxProvider {
40    /// Telnyx API key
41    api_key: Option<String>,
42    /// HTTP client for API requests
43    client: Client,
44}
45
46impl TelnyxProvider {
47    /// Create a new Telnyx AI provider.
48    ///
49    /// The API key can be provided directly or will be resolved from:
50    /// 1. `TELNYX_API_KEY` environment variable
51    /// 2. `ZEROCLAW_API_KEY` environment variable (fallback)
52    pub fn new(api_key: Option<&str>) -> Self {
53        let resolved_key = resolve_telnyx_api_key(api_key);
54        Self {
55            api_key: resolved_key,
56            client: Client::builder()
57                .timeout(std::time::Duration::from_secs(120))
58                .connect_timeout(std::time::Duration::from_secs(10))
59                .build()
60                .unwrap_or_else(|_| Client::new()),
61        }
62    }
63
64    /// Create a provider with a custom base URL (for testing or proxies).
65    pub fn with_base_url(api_key: Option<&str>, _base_url: &str) -> Self {
66        // Note: custom base URL support for testing
67        Self::new(api_key)
68    }
69
70    /// List available models from Telnyx AI.
71    ///
72    /// Returns a list of model IDs that can be used with the chat API.
73    pub async fn list_models(&self) -> anyhow::Result<Vec<String>> {
74        let api_key = self.api_key.as_ref().ok_or_else(|| {
75            anyhow::anyhow!("Telnyx API key not set. Set TELNYX_API_KEY environment variable.")
76        })?;
77
78        let response = self
79            .client
80            .get(format!("{}/models", BASE_URL))
81            .header("Authorization", format!("Bearer {}", api_key))
82            .send()
83            .await?;
84
85        if !response.status().is_success() {
86            let error = response.text().await?;
87            anyhow::bail!("Failed to list Telnyx models: {}", error);
88        }
89
90        let models_response: ModelsResponse = response.json().await?;
91        Ok(models_response.data.into_iter().map(|m| m.id).collect())
92    }
93
94    /// Build the chat completions URL
95    fn chat_url(&self) -> String {
96        format!("{}/chat/completions", BASE_URL)
97    }
98}

100/// Resolve Telnyx API key from parameter or environment.
101fn resolve_telnyx_api_key(api_key: Option<&str>) -> Option<String> {
102    if let Some(key) = api_key.map(str::trim).filter(|k| !k.is_empty()) {
103        return Some(key.to_string());
104    }
105
106    // Try Telnyx-specific env var first
107    if let Ok(key) = std::env::var("TELNYX_API_KEY") {
108        let key = key.trim();
109        if !key.is_empty() {
110            return Some(key.to_string());
111        }
112    }
113
114    // Fall back to generic env vars
115    for env_var in ["ZEROCLAW_API_KEY", "API_KEY"] {
116        if let Ok(key) = std::env::var(env_var) {
117            let key = key.trim();
118            if !key.is_empty() {
119                return Some(key.to_string());
120            }
121        }
122    }
123
124    None
125}

169impl Provider for TelnyxProvider {
170    // ── Provider-family defaults ──
171    fn default_base_url(&self) -> Option<&str> {
172        Some(BASE_URL)
173    }
174
175    async fn chat_with_system(
176        &self,
177        system_prompt: Option<&str>,
178        message: &str,
179        model: &str,
180        temperature: Option<f64>,
181    ) -> anyhow::Result<String> {
182        let temperature = temperature.unwrap_or(self.default_temperature());
183        let api_key = self.api_key.as_ref().ok_or_else(|| {
184            anyhow::anyhow!(
185                "Telnyx API key not set. Set TELNYX_API_KEY environment variable or run `zeroclaw onboard`."
186            )
187        })?;
188
189        let mut messages = Vec::new();
190
191        if let Some(sys) = system_prompt {
192            messages.push(Message {
193                role: "system".to_string(),
194                content: sys.to_string(),
195            });
196        }
197
198        messages.push(Message {
199            role: "user".to_string(),
200            content: message.to_string(),
201        });
202
203        let request = ChatRequest {
204            model: model.to_string(),
205            messages,
206            temperature,
207        };
208
209        let response = self
210            .client
211            .post(self.chat_url())
212            .header("Authorization", format!("Bearer {}", api_key))
213            .header("Content-Type", "application/json")
214            .json(&request)
215            .send()
216            .await?;
217
218        if !response.status().is_success() {
219            let status = response.status();
220            let error = response.text().await?;
221            let sanitized = super::sanitize_api_error(&error);
222            anyhow::bail!("Telnyx API error ({}): {}", status, sanitized);
223        }
224
225        let chat_response: ChatResponse = response.json().await?;
226
227        chat_response
228            .choices
229            .into_iter()
230            .next()
231            .map(|c| c.message.content)
232            .ok_or_else(|| anyhow::anyhow!("No response from Telnyx"))
233    }
234
235    async fn chat_with_history(
236        &self,
237        messages: &[ChatMessage],
238        model: &str,
239        temperature: Option<f64>,
240    ) -> anyhow::Result<String> {
241        let temperature = temperature.unwrap_or(self.default_temperature());
242        let api_key = self.api_key.as_ref().ok_or_else(|| {
243            anyhow::anyhow!(
244                "Telnyx API key not set. Set TELNYX_API_KEY environment variable or run `zeroclaw onboard`."
245            )
246        })?;
247
248        let api_messages: Vec<Message> = messages
249            .iter()
250            .map(|m| Message {
251                role: m.role.clone(),
252                content: m.content.clone(),
253            })
254            .collect();
255
256        let request = ChatRequest {
257            model: model.to_string(),
258            messages: api_messages,
259            temperature,
260        };
261
262        let response = self
263            .client
264            .post(self.chat_url())
265            .header("Authorization", format!("Bearer {}", api_key))
266            .header("Content-Type", "application/json")
267            .json(&request)
268            .send()
269            .await?;
270
271        if !response.status().is_success() {
272            let status = response.status();
273            let error = response.text().await?;
274            let sanitized = super::sanitize_api_error(&error);
275            anyhow::bail!("Telnyx API error ({}): {}", status, sanitized);
276        }
277
278        let chat_response: ChatResponse = response.json().await?;
279
280        chat_response
281            .choices
282            .into_iter()
283            .next()
284            .map(|c| c.message.content)
285            .ok_or_else(|| anyhow::anyhow!("No response from Telnyx"))
286    }
287
288    async fn warmup(&self) -> anyhow::Result<()> {
289        // Pre-warm the connection pool
290        let _ = self.client.get(format!("{}/models", BASE_URL)).send().await;
291        Ok(())
292    }
