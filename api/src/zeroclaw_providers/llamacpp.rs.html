128    fn responses_url(&self) -> String {
129        let base = &self.base_url;
130        // Already pointing at the responses endpoint.
131        if base.ends_with("/responses") {
132            return base.clone();
133        }
134        // Derive sibling /responses from /chat/completions.
135        if let Some(prefix) = base.strip_suffix("/chat/completions") {
136            return format!("{prefix}/responses");
137        }
138        // If the base is a bare host (http://host or http://host:port with no
139        // path), add the standard /v1 prefix that llama-server uses.
140        let after_scheme = base.split_once("://").map(|(_, r)| r).unwrap_or(base);
141        if !after_scheme.contains('/') {
142            return format!("{base}/v1/responses");
143        }
144        format!("{base}/responses")
145    }

158    /// HTTP client with total timeout — for non-streaming requests only.
159    fn http_client(&self) -> Client {
160        let timeout = self.timeout_secs;
161        if self.extra_headers.is_empty() {
162            return zeroclaw_config::schema::build_runtime_proxy_client_with_timeouts(
163                "provider.llamacpp",
164                timeout,
165                10,
166            );
167        }
168        let mut headers = HeaderMap::new();
169        for (key, value) in &self.extra_headers {
170            match (
171                reqwest::header::HeaderName::from_bytes(key.as_bytes()),
172                HeaderValue::from_str(value),
173            ) {
174                (Ok(name), Ok(val)) => {
175                    headers.insert(name, val);
176                }
177                _ => {
178                    warn!(header = key, "Skipping invalid extra header");
179                }
180            }
181        }
182        let builder = Client::builder()
183            .timeout(std::time::Duration::from_secs(timeout))
184            .connect_timeout(std::time::Duration::from_secs(10))
185            .default_headers(headers);
186        let builder =
187            zeroclaw_config::schema::apply_runtime_proxy_to_builder(builder, "provider.llamacpp");
188        builder.build().unwrap_or_else(|e| {
189            warn!("Failed to build llama.cpp HTTP client: {e}");
190            Client::new()
191        })
192    }
193
194    /// HTTP client with connect timeout only — for streaming SSE sessions.
195    fn streaming_http_client(&self) -> Client {
196        if self.extra_headers.is_empty() {
197            let builder = Client::builder().connect_timeout(std::time::Duration::from_secs(10));
198            let builder = zeroclaw_config::schema::apply_runtime_proxy_to_builder(
199                builder,
200                "provider.llamacpp",
201            );
202            return builder.build().unwrap_or_else(|e| {
203                warn!("Failed to build llama.cpp streaming client: {e}");
204                Client::new()
205            });
206        }
207        let mut headers = HeaderMap::new();
208        for (key, value) in &self.extra_headers {
209            if let (Ok(name), Ok(val)) = (
210                reqwest::header::HeaderName::from_bytes(key.as_bytes()),
211                HeaderValue::from_str(value),
212            ) {
213                headers.insert(name, val);
214            }
215        }
216        let builder = Client::builder()
217            .connect_timeout(std::time::Duration::from_secs(10))
218            .default_headers(headers);
219        let builder =
220            zeroclaw_config::schema::apply_runtime_proxy_to_builder(builder, "provider.llamacpp");
221        builder.build().unwrap_or_else(|e| {
222            warn!("Failed to build llama.cpp streaming client: {e}");
223            Client::new()
224        })
225    }

227    fn should_sanitize_tool_schema(model: &str) -> bool {
228        let lower = model.to_ascii_lowercase();
229        model.is_empty() || lower.contains("gemma-4") || lower.contains("gemma4")
230    }

232    fn sanitize_tool_payload_for_model(
233        tools: Vec<serde_json::Value>,
234        model: &str,
235    ) -> Vec<serde_json::Value> {
236        if !Self::should_sanitize_tool_schema(model) {
237            return tools;
238        }
239
240        tools
241            .into_iter()
242            .map(|mut tool| {
243                let Some(raw_parameters) = tool.get("parameters").cloned() else {
244                    return tool;
245                };
246
247                let cleaned_parameters = zeroclaw_api::schema::SchemaCleanr::clean(
248                    raw_parameters,
249                    zeroclaw_api::schema::CleaningStrategy::Conservative,
250                );
251
252                if let Some(tool_obj) = tool.as_object_mut() {
253                    tool_obj.insert("parameters".to_string(), cleaned_parameters);
254                }
255
256                tool
257            })
258            .collect()
259    }
260
261    fn convert_tools(
262        tools: Option<&[zeroclaw_api::tool::ToolSpec]>,
263        model: &str,
264    ) -> Option<Vec<serde_json::Value>> {
265        tools.map(|items| {
266            let converted = items
267                .iter()
268                .map(|tool| {
269                    let params = zeroclaw_api::schema::SchemaCleanr::clean_for_openai(
270                        tool.parameters.clone(),
271                    );
272                    serde_json::json!({
273                        "type": "function",
274                        "name": tool.name,
275                        "description": tool.description,
276                        "parameters": params,
277                    })
278                })
279                .collect::<Vec<_>>();
280
281            Self::sanitize_tool_payload_for_model(converted, model)
282        })
283    }

287    async fn do_chat(
288        &self,
289        messages: &[ChatMessage],
290        model: &str,
291        temperature: Option<f64>,
292        tools: Option<Vec<serde_json::Value>>,
293    ) -> anyhow::Result<ProviderChatResponse> {
294        let (raw_instructions, input) = build_prompt(messages);
295        if input.is_empty() {
296            anyhow::bail!("llama.cpp: at least one non-system message is required");
297        }
298        let instructions = if tools.as_ref().is_some_and(|t| !t.is_empty()) {
299            strip_tools_section(raw_instructions)
300        } else {
301            raw_instructions
302        };
303        let request = ResponsesRequest {
304            model: model.to_string(),
305            input,
306            instructions,
307            stream: Some(false),
308            temperature: Some(temperature.unwrap_or(self.default_temperature())),
309            tools,
310            enable_thinking: self.think,
311            max_output_tokens: self.max_tokens,
312            chat_template_kwargs: self.chat_template_kwargs.clone(),
313        };
314        let url = self.responses_url();
315        let response = self
316            .auth_header(self.http_client().post(&url).json(&request))
317            .send()
318            .await?;
319        if !response.status().is_success() {
320            return Err(super::api_error("llama.cpp", response).await);
321        }
322        let body = response.text().await?;
323        parse_response_body(&body)
324    }

326    fn do_stream(
327        &self,
328        messages: Vec<ChatMessage>,
329        model: &str,
330        temperature: Option<f64>,
331        tools: Option<Vec<serde_json::Value>>,
332        count_tokens: bool,
333    ) -> stream::BoxStream<'static, StreamResult<StreamEvent>> {
334        let (raw_instructions, input) = build_prompt(&messages);
335        if input.is_empty() {
336            return stream::once(async {
337                Err(StreamError::Provider(
338                    "llama.cpp: at least one non-system message is required".into(),
339                ))
340            })
341            .boxed();
342        }
343        let instructions = if tools.as_ref().is_some_and(|t| !t.is_empty()) {
344            strip_tools_section(raw_instructions)
345        } else {
346            raw_instructions
347        };
348        let req_body = ResponsesRequest {
349            model: model.to_string(),
350            input,
351            instructions,
352            stream: Some(true),
353            temperature: Some(temperature.unwrap_or(self.default_temperature())),
354            tools,
355            enable_thinking: self.think,
356            max_output_tokens: self.max_tokens,
357            chat_template_kwargs: self.chat_template_kwargs.clone(),
358        };
359        let payload = match serde_json::to_value(req_body) {
360            Ok(p) => p,
361            Err(e) => return stream::once(async move { Err(StreamError::Json(e)) }).boxed(),
362        };
363        let payload_bytes = payload.to_string().len();
364        let instructions_bytes = payload
365            .get("instructions")
366            .and_then(|v| v.as_str())
367            .map(str::len)
368            .unwrap_or(0);
369        let tools_bytes = payload
370            .get("tools")
371            .map(|v| v.to_string().len())
372            .unwrap_or(0);
373        let input_bytes = payload
374            .get("input")
375            .map(|v| v.to_string().len())
376            .unwrap_or(0);
377        debug!(
378            "llama.cpp stream request payload size={payload_bytes} bytes \
379             (instructions={instructions_bytes}, tools={tools_bytes}, input={input_bytes})"
380        );
381        let url = self.responses_url();
382        let client = self.streaming_http_client();
383        let credential = self.credential.clone();
384
385        let (tx, rx) = tokio::sync::mpsc::channel::<StreamResult<StreamEvent>>(100);
386        tokio::spawn(async move {
387            let mut req = client.post(&url).json(&payload);
388            if let Some(key) = credential.as_deref().filter(|k| !k.is_empty()) {
389                req = req.header("Authorization", format!("Bearer {key}"));
390            }
391            req = req.header("Accept", "text/event-stream");
392
393            let response = match req.send().await {
394                Ok(r) => r,
395                Err(e) => {
396                    let _ = tx.send(Err(StreamError::Http(e.to_string()))).await;
397                    return;
398                }
399            };
400            if !response.status().is_success() {
401                let status = response.status();
402                let error = response
403                    .text()
404                    .await
405                    .unwrap_or_else(|_| format!("HTTP {status}"));
406                let sanitized = super::sanitize_api_error(&error);
407                let _ = tx
408                    .send(Err(StreamError::Provider(format!("{status}: {sanitized}"))))
409                    .await;
410                return;
411            }
412            let mut events = parse_sse_responses(response, count_tokens);
413            while let Some(event) = events.next().await {
414                if tx.send(event).await.is_err() {
415                    break;
416                }
417            }
418        });
419        stream::unfold(rx, |mut rx| async move { rx.recv().await.map(|e| (e, rx)) }).boxed()
420    }
421}

580/// Remove the `## Tools` section from the system prompt when tools are already
581/// sent as structured data in the request body. The section contains full JSON
582/// schemas for every tool, which can be hundreds of KB and is redundant when
583/// the model receives the same information via the `tools` field.
584fn strip_tools_section(instructions: Option<String>) -> Option<String> {
585    let s = instructions?;
586    // Match "## Tools\n" at start or after a newline.
587    let needle = "## Tools\n";
588    let (prefix, rest) = if let Some(rest) = s.strip_prefix(needle) {
589        ("", rest)
590    } else if let Some(pos) = s.find(&format!("\n{needle}")) {
591        (&s[..pos], &s[pos + 1 + needle.len()..])
592    } else {
593        return Some(s);
594    };
595    // Find the next top-level section header or end of string.
596    let suffix = if let Some(next) = rest.find("\n## ") {
597        &rest[next + 1..]
598    } else {
599        ""
600    };
601    let result = format!("{prefix}\n\n{suffix}").trim().to_string();
602    if result.is_empty() {
603        None
604    } else {
605        Some(result)
606    }
607}

609fn build_prompt(messages: &[ChatMessage]) -> (Option<String>, Vec<serde_json::Value>) {
610    let mut sys_parts: Vec<String> = Vec::new();
611    let mut input: Vec<serde_json::Value> = Vec::new();
612
613    for message in messages {
614        if message.content.trim().is_empty() {
615            continue;
616        }
617        if message.role == "system" {
618            sys_parts.push(message.content.clone());
619            continue;
620        }
621
622        let item: serde_json::Value = match message.role.as_str() {
623            "assistant" => {
624                if let Ok(value) = serde_json::from_str::<serde_json::Value>(&message.content)
625                    && let Some(calls) = value.get("tool_calls").and_then(|v| v.as_array())
626                    && !calls.is_empty()
627                {
628                    let text = value
629                        .get("content")
630                        .and_then(|v| v.as_str())
631                        .unwrap_or("")
632                        .trim()
633                        .to_string();
634                    if !text.is_empty() {
635                        input.push(serde_json::json!({
636                            "role": "assistant",
637                            "type": "message",
638                            "content": [{"type": "output_text", "text": text}]
639                        }));
640                    }
641                    for call in calls {
642                        let call_id = call
643                            .get("id")
644                            .and_then(|v| v.as_str())
645                            .unwrap_or("")
646                            .to_string();
647                        let name = call
648                            .get("name")
649                            .and_then(|v| v.as_str())
650                            .unwrap_or("")
651                            .to_string();
652                        let arguments = call
653                            .get("arguments")
654                            .and_then(|v| v.as_str())
655                            .unwrap_or("{}")
656                            .to_string();
657                        input.push(serde_json::json!({
658                            "type": "function_call",
659                            "call_id": call_id,
660                            "name": name,
661                            "arguments": arguments,
662                        }));
663                    }
664                    continue;
665                }
666                serde_json::json!({
667                    "role": "assistant",
668                    "type": "message",
669                    "content": [{"type": "output_text", "text": message.content}]
670                })
671            }
672            "tool" => {
673                if let Ok(value) = serde_json::from_str::<serde_json::Value>(&message.content) {
674                    let call_id = value
675                        .get("tool_call_id")
676                        .and_then(|v| v.as_str())
677                        .unwrap_or("")
678                        .to_string();
679                    let output = value
680                        .get("content")
681                        .and_then(|v| v.as_str())
682                        .map(str::to_string)
683                        .unwrap_or_else(|| message.content.clone());
684                    serde_json::json!({
685                        "type": "function_call_output",
686                        "call_id": call_id,
687                        "output": output,
688                    })
689                } else {
690                    serde_json::json!({
691                        "role": "assistant",
692                        "type": "message",
693                        "content": [{"type": "output_text", "text": message.content}]
694                    })
695                }
696            }
697            _ => serde_json::json!({"role": "user", "content": message.content}),
698        };
699        input.push(item);
700    }
701
702    let instructions = if sys_parts.is_empty() {
703        None
704    } else {
705        Some(sys_parts.join("\n\n"))
706    };
707    (instructions, input)
708}

711
712fn parse_response_body(body: &str) -> anyhow::Result<ProviderChatResponse> {
713    debug!("llama.cpp response body: {body}");
714    let resp = serde_json::from_str::<ResponsesResponse>(body).map_err(|e| {
715        let snippet = super::sanitize_api_error(body);
716        anyhow::anyhow!("llama.cpp responses API returned unexpected payload: {e}; body={snippet}")
717    })?;
718
719    let mut tool_calls: Vec<ProviderToolCall> = Vec::new();
720    let mut text: Option<String> = resp.output_text.as_deref().and_then(nonempty);
721
722    for item in resp.output {
723        match item.kind.as_deref() {
724            Some("function_call") => {
725                if let Some(name) = item.name {
726                    let arguments = item.arguments.unwrap_or_else(|| "{}".to_string());
727                    let id = item
728                        .call_id
729                        .unwrap_or_else(|| uuid::Uuid::new_v4().to_string());
730                    tool_calls.push(ProviderToolCall {
731                        id,
732                        name,
733                        arguments,
734                        extra_content: None,
735                    });
736                }
737            }
738            // Skip chain-of-thought reasoning items; the answer is in the message item.
739            Some("reasoning") => {}
740            _ => {
741                if text.is_none() {
742                    for content in &item.content {
743                        if let Some(t) = content.text.as_deref().and_then(nonempty) {
744                            text = Some(t);
745                            break;
746                        }
747                    }
748                }
749            }
750        }
751    }
752
753    Ok(ProviderChatResponse {
754        text,
755        tool_calls,
756        usage: None,
757        reasoning_content: None,
758    })
759}

762    let t = s.trim();
763    if t.is_empty() {
764        None
765    } else {
766        Some(t.to_string())
767    }

770// ── SSE stream parser ───────────────────────────────────────────────────────
771
772fn parse_sse_responses(
773    response: reqwest::Response,
774    count_tokens: bool,
775) -> stream::BoxStream<'static, StreamResult<StreamEvent>> {
776    use crate::traits::StreamChunk;
777
778    let (tx, rx) = tokio::sync::mpsc::channel::<StreamResult<StreamEvent>>(100);
779
780    tokio::spawn(async move {
781        if let Err(e) = response.error_for_status_ref() {
782            let _ = tx.send(Err(StreamError::Http(e.to_string()))).await;
783            return;
784        }
785
786        let mut pending: HashMap<String, (String, String)> = HashMap::new();
787        // Track item_ids that have already emitted at least one delta (text or reasoning),
788        // so the corresponding *done events can serve as fallbacks for non-streaming models.
789        let mut text_delta_seen: HashSet<String> = HashSet::new();
790        let mut reasoning_delta_seen: HashSet<String> = HashSet::new();
791        let mut buffer = String::new();
792        let mut utf8_buf: Vec<u8> = Vec::new();
793        let mut bytes_stream = response.bytes_stream();
794
795        'outer: while let Some(item) = bytes_stream.next().await {
796            let bytes = match item {
797                Ok(b) => b,
798                Err(e) => {
799                    let _ = tx.send(Err(StreamError::Http(e.to_string()))).await;
800                    return;
801                }
802            };
803
804            utf8_buf.extend_from_slice(&bytes);
805            let text = match std::str::from_utf8(&utf8_buf) {
806                Ok(s) => {
807                    let owned = s.to_string();
808                    utf8_buf.clear();
809                    owned
810                }
811                Err(e) => {
812                    let valid_up_to = e.valid_up_to();
813                    if valid_up_to == 0 && utf8_buf.len() < 4 {
814                        continue;
815                    }
816                    let valid = String::from_utf8_lossy(&utf8_buf[..valid_up_to]).into_owned();
817                    utf8_buf.drain(..valid_up_to);
818                    valid
819                }
820            };
821            if text.is_empty() {
822                continue;
823            }
824
825            buffer.push_str(&text);
826
827            while let Some(pos) = buffer.find('\n') {
828                let line = buffer[..pos].to_string();
829                buffer.drain(..=pos);
830                let line = line.trim();
831                if line.is_empty() || line.starts_with("event:") || line.starts_with(':') {
832                    continue;
833                }
834                let Some(data) = line.strip_prefix("data:") else {
835                    continue;
836                };
837                let data = data.trim();
838                if data == "[DONE]" {
839                    break 'outer;
840                }
841
842                let event: serde_json::Value = match serde_json::from_str(data) {
843                    Ok(v) => v,
844                    Err(e) => {
845                        let _ = tx.send(Err(StreamError::Json(e))).await;
846                        return;
847                    }
848                };
849
850                let kind = event.get("type").and_then(|v| v.as_str()).unwrap_or("");
851                debug!("llama.cpp SSE event type={kind:?}");
852
853                match kind {
854                    // Response lifecycle bookkeeping — no action needed.
855                    "response.created" | "response.in_progress" => {}
856                    // A new output_text part is starting; actual content follows via delta/done.
857                    "response.content_part.added" => {}
858                    "response.output_text.delta" => {
859                        if let Some(item_id) = event.get("item_id").and_then(|v| v.as_str()) {
860                            text_delta_seen.insert(item_id.to_string());
861                        }
862                        match event.get("delta").and_then(|v| v.as_str()) {
863                            Some(delta) if !delta.is_empty() => {
864                                let mut chunk = StreamChunk::delta(delta.to_string());
865                                if count_tokens {
866                                    chunk = chunk.with_token_estimate();
867                                }
868                                if tx.send(Ok(StreamEvent::TextDelta(chunk))).await.is_err() {
869                                    return;
870                                }
871                            }
872                            _ => debug!("llama.cpp output_text.delta had no string delta: {event}"),
873                        }
874                    }
875                    // Emitted when a text part is complete.  Fallback for models that don't
876                    // stream character-by-character deltas.
877                    "response.output_text.done" => {
878                        let item_id = event.get("item_id").and_then(|v| v.as_str()).unwrap_or("");
879                        if !text_delta_seen.contains(item_id)
880                            && let Some(text) = event.get("text").and_then(|v| v.as_str())
881                            && !text.is_empty()
882                        {
883                            let mut chunk = StreamChunk::delta(text.to_string());
884                            if count_tokens {
885                                chunk = chunk.with_token_estimate();
886                            }
887                            if tx.send(Ok(StreamEvent::TextDelta(chunk))).await.is_err() {
888                                return;
889                            }
890                            text_delta_seen.insert(item_id.to_string());
891                        }
892                    }
893                    // content_part.done duplicates output_text.done for text parts; discard.
894                    "response.content_part.done" => {}
895                    // Chain-of-thought reasoning/thinking content from the model.
896                    "response.reasoning_text.delta"
897                    | "response.reasoning_summary_text.delta"
898                    | "response.reasoning.delta" => {
899                        if let Some(item_id) = event.get("item_id").and_then(|v| v.as_str()) {
900                            reasoning_delta_seen.insert(item_id.to_string());
901                        }
902                        if let Some(delta) = event.get("delta").and_then(|v| v.as_str())
903                            && !delta.is_empty()
904                            && tx
905                                .send(Ok(StreamEvent::TextDelta(StreamChunk::reasoning(
906                                    delta.to_string(),
907                                ))))
908                                .await
909                                .is_err()
910                        {
911                            return;
912                        }
913                    }
914                    // Fallback for non-streaming reasoning output.
915                    "response.reasoning_text.done" | "response.reasoning.done" => {
916                        let item_id = event.get("item_id").and_then(|v| v.as_str()).unwrap_or("");
917                        if !reasoning_delta_seen.contains(item_id)
918                            && let Some(text) = event.get("text").and_then(|v| v.as_str())
919                            && !text.is_empty()
920                        {
921                            if tx
922                                .send(Ok(StreamEvent::TextDelta(StreamChunk::reasoning(
923                                    text.to_string(),
924                                ))))
925                                .await
926                                .is_err()
927                            {
928                                return;
929                            }
930                            reasoning_delta_seen.insert(item_id.to_string());
931                        }
932                    }
933                    "response.output_item.added" => {
934                        if let Some(item) = event.get("item")
935                            && item.get("type").and_then(|v| v.as_str()) == Some("function_call")
936                        {
937                            let call_id = item
938                                .get("call_id")
939                                .and_then(|v| v.as_str())
940                                .unwrap_or("")
941                                .to_string();
942                            let name = item
943                                .get("name")
944                                .and_then(|v| v.as_str())
945                                .unwrap_or("")
946                                .to_string();
947                            pending.insert(call_id, (name, String::new()));
948                        }
949                    }
950                    "response.function_call_arguments.delta" => {
951                        if let (Some(call_id), Some(delta)) = (
952                            event.get("call_id").and_then(|v| v.as_str()),
953                            event.get("delta").and_then(|v| v.as_str()),
954                        ) && let Some((_, args)) = pending.get_mut(call_id)
955                        {
956                            args.push_str(delta);
957                        }
958                    }
959                    "response.output_item.done" => {
960                        if let Some(item) = event.get("item")
961                            && item.get("type").and_then(|v| v.as_str()) == Some("function_call")
962                        {
963                            let call_id =
964                                item.get("call_id").and_then(|v| v.as_str()).unwrap_or("");
965                            if let Some((name, accumulated_args)) = pending.remove(call_id) {
966                                let arguments = item
967                                    .get("arguments")
968                                    .and_then(|v| v.as_str())
969                                    .map(str::to_string)
970                                    .unwrap_or(accumulated_args);
971                                let arguments = if arguments.trim().is_empty() {
972                                    "{}".to_string()
973                                } else {
974                                    arguments
975                                };
976                                let tool_call = ProviderToolCall {
977                                    id: call_id.to_string(),
978                                    name,
979                                    arguments,
980                                    extra_content: None,
981                                };
982                                if tx.send(Ok(StreamEvent::ToolCall(tool_call))).await.is_err() {
983                                    return;
984                                }
985                            }
986                        }
987                    }
988                    "response.completed" => break 'outer,
989                    _ => debug!("llama.cpp unhandled SSE event type={kind:?} full={event}"),
990                }
991            }
992        }
993
994        // Flush any tool calls whose done event was missed.
995        for (call_id, (name, arguments)) in pending.drain() {
996            let arguments = if arguments.trim().is_empty() {
997                "{}".to_string()
998            } else {
999                arguments
1000            };
1001            let _ = tx
1002                .send(Ok(StreamEvent::ToolCall(ProviderToolCall {
1003                    id: call_id,
1004                    name,
1005                    arguments,
1006                    extra_content: None,
1007                })))
1008                .await;
1009        }
1010
1011        let _ = tx.send(Ok(StreamEvent::Final)).await;
1012    });
1013
1014    stream::unfold(rx, |mut rx| async move { rx.recv().await.map(|e| (e, rx)) }).boxed()
1015}

1017/// Convert a `StreamEvent` stream into a `StreamChunk` stream (text only).
1018fn text_chunks(
1019    events: stream::BoxStream<'static, StreamResult<StreamEvent>>,
1020) -> stream::BoxStream<'static, StreamResult<StreamChunk>> {
1021    use crate::traits::StreamChunk;
1022    let (tx, rx) = tokio::sync::mpsc::channel::<StreamResult<StreamChunk>>(100);
1023    tokio::spawn(async move {
1024        let mut events = events;
1025        while let Some(event) = events.next().await {
1026            match event {
1027                Ok(StreamEvent::TextDelta(chunk)) => {
1028                    if tx.send(Ok(chunk)).await.is_err() {
1029                        return;
1030                    }
1031                }
1032                Ok(StreamEvent::Final) => {
1033                    let _ = tx.send(Ok(StreamChunk::final_chunk())).await;
1034                    return;
1035                }
1036                Ok(_) => {}
1037                Err(e) => {
1038                    let _ = tx.send(Err(e)).await;
1039                    return;
1040                }
1041            }
1042        }
1043        let _ = tx.send(Ok(StreamChunk::final_chunk())).await;
1044    });
1045    stream::unfold(rx, |mut rx| async move { rx.recv().await.map(|c| (c, rx)) }).boxed()
1046}
