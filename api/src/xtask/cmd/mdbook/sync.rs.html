5pub fn run(
6    locale: Option<&str>,
7    force: bool,
8    provider: Option<&str>,
9    batch: Option<usize>,
10) -> anyhow::Result<()> {
11    let root = repo_root();
12    require_tool(
13        "mdbook-xgettext",
14        "cargo install mdbook-i18n-helpers --locked",
15    )?;
16    require_tool("msgmerge", "apt install gettext / brew install gettext")?;
17    require_tool("msginit", "apt install gettext / brew install gettext")?;
18    require_tool("msgfmt", "apt install gettext / brew install gettext")?;
19    require_tool("msgattrib", "apt install gettext / brew install gettext")?;
20
21    let book = book_dir(&root);
22    let po_dir = po_dir(&root);
23    let pot = pot_file(&root);
24
25    std::fs::create_dir_all(&po_dir)?;
26
27    // Step 1: extract English msgids
28    println!("==> Extracting English msgids → {}", pot.display());
29    crate::cmd::mdbook::build::inject_lang_switcher_locales(&book, &locale_entries())?;
30    run_cmd(
31        Command::new(mdbook_program()?)
32            .args(["build", "-d", "po-extract"])
33            .env("MDBOOK_OUTPUT__XGETTEXT__POT_FILE", "messages.pot")
34            .current_dir(&book),
35    )?;
36
37    let extracted = book.join("po-extract/xgettext/messages.pot");
38    if extracted.exists() {
39        std::fs::rename(&extracted, &pot)?;
40    }
41    let _ = std::fs::remove_dir_all(book.join("po-extract"));
42
43    if !pot.exists() {
44        anyhow::bail!(
45            "messages.pot not generated — is mdbook-i18n-helpers installed?\n  \
46             cargo install mdbook-i18n-helpers --locked"
47        );
48    }
49
50    // Step 2+3: per-locale merge + AI fill
51    let targets: Vec<String> = match locale {
52        Some(l) => vec![l.to_string()],
53        None => locales().into_iter().filter(|l| l != "en").collect(),
54    };
55
56    for locale in &targets {
57        if locale == "en" {
58            continue;
59        }
60        let po_file = po_dir.join(format!("{locale}.po"));
61
62        if !po_file.exists() {
63            println!("==> {locale}: bootstrapping new .po from template");
64            run_cmd(
65                Command::new("msginit")
66                    .args(["--no-translator", &format!("--locale={locale}"), "--input"])
67                    .arg(&pot)
68                    .arg("--output")
69                    .arg(&po_file),
70            )?;
71        } else {
72            println!("==> {locale}: msgmerge");
73            run_cmd(
74                Command::new("msgmerge")
75                    .args(["--update", "--backup=none", "--no-fuzzy-matching"])
76                    .arg(&po_file)
77                    .arg(&pot),
78            )?;
79            // Strip obsolete (#~) entries that msgmerge leaves behind for removed source strings.
80            run_cmd(
81                Command::new("msgattrib")
82                    .arg("--no-obsolete")
83                    .arg("--output-file")
84                    .arg(&po_file)
85                    .arg(&po_file),
86            )?;
87        }
88
89        if force {
90            if let Some(p) = provider {
91                println!("==> {locale}: --force: re-translating all entries");
92                fill(&root, &po_file, locale, true, p, batch)?;
93            } else {
94                println!(
95                    "==> {locale}: --force requested but no --provider specified — skipping AI step"
96                );
97            }
98        } else {
99            let delta = count_delta(&po_file)?;
100            if delta > 0 {
101                if let Some(p) = provider {
102                    println!("==> {locale}: AI-filling {delta} entries");
103                    fill(&root, &po_file, locale, false, p, batch)?;
104                } else {
105                    println!(
106                        "==> {locale}: {delta} entries need translation (use --provider <name> to auto-fill)"
107                    );
108                }
109            } else {
110                println!("==> {locale}: up to date, skipping AI step");
111            }
112        }
113    }
114
115    println!("\n==> Translation summary:");
116    for locale in &targets {
117        if locale == "en" {
118            continue;
119        }
120        let po_file = po_dir.join(format!("{locale}.po"));
121        if !po_file.exists() {
122            continue;
123        }
124        print!("    {locale:<8} ");
125        let out = Command::new("msgfmt")
126            .args(["--statistics", "-o", "/dev/null"])
127            .arg(&po_file)
128            .env("LANG", "C")
129            .output()?;
130        print!("{}", String::from_utf8_lossy(&out.stderr));
131    }
132    Ok(())
133}

135fn fill(
136    root: &Path,
137    po_file: &Path,
138    locale: &str,
139    force: bool,
140    provider: &str,
141    batch: Option<usize>,
142) -> anyhow::Result<()> {
143    // Build and invoke the binary directly — `cargo run` wraps the child in a way that
144    // breaks Ctrl-C propagation, leaving the translator orphaned in the terminal.
145    let bin = root.join("target/release/fill-translations");
146    run_cmd(
147        Command::new("cargo")
148            .args(["build", "--release", "-q", "--manifest-path"])
149            .arg(root.join("tools/fill-translations/Cargo.toml")),
150    )?;
151    let mut cmd = Command::new(&bin);
152    cmd.args(["--po"])
153        .arg(po_file)
154        .args(["--locale", locale])
155        .args(["--provider", provider]);
156    if let Some(b) = batch {
157        cmd.args(["--batch", &b.to_string()]);
158    }
159    if force {
160        cmd.arg("--force");
161    }
162    run_cmd(&mut cmd)
163}

165pub fn count_delta(po_file: &Path) -> anyhow::Result<u32> {
166    let out = Command::new("msgfmt")
167        .args(["--statistics", "-o", "/dev/null"])
168        .arg(po_file)
169        .env("LANG", "C")
170        .output()?;
171    let text = String::from_utf8_lossy(&out.stderr);
172    let mut total = 0u32;
173    let words: Vec<&str> = text.split_whitespace().collect();
174    for i in 0..words.len().saturating_sub(1) {
175        if let Ok(n) = words[i].parse::<u32>() {
176            let next = words[i + 1];
177            if next.starts_with("fuzzy") || next.starts_with("untranslated") {
178                total += n;
179            }
180        }
181    }
182    Ok(total)
183}
