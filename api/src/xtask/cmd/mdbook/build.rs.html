6pub fn run() -> anyhow::Result<()> {
7    let root = repo_root();
8    require_tool("cargo", "https://rustup.rs")?;
9    ensure_cargo_tool("mdbook", "mdbook")?;
10    ensure_cargo_tool("mdbook-xgettext", "mdbook-i18n-helpers")?;
11    ensure_cargo_tool("mdbook-gettext", "mdbook-i18n-helpers")?;
12    ensure_cargo_tool("mdbook-mermaid", "mdbook-mermaid")?;
13
14    build_refs(&root)?;
15    build_api(&root)?;
16    build_locales(&root)?;
17    assemble(&root)?;
18    println!(
19        "==> Done. Open: {}",
20        book_dir(&root).join("book/index.html").display()
21    );
22    Ok(())
23}

25pub fn build_locales(root: &std::path::Path) -> anyhow::Result<()> {
26    let book = book_dir(root);
27    let entries = locale_entries();
28    println!(
29        "==> Building mdBook for locales: {}",
30        entries
31            .iter()
32            .map(|e| e.code.as_str())
33            .collect::<Vec<_>>()
34            .join(" ")
35    );
36    inject_lang_switcher_locales(&book, &entries)?;
37    let mdbook = mdbook_program()?;
38    for entry in &entries {
39        run_cmd(
40            Command::new(&mdbook)
41                .args(["build", "-d", &format!("book/{}", entry.code)])
42                .env("MDBOOK_BOOK__LANGUAGE", &entry.code)
43                .current_dir(&book),
44        )?;
45    }
46    Ok(())
47}

49/// Render `theme/lang-switcher.js.tpl` into `theme/lang-switcher.js` with the
50/// `LOCALES` array filled from `locales.toml`. The `.js` output is gitignored
51/// (every locale add/remove rewrites it); the `.tpl` source is the tracked
52/// truth. Errors loudly when the template is missing — silently skipping
53/// would let mdBook fail later with a confusing "missing additional-js"
54/// message.
55pub fn inject_lang_switcher_locales(book: &Path, entries: &[LocaleEntry]) -> anyhow::Result<()> {
56    let tpl_path = book.join("theme/lang-switcher.js.tpl");
57    let js_path = book.join("theme/lang-switcher.js");
58    let src = std::fs::read_to_string(&tpl_path).map_err(|e| {
59        anyhow::anyhow!(
60            "lang-switcher.js.tpl missing at {}: {e}. The template is the tracked source of \
61             truth for the locale switcher; do not delete it.",
62            tpl_path.display(),
63        )
64    })?;
65    let locale_lines: String = entries
66        .iter()
67        .map(|e| format!("    {{ code: {:?}, label: {:?} }},", e.code, e.label))
68        .collect::<Vec<_>>()
69        .join("\n");
70    let new_block = format!("const LOCALES = [\n{locale_lines}\n  ];");
71
72    let start = src
73        .find("const LOCALES = [")
74        .ok_or_else(|| anyhow::anyhow!("lang-switcher.js.tpl: LOCALES array not found"))?;
75    let end = src[start..]
76        .find("];")
77        .ok_or_else(|| anyhow::anyhow!("lang-switcher.js.tpl: LOCALES closing ]; not found"))?;
78    let updated = format!("{}{}{}", &src[..start], new_block, &src[start + end + 2..]);
79    std::fs::write(&js_path, updated)?;
80    Ok(())
81}

88pub fn assemble(root: &std::path::Path) -> anyhow::Result<()> {
89    println!("==> Assembling site (rustdoc + locale redirect)");
90    let book = book_dir(root);
91    let api_dest = book.join("book/api");
92    let _ = std::fs::remove_dir_all(&api_dest);
93    copy_dir_all(root.join("target/doc"), &api_dest)?;
94
95    const INDEX_HTML: &str = "\
96<!doctype html>
97<meta charset=\"utf-8\">
98<meta http-equiv=\"refresh\" content=\"0; url=./en/\">
99<link rel=\"canonical\" href=\"./en/\">
100<title>ZeroClaw Docs</title>
101";
102    std::fs::write(book.join("book/index.html"), INDEX_HTML)?;
103    Ok(())
104}
