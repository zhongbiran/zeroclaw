12fn resolve_backend(provider: &str) -> anyhow::Result<Backend> {
13    let cfg = read_provider_config(provider)?;
14    let model = cfg.model
15        .ok_or_else(|| anyhow::anyhow!("No model set for provider '{provider}' — add `model = \"...\"` to [providers.models.{provider}] in config.toml"))?;
16    Ok(Backend {
17        base_url: cfg.base_url,
18        model,
19        api_key: cfg.api_key,
20    })
21}

23pub fn run(
24    locale: Option<&str>,
25    force: bool,
26    provider: Option<&str>,
27    batch: Option<usize>,
28) -> anyhow::Result<()> {
29    let root = repo_root();
30    let locales_dir = fluent_locales_dir(&root);
31    let en_dir = locales_dir.join("en");
32
33    if !en_dir.exists() {
34        anyhow::bail!("English locale dir not found: {}", en_dir.display());
35    }
36
37    let targets: Vec<String> = match locale {
38        Some(l) => vec![l.to_string()],
39        None => locales().into_iter().filter(|l| l != "en").collect(),
40    };
41
42    let provider_name = provider.ok_or_else(|| {
43        anyhow::anyhow!(
44            "--provider <name> is required (configured in [providers.models.<name>] in config.toml)"
45        )
46    })?;
47    let backend = resolve_backend(provider_name)?;
48    let batch_size = batch.unwrap_or(DEFAULT_BATCH_SIZE).max(1);
49
50    for target_locale in &targets {
51        let target_dir = locales_dir.join(target_locale);
52        std::fs::create_dir_all(&target_dir)?;
53
54        for ftl_path in ftl_files_in(&en_dir)? {
55            let filename = ftl_path.file_name().unwrap();
56            let target_ftl = target_dir.join(filename);
57
58            fill_ftl_file(
59                &ftl_path,
60                &target_ftl,
61                target_locale,
62                force,
63                &backend,
64                batch_size,
65            )?;
66        }
67    }
68
69    Ok(())
70}

72fn fill_ftl_file(
73    en_path: &Path,
74    target_path: &PathBuf,
75    locale: &str,
76    force: bool,
77    backend: &Backend,
78    batch_size: usize,
79) -> anyhow::Result<()> {
80    let en_entries = parse_ftl(&std::fs::read_to_string(en_path)?);
81    let mut target_entries: Vec<(String, String)> = if target_path.exists() {
82        parse_ftl(&std::fs::read_to_string(target_path)?)
83    } else {
84        vec![]
85    };
86
87    let existing_keys: std::collections::HashSet<String> =
88        target_entries.iter().map(|(k, _)| k.clone()).collect();
89
90    let to_translate: Vec<(String, String)> = en_entries
91        .iter()
92        .filter(|(key, _)| force || !existing_keys.contains(key))
93        .cloned()
94        .collect();
95
96    if to_translate.is_empty() {
97        println!(
98            "==> {locale}/{}: up to date, skipping AI step",
99            en_path.file_name().unwrap().to_string_lossy()
100        );
101        return Ok(());
102    }
103
104    println!(
105        "==> {locale}/{}: AI-filling {} entries",
106        en_path.file_name().unwrap().to_string_lossy(),
107        to_translate.len()
108    );
109
110    let locale_name = locale_display_name(locale);
111
112    for chunk in to_translate.chunks(batch_size) {
113        let translated = call_api(backend, locale_name, chunk)?;
114
115        // Merge: update existing or append new
116        for (key, value) in translated {
117            if let Some(entry) = target_entries.iter_mut().find(|(k, _)| k == &key) {
118                entry.1 = value;
119            } else {
120                target_entries.push((key, value));
121            }
122        }
123
124        // Write after each batch for incremental safety
125        write_ftl(target_path, &en_entries, &target_entries)?;
126        println!("    wrote {}", target_path.display());
127    }
128
129    Ok(())
130}

132fn call_api(
133    backend: &Backend,
134    locale_name: &str,
135    entries: &[(String, String)],
136) -> anyhow::Result<Vec<(String, String)>> {
137    let input_obj: serde_json::Map<String, serde_json::Value> = entries
138        .iter()
139        .map(|(k, v)| (k.clone(), serde_json::Value::String(v.clone())))
140        .collect();
141
142    let system = format!(
143        "You are a translator. Translate UI strings from English to {locale_name}.\n\
144         Rules:\n\
145         - Return ONLY a JSON object mapping each key to its translated string value.\n\
146         - Do NOT translate: proper nouns, brand names (e.g. ZeroClaw, Anthropic), command names, \
147           technical identifiers, or code examples.\n\
148         - Preserve exactly: Fluent placeholders ({{ variable }}), special syntax, and escape sequences.\n\
149         - Do NOT wrap output in markdown code fences."
150    );
151
152    let rt = tokio::runtime::Builder::new_current_thread()
153        .enable_all()
154        .build()?;
155
156    let (status, text) = rt.block_on(async {
157        let client = reqwest::Client::new();
158        let user_content = serde_json::to_string(&input_obj)?;
159
160        let body = serde_json::json!({
161            "model": backend.model,
162            "messages": [
163                {"role": "system", "content": system},
164                {"role": "user", "content": user_content}
165            ],
166            "reasoning_effort": "none"
167        });
168        let mut req = client
169            .post(format!("{}/v1/chat/completions", backend.base_url))
170            .json(&body);
171        if let Some(key) = &backend.api_key {
172            req = req.header("Authorization", format!("Bearer {key}"));
173        }
174        let response = req.send().await?;
175
176        let status = response.status();
177        let text = response.text().await?;
178        Ok::<_, anyhow::Error>((status, text))
179    })?;
180
181    if !status.is_success() {
182        anyhow::bail!("API error {status}: {text}");
183    }
184
185    let parsed: serde_json::Value = serde_json::from_str(&text)
186        .map_err(|e| anyhow::anyhow!("Failed to parse API response: {e}\n{text}"))?;
187
188    let content = parsed["choices"][0]["message"]["content"]
189        .as_str()
190        .ok_or_else(|| anyhow::anyhow!("No content in response: {text}"))?;
191
192    // Strip markdown code fences and stray inline backticks. Models sometimes
193    // wrap a single-line JSON response in `` `…` `` instead of a fenced block,
194    // so handle both shapes.
195    let json_str = content
196        .trim()
197        .trim_start_matches("```json")
198        .trim_start_matches("```")
199        .trim_end_matches("```")
200        .trim()
201        .trim_start_matches('`')
202        .trim_end_matches('`')
203        .trim();
204
205    let translations: serde_json::Value = serde_json::from_str(json_str)
206        .map_err(|e| anyhow::anyhow!("Failed to parse translation JSON: {e}\n{json_str}"))?;
207
208    let obj = translations
209        .as_object()
210        .ok_or_else(|| anyhow::anyhow!("Expected JSON object in translation response"))?;
211
212    let mut result = vec![];
213    for (key, value) in obj {
214        if let Some(translated) = value.as_str() {
215            result.push((key.clone(), translated.to_string()));
216        }
217    }
218
219    Ok(result)
220}

222/// Parse an FTL file into an ordered list of (key, value) pairs.
223/// Handles both single-line (`key = value`) and multi-line (indented continuation) FTL values.
224fn parse_ftl(src: &str) -> Vec<(String, String)> {
225    let mut entries: Vec<(String, String)> = vec![];
226    let mut current_key: Option<String> = None;
227    let mut current_lines: Vec<String> = vec![];
228
229    for line in src.lines() {
230        // Continuation line of a multi-line value (4-space or tab indent)
231        if (line.starts_with("    ") || line.starts_with('\t')) && current_key.is_some() {
232            current_lines.push(line.trim().to_string());
233            continue;
234        }
235
236        let trimmed = line.trim();
237
238        // Blank line: part of multi-line value if inside one, otherwise ignored
239        if trimmed.is_empty() {
240            if current_key.is_some() {
241                current_lines.push(String::new());
242            }
243            continue;
244        }
245
246        // Comment or term: flush pending entry and skip
247        if trimmed.starts_with('#') || trimmed.starts_with('-') {
248            flush_entry(&mut entries, &mut current_key, &mut current_lines);
249            continue;
250        }
251
252        // New `key = value` or `key =` (multi-line) line
253        let parsed_kv = if let Some((k, v)) = trimmed.split_once(" = ") {
254            Some((k.trim(), v.trim()))
255        } else {
256            trimmed.strip_suffix(" =").map(|k| (k.trim(), ""))
257        };
258        if let Some((key, value)) = parsed_kv {
259            flush_entry(&mut entries, &mut current_key, &mut current_lines);
260            current_key = Some(key.to_string());
261            if !value.is_empty() {
262                current_lines.push(value.to_string());
263            }
264        }
265    }
266
267    flush_entry(&mut entries, &mut current_key, &mut current_lines);
268    entries
269}

271fn flush_entry(
272    entries: &mut Vec<(String, String)>,
273    key: &mut Option<String>,
274    lines: &mut Vec<String>,
275) {
276    if let Some(k) = key.take() {
277        while lines.last().is_some_and(|l| l.is_empty()) {
278            lines.pop();
279        }
280        let value = lines.join("\n");
281        if !value.is_empty() {
282            entries.push((k, value));
283        }
284        lines.clear();
285    }
286}

288fn write_ftl_entry(out: &mut String, key: &str, value: &str) {
289    if value.contains('\n') {
290        out.push_str(&format!("{key} =\n"));
291        for line in value.lines() {
292            if line.is_empty() {
293                out.push('\n');
294            } else {
295                out.push_str(&format!("    {line}\n"));
296            }
297        }
298    } else {
299        out.push_str(&format!("{key} = {value}\n"));
300    }
301}

303/// Write a locale FTL file, using en_entries to preserve key order and comments.
304/// Keys not in en_entries (locale-only additions) are appended at the end.
305fn write_ftl(
306    path: &Path,
307    en_entries: &[(String, String)],
308    locale_entries: &[(String, String)],
309) -> anyhow::Result<()> {
310    let locale_map: std::collections::HashMap<&str, &str> = locale_entries
311        .iter()
312        .map(|(k, v)| (k.as_str(), v.as_str()))
313        .collect();
314
315    let mut out = String::new();
316
317    for (key, _en_value) in en_entries {
318        if let Some(translated) = locale_map.get(key.as_str()) {
319            write_ftl_entry(&mut out, key, translated);
320        }
321        // Keys not yet translated are omitted (runtime falls back to English)
322    }
323
324    // Append any locale-only keys not in en (shouldn't normally exist, but be safe)
325    let en_set: std::collections::HashSet<&str> =
326        en_entries.iter().map(|(k, _)| k.as_str()).collect();
327    for (key, value) in locale_entries {
328        if !en_set.contains(key.as_str()) {
329            write_ftl_entry(&mut out, key, value);
330        }
331    }
332
333    if let Some(parent) = path.parent() {
334        std::fs::create_dir_all(parent)?;
335    }
336    std::fs::write(path, out)?;
337    Ok(())
338}
