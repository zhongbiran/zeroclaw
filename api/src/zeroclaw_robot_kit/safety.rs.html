1//! Safety System - Collision avoidance, watchdogs, and emergency stops
2//!
3//! This module runs INDEPENDENTLY of the AI brain to ensure safety
4//! even if the LLM makes bad decisions or hangs.
5//!
6//! ## Safety Layers
7//!
8//! 1. **Pre-move checks** - Verify path clear before any movement
9//! 2. **Active monitoring** - Continuous sensor polling during movement
10//! 3. **Reactive stops** - Instant halt on obstacle detection
11//! 4. **Watchdog timer** - Auto-stop if no commands for N seconds
12//! 5. **Hardware E-stop** - Physical button overrides everything
13//!
14//! ## Design Philosophy
15//!
16//! The AI can REQUEST movement, but the safety system ALLOWS it.
17//! Safety always wins.
18

28/// Safety events broadcast to all listeners
29#[derive(Debug, Clone)]
30pub enum SafetyEvent {
31    /// Obstacle detected, movement blocked
32    ObstacleDetected { distance: f64, angle: u16 },
33    /// Emergency stop triggered
34    EmergencyStop { reason: String },
35    /// Watchdog timeout - no activity
36    WatchdogTimeout,
37    /// Movement approved
38    MovementApproved,
39    /// Movement denied with reason
40    MovementDenied { reason: String },
41    /// Bump sensor triggered
42    BumpDetected { sensor: String },
43    /// System recovered, ready to move again
44    Recovered,
45}
46
47/// Real-time safety state
48pub struct SafetyState {
49    /// Is it safe to move?
50    pub can_move: AtomicBool,
51    /// Emergency stop active?
52    pub estop_active: AtomicBool,
53    /// Last movement command timestamp (ms since epoch)
54    pub last_command_ms: AtomicU64,
55    /// Current minimum distance to obstacle
56    pub min_obstacle_distance: RwLock<f64>,
57    /// Reason movement is blocked (if any)
58    pub block_reason: RwLock<Option<String>>,
59    /// Speed multiplier based on proximity (0.0 - 1.0)
60    pub speed_limit: RwLock<f64>,
61}
62
63impl Default for SafetyState {
64    fn default() -> Self {
65        Self {
66            can_move: AtomicBool::new(true),
67            estop_active: AtomicBool::new(false),
68            last_command_ms: AtomicU64::new(0),
69            min_obstacle_distance: RwLock::new(999.0),
70            block_reason: RwLock::new(None),
71            speed_limit: RwLock::new(1.0),
72        }
73    }
74}

76/// Safety monitor - runs as background task
77pub struct SafetyMonitor {
78    config: SafetyConfig,
79    state: Arc<SafetyState>,
80    event_tx: broadcast::Sender<SafetyEvent>,
81    shutdown: AtomicBool,
82}
83
84impl SafetyMonitor {
85    pub fn new(config: SafetyConfig) -> (Self, broadcast::Receiver<SafetyEvent>) {
86        let (event_tx, event_rx) = broadcast::channel(64);
87        let monitor = Self {
88            config,
89            state: Arc::new(SafetyState::default()),
90            event_tx,
91            shutdown: AtomicBool::new(false),
92        };
93        (monitor, event_rx)
94    }
95
96    pub fn state(&self) -> Arc<SafetyState> {
97        self.state.clone()
98    }
99
100    pub fn subscribe(&self) -> broadcast::Receiver<SafetyEvent> {
101        self.event_tx.subscribe()
102    }
103
104    /// Check if movement is currently allowed
105    pub async fn can_move(&self) -> bool {
106        if self.state.estop_active.load(Ordering::SeqCst) {
107            return false;
108        }
109        self.state.can_move.load(Ordering::SeqCst)
110    }
111
112    /// Get current speed limit multiplier (0.0 - 1.0)
113    pub async fn speed_limit(&self) -> f64 {
114        *self.state.speed_limit.read().await
115    }
116
117    /// Request permission to move - returns allowed speed multiplier or error
118    pub async fn request_movement(&self, direction: &str, distance: f64) -> Result<f64, String> {
119        // Check E-stop
120        if self.state.estop_active.load(Ordering::SeqCst) {
121            return Err("Emergency stop active".to_string());
122        }
123
124        // Check general movement permission
125        if !self.state.can_move.load(Ordering::SeqCst) {
126            let reason = self.state.block_reason.read().await;
127            return Err(reason
128                .clone()
129                .unwrap_or_else(|| "Movement blocked".to_string()));
130        }
131
132        // Check obstacle distance in movement direction
133        let min_dist = *self.state.min_obstacle_distance.read().await;
134        if min_dist < self.config.min_obstacle_distance {
135            let msg = format!(
136                "Obstacle too close: {:.2}m (min: {:.2}m)",
137                min_dist, self.config.min_obstacle_distance
138            );
139            let _ = self.event_tx.send(SafetyEvent::MovementDenied {
140                reason: msg.clone(),
141            });
142            return Err(msg);
143        }
144
145        // Check if requested distance would hit obstacle
146        if distance > min_dist - self.config.min_obstacle_distance {
147            let safe_distance = (min_dist - self.config.min_obstacle_distance).max(0.0);
148            if safe_distance < 0.1 {
149                return Err(format!(
150                    "Cannot move {}: obstacle at {:.2}m",
151                    direction, min_dist
152                ));
153            }
154            // Allow reduced distance
155            tracing::warn!(
156                "Reducing {} distance from {:.2}m to {:.2}m due to obstacle",
157                direction,
158                distance,
159                safe_distance
160            );
161        }
162
163        // Update last command time
164        let now_ms = std::time::SystemTime::now()
165            .duration_since(std::time::UNIX_EPOCH)
166            .unwrap()
167            .as_millis() as u64;
168        self.state.last_command_ms.store(now_ms, Ordering::SeqCst);
169
170        // Calculate speed limit based on proximity
171        let speed_mult = self.calculate_speed_limit(min_dist).await;
172
173        let _ = self.event_tx.send(SafetyEvent::MovementApproved);
174        Ok(speed_mult)
175    }
176
177    /// Calculate safe speed based on obstacle proximity
178    async fn calculate_speed_limit(&self, obstacle_distance: f64) -> f64 {
179        let min_dist = self.config.min_obstacle_distance;
180        let slow_zone = min_dist * 3.0; // Start slowing at 3x minimum distance
181
182        let limit = if obstacle_distance >= slow_zone {
183            1.0 // Full speed
184        } else if obstacle_distance <= min_dist {
185            0.0 // Stop
186        } else {
187            // Linear interpolation between stop and full speed
188            (obstacle_distance - min_dist) / (slow_zone - min_dist)
189        };
190
191        *self.state.speed_limit.write().await = limit;
192        limit
193    }
194
195    /// Trigger emergency stop
196    pub async fn emergency_stop(&self, reason: &str) {
197        tracing::error!("EMERGENCY STOP: {}", reason);
198        self.state.estop_active.store(true, Ordering::SeqCst);
199        self.state.can_move.store(false, Ordering::SeqCst);
200        *self.state.block_reason.write().await = Some(reason.to_string());
201
202        let _ = self.event_tx.send(SafetyEvent::EmergencyStop {
203            reason: reason.to_string(),
204        });
205    }
206
207    /// Reset emergency stop (requires explicit action)
208    pub async fn reset_estop(&self) {
209        tracing::info!("E-STOP RESET");
210        self.state.estop_active.store(false, Ordering::SeqCst);
211        self.state.can_move.store(true, Ordering::SeqCst);
212        *self.state.block_reason.write().await = None;
213
214        let _ = self.event_tx.send(SafetyEvent::Recovered);
215    }
216
217    /// Update obstacle distance (call from sensor loop)
218    pub async fn update_obstacle_distance(&self, distance: f64, angle: u16) {
219        // Update minimum distance tracking
220        {
221            let mut min_dist = self.state.min_obstacle_distance.write().await;
222            // Always update to current reading (not just if closer)
223            *min_dist = distance;
224        }
225
226        // Recalculate speed limit based on new distance
227        self.calculate_speed_limit(distance).await;
228
229        // Check if too close
230        if distance < self.config.min_obstacle_distance {
231            self.state.can_move.store(false, Ordering::SeqCst);
232            *self.state.block_reason.write().await =
233                Some(format!("Obstacle at {:.2}m ({}°)", distance, angle));
234
235            let _ = self
236                .event_tx
237                .send(SafetyEvent::ObstacleDetected { distance, angle });
238        } else if !self.state.estop_active.load(Ordering::SeqCst) {
239            // Clear block if obstacle moved away and no E-stop
240            self.state.can_move.store(true, Ordering::SeqCst);
241            *self.state.block_reason.write().await = None;
242        }
243    }
244
245    /// Report bump sensor triggered
246    pub async fn bump_detected(&self, sensor: &str) {
247        tracing::warn!("BUMP DETECTED: {}", sensor);
248
249        // Immediate stop
250        self.state.can_move.store(false, Ordering::SeqCst);
251        *self.state.block_reason.write().await = Some(format!("Bump: {}", sensor));
252
253        let _ = self.event_tx.send(SafetyEvent::BumpDetected {
254            sensor: sensor.to_string(),
255        });
256
257        // Auto-recover after brief pause (robot should back up)
258        tokio::spawn({
259            let state = self.state.clone();
260            let event_tx = self.event_tx.clone();
261            async move {
262                tokio::time::sleep(Duration::from_secs(2)).await;
263                if !state.estop_active.load(Ordering::SeqCst) {
264                    state.can_move.store(true, Ordering::SeqCst);
265                    *state.block_reason.write().await = None;
266                    let _ = event_tx.send(SafetyEvent::Recovered);
267                }
268            }
269        });
270    }
271
272    /// Shutdown the monitor
273    pub fn shutdown(&self) {
274        self.shutdown.store(true, Ordering::SeqCst);
275    }

277    /// Run the safety monitor loop (call in background task)
278    pub async fn run(&self, mut sensor_rx: tokio::sync::mpsc::Receiver<SensorReading>) {
279        let watchdog_timeout = Duration::from_secs(self.config.max_drive_duration);
280        let mut last_sensor_update = Instant::now();
281
282        while !self.shutdown.load(Ordering::SeqCst) {
283            tokio::select! {
284                // Process sensor readings
285                Some(reading) = sensor_rx.recv() => {
286                    last_sensor_update = Instant::now();
287                    match reading {
288                        SensorReading::Lidar { distance, angle } => {
289                            self.update_obstacle_distance(distance, angle).await;
290                        }
291                        SensorReading::Bump { sensor } => {
292                            self.bump_detected(&sensor).await;
293                        }
294                        SensorReading::Estop { pressed } => {
295                            if pressed {
296                                self.emergency_stop("Hardware E-stop pressed").await;
297                            }
298                        }
299                    }
300                }
301
302                // Watchdog check every second
303                _ = tokio::time::sleep(Duration::from_secs(1)) => {
304                    // Check for sensor timeout
305                    if last_sensor_update.elapsed() > Duration::from_secs(5) {
306                        tracing::warn!("Sensor data stale - blocking movement");
307                        self.state.can_move.store(false, Ordering::SeqCst);
308                        *self.state.block_reason.write().await =
309                            Some("Sensor data stale".to_string());
310                    }
311
312                    // Check watchdog (auto-stop if no commands)
313                    let last_cmd_ms = self.state.last_command_ms.load(Ordering::SeqCst);
314                    if last_cmd_ms > 0 {
315                        let now_ms = std::time::SystemTime::now()
316                            .duration_since(std::time::UNIX_EPOCH)
317                            .unwrap()
318                            .as_millis() as u64;
319
320                        let elapsed = Duration::from_millis(now_ms - last_cmd_ms);
321                        if elapsed > watchdog_timeout {
322                            tracing::info!("Watchdog timeout - no commands for {:?}", elapsed);
323                            let _ = self.event_tx.send(SafetyEvent::WatchdogTimeout);
324                            // Don't block movement, just notify
325                        }
326                    }
327                }
328            }
329        }
330    }
331}

341/// Safety-aware drive wrapper
342/// Wraps the drive tool to enforce safety limits
343pub struct SafeDrive {
344    inner_drive: Arc<dyn crate::traits::Tool>,
345    safety: Arc<SafetyMonitor>,
346}
347
348impl SafeDrive {
349    pub fn new(drive: Arc<dyn crate::traits::Tool>, safety: Arc<SafetyMonitor>) -> Self {
350        Self {
351            inner_drive: drive,
352            safety,
353        }
354    }
355}
356
357#[async_trait::async_trait]
358impl crate::traits::Tool for SafeDrive {
359    fn name(&self) -> &str {
360        "drive"
361    }
362
363    fn description(&self) -> &str {
364        "Move the robot (with safety limits enforced)"
365    }
366
367    fn parameters_schema(&self) -> serde_json::Value {
368        self.inner_drive.parameters_schema()
369    }
370
371    async fn execute(&self, args: serde_json::Value) -> Result<ToolResult> {
372        // ToolResult imported at top of file
373
374        let action = args["action"].as_str().unwrap_or("unknown");
375        let distance = args["distance"].as_f64().unwrap_or(0.5);
376
377        // Always allow stop
378        if action == "stop" {
379            return self.inner_drive.execute(args).await;
380        }
381
382        // Request permission from safety system
383        match self.safety.request_movement(action, distance).await {
384            Ok(speed_mult) => {
385                // Modify speed in args
386                let mut modified_args = args.clone();
387                let original_speed = args["speed"].as_f64().unwrap_or(0.5);
388                modified_args["speed"] = serde_json::json!(original_speed * speed_mult);
389
390                if speed_mult < 1.0 {
391                    tracing::info!(
392                        "Safety: Reducing speed to {:.0}% due to obstacle proximity",
393                        speed_mult * 100.0
394                    );
395                }
396
397                self.inner_drive.execute(modified_args).await
398            }
399            Err(reason) => Ok(ToolResult {
400                success: false,
401                output: String::new(),
402                error: Some(format!("Safety blocked movement: {}", reason)),
403            }),
404        }
405    }
406}

408/// Pre-flight safety check before any operation
409pub async fn preflight_check(config: &RobotConfig) -> Result<Vec<String>> {
410    let mut warnings = Vec::new();
411
412    // Check safety config
413    if config.safety.min_obstacle_distance < 0.1 {
414        warnings.push("WARNING: min_obstacle_distance < 0.1m is dangerously low".to_string());
415    }
416
417    if config.safety.max_drive_duration > 60 {
418        warnings.push("WARNING: max_drive_duration > 60s may allow runaway".to_string());
419    }
420
421    if config.drive.max_speed > 1.0 {
422        warnings.push("WARNING: max_speed > 1.0 m/s is very fast for indoor use".to_string());
423    }
424
425    if config.safety.estop_pin.is_none() {
426        warnings.push(
427            "WARNING: No E-stop pin configured. Recommend wiring a hardware stop button."
428                .to_string(),
429        );
430    }
431
432    // Check for sensor availability
433    if config.sensors.lidar_type == "mock" {
434        warnings.push("NOTICE: LIDAR in mock mode - no real obstacle detection".to_string());
435    }
436
437    Ok(warnings)
438}
