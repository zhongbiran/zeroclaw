36/// List configured boards from config (no connection yet).
37pub fn list_configured_boards(config: &PeripheralsConfig) -> Vec<&PeripheralBoardConfig> {
38    if !config.enabled {
39        return Vec::new();
40    }
41    config.boards.iter().collect()
42}

44/// Create and connect peripherals from config, returning their tools.
45/// Returns empty vec if peripherals disabled or hardware feature off.
46#[cfg(feature = "hardware")]
47pub async fn create_peripheral_tools(config: &PeripheralsConfig) -> Result<Vec<Box<dyn Tool>>> {
48    if !config.enabled || config.boards.is_empty() {
49        return Ok(Vec::new());
50    }
51
52    let mut tools: Vec<Box<dyn Tool>> = Vec::new();
53    let mut serial_transports: Vec<(String, std::sync::Arc<serial::SerialTransport>)> = Vec::new();
54
55    for board in &config.boards {
56        // Arduino Uno Q: Bridge transport (socket to local Bridge app)
57        if board.transport == "bridge" && (board.board == "arduino-uno-q" || board.board == "uno-q")
58        {
59            tools.push(Box::new(uno_q_bridge::UnoQGpioReadTool));
60            tools.push(Box::new(uno_q_bridge::UnoQGpioWriteTool));
61            tracing::info!(board = %board.board, "Uno Q Bridge GPIO tools added");
62            continue;
63        }
64
65        // Native transport: RPi GPIO (Linux only)
66        #[cfg(all(feature = "peripheral-rpi", target_os = "linux"))]
67        if board.transport == "native"
68            && (board.board == "rpi-gpio" || board.board == "raspberry-pi")
69        {
70            match rpi::RpiGpioPeripheral::connect_from_config(board).await {
71                Ok(peripheral) => {
72                    tools.extend(peripheral.tools());
73                    tracing::info!(board = %board.board, "RPi GPIO peripheral connected");
74                }
75                Err(e) => {
76                    tracing::warn!("Failed to connect RPi GPIO {}: {}", board.board, e);
77                }
78            }
79            continue;
80        }
81
82        // Serial transport (STM32, ESP32, Arduino, etc.)
83        if board.transport != "serial" {
84            continue;
85        }
86        if board.path.is_none() {
87            tracing::warn!("Skipping serial board {}: no path", board.board);
88            continue;
89        }
90
91        match serial::SerialPeripheral::connect(board).await {
92            Ok(peripheral) => {
93                let mut p = peripheral;
94                if p.connect().await.is_err() {
95                    tracing::warn!("Peripheral {} connect warning (continuing)", p.name());
96                }
97                serial_transports.push((board.board.clone(), p.transport()));
98                tools.extend(p.tools());
99                if board.board == "arduino-uno"
100                    && let Some(ref path) = board.path
101                {
102                    tools.push(Box::new(arduino_upload::ArduinoUploadTool::new(
103                        path.clone(),
104                    )));
105                    tracing::info!("Arduino upload tool added (port: {})", path);
106                }
107                tracing::info!(board = %board.board, "Serial peripheral connected");
108            }
109            Err(e) => {
110                tracing::warn!("Failed to connect {}: {}", board.board, e);
111            }
112        }
113    }
114
115    // Phase B: Add hardware tools when any boards configured
116    if !tools.is_empty() {
117        let board_names: Vec<String> = config.boards.iter().map(|b| b.board.clone()).collect();
118        tools.push(Box::new(HardwareMemoryMapTool::new(board_names.clone())));
119        tools.push(Box::new(
120            zeroclaw_tools::hardware_board_info::HardwareBoardInfoTool::new(board_names.clone()),
121        ));
122        tools.push(Box::new(
123            zeroclaw_tools::hardware_memory_read::HardwareMemoryReadTool::new(board_names),
124        ));
125    }
126
127    // Phase C: Add hardware_capabilities tool when any serial boards
128    if !serial_transports.is_empty() {
129        tools.push(Box::new(capabilities_tool::HardwareCapabilitiesTool::new(
130            serial_transports,
131        )));
132    }
133
134    Ok(tools)
135}

143/// Create probe-rs / static board info tools (hardware_board_info, hardware_memory_map,
144/// hardware_memory_read). These use USB/probe-rs or static datasheet data — they never
145/// open a serial port, so they are safe to register regardless of the `hardware` feature.
146#[cfg(feature = "hardware")]
147pub fn create_board_info_tools(config: &PeripheralsConfig) -> Vec<Box<dyn Tool>> {
148    if !config.enabled || config.boards.is_empty() {
149        return Vec::new();
150    }
151    let board_names: Vec<String> = config.boards.iter().map(|b| b.board.clone()).collect();
152    vec![
153        Box::new(
154            zeroclaw_tools::hardware_memory_map::HardwareMemoryMapTool::new(board_names.clone()),
155        ),
156        Box::new(
157            zeroclaw_tools::hardware_board_info::HardwareBoardInfoTool::new(board_names.clone()),
158        ),
159        Box::new(zeroclaw_tools::hardware_memory_read::HardwareMemoryReadTool::new(board_names)),
160    ]
161}

168#[cfg(test)]
169mod tests {
170    use super::*;
171    use zeroclaw_config::schema::{PeripheralBoardConfig, PeripheralsConfig};

173    #[test]
174    fn list_configured_boards_when_disabled_returns_empty() {
175        let config = PeripheralsConfig {
176            enabled: false,
177            boards: vec![PeripheralBoardConfig {
178                board: "nucleo-f401re".into(),
179                transport: "serial".into(),
180                path: Some("/dev/ttyACM0".into()),
181                baud: 115_200,
182            }],
183            datasheet_dir: None,
184        };
185        let result = list_configured_boards(&config);
186        assert!(
187            result.is_empty(),
188            "disabled peripherals should return no boards"
189        );
190    }
191
192    #[test]
193    fn list_configured_boards_when_enabled_with_boards() {
194        let config = PeripheralsConfig {
195            enabled: true,
196            boards: vec![
197                PeripheralBoardConfig {
198                    board: "nucleo-f401re".into(),
199                    transport: "serial".into(),
200                    path: Some("/dev/ttyACM0".into()),
201                    baud: 115_200,
202                },
203                PeripheralBoardConfig {
204                    board: "rpi-gpio".into(),
205                    transport: "native".into(),
206                    path: None,
207                    baud: 115_200,
208                },
209            ],
210            datasheet_dir: None,
211        };
212        let result = list_configured_boards(&config);
213        assert_eq!(result.len(), 2);
214        assert_eq!(result[0].board, "nucleo-f401re");
215        assert_eq!(result[1].board, "rpi-gpio");
216    }
217
