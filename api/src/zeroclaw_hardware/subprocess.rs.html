1//! SubprocessTool — wraps any external binary as a [`Tool`].
2//!
3//! Plugins do not need to be written in Rust. Any executable that follows the
4//! ZeroClaw subprocess protocol is a valid tool:
5//!
6//! **Protocol (stdin/stdout, one line each):**
7//! ```text
8//! Host → binary stdin:  {"device":"pico0","pin":5}\n
9//! Binary → stdout:      {"success":true,"output":"done","error":null}\n
10//! ```
11//!
12//! Error protocol:
13//! - **Timeout (10 s)** — process is killed; `ToolResult::error` contains timeout message.
14//! - **Non-zero exit** — process is killed; `ToolResult::error` contains stderr.
15//! - **Empty / unparseable stdout** — `ToolResult::error` describes the failure.
16//!
17//! The schema advertised to the LLM is auto-generated from [`ToolManifest::parameters`].
18

28/// Subprocess timeout — kill the child process after this many seconds.
29const SUBPROCESS_TIMEOUT_SECS: u64 = 10;
30
31/// Timeout for waiting on child process exit after stdout has been read.
32/// Prevents a hung cleanup phase from blocking indefinitely.
33const PROCESS_EXIT_TIMEOUT_SECS: u64 = 5;
34
35/// A tool backed by an external subprocess.
36///
37/// The binary receives the LLM-supplied JSON arguments on stdin (one line,
38/// `\n`-terminated) and must write a single `ToolResult`-compatible JSON
39/// object to stdout before exiting.
40pub struct SubprocessTool {
41    /// Parsed plugin manifest (tool metadata + parameter definitions).
42    manifest: ToolManifest,
43    /// Resolved absolute path to the entry-point binary.
44    binary_path: PathBuf,
45}
46
47impl SubprocessTool {
48    /// Create a new `SubprocessTool` from a manifest and resolved binary path.
49    pub fn new(manifest: ToolManifest, binary_path: PathBuf) -> Self {
50        Self {
51            manifest,
52            binary_path,
53        }
54    }
55
56    /// Build JSON Schema `properties` and `required` arrays from the manifest.
57    fn build_schema_properties(
58        &self,
59    ) -> (
60        serde_json::Map<String, serde_json::Value>,
61        Vec<serde_json::Value>,
62    ) {
63        let mut properties = serde_json::Map::new();
64        let mut required = Vec::new();
65
66        for param in &self.manifest.parameters {
67            let mut prop = json!({
68                "type": param.r#type,
69                "description": param.description,
70            });
71
72            if let Some(default) = &param.default {
73                prop["default"] = default.clone();
74            }
75
76            properties.insert(param.name.clone(), prop);
77
78            if param.required {
79                required.push(serde_json::Value::String(param.name.clone()));
80            }
81        }
82
83        (properties, required)
84    }
85}

87#[async_trait]
88impl Tool for SubprocessTool {
89    fn name(&self) -> &str {
90        &self.manifest.tool.name
91    }
92
93    fn description(&self) -> &str {
94        &self.manifest.tool.description
95    }
96
97    /// JSON Schema Draft 7 — auto-generated from `manifest.parameters`.
98    fn parameters_schema(&self) -> serde_json::Value {
99        let (properties, required) = self.build_schema_properties();
100        json!({
101            "type": "object",
102            "properties": properties,
103            "required": required,
104        })
105    }
106
107    /// Spawn the binary, write args to stdin, read `ToolResult` from stdout.
108    ///
109    /// Steps:
110    /// 1. Serialize `args` to a JSON string.
111    /// 2. Spawn `binary_path` with piped stdin/stdout/stderr.
112    /// 3. Write `<json>\n` to child stdin; close stdin (signal EOF).
113    /// 4. Read one line from child stdout (10 s timeout).
114    /// 5. Kill the child process.
115    /// 6. Deserialize the line to `ToolResult`.
116    /// 7. On timeout → return error `ToolResult`; on empty/bad output → error.
117    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
118        let args_json = serde_json::to_string(&args)
119            .map_err(|e| anyhow::anyhow!("failed to serialise args: {}", e))?;
120
121        // Spawn child process.
122        let mut child = Command::new(&self.binary_path)
123            .stdin(std::process::Stdio::piped())
124            .stdout(std::process::Stdio::piped())
125            .stderr(std::process::Stdio::piped())
126            .spawn()
127            .map_err(|e| {
128                anyhow::anyhow!(
129                    "failed to spawn plugin '{}' at {}: {}",
130                    self.manifest.tool.name,
131                    self.binary_path.display(),
132                    e
133                )
134            })?;
135
136        // Write JSON args + newline to stdin, then drop stdin to signal EOF.
137        // BrokenPipe is tolerated — the child may exit before reading stdin
138        // (e.g. tools that only use command-line args or produce fixed output).
139        if let Some(mut stdin) = child.stdin.take() {
140            let write_result = async {
141                stdin.write_all(args_json.as_bytes()).await?;
142                stdin.write_all(b"\n").await?;
143                Ok::<(), std::io::Error>(())
144            }
145            .await;
146            if let Err(e) = write_result
147                && e.kind() != std::io::ErrorKind::BrokenPipe
148            {
149                let _ = child.kill().await;
150                return Err(anyhow::anyhow!(
151                    "failed to write args to plugin '{}' stdin: {}",
152                    self.manifest.tool.name,
153                    e
154                ));
155            }
156            // stdin dropped here → child receives EOF
157        }
158
159        // Take stdout and stderr handles before we move `child`.
160        let stdout_handle = child.stdout.take();
161        let stderr_handle = child.stderr.take();
162
163        // Read one line from stdout with a hard timeout.
164        let read_result = match stdout_handle {
165            None => {
166                // No stdout — kill and error.
167                let _ = child.kill().await;
168                return Ok(ToolResult {
169                    success: false,
170                    output: String::new(),
171                    error: Some(format!(
172                        "plugin '{}': could not attach stdout pipe",
173                        self.manifest.tool.name
174                    )),
175                });
176            }
177            Some(stdout) => {
178                let mut reader = BufReader::new(stdout);
179                let mut line = String::new();
180                timeout(
181                    Duration::from_secs(SUBPROCESS_TIMEOUT_SECS),
182                    reader.read_line(&mut line),
183                )
184                .await
185                .map(|inner| inner.map(|_| line))
186            }
187        };
188
189        match read_result {
190            // ── Timeout ────────────────────────────────────────────────────
191            // The read deadline elapsed — force-kill the plugin and collect
192            // any stderr it emitted before dying.
193            Err(_elapsed) => {
194                let _ = child.kill().await;
195                let _ = child.wait().await;
196                let stderr_msg = collect_stderr(stderr_handle).await;
197                Ok(ToolResult {
198                    success: false,
199                    output: String::new(),
200                    error: Some(format!(
201                        "plugin '{}' timed out after {}s{}",
202                        self.manifest.tool.name,
203                        SUBPROCESS_TIMEOUT_SECS,
204                        if stderr_msg.is_empty() {
205                            String::new()
206                        } else {
207                            format!("; stderr: {}", stderr_msg)
208                        }
209                    )),
210                })
211            }
212
213            // ── I/O error reading stdout ───────────────────────────────────
214            Ok(Err(io_err)) => {
215                let _ = child.kill().await;
216                let _ = child.wait().await;
217                let stderr_msg = collect_stderr(stderr_handle).await;
218                Ok(ToolResult {
219                    success: false,
220                    output: String::new(),
221                    error: Some(format!(
222                        "plugin '{}': I/O error reading stdout: {}{}",
223                        self.manifest.tool.name,
224                        io_err,
225                        if stderr_msg.is_empty() {
226                            String::new()
227                        } else {
228                            format!("; stderr: {}", stderr_msg)
229                        }
230                    )),
231                })
232            }
233
234            // ── Got a line ────────────────────────────────────────────────
235            // Let the process finish naturally — plugins that write their
236            // result and then do cleanup should not be interrupted.
237            Ok(Ok(line)) => {
238                let child_status =
239                    timeout(Duration::from_secs(PROCESS_EXIT_TIMEOUT_SECS), child.wait())
240                        .await
241                        .ok()
242                        .and_then(|r| r.ok());
243                let stderr_msg = collect_stderr(stderr_handle).await;
244                let line = line.trim();
245
246                if line.is_empty() {
247                    return Ok(ToolResult {
248                        success: false,
249                        output: String::new(),
250                        error: Some(format!(
251                            "plugin '{}': empty stdout{}",
252                            self.manifest.tool.name,
253                            if stderr_msg.is_empty() {
254                                String::new()
255                            } else {
256                                format!("; stderr: {}", stderr_msg)
257                            }
258                        )),
259                    });
260                }
261
262                match serde_json::from_str::<ToolResult>(line) {
263                    Ok(result) => {
264                        // Non-zero exit overrides a parsed result: the plugin
265                        // signalled failure even if it wrote a success line.
266                        if let Some(status) = child_status
267                            && !status.success()
268                        {
269                            return Ok(ToolResult {
270                                success: false,
271                                output: String::new(),
272                                error: Some(format!(
273                                    "plugin '{}' exited with {}{}",
274                                    self.manifest.tool.name,
275                                    status,
276                                    if stderr_msg.is_empty() {
277                                        String::new()
278                                    } else {
279                                        format!("; stderr: {}", stderr_msg)
280                                    }
281                                )),
282                            });
283                        }
284                        Ok(result)
285                    }
286                    Err(parse_err) => Ok(ToolResult {
287                        success: false,
288                        output: String::new(),
289                        error: Some(format!(
290                            "plugin '{}': failed to parse output as ToolResult: {} (got: {:?})",
291                            self.manifest.tool.name,
292                            parse_err,
293                            // Truncate oversized output in the error message.
294                            // Use char-based truncation to avoid panic on multi-byte UTF-8.
295                            if line.chars().count() > 200 {
296                                let truncated: String = line.chars().take(200).collect();
297                                format!("{}...", truncated)
298                            } else {
299                                line.to_string()
300                            }
301                        )),
302                    }),
303                }
304            }
305        }
306    }
307}

309/// Collect up to 512 bytes from an optional stderr handle.
310/// Used to enrich error messages when a plugin writes nothing to stdout.
311async fn collect_stderr(handle: Option<tokio::process::ChildStderr>) -> String {
312    use tokio::io::AsyncReadExt;
313    let Some(mut stderr) = handle else {
314        return String::new();
315    };
316    let mut buf = vec![0u8; 512];
317    match stderr.read(&mut buf).await {
318        Ok(n) if n > 0 => String::from_utf8_lossy(&buf[..n]).trim().to_string(),
319        _ => String::new(),
320    }
321}

328    fn make_manifest(name: &str, params: Vec<ParameterDef>) -> ToolManifest {
329        ToolManifest {

344        ParameterDef {
345            name: name.to_string(),
346            r#type: ty.to_string(),
347            description: format!("param {}", name),
348            required,
349            default: None,
350        }

363        let params = vec![
364            make_param("device", "string", true),
365            make_param("pin", "integer", true),
366            make_param("value", "integer", false),
367        ];
368        let m = make_manifest("gpio_write", params);

370        let schema = tool.parameters_schema();

374        assert_eq!(schema["properties"]["pin"]["type"], "integer");

376        let required = schema["required"].as_array().unwrap();
377        let req_names: Vec<&str> = required.iter().map(|v| v.as_str().unwrap()).collect();
378        assert!(req_names.contains(&"device"));
379        assert!(req_names.contains(&"pin"));
380        assert!(!req_names.contains(&"value"));

392    /// Verify that a binary which exits 0 with valid ToolResult JSON on stdout
393    /// is deserialised correctly.
394    #[tokio::test]
395    async fn execute_successful_subprocess() {
396        // Use `echo` to emit a valid ToolResult on stdout.
397        // `echo` prints its argument + newline and exits 0.
398        let result_json = r#"{"success":true,"output":"ok","error":null}"#;
399
400        // Build a manifest pointing at `echo`.
401        let m = make_manifest("echo_tool", vec![]);
402
403        // Construct an `echo` invocation as the binary with the JSON pre-set.
404        // We use `sh -c 'echo <json>'` because the SubprocessTool feeds the
405        // manifest binary with args on stdin — echo just ignores stdin.
406        let script = format!("echo '{}'", result_json);
407        let binary = PathBuf::from("sh");
408        // Override binary to `sh` and pass `-c` + script via a wrapper.
409        // Simpler: write a temp script.
410        let dir = tempfile::tempdir().unwrap();
411        let script_path = dir.path().join("tool.sh");
412        std::fs::write(
413            &script_path,
414            format!("#!/bin/sh\ncat > /dev/null\necho '{}'\n", result_json),
415        )
416        .unwrap();
417        #[cfg(unix)]
418        {
419            use std::os::unix::fs::PermissionsExt;
420            std::fs::set_permissions(&script_path, std::fs::Permissions::from_mode(0o755)).unwrap();
421        }
422
423        let tool = SubprocessTool::new(m, script_path.clone());
424        let result = tool
425            .execute(serde_json::json!({}))
426            .await
427            .expect("execute should not return Err");
428
429        assert!(result.success, "expected success=true, got: {:?}", result);
430        assert_eq!(result.output, "ok");
431        assert!(result.error.is_none());
432
433        let _ = script;
434        let _ = binary;
435    }
