5/// Allowed serial device path prefixes — reject arbitrary paths for security.
6/// Used by hardware serial transport and peripherals.
7const SERIAL_ALLOWED_PATH_PREFIXES: &[&str] = &[
8    "/dev/ttyACM",
9    "/dev/ttyUSB",
10    "/dev/tty.usbmodem",
11    "/dev/cu.usbmodem",
12    "/dev/tty.usbserial",
13    "/dev/cu.usbserial", // Arduino Uno (FTDI), clones
14    "COM",               // Windows
15];
16
17/// Returns true if the path is an allowed serial device (USB CDC, FTDI, etc.).
18/// Rejects arbitrary paths like /etc/passwd or /dev/sda.
19pub fn is_serial_path_allowed(path: &str) -> bool {
20    SERIAL_ALLOWED_PATH_PREFIXES
21        .iter()
22        .any(|prefix| path.starts_with(prefix))
23}

25/// Truncate a string to at most `max_chars` characters, appending "..." if truncated.
26///
27/// This function safely handles multi-byte UTF-8 characters (emoji, CJK, accented characters)
28/// by using character boundaries instead of byte indices.
29///
30/// # Arguments
31/// * `s` - The string to truncate
32/// * `max_chars` - Maximum number of characters to keep (excluding "...")
33///
34/// # Returns
35/// * Original string if length <= `max_chars`
36/// * Truncated string with "..." appended if length > `max_chars`
37///
38/// # Examples
39/// ```ignore
40/// use zeroclaw::util::truncate_with_ellipsis;
41///
42/// // ASCII string - no truncation needed
43/// assert_eq!(truncate_with_ellipsis("hello", 10), "hello");
44///
45/// // ASCII string - truncation needed
46/// assert_eq!(truncate_with_ellipsis("hello world", 5), "hello...");
47///
48/// // Multi-byte UTF-8 (emoji) - safe truncation
49/// assert_eq!(truncate_with_ellipsis("Hello 🦀 World", 8), "Hello 🦀...");
50/// assert_eq!(truncate_with_ellipsis("😀😀😀😀", 2), "😀😀...");
51///
52/// // Empty string
53/// assert_eq!(truncate_with_ellipsis("", 10), "");
54/// ```
55pub fn truncate_with_ellipsis(s: &str, max_chars: usize) -> String {
56    match s.char_indices().nth(max_chars) {
57        Some((idx, _)) => {
58            let truncated = &s[..idx];
59            // Trim trailing whitespace for cleaner output
60            format!("{}...", truncated.trim_end())
61        }
62        None => s.to_string(),
63    }
64}

73#[cfg(test)]
74mod tests {

77    #[test]
78    fn test_truncate_ascii_no_truncation() {
79        // ASCII string shorter than limit - no change
80        assert_eq!(truncate_with_ellipsis("hello", 10), "hello");
81        assert_eq!(truncate_with_ellipsis("hello world", 50), "hello world");
82    }
83
84    #[test]
85    fn test_truncate_ascii_with_truncation() {
86        // ASCII string longer than limit - truncates
87        assert_eq!(truncate_with_ellipsis("hello world", 5), "hello...");
88        assert_eq!(
89            truncate_with_ellipsis("This is a long message", 10),
90            "This is a..."
91        );
92    }
93
94    #[test]
95    fn test_truncate_empty_string() {
96        assert_eq!(truncate_with_ellipsis("", 10), "");
97    }
98
99    #[test]
100    fn test_truncate_at_exact_boundary() {
101        // String exactly at boundary - no truncation
102        assert_eq!(truncate_with_ellipsis("hello", 5), "hello");
103    }
104
105    #[test]
106    fn test_truncate_emoji_single() {
107        // Single emoji (4 bytes) - should not panic
108        let s = "🦀";
109        assert_eq!(truncate_with_ellipsis(s, 10), s);
110        assert_eq!(truncate_with_ellipsis(s, 1), s);
111    }
112
113    #[test]
114    fn test_truncate_emoji_multiple() {
115        // Multiple emoji - safe truncation at character boundary
116        let s = "😀😀😀😀"; // 4 emoji, each 4 bytes = 16 bytes total
117        assert_eq!(truncate_with_ellipsis(s, 2), "😀😀...");
118        assert_eq!(truncate_with_ellipsis(s, 3), "😀😀😀...");
119    }

121    #[test]
122    fn test_truncate_mixed_ascii_emoji() {
123        // Mixed ASCII and emoji
124        assert_eq!(truncate_with_ellipsis("Hello 🦀 World", 8), "Hello 🦀...");
125        assert_eq!(truncate_with_ellipsis("Hi 😊", 10), "Hi 😊");
126    }

128    #[test]
129    fn test_truncate_cjk_characters() {
130        // CJK characters (Chinese - each is 3 bytes)
131        let s = "这是一个测试消息用来触发崩溃的中文"; // 21 characters
132        let result = truncate_with_ellipsis(s, 16);

142    }

144    #[test]

147        let s = "aé你好🦀"; // 1 + 1 + 2 + 2 + 4 bytes = 10 bytes, 5 chars
148        assert_eq!(truncate_with_ellipsis(s, 3), "aé你...");

151    #[test]
152    fn test_truncate_long_string() {
153        // Long ASCII string
154        let s = "a".repeat(200);
155        let result = truncate_with_ellipsis(&s, 50);
156        assert_eq!(result.len(), 53); // 50 + "..."
