6/// Pinggy Tunnel — uses SSH to expose a local port via pinggy.io.
7///
8/// No separate binary required — uses the system `ssh` command.
9/// Free tier works without a token; Pro features require a token
10/// from dashboard.pinggy.io.
11pub struct PinggyTunnel {
12    token: Option<String>,
13    region: Option<String>,
14    proc: SharedProcess,
15}
16
17impl PinggyTunnel {
18    pub fn new(token: Option<String>, region: Option<String>) -> Self {
19        Self {
20            token,
21            region,
22            proc: new_shared_process(),
23        }
24    }
25}

27#[async_trait::async_trait]
28impl Tunnel for PinggyTunnel {
29    fn name(&self) -> &str {
30        "pinggy"
31    }
32
33    async fn start(&self, local_host: &str, local_port: u16) -> Result<String> {
34        // Pro tokens use pro.pinggy.io; free tier uses free.pinggy.io.
35        let base = match self.token.as_deref() {
36            Some(t) if !t.is_empty() => "pro.pinggy.io",
37            _ => "free.pinggy.io",
38        };
39        let server_host = match self.region.as_deref() {
40            Some(r) if !r.is_empty() => format!("{}.{base}", r.to_ascii_lowercase()),
41            _ => base.into(),
42        };
43
44        // Build the SSH user portion: TOKEN@ or empty for free tier
45        let destination = match self.token.as_deref() {
46            Some(t) if !t.is_empty() => format!("{t}@{server_host}"),
47            _ => server_host,
48        };
49
50        // Use the caller-provided local_host for forwarding target.
51        let forward_spec = format!("0:{local_host}:{local_port}");
52
53        let mut child = Command::new("ssh")
54            .args([
55                "-T",
56                "-p",
57                "443",
58                "-R",
59                &forward_spec,
60                "-o",
61                "StrictHostKeyChecking=accept-new",
62                "-o",
63                "ServerAliveInterval=30",
64                &destination,
65            ])
66            .stdin(std::process::Stdio::null())
67            .stdout(std::process::Stdio::piped())
68            .stderr(std::process::Stdio::piped())
69            .kill_on_drop(true)
70            .spawn()?;
71
72        // Pinggy may print the tunnel URL to stdout or stderr depending on
73        // SSH mode; read both streams concurrently to catch it either way.
74        let stdout = child
75            .stdout
76            .take()
77            .ok_or_else(|| anyhow::anyhow!("Failed to capture pinggy stdout"))?;
78        let stderr = child
79            .stderr
80            .take()
81            .ok_or_else(|| anyhow::anyhow!("Failed to capture pinggy stderr"))?;
82
83        let mut stdout_lines = tokio::io::BufReader::new(stdout).lines();
84        let mut stderr_lines = tokio::io::BufReader::new(stderr).lines();
85        let mut public_url = String::new();
86
87        // Tag each stream line so we know which stream produced EOF.
88        enum StreamLine {
89            Stdout(std::io::Result<Option<String>>),
90            Stderr(std::io::Result<Option<String>>),
91        }
92
93        let mut stdout_done = false;
94        let mut stderr_done = false;
95
96        // Wait up to 15s for the tunnel URL to appear on either stream
97        let deadline = tokio::time::Instant::now() + tokio::time::Duration::from_secs(15);
98        while tokio::time::Instant::now() < deadline && !(stdout_done && stderr_done) {
99            let stream_line = tokio::time::timeout(tokio::time::Duration::from_secs(3), async {
100                tokio::select! {
101                    biased;
102                    l = stdout_lines.next_line(), if !stdout_done => StreamLine::Stdout(l),
103                    l = stderr_lines.next_line(), if !stderr_done => StreamLine::Stderr(l),
104                }
105            })
106            .await;
107
108            match stream_line {
109                Ok(StreamLine::Stdout(Ok(Some(l))) | StreamLine::Stderr(Ok(Some(l)))) => {
110                    tracing::debug!("pinggy: {l}");
111                    // Pinggy prints tunnel URLs like: https://xxxxx.a.free.pinggy.link
112                    // Skip non-tunnel URLs (e.g. dashboard.pinggy.io promo links).
113                    if let Some(idx) = l.find("https://") {
114                        let url_part = &l[idx..];
115                        let end = url_part
116                            .find(|c: char| c.is_whitespace())
117                            .unwrap_or(url_part.len());
118                        let candidate = &url_part[..end];
119                        if candidate.contains(".pinggy.link") {
120                            public_url = candidate.to_string();
121                            break;
122                        }
123                    }
124                }
125                Ok(StreamLine::Stdout(Ok(None))) => stdout_done = true,
126                Ok(StreamLine::Stderr(Ok(None))) => stderr_done = true,
127                Ok(StreamLine::Stdout(Err(e)) | StreamLine::Stderr(Err(e))) => {
128                    bail!("Error reading pinggy output: {e}")
129                }
130                Err(_) => {} // timeout — retry
131            }
132        }
133
134        if public_url.is_empty() {
135            child.kill().await.ok();
136            child.wait().await.ok();
137            bail!(
138                "pinggy did not produce a public URL within 15s. Is SSH available and the token valid?"
139            );
140        }
141
142        let mut guard = self.proc.lock().await;
143        *guard = Some(TunnelProcess {
144            child,
145            public_url: public_url.clone(),
146        });
147
148        Ok(public_url)
149    }
150
151    async fn stop(&self) -> Result<()> {
152        kill_shared(&self.proc).await
153    }
154
155    async fn health_check(&self) -> bool {
156        let mut guard = self.proc.lock().await;
157        match guard.as_mut() {
158            Some(tp) => match tp.child.try_wait() {
159                Ok(None) => true,              // still running
160                Ok(Some(_)) | Err(_) => false, // exited or error
161            },
162            None => false,
163        }
164    }
165
166    fn public_url(&self) -> Option<String> {
167        self.proc
168            .try_lock()
169            .ok()
170            .and_then(|g| g.as_ref().map(|tp| tp.public_url.clone()))
171    }
172}
