6/// Try to extract a real tunnel URL from a cloudflared log line.
7///
8/// Returns `Some(url)` when the line contains a genuine tunnel endpoint,
9/// skipping documentation and warning URLs (quic-go GitHub links,
10/// Cloudflare docs pages, etc.).
11fn extract_tunnel_url(line: &str) -> Option<String> {
12    let idx = line.find("https://")?;
13    let url_part = &line[idx..];
14    let end = url_part
15        .find(|c: char| c.is_whitespace())
16        .unwrap_or(url_part.len());
17    let candidate = &url_part[..end];
18
19    let is_tunnel_line = line.contains("Visit it at")
20        || line.contains("Route at")
21        || line.contains("Registered tunnel connection");
22    let is_tunnel_domain = candidate.contains(".trycloudflare.com");
23    let is_docs_url = candidate.contains("github.com")
24        || candidate.contains("cloudflare.com/docs")
25        || candidate.contains("developers.cloudflare.com");
26
27    if is_tunnel_line || is_tunnel_domain || !is_docs_url {
28        Some(candidate.to_string())
29    } else {
30        None
31    }
32}

34/// Cloudflare Tunnel — wraps the `cloudflared` binary.
35///
36/// Requires `cloudflared` installed and a tunnel token from the
37/// Cloudflare Zero Trust dashboard.
38pub struct CloudflareTunnel {
39    token: String,
40    proc: SharedProcess,
41}

43impl CloudflareTunnel {

52#[async_trait::async_trait]
53impl Tunnel for CloudflareTunnel {
54    fn name(&self) -> &str {
55        "cloudflare"
56    }
57
58    async fn start(&self, _local_host: &str, local_port: u16) -> Result<String> {
59        // cloudflared tunnel --no-autoupdate run --token <TOKEN> --url http://localhost:<port>
60        let mut child = Command::new("cloudflared")
61            .args([
62                "tunnel",
63                "--no-autoupdate",
64                "run",
65                "--token",
66                &self.token,
67                "--url",
68                &format!("http://localhost:{local_port}"),
69            ])
70            .stdout(std::process::Stdio::piped())
71            .stderr(std::process::Stdio::piped())
72            .kill_on_drop(true)
73            .spawn()?;
74
75        // Read stderr to find the public URL (cloudflared prints it there)
76        let stderr = child
77            .stderr
78            .take()
79            .ok_or_else(|| anyhow::anyhow!("Failed to capture cloudflared stderr"))?;
80
81        let mut reader = tokio::io::BufReader::new(stderr).lines();
82        let mut public_url = String::new();
83
84        // Wait up to 30s for the tunnel URL to appear
85        let deadline = tokio::time::Instant::now() + tokio::time::Duration::from_secs(30);
86        while tokio::time::Instant::now() < deadline {
87            let line =
88                tokio::time::timeout(tokio::time::Duration::from_secs(5), reader.next_line()).await;
89
90            match line {
91                Ok(Ok(Some(l))) => {
92                    tracing::debug!("cloudflared: {l}");
93                    if let Some(url) = extract_tunnel_url(&l) {
94                        public_url = url;
95                        break;
96                    }
97                }
98                Ok(Ok(None)) => break,
99                Ok(Err(e)) => bail!("Error reading cloudflared output: {e}"),
100                Err(_) => {} // timeout on this line, keep trying
101            }
102        }
103
104        if public_url.is_empty() {
105            child.kill().await.ok();
106            bail!("cloudflared did not produce a public URL within 30s. Is the token valid?");
107        }
108
109        let mut guard = self.proc.lock().await;
110        *guard = Some(TunnelProcess {
111            child,
112            public_url: public_url.clone(),
113        });
114
115        Ok(public_url)
116    }
117
118    async fn stop(&self) -> Result<()> {
119        kill_shared(&self.proc).await
120    }
121
122    async fn health_check(&self) -> bool {
123        let guard = self.proc.lock().await;
124        guard.as_ref().is_some_and(|tp| tp.child.id().is_some())
125    }
126
127    fn public_url(&self) -> Option<String> {
128        // Can't block on async lock in a sync fn, so we try_lock
129        self.proc
130            .try_lock()
131            .ok()
132            .and_then(|g| g.as_ref().map(|tp| tp.public_url.clone()))
133    }
134}
