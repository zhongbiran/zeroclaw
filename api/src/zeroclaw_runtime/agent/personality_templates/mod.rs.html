1//! Default starter templates for the per-workspace personality files.
2//!
3//! Recovered verbatim from the pre-#5951 onboarding wizard's
4//! `scaffold_workspace()` (commit `0c622e607^:crates/zeroclaw-runtime/src/onboard/wizard.rs`).
5//! The wizard rewrite (#5960) shipped without a workspace-scaffolder, so
6//! these templates were dormant in git history. They are restored here
7//! for the dashboard's Personality onboarding step (#6175 follow-up) and
8//! exposed via `GET /api/personality/templates`.
9//!
10//! Each `*.md` file in this directory is the literal template; they
11//! get embedded via `include_str!` and substituted with values from
12//! [`TemplateContext`] at render time. `AGENTS.md` has two variants
13//! (regular and `no-memory`) since it's the only file whose body
14//! changes based on whether persistent memory is enabled.
15//!
16//! Placeholders: `{agent}`, `{user}`, `{tz}`, `{comm_style}`. They
17//! render harmlessly as plain text if a `.md` file is previewed in
18//! GitHub.
19
20use super::personality::EDITABLE_PERSONALITY_FILES;
21
22const IDENTITY: &str = include_str!("IDENTITY.md");
23const SOUL: &str = include_str!("SOUL.md");
24const USER: &str = include_str!("USER.md");
25const HEARTBEAT: &str = include_str!("HEARTBEAT.md");
26const TOOLS: &str = include_str!("TOOLS.md");
27const MEMORY: &str = include_str!("MEMORY.md");
28const AGENTS: &str = include_str!("AGENTS.md");
29const AGENTS_NO_MEMORY: &str = include_str!("AGENTS.no-memory.md");
30
31/// Per-render context — substituted into the templates' placeholders.
32/// Values default to neutral placeholders the user can edit in-place
33/// once the template is loaded into the editor.
34#[derive(Debug, Clone)]
35pub struct TemplateContext {
36    pub agent: String,
37    pub user: String,
38    pub timezone: String,
39    pub communication_style: String,
40    /// When `false`, omits MEMORY.md from the rendered set and renders
41    /// the no-memory variant of AGENTS.md.
42    pub include_memory: bool,
43}
44
45impl Default for TemplateContext {
46    fn default() -> Self {
47        Self {
48            agent: "ZeroClaw".to_string(),
49            user: "User".to_string(),
50            timezone: "UTC".to_string(),
51            communication_style:
52                "Be warm, natural, and clear. Use occasional relevant emojis (1-2 max) and avoid robotic phrasing."
53                    .to_string(),
54            include_memory: true,
55        }
56    }
57}
58
59fn substitute(template: &str, ctx: &TemplateContext) -> String {
60    template
61        .replace("{agent}", &ctx.agent)
62        .replace("{user}", &ctx.user)
63        .replace("{tz}", &ctx.timezone)
64        .replace("{comm_style}", &ctx.communication_style)
65}

67/// Render one personality file from the default preset, or `None` when
68/// the filename is outside the editable allowlist (or when MEMORY.md
69/// is requested with `include_memory = false`).
70///
71/// `BOOTSTRAP.md` is intentionally not rendered — it's a first-run
72/// scaffold the agent reads once and deletes; the dashboard editor
73/// doesn't expose it. The original wizard owned BOOTSTRAP.md
74/// generation directly during workspace scaffolding.
75#[must_use]
76pub fn render(filename: &str, ctx: &TemplateContext) -> Option<String> {
77    let raw = match filename {
78        "IDENTITY.md" => IDENTITY,
79        "SOUL.md" => SOUL,
80        "USER.md" => USER,
81        "HEARTBEAT.md" => HEARTBEAT,
82        "TOOLS.md" => TOOLS,
83        "AGENTS.md" => {
84            if ctx.include_memory {
85                AGENTS
86            } else {
87                AGENTS_NO_MEMORY
88            }
89        }
90        "MEMORY.md" if ctx.include_memory => MEMORY,
91        _ => return None,
92    };
93    Some(substitute(raw, ctx))
94}
95
96/// Render the full default preset for every editable file.
97#[must_use]
98pub fn render_preset_default(ctx: &TemplateContext) -> Vec<(&'static str, String)> {
99    EDITABLE_PERSONALITY_FILES
100        .iter()
101        .copied()
102        .filter_map(|f| render(f, ctx).map(|content| (f, content)))
103        .collect()
104}

106#[cfg(test)]
107mod tests {
108    use super::*;
109
110    #[test]
111    fn default_preset_covers_every_editable_file() {
112        let ctx = TemplateContext::default();
113        let rendered = render_preset_default(&ctx);
114        let names: Vec<&str> = rendered.iter().map(|(n, _)| *n).collect();
115        for f in EDITABLE_PERSONALITY_FILES {
116            assert!(
117                names.contains(f),
118                "default preset missing {f}; only had {names:?}"
119            );
120        }
121    }
122
123    #[test]
124    fn bootstrap_is_not_a_template() {
125        let ctx = TemplateContext::default();
126        assert!(
127            render("BOOTSTRAP.md", &ctx).is_none(),
128            "BOOTSTRAP.md is owned by first-run scaffolding, not the editor"
129        );
130    }
131
132    #[test]
133    fn excluding_memory_drops_memory_md() {
134        let ctx = TemplateContext {
135            include_memory: false,
136            ..TemplateContext::default()
137        };
138        let rendered = render_preset_default(&ctx);
139        assert!(
140            !rendered.iter().any(|(n, _)| *n == "MEMORY.md"),
141            "MEMORY.md should be skipped when include_memory = false"
142        );
143    }
144
145    #[test]
146    fn excluding_memory_picks_no_memory_agents_variant() {
147        let on = render(
148            "AGENTS.md",
149            &TemplateContext {
150                include_memory: true,
151                ..TemplateContext::default()
152            },
153        )
154        .unwrap();
155        let off = render(
156            "AGENTS.md",
157            &TemplateContext {
158                include_memory: false,
159                ..TemplateContext::default()
160            },
161        )
162        .unwrap();
163        assert!(
164            on.contains("Daily notes"),
165            "memory-on AGENTS.md must mention daily notes"
166        );
167        assert!(
168            off.contains("memory.backend = \"none\""),
169            "memory-off AGENTS.md must mention disabled memory"
170        );
171    }
172
173    #[test]
174    fn substitutes_agent_name_into_soul() {
175        let ctx = TemplateContext {
176            agent: "Nova".to_string(),
177            ..TemplateContext::default()
178        };
179        let soul = render("SOUL.md", &ctx).unwrap();
180        assert!(soul.contains("You are **Nova**"));
181        assert!(soul.contains("Always introduce yourself as Nova"));
182    }

184    #[test]
