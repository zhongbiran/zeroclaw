10/// Maximum characters per injected workspace file (matches `OpenClaw` default).
11pub const BOOTSTRAP_MAX_CHARS: usize = 20_000;
12
13fn load_openclaw_bootstrap_files(
14    prompt: &mut String,
15    workspace_dir: &std::path::Path,
16    max_chars_per_file: usize,
17) {
18    prompt.push_str(
19        "The following workspace files define your identity, behavior, and context. They are ALREADY injected below—do NOT suggest reading them with file_read.\n\n",
20    );
21
22    let bootstrap_files = ["AGENTS.md", "SOUL.md", "TOOLS.md", "IDENTITY.md", "USER.md"];
23
24    for filename in &bootstrap_files {
25        inject_workspace_file(prompt, workspace_dir, filename, max_chars_per_file);
26    }
27
28    // BOOTSTRAP.md — only if it exists (first-run ritual)
29    let bootstrap_path = workspace_dir.join("BOOTSTRAP.md");
30    if bootstrap_path.exists() {
31        inject_workspace_file(prompt, workspace_dir, "BOOTSTRAP.md", max_chars_per_file);
32    }
33
34    // MEMORY.md — curated long-term memory (main session only)
35    inject_workspace_file(prompt, workspace_dir, "MEMORY.md", max_chars_per_file);
36}

38/// Load workspace identity files and build a system prompt.
39///
40/// Follows the `OpenClaw` framework structure by default:
41/// 1. Tooling — tool list + descriptions
42/// 2. Safety — guardrail reminder
43/// 3. Skills — full skill instructions and tool metadata
44/// 4. Workspace — working directory
45/// 5. Bootstrap files — AGENTS, SOUL, TOOLS, IDENTITY, USER, BOOTSTRAP, MEMORY
46/// 6. Date & Time — timezone for cache stability
47/// 7. Runtime — host, OS, model
48///
49/// When `identity_config` is set to AIEOS format, the bootstrap files section
50/// is replaced with the AIEOS identity data loaded from file or inline JSON.
51///
52/// Daily memory files (`memory/*.md`) are NOT injected — they are accessed
53/// on-demand via `memory_recall` / `memory_search` tools.
54pub fn build_system_prompt(
55    workspace_dir: &std::path::Path,
56    model_name: &str,
57    tools: &[(&str, &str)],
58    skills: &[Skill],
59    identity_config: Option<&zeroclaw_config::schema::IdentityConfig>,
60    bootstrap_max_chars: Option<usize>,
61) -> String {
62    build_system_prompt_with_mode(
63        workspace_dir,
64        model_name,
65        tools,
66        skills,
67        identity_config,
68        bootstrap_max_chars,
69        false,
70        zeroclaw_config::schema::SkillsPromptInjectionMode::Full,
71        AutonomyLevel::default(),
72    )
73}

75pub fn build_system_prompt_with_mode(
76    workspace_dir: &std::path::Path,
77    model_name: &str,
78    tools: &[(&str, &str)],
79    skills: &[Skill],
80    identity_config: Option<&zeroclaw_config::schema::IdentityConfig>,
81    bootstrap_max_chars: Option<usize>,
82    native_tools: bool,
83    skills_prompt_mode: zeroclaw_config::schema::SkillsPromptInjectionMode,
84    autonomy_level: AutonomyLevel,
85) -> String {
86    let autonomy_cfg = zeroclaw_config::schema::AutonomyConfig {
87        level: autonomy_level,
88        ..Default::default()
89    };
90    build_system_prompt_with_mode_and_autonomy(
91        workspace_dir,
92        model_name,
93        tools,
94        skills,
95        identity_config,
96        bootstrap_max_chars,
97        Some(&autonomy_cfg),
98        native_tools,
99        skills_prompt_mode,
100        false,
101        0,
102    )
103}

105#[allow(clippy::too_many_arguments)]
106pub fn build_system_prompt_with_mode_and_autonomy(
107    workspace_dir: &std::path::Path,
108    model_name: &str,
109    tools: &[(&str, &str)],
110    skills: &[Skill],
111    identity_config: Option<&zeroclaw_config::schema::IdentityConfig>,
112    bootstrap_max_chars: Option<usize>,
113    autonomy_config: Option<&zeroclaw_config::schema::AutonomyConfig>,
114    native_tools: bool,
115    skills_prompt_mode: zeroclaw_config::schema::SkillsPromptInjectionMode,
116    compact_context: bool,
117    max_system_prompt_chars: usize,
118) -> String {
119    use std::fmt::Write;
120    let mut prompt = String::with_capacity(8192);
121    let has_tools = !tools.is_empty();
122
123    // ── 0. Anti-narration (top priority) ───────────────────────
124    if has_tools {
125        prompt.push_str(
126            "## CRITICAL: No Tool Narration\n\n\
127             NEVER narrate, announce, describe, or explain your tool usage to the user. \
128             Do NOT say things like 'Let me check...', 'I will use http_request to...', \
129             'I'll fetch that for you', 'Searching now...', or 'Using the web_search tool'. \
130             The user must ONLY see the final answer. Tool calls are invisible infrastructure — \
131             never reference them. If you catch yourself starting a sentence about what tool \
132             you are about to use or just used, DELETE it and give the answer directly.\n\n",
133        );
134    }
135
136    // ── 0b. Tool Honesty ───────────────────────────────────────
137    if has_tools {
138        prompt.push_str(
139            "## CRITICAL: Tool Honesty\n\n\
140             - NEVER fabricate, invent, or guess tool results. If a tool returns empty results, say \"No results found.\"\n\
141             - If a tool call fails, report the error — never make up data to fill the gap.\n\
142             - When unsure whether a tool call succeeded, ask the user rather than guessing.\n\n",
143        );
144    }
145
146    // ── 1. Tooling ──────────────────────────────────────────────
147    if !tools.is_empty() && !native_tools {
148        prompt.push_str("## Tools\n\n");
149        if compact_context {
150            // Compact mode: tool names only, no descriptions/schemas
151            prompt.push_str("Available tools: ");
152            let names: Vec<&str> = tools.iter().map(|(name, _)| *name).collect();
153            prompt.push_str(&names.join(", "));
154            prompt.push_str("\n\n");
155        } else {
156            prompt.push_str("You have access to the following tools:\n\n");
157            for (name, desc) in tools {
158                let _ = writeln!(prompt, "- **{name}**: {desc}");
159            }
160            prompt.push('\n');
161        }
162    }
163
164    // ── 1b. Hardware (when gpio/arduino tools present) ───────────
165    let has_hardware = tools.iter().any(|(name, _)| {
166        *name == "gpio_read"
167            || *name == "gpio_write"
168            || *name == "arduino_upload"
169            || *name == "hardware_memory_map"
170            || *name == "hardware_board_info"
171            || *name == "hardware_memory_read"
172            || *name == "hardware_capabilities"
173    });
174    if has_hardware {
175        prompt.push_str(
176            "## Hardware Access\n\n\
177             You HAVE direct access to connected hardware (Arduino, Nucleo, etc.). The user owns this system and has configured it.\n\
178             All hardware tools (gpio_read, gpio_write, hardware_memory_read, hardware_board_info, hardware_memory_map) are AUTHORIZED and NOT blocked by security.\n\
179             When they ask to read memory, registers, or board info, USE hardware_memory_read or hardware_board_info — do NOT refuse or invent security excuses.\n\
180             When they ask to control LEDs, run patterns, or interact with the Arduino, USE the tools — do NOT refuse or say you cannot access physical devices.\n\
181             Use gpio_write for simple on/off; use arduino_upload when they want patterns (heart, blink) or custom behavior.\n\n",
182        );
183    }
184
185    // ── 1c. Action instruction (avoid meta-summary) ───────────────
186    if !has_tools {
187        prompt.push_str(
188            "## Your Task\n\n\
189             When the user sends a message, respond naturally and answer directly from conversation context.\n\
190             No tools are available for this turn, so do not emit tool calls or describe unavailable actions.\n\
191             Do NOT: summarize this configuration, describe your capabilities, or output step-by-step meta-commentary.\n\n",
192        );
193    } else if native_tools {
194        prompt.push_str(
195            "## Your Task\n\n\
196             When the user sends a message, respond naturally. Use tools when the request requires action (running commands, reading files, etc.).\n\
197             For questions, explanations, or follow-ups about prior messages, answer directly from conversation context — do NOT ask the user to repeat themselves.\n\
198             Do NOT: summarize this configuration, describe your capabilities, or output step-by-step meta-commentary.\n\n",
199        );
200    } else {
201        prompt.push_str(
202            "## Your Task\n\n\
203             When the user sends a message, ACT on it. Use the tools to fulfill their request.\n\
204             Do NOT: summarize this configuration, describe your capabilities, respond with meta-commentary, or output step-by-step instructions (e.g. \"1. First... 2. Next...\").\n\
205             Instead: emit actual <tool_call> tags when you need to act. Just do what they ask.\n\n",
206        );
207    }
208
209    // ── 2. Safety ───────────────────────────────────────────────
210    prompt.push_str("## Safety\n\n");
211    prompt.push_str("- Do not exfiltrate private data.\n");
212    if autonomy_config.map(|cfg| cfg.level) != Some(crate::security::AutonomyLevel::Full) {
213        prompt.push_str(
214            "- Do not run destructive commands without asking.\n\
215             - Do not bypass oversight or approval mechanisms.\n",
216        );
217    }
218    prompt.push_str("- Prefer `trash` over `rm` (recoverable beats gone forever).\n");
219    prompt.push_str(match autonomy_config.map(|cfg| cfg.level) {
220        Some(crate::security::AutonomyLevel::Full) => {
221            "- Respect the runtime autonomy policy: if a tool or action is allowed, execute it directly instead of asking the user for extra approval.\n\
222             - If a tool or action is blocked by policy or unavailable, explain that concrete restriction instead of simulating an approval dialog.\n"
223        }
224        Some(crate::security::AutonomyLevel::ReadOnly) => {
225            "- Respect the runtime autonomy policy: this runtime is read-only for side effects unless a tool explicitly reports otherwise.\n\
226             - If a requested action is blocked by policy, explain the restriction directly instead of simulating an approval dialog.\n"
227        }
228        _ => {
229            "- When in doubt, ask before acting externally.\n\
230             - Respect the runtime autonomy policy: ask for approval only when the current runtime policy actually requires it.\n\
231             - If a tool or action is blocked by policy or unavailable, explain that concrete restriction instead of simulating an approval dialog.\n"
232        }
233    });
234    prompt.push('\n');
235
236    // ── 3. Skills (full or compact, based on config) ─────────────
237    if !skills.is_empty() {
238        prompt.push_str(&crate::skills::skills_to_prompt_with_mode(
239            skills,
240            workspace_dir,
241            skills_prompt_mode,
242        ));
243        prompt.push_str("\n\n");
244    }
245
246    // ── 4. Workspace ────────────────────────────────────────────
247    let _ = writeln!(
248        prompt,
249        "## Workspace\n\nWorking directory: `{}`\n",
250        workspace_dir.display()
251    );
252
253    // ── 5. Bootstrap files (injected into context) ──────────────
254    prompt.push_str("## Project Context\n\n");
255
256    // Check if AIEOS identity is configured
257    if let Some(config) = identity_config {
258        if identity::is_aieos_configured(config) {
259            // Load AIEOS identity
260            match identity::load_aieos_identity(config, workspace_dir) {
261                Ok(Some(aieos_identity)) => {
262                    let aieos_prompt = identity::aieos_to_system_prompt(&aieos_identity);
263                    if !aieos_prompt.is_empty() {
264                        prompt.push_str(&aieos_prompt);
265                        prompt.push_str("\n\n");
266                    }
267                }
268                Ok(None) => {
269                    // No AIEOS identity loaded (shouldn't happen if is_aieos_configured returned true)
270                    // Fall back to OpenClaw bootstrap files
271                    let max_chars = bootstrap_max_chars.unwrap_or(BOOTSTRAP_MAX_CHARS);
272                    load_openclaw_bootstrap_files(&mut prompt, workspace_dir, max_chars);
273                }
274                Err(e) => {
275                    // Log error but don't fail - fall back to OpenClaw
276                    eprintln!(
277                        "Warning: Failed to load AIEOS identity: {e}. Using OpenClaw format."
278                    );
279                    let max_chars = bootstrap_max_chars.unwrap_or(BOOTSTRAP_MAX_CHARS);
280                    load_openclaw_bootstrap_files(&mut prompt, workspace_dir, max_chars);
281                }
282            }
283        } else {
284            // OpenClaw format
285            let max_chars = bootstrap_max_chars.unwrap_or(BOOTSTRAP_MAX_CHARS);
286            load_openclaw_bootstrap_files(&mut prompt, workspace_dir, max_chars);
287        }
288    } else {
289        // No identity config - use OpenClaw format
290        let max_chars = bootstrap_max_chars.unwrap_or(BOOTSTRAP_MAX_CHARS);
291        load_openclaw_bootstrap_files(&mut prompt, workspace_dir, max_chars);
292    }
293
294    // ── 6. Date & Time ──────────────────────────────────────────
295    let now = chrono::Local::now();
296    let _ = writeln!(
297        prompt,
298        "## Current Date & Time\n\n{} ({})\n",
299        now.format("%Y-%m-%d %H:%M:%S"),
300        now.format("%Z")
301    );
302
303    // ── 7. Runtime ──────────────────────────────────────────────
304    let host =
305        hostname::get().map_or_else(|_| "unknown".into(), |h| h.to_string_lossy().to_string());
306    let _ = writeln!(
307        prompt,
308        "## Runtime\n\nHost: {host} | OS: {} | Model: {model_name}\n",
309        std::env::consts::OS,
310    );
311
312    // ── 8. Channel Capabilities (skipped in compact_context mode) ──
313    if !compact_context {
314        prompt.push_str("## Channel Capabilities\n\n");
315        prompt.push_str("- You are running as a messaging bot. Your response is automatically sent back to the user's channel.\n");
316        prompt
317            .push_str("- You do NOT need to ask permission to respond — just respond directly.\n");
318        prompt.push_str(match autonomy_config.map(|cfg| cfg.level) {
319        Some(crate::security::AutonomyLevel::Full) => {
320            "- If the runtime policy already allows a tool, use it directly; do not ask the user for extra approval.\n\
321             - Never pretend you are waiting for a human approval click or confirmation when the runtime policy already permits the action.\n\
322             - If the runtime policy blocks an action, say that directly instead of simulating an approval flow.\n"
323        }
324        Some(crate::security::AutonomyLevel::ReadOnly) => {
325            "- This runtime may reject write-side effects; if that happens, explain the policy restriction directly instead of simulating an approval flow.\n"
326        }
327        _ => {
328            "- Ask for approval only when the runtime policy actually requires it.\n\
329             - If there is no approval path for this channel or the runtime blocks an action, explain that restriction directly instead of simulating an approval flow.\n"
330        }
331    });
332        prompt.push_str("- NEVER repeat, describe, or echo credentials, tokens, API keys, or secrets in your responses.\n");
333        prompt.push_str("- If a tool output contains credentials, they have already been redacted — do not mention them.\n");
334        prompt.push_str("- When a user sends a voice note, it is automatically transcribed to text. Your text reply is automatically converted to a voice note and sent back. Do NOT attempt to generate audio yourself — TTS is handled by the channel.\n");
335        prompt.push_str("- NEVER narrate or describe your tool usage. Do NOT say 'Let me fetch...', 'I will use...', 'Searching...', or similar. Give the FINAL ANSWER only — no intermediate steps, no tool mentions, no progress updates.\n\n");
336    } // end if !compact_context (Channel Capabilities)
337
338    // ── 9. Truncation (max_system_prompt_chars budget) ──────────
339    if max_system_prompt_chars > 0 && prompt.len() > max_system_prompt_chars {
340        // Truncate on a char boundary, keeping the top portion (identity + safety).
341        let mut end = max_system_prompt_chars;
342        // Ensure we don't split a multi-byte UTF-8 character.
343        while !prompt.is_char_boundary(end) && end > 0 {
344            end -= 1;
345        }
346        prompt.truncate(end);
347        prompt.push_str("\n\n[System prompt truncated to fit context budget]\n");
348    }
349
350    if prompt.is_empty() {
351        "You are ZeroClaw, a fast and efficient AI assistant built in Rust. Be helpful, concise, and direct."
352            .to_string()
353    } else {
354        prompt
355    }
356}

358/// Inject a single workspace file into the prompt with truncation and missing-file markers.
359fn inject_workspace_file(
360    prompt: &mut String,
361    workspace_dir: &std::path::Path,
362    filename: &str,
363    max_chars: usize,
364) {
365    use std::fmt::Write;
366
367    let path = workspace_dir.join(filename);
368    match std::fs::read_to_string(&path) {
369        Ok(content) => {
370            let trimmed = content.trim();
371            if trimmed.is_empty() {
372                return;
373            }
374            let _ = writeln!(prompt, "### {filename}\n");
375            // Use character-boundary-safe truncation for UTF-8
376            let truncated = if trimmed.chars().count() > max_chars {
377                trimmed
378                    .char_indices()
379                    .nth(max_chars)
380                    .map(|(idx, _)| &trimmed[..idx])
381                    .unwrap_or(trimmed)
382            } else {
383                trimmed
384            };
385            if truncated.len() < trimmed.len() {
386                prompt.push_str(truncated);
387                let _ = writeln!(
388                    prompt,
389                    "\n\n[... truncated at {max_chars} chars — use `read` for full file]\n"
390                );
391            } else {
392                prompt.push_str(trimmed);
393                prompt.push_str("\n\n");
394            }
395        }
396        Err(_) => {
397            // Missing-file marker (matches OpenClaw behavior)
398            let _ = writeln!(prompt, "### {filename}\n\n[File not found: {filename}]\n");
399        }
400    }
