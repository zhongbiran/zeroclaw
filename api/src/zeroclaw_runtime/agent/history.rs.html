18/// Find the largest byte index `<= i` that is a valid char boundary.
19/// MSRV-compatible replacement for `str::floor_char_boundary` (stable in 1.91).
20pub fn floor_char_boundary(s: &str, i: usize) -> usize {
21    if i >= s.len() {
22        return s.len();
23    }
24    let mut pos = i;
25    while pos > 0 && !s.is_char_boundary(pos) {
26        pos -= 1;
27    }
28    pos
29}

31/// Indicates which side of a truncated string a boundary belongs to when
32/// nudging it away from a half-cut `[IMAGE:...]` marker.
33#[derive(Clone, Copy)]
34enum TruncationSide {
35    /// Boundary is the end of the kept head; nudge backward (out of the marker).
36    Head,
37    /// Boundary is the start of the kept tail; nudge forward (out of the marker).
38    Tail,
39}
40
41/// If `boundary` falls inside an `[IMAGE:...]` marker (i.e. between an
42/// unclosed `[IMAGE:` and its closing `]`), nudge it onto the nearest
43/// complete-marker boundary. The malformed half-marker is dropped into the
44/// truncated middle rather than emitted to the regex, which would otherwise
45/// silently fail to match and quietly lose the image.
46fn nudge_around_image_marker(s: &str, boundary: usize, side: TruncationSide) -> usize {
47    const OPEN: &str = "[IMAGE:";
48    if boundary == 0 || boundary >= s.len() {
49        return boundary;
50    }
51
52    // Walk forward to find the most recent `[IMAGE:` whose `[` is strictly
53    // before `boundary`. Searching forward (rather than `rfind` on a prefix)
54    // correctly handles the case where `boundary` itself splits the literal
55    // `[IMAGE:` token.
56    let mut search_from = 0usize;
57    let mut last_open: Option<usize> = None;
58    while let Some(rel) = s[search_from..].find(OPEN) {
59        let open_idx = search_from + rel;
60        if open_idx >= boundary {
61            break;
62        }
63        last_open = Some(open_idx);
64        search_from = open_idx + OPEN.len();
65    }
66    let Some(open_idx) = last_open else {
67        return boundary;
68    };
69
70    // First `]` after the opener closes the marker (canonicalize regex
71    // forbids `]` inside paths, so this is unambiguous in practice).
72    let close_idx = match s[open_idx..].find(']') {
73        Some(rel) => open_idx + rel,
74        None => return boundary, // malformed input — leave the boundary alone
75    };
76
77    if close_idx < boundary {
78        return boundary; // marker fully closed before boundary — safe
79    }
80
81    match side {
82        TruncationSide::Head => open_idx,
83        TruncationSide::Tail => (close_idx + 1).min(s.len()),
84    }
85}

87/// Truncate a tool result to `max_chars`, keeping head (2/3) + tail (1/3)
88/// with a marker in the middle. Returns input unchanged if within limit or
89/// `max_chars == 0` (disabled).
90///
91/// Boundaries are nudged inward when they would split an `[IMAGE:...]`
92/// marker, so the multimodal regex never sees a half-marker in the
93/// surviving head/tail. This matches the canonicalization step that runs
94/// immediately before truncation in `run_tool_call_loop`.
95pub fn truncate_tool_result(output: &str, max_chars: usize) -> String {
96    if max_chars == 0 || output.len() <= max_chars {
97        return output.to_string();
98    }
99    let head_len = max_chars * 2 / 3;
100    let tail_len = max_chars.saturating_sub(head_len);
101    let head_end = floor_char_boundary(output, head_len);
102    // ceil_char_boundary: find smallest byte index >= i on a char boundary
103    let tail_start_raw = output.len().saturating_sub(tail_len);
104    let tail_start = if tail_start_raw >= output.len() {
105        output.len()
106    } else {
107        let mut pos = tail_start_raw;
108        while pos < output.len() && !output.is_char_boundary(pos) {
109            pos += 1;
110        }
111        pos
112    };
113
114    // Step boundaries away from any `[IMAGE:...]` marker they would bisect.
115    // `[IMAGE:` and `]` are pure ASCII, so the adjusted indices land on
116    // valid UTF-8 char boundaries.
117    let head_end = nudge_around_image_marker(output, head_end, TruncationSide::Head);
118    let tail_start = nudge_around_image_marker(output, tail_start, TruncationSide::Tail);
119
120    // Guard against overlap when max_chars is very small
121    if head_end >= tail_start {
122        return output[..floor_char_boundary(output, max_chars)].to_string();
123    }
124    let truncated_chars = tail_start - head_end;
125    format!(
126        "{}\n\n[... {} characters truncated ...]\n\n{}",
127        &output[..head_end],
128        truncated_chars,
129        &output[tail_start..]
130    )
131}

133fn is_existing_local_image_path(path: &str) -> bool {
134    let candidate = Path::new(path);
135    candidate.is_absolute()
136        && candidate.is_file()
137        && candidate
138            .extension()
139            .and_then(|ext| ext.to_str())
140            .is_some_and(|ext| {
141                matches!(
142                    ext.to_ascii_lowercase().as_str(),
143                    "png" | "jpg" | "jpeg" | "webp" | "gif" | "bmp"
144                )
145            })
146}

148/// Rewrite real local image file paths in tool output into `[IMAGE:...]`
149/// markers so the multimodal pipeline can normalize them before the next
150/// provider call. This targets shell/skill outputs that print filesystem
151/// paths directly rather than returning explicit media markers.
152pub fn canonicalize_tool_result_media_markers(output: &str) -> String {
153    let mut rewritten = String::with_capacity(output.len());
154    let mut cursor = 0usize;
155    let mut changed = false;
156
157    for mat in LOCAL_IMAGE_PATH_RE.find_iter(output) {
158        let start = mat.start();
159        let end = mat.end();
160        let path = &output[start..end];
161
162        // Skip paths that are already part of an explicit media marker.
163        if output[..start].ends_with("[IMAGE:") {
164            continue;
165        }
166
167        if !is_existing_local_image_path(path) {
168            continue;
169        }
170
171        rewritten.push_str(&output[cursor..start]);
172        rewritten.push_str("[IMAGE:");
173        rewritten.push_str(path);
174        rewritten.push(']');
175        cursor = end;
176        changed = true;
177    }
178
179    if !changed {
180        return output.to_string();
181    }
182
183    rewritten.push_str(&output[cursor..]);
184    rewritten
185}

187/// Truncate a tool message's content, preserving JSON structure when the
188/// message stores `tool_call_id` alongside `content` (native tool-call
189/// format). Without this, `truncate_tool_result` destroys the JSON envelope
190/// and downstream providers receive a `null` `call_id` (#5425).
191pub fn truncate_tool_message(msg_content: &str, max_chars: usize) -> String {
192    if max_chars == 0 || msg_content.len() <= max_chars {
193        return msg_content.to_string();
194    }
195    if let Ok(mut obj) =
196        serde_json::from_str::<serde_json::Map<String, serde_json::Value>>(msg_content)
197        && obj.contains_key("tool_call_id")
198        && let Some(serde_json::Value::String(inner)) = obj.get("content")
199    {
200        let truncated = truncate_tool_result(inner, max_chars);
201        obj.insert("content".to_string(), serde_json::Value::String(truncated));
202        return serde_json::to_string(&obj).unwrap_or_else(|_| msg_content.to_string());
203    }
204    truncate_tool_result(msg_content, max_chars)
205}

207/// Aggressively trim old tool result messages in history to recover from
208/// context overflow. Keeps the last `protect_last_n` messages untouched.
209/// Returns total characters saved.
210pub fn fast_trim_tool_results(
211    history: &mut [zeroclaw_providers::ChatMessage],
212    protect_last_n: usize,
213) -> usize {
214    let trim_to = 2000;
215    let mut saved = 0;
216    let cutoff = history.len().saturating_sub(protect_last_n);
217    for msg in &mut history[..cutoff] {
218        if msg.role == "tool" && msg.content.len() > trim_to {
219            let original_len = msg.content.len();
220            msg.content = truncate_tool_message(&msg.content, trim_to);
221            saved += original_len - msg.content.len();
222        }
223    }
224    saved
225}

227/// Emergency: drop oldest non-system, non-recent messages from history.
228/// Tool groups (assistant + consecutive tool messages) are dropped
229/// atomically to preserve tool_use/tool_result pairing. See #4810.
230/// Returns number of messages dropped.
231pub fn emergency_history_trim(
232    history: &mut Vec<zeroclaw_providers::ChatMessage>,
233    keep_recent: usize,
234) -> usize {
235    let mut dropped = 0;
236    let target_drop = history.len() / 3;
237    let mut i = 0;
238    while dropped < target_drop && i < history.len().saturating_sub(keep_recent) {
239        if history[i].role == "system" {
240            i += 1;
241        } else if history[i].role == "assistant" {
242            // Count following tool messages — drop as atomic group
243            let mut tool_count = 0;
244            while i + 1 + tool_count < history.len().saturating_sub(keep_recent)
245                && history[i + 1 + tool_count].role == "tool"
246            {
247                tool_count += 1;
248            }
249            for _ in 0..=tool_count {
250                history.remove(i);
251                dropped += 1;
252            }
253        } else {
254            history.remove(i);
255            dropped += 1;
256        }
257    }
258    dropped += remove_orphaned_tool_messages(history);
259    dropped
260}

262/// Estimate token count for a message history using ~4 chars/token heuristic.
263/// Includes a small overhead per message for role/framing tokens.
264pub fn estimate_history_tokens(history: &[ChatMessage]) -> usize {
265    history
266        .iter()
267        .map(|m| {
268            // ~4 chars per token + ~4 framing tokens per message (role, delimiters)
269            m.content.len().div_ceil(4) + 4
270        })
271        .sum()
272}

274pub fn normalize_system_messages(history: &mut Vec<ChatMessage>) {
275    let mut saw_system = false;
276    let mut system_content = String::new();
277    let mut non_system = Vec::with_capacity(history.len());
278
279    for message in history.drain(..) {
280        if message.role == "system" {
281            saw_system = true;
282            if !message.content.is_empty() {
283                if !system_content.is_empty() {
284                    system_content.push_str("\n\n");
285                }
286                system_content.push_str(&message.content);
287            }
288        } else {
289            non_system.push(message);
290        }
291    }
292
293    if saw_system && !system_content.is_empty() {
294        history.push(ChatMessage::system(system_content));
295    }
296    history.extend(non_system);
297}

299pub fn append_or_merge_system_message(history: &mut Vec<ChatMessage>, content: impl Into<String>) {
300    let content = content.into();
301    if content.is_empty() {
302        normalize_system_messages(history);
303        return;
304    }
305
306    if let Some(system_message) = history.iter_mut().find(|message| message.role == "system") {
307        if !system_message.content.is_empty() {
308            system_message.content.push_str("\n\n");
309        }
310        system_message.content.push_str(&content);
311    } else {
312        history.insert(0, ChatMessage::system(content));
313    }
314    normalize_system_messages(history);
315}

317/// Trim conversation history to prevent unbounded growth.
318/// Preserves the system prompt (first message if role=system) and the most recent messages.
319pub fn trim_history(history: &mut Vec<ChatMessage>, max_history: usize) {
320    // Nothing to trim if within limit
321    let has_system = history.first().is_some_and(|m| m.role == "system");
322    let non_system_count = if has_system {
323        history.len() - 1
324    } else {
325        history.len()
326    };
327
328    if non_system_count <= max_history {
329        return;
330    }
331
332    let start = if has_system { 1 } else { 0 };
333    let to_remove = non_system_count - max_history;
334    history.drain(start..start + to_remove);
335    remove_orphaned_tool_messages(history);
336    normalize_system_messages(history);
337}

354pub fn load_interactive_session_history(
355    path: &Path,
356    system_prompt: &str,
357) -> Result<Vec<ChatMessage>> {
358    if !path.exists() {
359        return Ok(vec![ChatMessage::system(system_prompt)]);
360    }
361
362    let raw = std::fs::read_to_string(path)?;
363    let mut state: InteractiveSessionState = serde_json::from_str(&raw)?;
364    if state.history.is_empty() {
365        state.history.push(ChatMessage::system(system_prompt));
366    } else if state.history.first().map(|msg| msg.role.as_str()) != Some("system") {
367        state.history.insert(0, ChatMessage::system(system_prompt));
368    }
369    normalize_system_messages(&mut state.history);
370    if state.history.first().map(|msg| msg.role.as_str()) != Some("system") {
371        state.history.insert(0, ChatMessage::system(system_prompt));
372    }
373
374    // Self-heal persisted sessions that were written with orphaned
375    // tool_result messages (e.g. a crash mid-compaction, or a trim that
376    // dropped the assistant tool_use block but left its tool_result).
377    // Without this the next API call fails with 400 "unexpected tool_use_id
378    // found in tool_result blocks" and the session stays bricked until the
379    // file is deleted. See #5813.
380    remove_orphaned_tool_messages(&mut state.history);
381
382    Ok(state.history)
383}

385pub fn save_interactive_session_history(path: &Path, history: &[ChatMessage]) -> Result<()> {
386    if let Some(parent) = path.parent() {
387        std::fs::create_dir_all(parent)?;
388    }
389
390    let payload = serde_json::to_string_pretty(&InteractiveSessionState::from_history(history))?;
391    std::fs::write(path, payload)?;
392    Ok(())
393}
