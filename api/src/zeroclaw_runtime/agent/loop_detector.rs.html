1//! Loop detection guardrail for the agent tool-call loop.
2//!
3//! Monitors a sliding window of recent tool calls and their results to detect
4//! three repetitive patterns that indicate the agent is stuck:
5//!
6//! 1. **Exact repeat** — same tool + args called 3+ times consecutively.
7//! 2. **Ping-pong** — two tools alternating (A->B->A->B) for 4+ cycles.
8//! 3. **No progress** — same tool called 5+ times with different args but
9//!    identical result hash each time.
10//!
11//! Detection triggers escalating responses: `Warning` -> `Block` -> `Break`.
12
13use std::collections::VecDeque;
14use std::collections::hash_map::DefaultHasher;
15use std::hash::{Hash, Hasher};
16
17// ── Configuration ────────────────────────────────────────────────
18
19/// Configuration for the loop detector, typically derived from
20/// `PacingConfig` fields at the call site.
21#[derive(Debug, Clone)]
22pub struct LoopDetectorConfig {
23    /// Master switch. When `false`, `record` always returns `Ok`.
24    pub enabled: bool,
25    /// Number of recent calls retained for pattern analysis.
26    pub window_size: usize,
27    /// How many consecutive exact-repeat calls before escalation starts.
28    pub max_repeats: usize,
29}
30
31impl Default for LoopDetectorConfig {
32    fn default() -> Self {
33        Self {
34            enabled: true,
35            window_size: 20,
36            max_repeats: 3,
37        }
38    }
39}

41// ── Result enum ──────────────────────────────────────────────────
42
43/// Outcome of a loop-detection check after recording a tool call.
44#[derive(Debug, Clone, PartialEq, Eq)]
45pub enum LoopDetectionResult {
46    /// No pattern detected — continue normally.
47    Ok,
48    /// A suspicious pattern was detected; the caller should inject a
49    /// system-level nudge message into the conversation.
50    Warning(String),
51    /// The tool call should be refused (output replaced with an error).
52    Block(String),
53    /// The agent turn should be terminated immediately.
54    Break(String),
55}

57// ── Internal types ───────────────────────────────────────────────
58
59/// A single recorded tool invocation inside the sliding window.
60#[derive(Debug, Clone)]
61struct ToolCallRecord {
62    /// Tool name.
63    name: String,
64    /// Hash of the serialised arguments.
65    args_hash: u64,
66    /// Hash of the tool's output/result.
67    result_hash: u64,
68}
69
70/// Produce a deterministic hash for a JSON value by recursively sorting
71/// object keys before serialisation.  This ensures `{"a":1,"b":2}` and
72/// `{"b":2,"a":1}` hash identically.
73fn hash_value(value: &serde_json::Value) -> u64 {
74    let mut hasher = DefaultHasher::new();
75    let canonical = serde_json::to_string(&canonicalise(value)).unwrap_or_default();
76    canonical.hash(&mut hasher);
77    hasher.finish()
78}
79
80/// Return a clone of `value` with all object keys sorted recursively.
81fn canonicalise(value: &serde_json::Value) -> serde_json::Value {
82    match value {
83        serde_json::Value::Object(map) => {
84            let mut sorted: Vec<(&String, &serde_json::Value)> = map.iter().collect();
85            sorted.sort_by_key(|(k, _)| *k);
86            let new_map: serde_json::Map<String, serde_json::Value> = sorted
87                .into_iter()
88                .map(|(k, v)| (k.clone(), canonicalise(v)))
89                .collect();
90            serde_json::Value::Object(new_map)
91        }
92        serde_json::Value::Array(arr) => {
93            serde_json::Value::Array(arr.iter().map(canonicalise).collect())
94        }
95        other => other.clone(),
96    }
97}
98
99fn hash_str(s: &str) -> u64 {
100    let mut hasher = DefaultHasher::new();
101    s.hash(&mut hasher);
102    hasher.finish()
103}

105// ── Detector ─────────────────────────────────────────────────────

107/// Stateful loop detector that lives for the duration of a single
108/// `run_tool_call_loop` invocation.
109pub struct LoopDetector {
110    config: LoopDetectorConfig,
111    window: VecDeque<ToolCallRecord>,
112}
113
114impl LoopDetector {
115    pub fn new(config: LoopDetectorConfig) -> Self {
116        Self {
117            window: VecDeque::with_capacity(config.window_size),
118            config,
119        }
120    }
121
122    /// Record a completed tool call and check for loop patterns.
123    ///
124    /// * `name` — tool name (e.g. `"shell"`, `"file_read"`).
125    /// * `args` — the arguments JSON value sent to the tool.
126    /// * `result` — the tool's textual output.
127    pub fn record(
128        &mut self,
129        name: &str,
130        args: &serde_json::Value,
131        result: &str,
132    ) -> LoopDetectionResult {
133        if !self.config.enabled {
134            return LoopDetectionResult::Ok;
135        }
136
137        let record = ToolCallRecord {
138            name: name.to_string(),
139            args_hash: hash_value(args),
140            result_hash: hash_str(result),
141        };
142
143        // Maintain sliding window.
144        if self.window.len() >= self.config.window_size {
145            self.window.pop_front();
146        }
147        self.window.push_back(record);
148
149        // Run detectors in escalation order (most severe first).
150        if let Some(result) = self.detect_exact_repeat() {
151            return result;
152        }
153        if let Some(result) = self.detect_ping_pong() {
154            return result;
155        }
156        if let Some(result) = self.detect_no_progress() {
157            return result;
158        }
159
160        LoopDetectionResult::Ok
161    }
162
163    /// Pattern 1: Same tool + same args called N+ times consecutively.
164    ///
165    /// Escalation:
166    /// - N == max_repeats     -> Warning
167    /// - N == max_repeats + 1 -> Block
168    /// - N >= max_repeats + 2 -> Break (circuit breaker)
169    fn detect_exact_repeat(&self) -> Option<LoopDetectionResult> {
170        let max = self.config.max_repeats;
171        if self.window.len() < max {
172            return None;
173        }
174
175        let last = self.window.back()?;
176        let consecutive = self
177            .window
178            .iter()
179            .rev()
180            .take_while(|r| r.name == last.name && r.args_hash == last.args_hash)
181            .count();
182
183        if consecutive >= max + 2 {
184            Some(LoopDetectionResult::Break(format!(
185                "Circuit breaker: tool '{}' called {} times consecutively with identical arguments",
186                last.name, consecutive
187            )))
188        } else if consecutive > max {
189            Some(LoopDetectionResult::Block(format!(
190                "Blocked: tool '{}' called {} times consecutively with identical arguments",
191                last.name, consecutive
192            )))
193        } else if consecutive >= max {
194            Some(LoopDetectionResult::Warning(format!(
195                "Warning: tool '{}' has been called {} times consecutively with identical arguments. \
196                 Try a different approach.",
197                last.name, consecutive
198            )))
199        } else {
200            None
201        }
202    }
203
204    /// Pattern 2: Two tools alternating (A->B->A->B) for 4+ full cycles
205    /// (i.e. 8 consecutive entries following the pattern).
206    fn detect_ping_pong(&self) -> Option<LoopDetectionResult> {
207        const MIN_CYCLES: usize = 4;
208        let needed = MIN_CYCLES * 2; // each cycle = 2 calls
209
210        if self.window.len() < needed {
211            return None;
212        }
213
214        let tail: Vec<&ToolCallRecord> = self.window.iter().rev().take(needed).collect();
215        // tail[0] is most recent; pattern: A, B, A, B, ...
216        let a_name = &tail[0].name;
217        let b_name = &tail[1].name;
218
219        if a_name == b_name {
220            return None;
221        }
222
223        let is_ping_pong = tail.iter().enumerate().all(|(i, r)| {
224            if i % 2 == 0 {
225                &r.name == a_name
226            } else {
227                &r.name == b_name
228            }
229        });
230
231        if !is_ping_pong {
232            return None;
233        }
234
235        // Count total alternating length for escalation.
236        let mut cycles = MIN_CYCLES;
237        let extended: Vec<&ToolCallRecord> = self.window.iter().rev().collect();
238        for extra_pair in extended.chunks(2).skip(MIN_CYCLES) {
239            if extra_pair.len() == 2
240                && &extra_pair[0].name == a_name
241                && &extra_pair[1].name == b_name
242            {
243                cycles += 1;
244            } else {
245                break;
246            }
247        }
248
249        if cycles >= MIN_CYCLES + 2 {
250            Some(LoopDetectionResult::Break(format!(
251                "Circuit breaker: tools '{}' and '{}' have been alternating for {} cycles",
252                a_name, b_name, cycles
253            )))
254        } else if cycles > MIN_CYCLES {
255            Some(LoopDetectionResult::Block(format!(
256                "Blocked: tools '{}' and '{}' have been alternating for {} cycles",
257                a_name, b_name, cycles
258            )))
259        } else {
260            Some(LoopDetectionResult::Warning(format!(
261                "Warning: tools '{}' and '{}' appear to be alternating ({} cycles). \
262                 Consider a different strategy.",
263                a_name, b_name, cycles
264            )))
265        }
266    }
267
268    /// Pattern 3: Same tool called 5+ times (with different args each time)
269    /// but producing the exact same result hash every time.
270    fn detect_no_progress(&self) -> Option<LoopDetectionResult> {
271        const MIN_CALLS: usize = 5;
272
273        if self.window.len() < MIN_CALLS {
274            return None;
275        }
276
277        let last = self.window.back()?;
278        let same_tool_same_result: Vec<&ToolCallRecord> = self
279            .window
280            .iter()
281            .rev()
282            .take_while(|r| r.name == last.name && r.result_hash == last.result_hash)
283            .collect();
284
285        let count = same_tool_same_result.len();
286        if count < MIN_CALLS {
287            return None;
288        }
289
290        // Verify they have *different* args (otherwise exact_repeat handles it).
291        let unique_args: std::collections::HashSet<u64> =
292            same_tool_same_result.iter().map(|r| r.args_hash).collect();
293        if unique_args.len() < 2 {
294            // All same args — this is exact-repeat territory, not no-progress.
295            return None;
296        }
297
298        if count >= MIN_CALLS + 2 {
299            Some(LoopDetectionResult::Break(format!(
300                "Circuit breaker: tool '{}' called {} times with different arguments but identical results — no progress",
301                last.name, count
302            )))
303        } else if count > MIN_CALLS {
304            Some(LoopDetectionResult::Block(format!(
305                "Blocked: tool '{}' called {} times with different arguments but identical results",
306                last.name, count
307            )))
308        } else {
309            Some(LoopDetectionResult::Warning(format!(
310                "Warning: tool '{}' called {} times with different arguments but identical results. \
311                 The current approach may not be making progress.",
312                last.name, count
313            )))
314        }
315    }
316}
