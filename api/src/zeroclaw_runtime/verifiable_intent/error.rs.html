1//! Machine-readable error taxonomy for Verifiable Intent operations.
2//!
3//! Every VI error carries a [`ViErrorKind`] discriminant so policy engines and
4//! tool gates can branch deterministically on failure reason without parsing
5//! human-readable messages.
6
7use std::fmt;
8
9/// Discriminant for VI error classification — used by policy engines to decide
10/// whether a transaction should be blocked, retried, or escalated.
11#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
12pub enum ViErrorKind {
13    // ── Credential structure errors ───────────────────────────────────
14    /// JWT header is malformed or missing required fields.
15    InvalidHeader,
16    /// JWT payload cannot be decoded or is missing required claims.
17    InvalidPayload,
18    /// SD-JWT disclosure is malformed or cannot be resolved.
19    InvalidDisclosure,
20    /// Credential has expired (`exp` < now).
21    Expired,
22    /// Credential is not yet valid (`iat` > now).
23    NotYetValid,
24
25    // ── Signature / key errors ────────────────────────────────────────
26    /// Cryptographic signature verification failed.
27    SignatureInvalid,
28    /// The signing key does not match the expected `cnf.jwk` binding.
29    KeyMismatch,
30    /// Key material is missing or in an unsupported format.
31    KeyUnsupported,
32
33    // ── Chain binding errors ──────────────────────────────────────────
34    /// `sd_hash` in L2/L3 does not match the hash of the parent layer.
35    SdHashMismatch,
36    /// `checkout_hash` / `transaction_id` cross-reference between L3a and L3b failed.
37    CrossReferenceMismatch,
38    /// `conditional_transaction_id` binding between payment and checkout mandates failed.
39    ReferenceBindingMismatch,
40
41    // ── Constraint violations ─────────────────────────────────────────
42    /// Transaction amount is outside the permitted range.
43    AmountOutOfRange,
44    /// Cumulative budget cap exceeded.
45    BudgetExceeded,
46    /// Currency in L3 does not match the constraint currency.
47    CurrencyMismatch,
48    /// Merchant is not in the allowed merchant list.
49    MerchantNotAllowed,
50    /// Payee is not in the allowed payee list.
51    PayeeNotAllowed,
52    /// Line items violate product selection or quantity constraints.
53    LineItemViolation,
54    /// Recurrence constraint violated.
55    RecurrenceViolation,
56    /// An unknown constraint type was encountered in strict mode.
57    UnknownConstraintType,
58
59    // ── Mode / structural mismatch ────────────────────────────────────
60    /// L2 contains `cnf` in Immediate mode (forbidden) or lacks it in Autonomous mode.
61    ModeMismatch,
62    /// Mandate VCT value is not recognized.
63    UnknownMandateType,
64    /// Mandate pair is incomplete (missing checkout or payment mandate).
65    IncompleteMandatePair,
66
67    // ── Issuance errors ───────────────────────────────────────────────
68    /// Issuance failed due to missing or invalid input parameters.
69    IssuanceInputInvalid,
70}
71
72/// A Verifiable Intent error with a machine-readable kind and human-readable context.
73#[derive(Debug, Clone)]
74pub struct ViError {
75    pub kind: ViErrorKind,
76    pub message: String,
77}
78
79impl ViError {
80    pub fn new(kind: ViErrorKind, message: impl Into<String>) -> Self {
81        Self {
82            kind,
83            message: message.into(),
84        }
85    }
86}
87
88impl fmt::Display for ViError {
89    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
90        write!(f, "VI/{:?}: {}", self.kind, self.message)
91    }
92}
93
94impl std::error::Error for ViError {}

96#[cfg(test)]
97mod tests {
98    use super::*;
99
100    #[test]
101    fn error_display_includes_kind_and_message() {
102        let err = ViError::new(ViErrorKind::AmountOutOfRange, "50000 > 40000 USD");
103        let s = format!("{err}");
104        assert!(s.contains("AmountOutOfRange"));
105        assert!(s.contains("50000 > 40000 USD"));
106    }
107
108    #[test]
109    fn error_kind_equality() {
110        assert_eq!(ViErrorKind::Expired, ViErrorKind::Expired);
111        assert_ne!(ViErrorKind::Expired, ViErrorKind::SignatureInvalid);
112    }
