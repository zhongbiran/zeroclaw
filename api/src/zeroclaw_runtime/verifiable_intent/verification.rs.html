1//! Chain verification, constraint checking, and binding integrity validation.
2//!
3//! Implements the normative verification algorithms from the VI specification:
4//! - Full credential chain verification (L1 → L2 → L3)
5//! - Per-constraint validation against fulfillment data
6//! - Cross-reference and hash binding integrity checks
7
8use crate::verifiable_intent::error::{ViError, ViErrorKind};
9use crate::verifiable_intent::types::{
10    CheckoutL3Mandate, Constraint, Entity, Fulfillment, LineItemEntry, MandateMode,
11    PaymentL3Mandate,
12};

14// ── Strictness mode ──────────────────────────────────────────────────
15
16/// Controls behavior when an unknown constraint type is encountered.
17#[derive(Debug, Clone, Copy, PartialEq, Eq)]
18pub enum StrictnessMode {
19    /// Unknown constraint types cause a violation (fail-closed).
20    Strict,
21    /// Unknown constraint types are skipped with a warning (fail-open).
22    Permissive,
23}
24
25// ── Chain verification result ────────────────────────────────────────
26
27/// Result of verifying the credential chain (L1 → L2 → optional L3).
28#[derive(Debug, Clone)]
29pub struct ChainVerificationResult {
30    pub valid: bool,
31    pub mode: Option<MandateMode>,
32    pub errors: Vec<ViError>,
33}
34
35impl ChainVerificationResult {
36    pub fn ok(mode: MandateMode) -> Self {
37        Self {
38            valid: true,
39            mode: Some(mode),
40            errors: vec![],
41        }
42    }
43
44    pub fn fail(errors: Vec<ViError>) -> Self {
45        Self {
46            valid: false,
47            mode: None,
48            errors,
49        }
50    }
51}

53// ── Constraint check result ──────────────────────────────────────────
54
55/// Result of evaluating a single constraint against fulfillment data.
56#[derive(Debug, Clone)]
57pub struct ConstraintCheckResult {
58    pub satisfied: bool,
59    pub constraint_type: String,
60    pub violations: Vec<ViError>,
61}
62
63impl ConstraintCheckResult {
64    pub fn ok(constraint_type: &str) -> Self {
65        Self {
66            satisfied: true,
67            constraint_type: constraint_type.into(),
68            violations: vec![],
69        }
70    }
71
72    pub fn violation(constraint_type: &str, err: ViError) -> Self {
73        Self {
74            satisfied: false,
75            constraint_type: constraint_type.into(),
76            violations: vec![err],
77        }
78    }
79}

81// ── Time validation ──────────────────────────────────────────────────
82
83const CLOCK_SKEW_SECS: i64 = 300;
84
85fn current_timestamp() -> i64 {
86    chrono::Utc::now().timestamp()
87}
88
89/// Verify `iat` and `exp` claims with a 300-second clock skew tolerance.
90pub fn verify_timestamps(iat: i64, exp: i64) -> Result<(), ViError> {
91    let now = current_timestamp();
92    if exp + CLOCK_SKEW_SECS < now {
93        return Err(ViError::new(
94            ViErrorKind::Expired,
95            format!("credential expired at {exp}, now {now}"),
96        ));
97    }
98    if iat - CLOCK_SKEW_SECS > now {
99        return Err(ViError::new(
100            ViErrorKind::NotYetValid,
101            format!("credential not valid until {iat}, now {now}"),
102        ));
103    }
104    Ok(())
105}

107// ── sd_hash binding ──────────────────────────────────────────────────
108
109/// Verify that `expected_hash` equals `B64U(SHA-256(ASCII(serialized_parent)))`.
110pub fn verify_sd_hash_binding(expected_hash: &str, serialized_parent: &str) -> Result<(), ViError> {
111    let computed = crate::verifiable_intent::crypto::sd_hash(serialized_parent);
112    if computed != expected_hash {
113        return Err(ViError::new(
114            ViErrorKind::SdHashMismatch,
115            format!("sd_hash mismatch: expected {expected_hash}, computed {computed}"),
116        ));
117    }
118    Ok(())
119}

121// ── L3 cross-reference binding ───────────────────────────────────────
122
123/// Verify that L3a `transaction_id` equals L3b `checkout_hash`.
124pub fn verify_l3_cross_reference(
125    l3a: &PaymentL3Mandate,
126    l3b: &CheckoutL3Mandate,
127) -> Result<(), ViError> {
128    if l3a.transaction_id != l3b.checkout_hash {
129        return Err(ViError::new(
130            ViErrorKind::CrossReferenceMismatch,
131            format!(
132                "L3a transaction_id ({}) != L3b checkout_hash ({})",
133                l3a.transaction_id, l3b.checkout_hash
134            ),
135        ));
136    }
137    Ok(())
138}
139
140/// Verify checkout_hash is `B64U(SHA-256(ASCII(checkout_jwt)))`.
141pub fn verify_checkout_hash_binding(
142    checkout_hash: &str,
143    checkout_jwt: &str,
144) -> Result<(), ViError> {
145    let computed = crate::verifiable_intent::crypto::sd_hash(checkout_jwt);
146    if computed != checkout_hash {
147        return Err(ViError::new(
148            ViErrorKind::CrossReferenceMismatch,
149            format!("checkout_hash mismatch: expected {checkout_hash}, computed {computed}"),
150        ));
151    }
152    Ok(())
153}

155// ── Mandate mode inference ───────────────────────────────────────────
156
157/// Infer the execution mode from mandate VCT values.
158pub fn infer_mode_from_vct(vct: &str) -> Result<MandateMode, ViError> {
159    match vct {
160        "mandate.checkout" | "mandate.payment" => Ok(MandateMode::Immediate),
161        "mandate.checkout.open" | "mandate.payment.open" => Ok(MandateMode::Autonomous),
162        _ => Err(ViError::new(
163            ViErrorKind::UnknownMandateType,
164            format!("unrecognized mandate VCT: {vct}"),
165        )),
166    }
167}

169// ── Constraint validation ────────────────────────────────────────────
170
171/// Evaluate all constraints against fulfillment data.
172pub fn check_constraints(
173    constraints: &[Constraint],
174    fulfillment: &Fulfillment,
175    strictness: StrictnessMode,
176) -> Vec<ConstraintCheckResult> {
177    constraints
178        .iter()
179        .map(|c| check_single_constraint(c, fulfillment, strictness))
180        .collect()
181}
182
183fn check_single_constraint(
184    constraint: &Constraint,
185    fulfillment: &Fulfillment,
186    _strictness: StrictnessMode,
187) -> ConstraintCheckResult {
188    match constraint {
189        Constraint::AllowedMerchant { allowed_merchants } => {
190            check_allowed_merchant(allowed_merchants, fulfillment)
191        }
192        Constraint::LineItems { items } => check_line_items(items, fulfillment),
193        Constraint::AllowedPayee { allowed_payees } => {
194            check_allowed_payee(allowed_payees, fulfillment)
195        }
196        Constraint::PaymentAmount { currency, min, max } => {
197            check_payment_amount(currency, *min, *max, fulfillment)
198        }
199        Constraint::PaymentBudget { currency, max } => {
200            check_payment_budget(currency, *max, fulfillment)
201        }
202        Constraint::PaymentReference {
203            conditional_transaction_id,
204        } => {
205            // Reference binding is verified structurally, not against fulfillment.
206            ConstraintCheckResult::ok(&format!(
207                "payment.reference({})",
208                &conditional_transaction_id[..8.min(conditional_transaction_id.len())]
209            ))
210        }
211        Constraint::PaymentRecurrence { .. } | Constraint::AgentRecurrence { .. } => {
212            // Recurrence constraints are informational for the payment network
213            // to enforce statefulness. Pass-through at the agent level.
214            ConstraintCheckResult::ok("recurrence")
215        }
216    }
217}

221fn check_allowed_merchant(
222    allowed_merchants: &[Entity],
223    fulfillment: &Fulfillment,
224) -> ConstraintCheckResult {
225    let ct = "mandate.checkout.allowed_merchant";
226    if allowed_merchants.is_empty() {
227        return ConstraintCheckResult::violation(
228            ct,
229            ViError::new(
230                ViErrorKind::MerchantNotAllowed,
231                "empty merchant allowlist is unsatisfiable",
232            ),
233        );
234    }
235    let Some(merchant) = &fulfillment.merchant else {
236        // No merchant info in fulfillment — cannot validate, skip per spec.
237        return ConstraintCheckResult::ok(ct);
238    };
239    if allowed_merchants.iter().any(|m| m.matches(merchant)) {
240        ConstraintCheckResult::ok(ct)
241    } else {
242        ConstraintCheckResult::violation(
243            ct,
244            ViError::new(
245                ViErrorKind::MerchantNotAllowed,
246                format!("merchant '{}' not in allowed list", merchant.name),
247            ),
248        )
249    }
250}
251
252fn check_allowed_payee(
253    allowed_payees: &[Entity],
254    fulfillment: &Fulfillment,
255) -> ConstraintCheckResult {
256    let ct = "payment.allowed_payee";
257    if allowed_payees.is_empty() {
258        return ConstraintCheckResult::violation(
259            ct,
260            ViError::new(
261                ViErrorKind::PayeeNotAllowed,
262                "empty payee allowlist is unsatisfiable",
263            ),
264        );
265    }
266    let Some(payee) = &fulfillment.payee else {
267        return ConstraintCheckResult::ok(ct);
268    };
269    if allowed_payees.iter().any(|p| p.matches(payee)) {
270        ConstraintCheckResult::ok(ct)
271    } else {
272        ConstraintCheckResult::violation(
273            ct,
274            ViError::new(
275                ViErrorKind::PayeeNotAllowed,
276                format!("payee '{}' not in allowed list", payee.name),
277            ),
278        )
279    }
280}

282fn check_payment_amount(
283    currency: &str,
284    min: Option<i64>,
285    max: Option<i64>,
286    fulfillment: &Fulfillment,
287) -> ConstraintCheckResult {
288    let ct = "payment.amount";
289    let Some(actual_amount) = fulfillment.amount else {
290        return ConstraintCheckResult::violation(
291            ct,
292            ViError::new(
293                ViErrorKind::AmountOutOfRange,
294                "missing payment amount in fulfillment",
295            ),
296        );
297    };
298    if let Some(actual_currency) = &fulfillment.currency
299        && actual_currency != currency
300    {
301        return ConstraintCheckResult::violation(
302            ct,
303            ViError::new(
304                ViErrorKind::CurrencyMismatch,
305                format!("expected {currency}, got {actual_currency}"),
306            ),
307        );
308    }
309    if let Some(max_val) = max
310        && actual_amount > max_val
311    {
312        return ConstraintCheckResult::violation(
313            ct,
314            ViError::new(
315                ViErrorKind::AmountOutOfRange,
316                format!("amount {actual_amount} > max {max_val} {currency}"),
317            ),
318        );
319    }
320    if let Some(min_val) = min
321        && actual_amount < min_val
322    {
323        return ConstraintCheckResult::violation(
324            ct,
325            ViError::new(
326                ViErrorKind::AmountOutOfRange,
327                format!("amount {actual_amount} < min {min_val} {currency}"),
328            ),
329        );
330    }
331    ConstraintCheckResult::ok(ct)
332}
333
334fn check_payment_budget(
335    currency: &str,
336    max: i64,
337    fulfillment: &Fulfillment,
338) -> ConstraintCheckResult {
339    let ct = "payment.budget";
340    let Some(actual_amount) = fulfillment.amount else {
341        return ConstraintCheckResult::violation(
342            ct,
343            ViError::new(
344                ViErrorKind::BudgetExceeded,
345                "missing payment amount in fulfillment",
346            ),
347        );
348    };
349    if let Some(actual_currency) = &fulfillment.currency
350        && actual_currency != currency
351    {
352        return ConstraintCheckResult::violation(
353            ct,
354            ViError::new(
355                ViErrorKind::CurrencyMismatch,
356                format!("expected {currency}, got {actual_currency}"),
357            ),
358        );
359    }
360    // Single-transaction check: amount must not exceed budget.
361    // Cumulative tracking is the payment network's responsibility.
362    if actual_amount > max {
363        return ConstraintCheckResult::violation(
364            ct,
365            ViError::new(
366                ViErrorKind::BudgetExceeded,
367                format!("amount {actual_amount} > budget max {max} {currency}"),
368            ),
369        );
370    }
371    ConstraintCheckResult::ok(ct)
372}

374fn check_line_items(
375    constraint_items: &[LineItemEntry],
376    fulfillment: &Fulfillment,
377) -> ConstraintCheckResult {
378    let ct = "mandate.checkout.line_items";
379    if constraint_items.is_empty() {
380        return ConstraintCheckResult::violation(
381            ct,
382            ViError::new(
383                ViErrorKind::LineItemViolation,
384                "empty items allowlist is unsatisfiable",
385            ),
386        );
387    }
388    let Some(fulfillment_items) = &fulfillment.line_items else {
389        return ConstraintCheckResult::violation(
390            ct,
391            ViError::new(
392                ViErrorKind::LineItemViolation,
393                "empty cart does not satisfy line_items constraint",
394            ),
395        );
396    };
397    if fulfillment_items.is_empty() {
398        return ConstraintCheckResult::violation(
399            ct,
400            ViError::new(
401                ViErrorKind::LineItemViolation,
402                "empty cart does not satisfy line_items constraint",
403            ),
404        );
405    }
406
407    // Total quantity check
408    let total_allowed: u32 = constraint_items.iter().map(|l| l.quantity).sum();
409    let total_actual: u32 = fulfillment_items.iter().map(|f| f.quantity).sum();
410    if total_actual > total_allowed {
411        return ConstraintCheckResult::violation(
412            ct,
413            ViError::new(
414                ViErrorKind::LineItemViolation,
415                format!("total quantity {total_actual} > allowed {total_allowed}"),
416            ),
417        );
418    }
419
420    // Per-item validation: each fulfillment item must be in at least one
421    // constraint entry's acceptable_items (unless acceptable_items is empty = wildcard).
422    for fi in fulfillment_items {
423        let allowed_by_any = constraint_items.iter().any(|entry| {
424            if entry.acceptable_items.is_empty() {
425                return true; // wildcard
426            }
427            entry.acceptable_items.iter().any(|ai| ai.id == fi.item_id)
428        });
429        if !allowed_by_any {
430            return ConstraintCheckResult::violation(
431                ct,
432                ViError::new(
433                    ViErrorKind::LineItemViolation,
434                    format!("item '{}' not in any acceptable_items list", fi.item_id),
435                ),
436            );
437        }
438    }
439
440    ConstraintCheckResult::ok(ct)
441}
