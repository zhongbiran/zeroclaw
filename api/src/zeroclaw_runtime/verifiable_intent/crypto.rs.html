1//! SD-JWT / KB-SD-JWT cryptographic primitives.
2//!
3//! Provides JWS signing/verification (ES256), SD-JWT disclosure hashing,
4//! `sd_hash` computation, and selective disclosure resolution.
5//!
6//! Uses `ring` for ECDSA P-256 (already a dependency) and `sha2`/`base64`
7//! for hashing and encoding (also existing dependencies).
8
9use base64::Engine;
10use base64::engine::general_purpose::URL_SAFE_NO_PAD;
11use ring::rand::SystemRandom;
12use ring::signature::{self, ECDSA_P256_SHA256_FIXED_SIGNING, EcdsaKeyPair, KeyPair};
13use sha2::{Digest, Sha256};
14
15use crate::verifiable_intent::error::{ViError, ViErrorKind};
16use crate::verifiable_intent::types::Jwk;

18// ── Base64url helpers ────────────────────────────────────────────────

20/// Encode bytes as base64url without padding.
21pub fn b64u_encode(data: &[u8]) -> String {
22    URL_SAFE_NO_PAD.encode(data)
23}
24
25/// Decode base64url without padding.
26pub fn b64u_decode(s: &str) -> Result<Vec<u8>, ViError> {
27    URL_SAFE_NO_PAD.decode(s).map_err(|e| {
28        ViError::new(
29            ViErrorKind::InvalidPayload,
30            format!("base64url decode: {e}"),
31        )
32    })
33}

35// ── Hashing ──────────────────────────────────────────────────────────

37/// Compute `B64U(SHA-256(ASCII(input)))` — used for `sd_hash`, `checkout_hash`,
38/// `transaction_id`, disclosure hashes, and `conditional_transaction_id`.
39pub fn sd_hash(input: &str) -> String {
40    let digest = Sha256::digest(input.as_bytes());
41    b64u_encode(&digest)
42}
43
44/// Compute raw SHA-256 hash of a byte slice.
45pub fn sha256(data: &[u8]) -> Vec<u8> {
46    Sha256::digest(data).to_vec()
47}

49// ── JWS / ES256 signing ─────────────────────────────────────────────

51/// Sign a JWS (compact serialization) over the given header and payload JSON.
52/// Returns the full `header.payload.signature` string.
53pub fn jws_sign(
54    header_json: &[u8],
55    payload_json: &[u8],
56    key_pair: &EcdsaKeyPair,
57) -> Result<String, ViError> {
58    let header_b64 = b64u_encode(header_json);
59    let payload_b64 = b64u_encode(payload_json);
60    let signing_input = format!("{header_b64}.{payload_b64}");
61
62    let rng = SystemRandom::new();
63    let sig = key_pair.sign(&rng, signing_input.as_bytes()).map_err(|e| {
64        ViError::new(
65            ViErrorKind::SignatureInvalid,
66            format!("signing failed: {e}"),
67        )
68    })?;
69
70    let sig_b64 = b64u_encode(sig.as_ref());
71    Ok(format!("{signing_input}.{sig_b64}"))
72}
73
74/// Verify an ES256 JWS compact-serialization string against a public key.
75pub fn jws_verify(compact: &str, public_key_bytes: &[u8]) -> Result<(), ViError> {
76    let parts: Vec<&str> = compact.splitn(3, '.').collect();
77    if parts.len() != 3 {
78        return Err(ViError::new(
79            ViErrorKind::InvalidHeader,
80            "JWS must have 3 dot-separated parts",
81        ));
82    }
83
84    let signing_input = format!("{}.{}", parts[0], parts[1]);
85    let sig_bytes = b64u_decode(parts[2])?;
86
87    let peer_public_key =
88        signature::UnparsedPublicKey::new(&signature::ECDSA_P256_SHA256_FIXED, public_key_bytes);
89
90    peer_public_key
91        .verify(signing_input.as_bytes(), &sig_bytes)
92        .map_err(|_| {
93            ViError::new(
94                ViErrorKind::SignatureInvalid,
95                "ES256 signature verification failed",
96            )
97        })
98}

100/// Decode the payload segment of a JWS compact string (the middle part).
101pub fn jws_decode_payload(compact: &str) -> Result<serde_json::Value, ViError> {
102    let parts: Vec<&str> = compact.splitn(3, '.').collect();
103    if parts.len() < 2 {
104        return Err(ViError::new(
105            ViErrorKind::InvalidPayload,
106            "JWS must have at least 2 dot-separated parts",
107        ));
108    }
109    let bytes = b64u_decode(parts[1])?;
110    serde_json::from_slice(&bytes)
111        .map_err(|e| ViError::new(ViErrorKind::InvalidPayload, format!("payload JSON: {e}")))
112}
113
114/// Decode the header segment of a JWS compact string (the first part).
115pub fn jws_decode_header(compact: &str) -> Result<serde_json::Value, ViError> {
116    let part = compact
117        .split('.')
118        .next()
119        .ok_or_else(|| ViError::new(ViErrorKind::InvalidHeader, "empty JWS"))?;
120    let bytes = b64u_decode(part)?;
121    serde_json::from_slice(&bytes)
122        .map_err(|e| ViError::new(ViErrorKind::InvalidHeader, format!("header JSON: {e}")))
123}

127/// Generate a fresh EC P-256 key pair.  Returns (pkcs8_document, Jwk_public).
128pub fn generate_ec_p256() -> Result<(Vec<u8>, Jwk), ViError> {
129    let rng = SystemRandom::new();
130    let pkcs8 = EcdsaKeyPair::generate_pkcs8(&ECDSA_P256_SHA256_FIXED_SIGNING, &rng)
131        .map_err(|e| ViError::new(ViErrorKind::KeyUnsupported, format!("keygen: {e}")))?;
132
133    let key_pair = EcdsaKeyPair::from_pkcs8(&ECDSA_P256_SHA256_FIXED_SIGNING, pkcs8.as_ref(), &rng)
134        .map_err(|e| ViError::new(ViErrorKind::KeyUnsupported, format!("parse pkcs8: {e}")))?;
135
136    let pub_bytes = key_pair.public_key().as_ref();
137    let jwk = ec_public_bytes_to_jwk(pub_bytes)?;
138
139    Ok((pkcs8.as_ref().to_vec(), jwk))
140}

142/// Load an `EcdsaKeyPair` from PKCS#8 DER bytes.
143pub fn load_key_pair(pkcs8_der: &[u8]) -> Result<EcdsaKeyPair, ViError> {
144    let rng = SystemRandom::new();
145    EcdsaKeyPair::from_pkcs8(&ECDSA_P256_SHA256_FIXED_SIGNING, pkcs8_der, &rng)
146        .map_err(|e| ViError::new(ViErrorKind::KeyUnsupported, format!("load pkcs8: {e}")))
147}

149/// Convert the raw uncompressed public key bytes (65 bytes: 0x04 || x || y)
150/// into a [`Jwk`].
151pub fn ec_public_bytes_to_jwk(pub_bytes: &[u8]) -> Result<Jwk, ViError> {
152    if pub_bytes.len() != 65 || pub_bytes[0] != 0x04 {
153        return Err(ViError::new(
154            ViErrorKind::KeyUnsupported,
155            "expected 65-byte uncompressed EC point (0x04 || x || y)",
156        ));
157    }
158    Ok(Jwk {
159        kty: "EC".into(),
160        crv: "P-256".into(),
161        x: b64u_encode(&pub_bytes[1..33]),
162        y: b64u_encode(&pub_bytes[33..65]),
163        d: None,
164    })
165}
166
167/// Convert a [`Jwk`] (public) back to raw uncompressed bytes (65 bytes).
168pub fn jwk_to_public_bytes(jwk: &Jwk) -> Result<Vec<u8>, ViError> {
169    if jwk.kty != "EC" || jwk.crv != "P-256" {
170        return Err(ViError::new(
171            ViErrorKind::KeyUnsupported,
172            format!("unsupported key type: {}:{}", jwk.kty, jwk.crv),
173        ));
174    }
175    let x = b64u_decode(&jwk.x)?;
176    let y = b64u_decode(&jwk.y)?;
177    if x.len() != 32 || y.len() != 32 {
178        return Err(ViError::new(
179            ViErrorKind::KeyUnsupported,
180            "x/y coordinates must be 32 bytes each",
181        ));
182    }
183    let mut bytes = Vec::with_capacity(65);
184    bytes.push(0x04);
185    bytes.extend_from_slice(&x);
186    bytes.extend_from_slice(&y);
187    Ok(bytes)
188}

192/// Create a single SD-JWT disclosure: `[salt, claim_name, claim_value]`.
193/// Returns `(disclosure_b64, disclosure_hash)`.
194pub fn create_disclosure(
195    claim_name: &str,
196    claim_value: &serde_json::Value,
197) -> Result<(String, String), ViError> {
198    let rng = SystemRandom::new();
199    let mut salt_bytes = [0u8; 16];
200    ring::rand::SecureRandom::fill(&rng, &mut salt_bytes)
201        .map_err(|e| ViError::new(ViErrorKind::IssuanceInputInvalid, format!("rng: {e}")))?;
202    let salt = b64u_encode(&salt_bytes);
203
204    let disclosure_json = serde_json::json!([salt, claim_name, claim_value]);
205    let disclosure_str = serde_json::to_string(&disclosure_json).map_err(|e| {
206        ViError::new(
207            ViErrorKind::IssuanceInputInvalid,
208            format!("disclosure JSON: {e}"),
209        )
210    })?;
211    let disclosure_b64 = b64u_encode(disclosure_str.as_bytes());
212    let hash = sd_hash(&disclosure_b64);
213    Ok((disclosure_b64, hash))
214}

216/// Serialize an SD-JWT: `issuer_jwt~disclosure1~disclosure2~...~kb_jwt`
217/// (omit `kb_jwt` for L1 which has no key-binding JWT).
218pub fn serialize_sd_jwt(issuer_jwt: &str, disclosures: &[String], kb_jwt: Option<&str>) -> String {
219    let mut result = issuer_jwt.to_string();
220    for d in disclosures {
221        result.push('~');
222        result.push_str(d);
223    }
224    result.push('~');
225    if let Some(kb) = kb_jwt {
226        result.push_str(kb);
227    }
228    result
229}
230
231/// Parse a serialized SD-JWT into (issuer_jwt, disclosures, optional_kb_jwt).
232pub fn parse_sd_jwt(serialized: &str) -> Result<(&str, Vec<&str>, Option<&str>), ViError> {
233    let parts: Vec<&str> = serialized.split('~').collect();
234    if parts.len() < 2 {
235        return Err(ViError::new(
236            ViErrorKind::InvalidDisclosure,
237            "SD-JWT must have at least issuer JWT and trailing ~",
238        ));
239    }
240    let issuer_jwt = parts[0];
241    let last = *parts.last().unwrap();
242    let kb_jwt = if last.is_empty() { None } else { Some(last) };
243
244    let disclosures = parts[1..parts.len() - 1].to_vec();
245
246    Ok((issuer_jwt, disclosures, kb_jwt))
247}
