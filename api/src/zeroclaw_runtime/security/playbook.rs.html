1//! Incident response playbook definitions and execution engine.
2//!
3//! Playbooks define structured response procedures for security incidents.
4//! Each playbook has named steps, some of which require human approval before
5//! execution. Playbooks are loaded from JSON files in the configured directory.
6
7use serde::{Deserialize, Serialize};
8use std::path::Path;
9
10/// A single step in an incident response playbook.
11#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
12pub struct PlaybookStep {
13    /// Machine-readable action identifier (e.g. "isolate_host", "block_ip").
14    pub action: String,
15    /// Human-readable description of what this step does.
16    pub description: String,
17    /// Whether this step requires explicit human approval before execution.
18    #[serde(default)]
19    pub requires_approval: bool,
20    /// Timeout in seconds for this step. Default: 300 (5 minutes).
21    #[serde(default = "default_timeout_secs")]
22    pub timeout_secs: u64,
23}
24
25fn default_timeout_secs() -> u64 {
26    300
27}

29/// An incident response playbook.
30#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
31pub struct Playbook {
32    /// Unique playbook name (e.g. "suspicious_login").
33    pub name: String,
34    /// Human-readable description.
35    pub description: String,
36    /// Ordered list of response steps.
37    pub steps: Vec<PlaybookStep>,
38    /// Minimum alert severity that triggers this playbook (low/medium/high/critical).
39    #[serde(default = "default_severity_filter")]
40    pub severity_filter: String,
41    /// Step indices (0-based) that can be auto-approved when below max_auto_severity.
42    #[serde(default)]
43    pub auto_approve_steps: Vec<usize>,
44}
45
46fn default_severity_filter() -> String {
47    "medium".into()
48}

50/// Result of executing a single playbook step.
51#[derive(Debug, Clone, Serialize, Deserialize)]
52pub struct StepExecutionResult {
53    pub step_index: usize,
54    pub action: String,
55    pub status: StepStatus,
56    pub message: String,
57}
58
59/// Status of a playbook step.
60#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
61pub enum StepStatus {
62    /// Step completed successfully.
63    Completed,
64    /// Step is waiting for human approval.
65    PendingApproval,
66    /// Step was skipped (e.g. not applicable).
67    Skipped,
68    /// Step failed with an error.
69    Failed,
70}
71
72impl std::fmt::Display for StepStatus {
73    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
74        match self {
75            Self::Completed => write!(f, "completed"),
76            Self::PendingApproval => write!(f, "pending_approval"),
77            Self::Skipped => write!(f, "skipped"),
78            Self::Failed => write!(f, "failed"),
79        }
80    }
81}

83/// Load all playbook definitions from a directory of JSON files.
84pub fn load_playbooks(dir: &Path) -> Vec<Playbook> {
85    let mut playbooks = Vec::new();
86
87    if !dir.exists() || !dir.is_dir() {
88        return builtin_playbooks();
89    }
90
91    if let Ok(entries) = std::fs::read_dir(dir) {
92        for entry in entries.flatten() {
93            let path = entry.path();
94            if path.extension().is_some_and(|ext| ext == "json") {
95                match std::fs::read_to_string(&path) {
96                    Ok(contents) => match serde_json::from_str::<Playbook>(&contents) {
97                        Ok(pb) => playbooks.push(pb),
98                        Err(e) => {
99                            tracing::warn!("Failed to parse playbook {}: {e}", path.display());
100                        }
101                    },
102                    Err(e) => {
103                        tracing::warn!("Failed to read playbook {}: {e}", path.display());
104                    }
105                }
106            }
107        }
108    }
109
110    // Merge built-in playbooks that aren't overridden by user-defined ones
111    for builtin in builtin_playbooks() {
112        if !playbooks.iter().any(|p| p.name == builtin.name) {
113            playbooks.push(builtin);
114        }
115    }
116
117    playbooks
118}

120/// Severity ordering for comparison: low < medium < high < critical.
121pub fn severity_level(severity: &str) -> u8 {
122    match severity.to_lowercase().as_str() {
123        "low" => 1,
124        "medium" => 2,
125        "high" => 3,
126        "critical" => 4,
127        // Deny-by-default: unknown severities get the highest level to prevent
128        // auto-approval of unrecognized severity labels.
129        _ => u8::MAX,
130    }
131}

133/// Check whether a step can be auto-approved given config constraints.
134pub fn can_auto_approve(
135    playbook: &Playbook,
136    step_index: usize,
137    alert_severity: &str,
138    max_auto_severity: &str,
139) -> bool {
140    // Never auto-approve if alert severity exceeds the configured max
141    if severity_level(alert_severity) > severity_level(max_auto_severity) {
142        return false;
143    }
144
145    // Only auto-approve steps explicitly listed in auto_approve_steps
146    playbook.auto_approve_steps.contains(&step_index)
147}

149/// Evaluate a playbook step. Returns the result with approval gating.
150///
151/// Steps that require approval and cannot be auto-approved will return
152/// `StepStatus::PendingApproval` without executing.
153pub fn evaluate_step(
154    playbook: &Playbook,
155    step_index: usize,
156    alert_severity: &str,
157    max_auto_severity: &str,
158    require_approval: bool,
159) -> StepExecutionResult {
160    let step = match playbook.steps.get(step_index) {
161        Some(s) => s,
162        None => {
163            return StepExecutionResult {
164                step_index,
165                action: "unknown".into(),
166                status: StepStatus::Failed,
167                message: format!("Step index {step_index} out of range"),
168            };
169        }
170    };
171
172    // Enforce approval gates: steps that require approval must either be
173    // auto-approved or wait for human approval. Never mark an unexecuted
174    // approval-gated step as Completed.
175    if step.requires_approval
176        && (!require_approval
177            || !can_auto_approve(playbook, step_index, alert_severity, max_auto_severity))
178    {
179        return StepExecutionResult {
180            step_index,
181            action: step.action.clone(),
182            status: StepStatus::PendingApproval,
183            message: format!(
184                "Step '{}' requires human approval (severity: {alert_severity})",
185                step.description
186            ),
187        };
188    }
189
190    // Step is approved (either doesn't require approval, or was auto-approved)
191    // Actual execution would be delegated to the appropriate tool/system
192    StepExecutionResult {
193        step_index,
194        action: step.action.clone(),
195        status: StepStatus::Completed,
196        message: format!("Executed: {}", step.description),
197    }
198}

349    #[test]
350    fn severity_level_ordering() {
351        assert!(severity_level("low") < severity_level("medium"));
352        assert!(severity_level("medium") < severity_level("high"));
353        assert!(severity_level("high") < severity_level("critical"));
354        assert_eq!(severity_level("unknown"), u8::MAX);
355    }

357    #[test]
358    fn auto_approve_respects_severity_cap() {
359        let pb = &builtin_playbooks()[0]; // suspicious_login
360
361        // Step 0 is in auto_approve_steps
362        assert!(can_auto_approve(pb, 0, "low", "low"));
363        assert!(can_auto_approve(pb, 0, "low", "medium"));
364
365        // Alert severity exceeds max -> cannot auto-approve
366        assert!(!can_auto_approve(pb, 0, "high", "low"));
367        assert!(!can_auto_approve(pb, 0, "critical", "medium"));
368
369        // Step 2 is NOT in auto_approve_steps
370        assert!(!can_auto_approve(pb, 2, "low", "critical"));
371    }

375        let pb = &builtin_playbooks()[0]; // suspicious_login

377        // Step 2 (notify_user) requires approval, high severity, max=low -> pending
378        let result = evaluate_step(pb, 2, "high", "low", true);
379        assert_eq!(result.status, StepStatus::PendingApproval);
