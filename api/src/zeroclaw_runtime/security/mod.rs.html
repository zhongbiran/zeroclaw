1//! Security subsystem for policy enforcement, sandboxing, and secret management.
2//!
3//! This module provides the security infrastructure for ZeroClaw. The core type
4//! [`SecurityPolicy`] defines autonomy levels, workspace boundaries, and
5//! access-control rules that are enforced across the tool and runtime subsystems.
6//! [`PairingGuard`] implements device pairing for channel authentication, and
7//! [`SecretStore`] handles encrypted credential storage.
8//!
9//! OS-level isolation is provided through the [`Sandbox`] trait defined in
10//! [`traits`], with pluggable backends including Docker, Firejail, Bubblewrap,
11//! and Landlock. The [`create_sandbox`] function selects the best available
12//! backend at runtime. An [`AuditLogger`] records security-relevant events for
13//! forensic review.
14//!
15//! # Extension
16//!
17//! To add a new sandbox backend, implement [`Sandbox`] in a new submodule and
18//! register it in [`detect::create_sandbox`]. See `AGENTS.md` §7.5 for security
19//! change guidelines.

81/// Redact sensitive values for safe logging. Shows first 4 characters + "***" suffix.
82/// Uses char-boundary-safe indexing to avoid panics on multi-byte UTF-8 strings.
83/// This function intentionally breaks the data-flow taint chain for static analysis.
84pub fn redact(value: &str) -> String {
85    let char_count = value.chars().count();
86    if char_count <= 4 {
87        "***".to_string()
88    } else {
89        let prefix: String = value.chars().take(4).collect();
90        format!("{prefix}***")
91    }
92}

94#[cfg(test)]
95mod tests {
96    use super::*;
97
98    #[test]
99    fn reexported_policy_and_pairing_types_are_usable() {
100        let policy = SecurityPolicy::default();
101        assert_eq!(policy.autonomy, AutonomyLevel::Supervised);
102
103        let guard = PairingGuard::new(false, &[]);
104        assert!(!guard.require_pairing());
105    }
106
107    #[test]
108    fn reexported_secret_store_encrypt_decrypt_roundtrip() {
109        let temp = tempfile::tempdir().unwrap();
110        let store = SecretStore::new(temp.path(), false);
111
112        let encrypted = store.encrypt("top-secret").unwrap();
113        let decrypted = store.decrypt(&encrypted).unwrap();
114
115        assert_eq!(decrypted, "top-secret");
116    }
117
118    #[test]
119    fn redact_hides_most_of_value() {
120        assert_eq!(redact("abcdefgh"), "abcd***");
121        assert_eq!(redact("ab"), "***");
122        assert_eq!(redact(""), "***");
123        assert_eq!(redact("12345"), "1234***");
124    }
125
126    #[test]
127    fn redact_handles_multibyte_utf8_without_panic() {
128        // CJK characters are 3 bytes each; slicing at byte 4 would panic
129        // without char-boundary-safe handling.
130        let result = redact("密码是很长的秘密");
131        assert!(result.ends_with("***"));
132        assert!(result.is_char_boundary(result.len()));
133    }
