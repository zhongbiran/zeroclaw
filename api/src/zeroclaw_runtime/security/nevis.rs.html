1//! Nevis IAM authentication provider for ZeroClaw.
2//!
3//! Integrates with Nevis Security Suite (Adnovum) for OAuth2/OIDC token
4//! validation, FIDO2/passkey verification, and session management. Maps Nevis
5//! roles to ZeroClaw tool permissions via [`super::iam_policy::IamPolicy`].
6
7use anyhow::{Context, Result, bail};
8use serde::{Deserialize, Serialize};
9use std::time::Duration;
10
11/// Identity resolved from a validated Nevis token or session.
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct NevisIdentity {
14    /// Unique user identifier from Nevis.
15    pub user_id: String,
16    /// Nevis roles assigned to this user.
17    pub roles: Vec<String>,
18    /// OAuth2 scopes granted to this session.
19    pub scopes: Vec<String>,
20    /// Whether the user completed MFA (FIDO2/passkey/OTP) in this session.
21    pub mfa_verified: bool,
22    /// When this session expires (seconds since UNIX epoch).
23    pub session_expiry: u64,
24}
25
26/// Token validation strategy.
27#[derive(Debug, Clone, Copy, PartialEq, Eq)]
28pub enum TokenValidationMode {
29    /// Validate JWT locally using cached JWKS keys.
30    Local,
31    /// Validate token by calling the Nevis introspection endpoint.
32    Remote,
33}
34
35impl TokenValidationMode {
36    pub fn from_str_config(s: &str) -> Result<Self> {
37        match s.to_ascii_lowercase().as_str() {
38            "local" => Ok(Self::Local),
39            "remote" => Ok(Self::Remote),
40            other => bail!("invalid token_validation mode '{other}': expected 'local' or 'remote'"),
41        }
42    }
43}

45/// Authentication provider backed by a Nevis instance.
46///
47/// Validates tokens, manages sessions, and resolves identities. The provider
48/// is designed to be shared across concurrent requests (`Send + Sync`).
49pub struct NevisAuthProvider {
50    /// Base URL of the Nevis instance (e.g. `https://nevis.example.com`).
51    instance_url: String,
52    /// Nevis realm to authenticate against.
53    realm: String,
54    /// OAuth2 client ID registered in Nevis.
55    client_id: String,
56    /// OAuth2 client secret (decrypted at startup).
57    client_secret: Option<String>,
58    /// Token validation strategy.
59    validation_mode: TokenValidationMode,
60    /// JWKS endpoint for local token validation.
61    jwks_url: Option<String>,
62    /// Whether MFA is required for all authentications.
63    require_mfa: bool,
64    /// Session timeout duration.
65    session_timeout: Duration,
66    /// HTTP client for Nevis API calls.
67    http_client: reqwest::Client,
68}

98impl NevisAuthProvider {
99    /// Create a new Nevis auth provider from config values.
100    ///
101    /// `client_secret` should already be decrypted by the config loader.
102    pub fn new(
103        instance_url: String,
104        realm: String,
105        client_id: String,
106        client_secret: Option<String>,
107        token_validation: &str,
108        jwks_url: Option<String>,
109        require_mfa: bool,
110        session_timeout_secs: u64,
111    ) -> Result<Self> {
112        let validation_mode = TokenValidationMode::from_str_config(token_validation)?;
113
114        if validation_mode == TokenValidationMode::Local && jwks_url.is_none() {
115            bail!(
116                "Nevis token_validation is 'local' but no jwks_url is configured. \
117                 Either set jwks_url or use token_validation = 'remote'."
118            );
119        }
120
121        let http_client = reqwest::Client::builder()
122            .timeout(Duration::from_secs(30))
123            .build()
124            .context("Failed to create HTTP client for Nevis")?;
125
126        Ok(Self {
127            instance_url,
128            realm,
129            client_id,
130            client_secret,
131            validation_mode,
132            jwks_url,
133            require_mfa,
134            session_timeout: Duration::from_secs(session_timeout_secs),
135            http_client,
136        })
137    }

139    /// Validate a bearer token and resolve the caller's identity.
140    ///
141    /// Returns `NevisIdentity` on success, or an error if the token is invalid,
142    /// expired, or MFA requirements are not met.
143    pub async fn validate_token(&self, token: &str) -> Result<NevisIdentity> {
144        if token.is_empty() {
145            bail!("empty bearer token");
146        }
147
148        let identity = match self.validation_mode {
149            TokenValidationMode::Local => self.validate_token_local(token).await?,
150            TokenValidationMode::Remote => self.validate_token_remote(token).await?,
151        };
152
153        if self.require_mfa && !identity.mfa_verified {
154            bail!(
155                "MFA is required but user '{}' has not completed MFA verification",
156                crate::security::redact(&identity.user_id)
157            );
158        }
159
160        let now = std::time::SystemTime::now()
161            .duration_since(std::time::UNIX_EPOCH)
162            .unwrap_or_default()
163            .as_secs();
164
165        if identity.session_expiry > 0 && identity.session_expiry < now {
166            bail!("Nevis session expired");
167        }
168
169        Ok(identity)
170    }

172    /// Validate token by calling the Nevis introspection endpoint.
173    async fn validate_token_remote(&self, token: &str) -> Result<NevisIdentity> {
174        let introspect_url = format!(
175            "{}/auth/realms/{}/protocol/openid-connect/token/introspect",
176            self.instance_url.trim_end_matches('/'),
177            self.realm,
178        );
179
180        let mut form = vec![("token", token), ("client_id", &self.client_id)];
181        // client_secret is optional (public clients don't need it)
182        let secret_ref;
183        if let Some(ref secret) = self.client_secret {
184            secret_ref = secret.as_str();
185            form.push(("client_secret", secret_ref));
186        }
187
188        let resp = self
189            .http_client
190            .post(&introspect_url)
191            .form(&form)
192            .send()
193            .await
194            .context("Failed to reach Nevis introspection endpoint")?;
195
196        if !resp.status().is_success() {
197            bail!(
198                "Nevis introspection returned HTTP {}",
199                resp.status().as_u16()
200            );
201        }
202
203        let body: IntrospectionResponse = resp
204            .json()
205            .await
206            .context("Failed to parse Nevis introspection response")?;
207
208        if !body.active {
209            bail!("Token is not active (revoked or expired)");
210        }
211
212        let user_id = body
213            .sub
214            .filter(|s| !s.trim().is_empty())
215            .context("Token has missing or empty `sub` claim")?;
216
217        let mut roles = body.realm_access.map(|ra| ra.roles).unwrap_or_default();
218        roles.sort();
219        roles.dedup();
220
221        Ok(NevisIdentity {
222            user_id,
223            roles,
224            scopes: body
225                .scope
226                .unwrap_or_default()
227                .split_whitespace()
228                .map(String::from)
229                .collect(),
230            mfa_verified: body.acr.as_deref() == Some("mfa")
231                || body
232                    .amr
233                    .iter()
234                    .flatten()
235                    .any(|m| m == "fido2" || m == "passkey" || m == "otp" || m == "webauthn"),
236            session_expiry: body.exp.unwrap_or(0),
237        })
238    }

240    /// Validate token locally using JWKS.
241    ///
242    /// Local JWT/JWKS validation is not yet implemented. Rather than silently
243    /// falling back to the remote introspection endpoint (which would hide a
244    /// misconfiguration), this returns an explicit error directing the operator
245    /// to use `token_validation = "remote"` until local JWKS support is added.
246    #[allow(clippy::unused_async)] // Will use async when JWKS validation is implemented
247    async fn validate_token_local(&self, token: &str) -> Result<NevisIdentity> {
248        // JWT structure check: header.payload.signature
249        let parts: Vec<&str> = token.split('.').collect();
250        if parts.len() != 3 {
251            bail!("Invalid JWT structure: expected 3 dot-separated parts");
252        }
253
254        bail!(
255            "Local JWKS token validation is not yet implemented. \
256             Set token_validation = \"remote\" to use the Nevis introspection endpoint."
257        );
258    }

260    /// Validate a Nevis session token (cookie-based sessions).
261    pub async fn validate_session(&self, session_token: &str) -> Result<NevisIdentity> {
262        if session_token.is_empty() {
263            bail!("empty session token");
264        }
265
266        let session_url = format!(
267            "{}/auth/realms/{}/protocol/openid-connect/userinfo",
268            self.instance_url.trim_end_matches('/'),
269            self.realm,
270        );
271
272        let resp = self
273            .http_client
274            .get(&session_url)
275            .bearer_auth(session_token)
276            .send()
277            .await
278            .context("Failed to reach Nevis userinfo endpoint")?;
279
280        if !resp.status().is_success() {
281            bail!(
282                "Nevis session validation returned HTTP {}",
283                resp.status().as_u16()
284            );
285        }
286
287        let body: UserInfoResponse = resp
288            .json()
289            .await
290            .context("Failed to parse Nevis userinfo response")?;
291
292        if body.sub.trim().is_empty() {
293            bail!("Userinfo response has missing or empty `sub` claim");
294        }
295
296        let now = std::time::SystemTime::now()
297            .duration_since(std::time::UNIX_EPOCH)
298            .unwrap_or_default()
299            .as_secs();
300
301        let mut roles = body.realm_access.map(|ra| ra.roles).unwrap_or_default();
302        roles.sort();
303        roles.dedup();
304
305        let identity = NevisIdentity {
306            user_id: body.sub,
307            roles,
308            scopes: body
309                .scope
310                .unwrap_or_default()
311                .split_whitespace()
312                .map(String::from)
313                .collect(),
314            mfa_verified: body.acr.as_deref() == Some("mfa")
315                || body
316                    .amr
317                    .iter()
318                    .flatten()
319                    .any(|m| m == "fido2" || m == "passkey" || m == "otp" || m == "webauthn"),
320            session_expiry: now + self.session_timeout.as_secs(),
321        };
322
323        if self.require_mfa && !identity.mfa_verified {
324            bail!(
325                "MFA is required but user '{}' has not completed MFA verification",
326                crate::security::redact(&identity.user_id)
327            );
328        }
329
330        Ok(identity)
331    }

333    /// Health check against the Nevis instance.
334    pub async fn health_check(&self) -> Result<()> {
335        let health_url = format!(
336            "{}/auth/realms/{}",
337            self.instance_url.trim_end_matches('/'),
338            self.realm,
339        );
340
341        let resp = self
342            .http_client
343            .get(&health_url)
344            .send()
345            .await
346            .context("Nevis health check failed: cannot reach instance")?;
347
348        if !resp.status().is_success() {
349            bail!("Nevis health check failed: HTTP {}", resp.status().as_u16());
350        }
351
352        Ok(())
353    }

368#[derive(Debug, Deserialize)]
369struct IntrospectionResponse {
370    active: bool,
371    sub: Option<String>,
372    scope: Option<String>,
373    exp: Option<u64>,
374    #[serde(rename = "realm_access")]
375    realm_access: Option<RealmAccess>,
376    /// Authentication Context Class Reference
377    acr: Option<String>,
378    /// Authentication Methods References
379    amr: Option<Vec<String>>,
380}
381
382#[derive(Debug, Deserialize)]
383struct RealmAccess {
384    #[serde(default)]
385    roles: Vec<String>,
386}
387
388#[derive(Debug, Deserialize)]
389struct UserInfoResponse {
390    sub: String,
391    #[serde(rename = "realm_access")]
392    realm_access: Option<RealmAccess>,
393    scope: Option<String>,
394    acr: Option<String>,
395    /// Authentication Methods References
396    amr: Option<Vec<String>>,
397}
