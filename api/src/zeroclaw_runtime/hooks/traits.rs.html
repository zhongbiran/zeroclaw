9/// Result of a modifying hook — continue with (possibly modified) data, or cancel.
10#[derive(Debug, Clone)]
11pub enum HookResult<T> {
12    Continue(T),
13    Cancel(String),
14}
15
16impl<T> HookResult<T> {
17    pub fn is_cancel(&self) -> bool {
18        matches!(self, HookResult::Cancel(_))
19    }
20}

22/// Trait for hook handlers. All methods have default no-op implementations.
23/// Implement only the events you care about.
24#[async_trait]
25pub trait HookHandler: Send + Sync {
26    fn name(&self) -> &str;
27    fn priority(&self) -> i32 {
28        0
29    }
30
31    // --- Void hooks (parallel, fire-and-forget) ---
32    async fn on_gateway_start(&self, _host: &str, _port: u16) {}
33    async fn on_gateway_stop(&self) {}
34    async fn on_session_start(&self, _session_id: &str, _channel: &str) {}
35    async fn on_session_end(&self, _session_id: &str, _channel: &str) {}
36    async fn on_llm_input(&self, _messages: &[ChatMessage], _model: &str) {}
37    async fn on_llm_output(&self, _response: &ChatResponse) {}
38    async fn on_after_tool_call(&self, _tool: &str, _result: &ToolResult, _duration: Duration) {}
39    async fn on_message_sent(&self, _channel: &str, _recipient: &str, _content: &str) {}
40    async fn on_heartbeat_tick(&self) {}
41
42    // --- Modifying hooks (sequential by priority, can cancel) ---
43    async fn before_model_resolve(
44        &self,
45        provider: String,
46        model: String,
47    ) -> HookResult<(String, String)> {
48        HookResult::Continue((provider, model))
49    }
50
51    async fn before_prompt_build(&self, prompt: String) -> HookResult<String> {
52        HookResult::Continue(prompt)
53    }
54
55    async fn before_llm_call(
56        &self,
57        messages: Vec<ChatMessage>,
58        model: String,
59    ) -> HookResult<(Vec<ChatMessage>, String)> {
60        HookResult::Continue((messages, model))
61    }
62
63    async fn before_tool_call(&self, name: String, args: Value) -> HookResult<(String, Value)> {
64        HookResult::Continue((name, args))
65    }
66
67    async fn on_message_received(&self, message: ChannelMessage) -> HookResult<ChannelMessage> {
68        HookResult::Continue(message)
69    }
70
71    async fn on_message_sending(
72        &self,
73        channel: String,
74        recipient: String,
75        content: String,
76    ) -> HookResult<(String, String, String)> {
77        HookResult::Continue((channel, recipient, content))
78    }
79}

109    #[test]
110    fn hook_result_is_cancel() {
111        let ok: HookResult<String> = HookResult::Continue("hi".into());
112        assert!(!ok.is_cancel());
113        let cancel: HookResult<String> = HookResult::Cancel("blocked".into());
114        assert!(cancel.is_cancel());
