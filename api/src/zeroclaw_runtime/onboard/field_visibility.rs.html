1//! Per-section field visibility + default-injection helpers.
2//!
3//! Used by both the CLI wizard (`onboard::offer_advanced_settings`) and the
4//! gateway HTTP endpoints (`/api/onboard/sections/.../items/...` for default
5//! injection, `/api/config/list` for filtering). One source of truth so the
6//! CLI and dashboard can't disagree about which fields apply to a given
7//! provider / memory backend / etc.
8//!
9//! Follow-up (#5960 deferral): a `#[configurable(family = "...")]` schema
10//! attribute would let the `Configurable` derive emit these filters
11//! automatically. Until that lands, the per-family lists below are the
12//! pragmatic stand-in.
13//!
14//! Issue #6175.
15

19/// Exclude list for `[providers.models.<name>]` field walks.
20///
21/// Returns leaf field names (kebab-case, the suffix after the
22/// `providers.models.<name>.` prefix in `prop_fields()`). A field absent
23/// from this list is shown for every provider.
24pub fn provider_family_excludes(provider: &str) -> Vec<&'static str> {
25    let mut out = Vec::new();
26    if provider != "azure_openai" {
27        out.push("azure-openai-resource");
28        out.push("azure-openai-deployment");
29        out.push("azure-openai-api-version");
30    }
31    if !matches!(provider, "openai" | "openai_codex") {
32        out.push("wire-api");
33        out.push("requires-openai-auth");
34    }
35    out
36}

38/// Exclude list for the top-level `[memory]` walk based on the active backend.
39///
40/// `MemoryConfig` carries fields and nested subsections for every backend
41/// (sqlite-only knobs, `[memory.qdrant]`, `[memory.postgres]`); only the
42/// active backend's surface is relevant. Each entry is a path SUFFIX after
43/// the `memory.` prefix in `prop_fields()`. Sub-table fields are matched
44/// by leading segment (`qdrant.`, `postgres.`).
45pub fn memory_backend_excludes(backend: &str) -> Vec<&'static str> {
46    let mut out = Vec::new();
47    if backend != "sqlite" {
48        out.push("sqlite-open-timeout-secs");
49        out.push("conversation-retention-days");
50    }
51    if backend != "qdrant" {
52        out.push("qdrant.");
53    }
54    if backend != "postgres" {
55        out.push("postgres.");
56    }
57    out
58}

60/// Compute the set of full property paths to hide when a client requests
61/// `prefix`. Returns an empty vec for prefixes that don't have visibility
62/// rules (most of the schema).
63///
64/// This is the single entry point the gateway's `/api/config/list` handler
65/// calls — it inspects the requested prefix, looks at the live config to
66/// resolve any state-dependent rules (e.g. `memory.backend`), and returns
67/// the absolute paths to drop from the response.
68pub fn excluded_paths(cfg: &Config, prefix: &str) -> Vec<String> {
69    if let Some(provider_key) = providers_models_key(prefix) {
70        return provider_family_excludes(provider_key)
71            .into_iter()
72            .map(|leaf| format!("providers.models.{provider_key}.{leaf}"))
73            .collect();
74    }
75
76    if prefix == "memory" || prefix.is_empty() {
77        let backend = if cfg.memory.backend.is_empty() {
78            "sqlite"
79        } else {
80            cfg.memory.backend.as_str()
81        };
82        return memory_backend_excludes(backend)
83            .into_iter()
84            .map(|leaf| {
85                if leaf.ends_with('.') {
86                    // Sub-table prefix — represent as a path-prefix marker
87                    // that callers match with `starts_with`.
88                    format!("memory.{leaf}")
89                } else {
90                    format!("memory.{leaf}")
91                }
92            })
93            .collect();
94    }
95
96    Vec::new()
97}

99/// Test whether `path` is one of the excluded entries returned from
100/// `excluded_paths`. Handles both exact matches and sub-table prefix
101/// markers (`"memory.qdrant."` matches every `memory.qdrant.*`).
102pub fn is_excluded(path: &str, excludes: &[String]) -> bool {
103    excludes
104        .iter()
105        .any(|e| path == e || (e.ends_with('.') && path.starts_with(e)))
106}

108/// Extract the `<key>` from a `providers.models.<key>` prefix. Returns
109/// `None` when the prefix isn't a per-model walk.
110pub fn providers_models_key(prefix: &str) -> Option<&str> {
111    let rest = prefix.strip_prefix("providers.models.")?;
112    if rest.is_empty() || rest.contains('.') {
113        return None;
114    }
115    Some(rest)
116}

118/// Walk a freshly-constructed defaults struct (from
119/// `zeroclaw_providers::default_provider_config`, etc.) and copy its
120/// populated values into `cfg` under `target_prefix`, but only for paths
121/// the user hasn't already set.
122///
123/// Schema-driven: enumerates fields via the source's `prop_fields()`
124/// macro output. Adding a new field to the source struct propagates
125/// automatically — no per-field plumbing here.
126///
127/// `source_prefix` is the source struct's `configurable_prefix()` (e.g.
128/// `providers.models` for `ModelProviderConfig`). Each leaf gets rebased
129/// onto `target_prefix` before writing.
130fn apply_typed_defaults<S>(cfg: &mut Config, defaults: &S, source_prefix: &str, target_prefix: &str)
131where
132    S: PropFieldSource,
133{
134    let source_dot = format!("{source_prefix}.");
135    let target_dot = format!("{target_prefix}.");
136    for field in defaults.prop_fields() {
137        // Only direct leaves of the defaults struct — nested sub-tables
138        // need their own walks (they have their own configurable_prefix).
139        let Some(leaf) = field.name.strip_prefix(&source_dot) else {
140            continue;
141        };
142        if leaf.contains('.') {
143            continue;
144        }
145        // Sparse defaults: an unset field on the typed default struct means
146        // the source has no opinion; leave the target alone.
147        if field.display_value == "<unset>" {
148            continue;
149        }
150        let target_path = format!("{target_dot}{leaf}");
151        let current = cfg.get_prop(&target_path).unwrap_or_default();
152        if !current.is_empty() && current != "<unset>" {
153            // User has already set this field — don't clobber.
154            continue;
155        }
156        let _ = cfg.set_prop(&target_path, &field.display_value);
157    }
158}

160/// Trait-object-friendly view onto any `Configurable`-derived struct's
161/// `prop_fields()`. Lets `apply_typed_defaults` accept any source struct
162/// without naming concrete types.
163trait PropFieldSource {
164    fn prop_fields(&self) -> Vec<zeroclaw_config::traits::PropFieldInfo>;
165}
166
167impl PropFieldSource for zeroclaw_config::schema::ModelProviderConfig {
168    fn prop_fields(&self) -> Vec<zeroclaw_config::traits::PropFieldInfo> {
169        zeroclaw_config::schema::ModelProviderConfig::prop_fields(self)
170    }
171}

173/// Pre-populate the per-provider entry under `prefix` with the named
174/// provider's trait-derived defaults. Idempotent: existing user-set
175/// values aren't touched.
176///
177/// Source of truth is `zeroclaw_providers::default_provider_config(name)`
178/// — a typed `ModelProviderConfig`. We walk its `prop_fields()` and
179/// rebase each leaf onto `prefix` so e.g. the source's
180/// `providers.models.base-url` becomes the target's
181/// `providers.models.<key>.base-url`.
182pub fn apply_provider_trait_defaults(cfg: &mut Config, provider: &str, prefix: &str) -> Result<()> {
183    let defaults = zeroclaw_providers::default_provider_config(provider);
184    let source_prefix = zeroclaw_config::schema::ModelProviderConfig::configurable_prefix();
185    apply_typed_defaults(cfg, &defaults, source_prefix, prefix);
186    Ok(())
187}
