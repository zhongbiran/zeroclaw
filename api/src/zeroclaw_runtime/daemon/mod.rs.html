12/// Why the daemon's main loop returned.
13///
14/// `Shutdown`: process exits cleanly. `Reload`: caller (typically `src/main.rs`)
15/// re-reads the config from disk and calls `daemon::run` again. The PID stays
16/// the same; only the in-process subsystems get torn down and re-instantiated.
17#[derive(Debug, Clone, Copy, PartialEq, Eq)]
18pub enum DaemonExit {
19    Shutdown,
20    Reload,
21}
22
23/// Wait for either a shutdown signal (SIGINT / SIGTERM / Ctrl+C) or an
24/// in-process reload signal (the gateway's `/admin/reload` writes `true`
25/// on the watch channel). Returns the reason so the outer loop can decide
26/// whether to re-init or exit. SIGHUP is ignored on Unix so the daemon
27/// survives terminal / SSH disconnects.
28///
29/// The reload trigger is a tokio watch channel (not an OS signal) so it
30/// works identically on Linux, macOS, and Windows. The Sender is owned by
31/// the daemon (created in `run`) and cloned to the gateway for AppState.
32async fn wait_for_exit_signal(
33    mut reload_rx: tokio::sync::watch::Receiver<bool>,
34) -> Result<DaemonExit> {
35    #[cfg(unix)]
36    {
37        use tokio::signal::unix::{SignalKind, signal};
38
39        let mut sigint = signal(SignalKind::interrupt())?;
40        let mut sigterm = signal(SignalKind::terminate())?;
41        let mut sighup = signal(SignalKind::hangup())?;
42
43        loop {
44            tokio::select! {
45                _ = sigint.recv() => {
46                    tracing::info!("Received SIGINT, shutting down...");
47                    return Ok(DaemonExit::Shutdown);
48                }
49                _ = sigterm.recv() => {
50                    tracing::info!("Received SIGTERM, shutting down...");
51                    return Ok(DaemonExit::Shutdown);
52                }
53                _ = sighup.recv() => {
54                    tracing::info!("Received SIGHUP, ignoring (daemon stays running)");
55                }
56                changed = reload_rx.changed() => {
57                    if changed.is_err() {
58                        // Sender dropped — treat as shutdown (shouldn't
59                        // happen in normal operation; the gateway holds a
60                        // clone for the lifetime of the daemon).
61                        tracing::warn!("Reload sender dropped; shutting down");
62                        return Ok(DaemonExit::Shutdown);
63                    }
64                    if *reload_rx.borrow_and_update() {
65                        tracing::info!("Reload requested via /admin/reload");
66                        return Ok(DaemonExit::Reload);
67                    }
68                }
69            }
70        }
71    }
72
73    #[cfg(not(unix))]
74    {
75        loop {
76            tokio::select! {
77                res = tokio::signal::ctrl_c() => {
78                    res?;
79                    tracing::info!("Received Ctrl+C, shutting down...");
80                    return Ok(DaemonExit::Shutdown);
81                }
82                changed = reload_rx.changed() => {
83                    if changed.is_err() {
84                        tracing::warn!("Reload sender dropped; shutting down");
85                        return Ok(DaemonExit::Shutdown);
86                    }
87                    if *reload_rx.borrow_and_update() {
88                        tracing::info!("Reload requested via /admin/reload");
89                        return Ok(DaemonExit::Reload);
90                    }
91                }
92            }
93        }
94    }
95}

137pub async fn run(
138    config: Config,
139    host: String,
140    port: u16,
141    subsystems: DaemonSubsystems,
142) -> Result<DaemonExit> {
143    let initial_backoff = config.reliability.channel_initial_backoff_secs.max(1);
144    let max_backoff = config
145        .reliability
146        .channel_max_backoff_secs
147        .max(initial_backoff);
148
149    crate::health::mark_component_ok("daemon");
150
151    // Shared broadcast channel so all daemon components (gateway, cron,
152    // heartbeat) can publish real-time events to dashboard clients.
153    let (event_tx, _rx) = tokio::sync::broadcast::channel::<serde_json::Value>(256);
154
155    if config.heartbeat.enabled {
156        let _ =
157            crate::heartbeat::engine::HeartbeatEngine::ensure_heartbeat_file(&config.workspace_dir)
158                .await;
159    }
160
161    let mut handles: Vec<JoinHandle<()>> = vec![spawn_state_writer(config.clone())];
162
163    // Reload channel: gateway's /admin/reload writes here; our wait loop
164    // (below) selects on it alongside OS signals. Cross-platform.
165    let (reload_tx, reload_rx) = tokio::sync::watch::channel::<bool>(false);
166
167    if let Some(gateway_start) = subsystems.gateway_start {
168        let gateway_cfg = config.clone();
169        let gateway_host = host.clone();
170        let gateway_event_tx = event_tx.clone();
171        let gateway_reload_tx = reload_tx.clone();
172        let gateway_start = std::sync::Arc::new(gateway_start);
173        handles.push(spawn_component_supervisor(
174            "gateway",
175            initial_backoff,
176            max_backoff,
177            Some(event_tx.clone()),
178            move || {
179                let cfg = gateway_cfg.clone();
180                let host = gateway_host.clone();
181                let tx = gateway_event_tx.clone();
182                let reload = gateway_reload_tx.clone();
183                let start = gateway_start.clone();
184                async move { start(host, port, cfg, Some(tx), Some(reload)).await }
185            },
186        ));
187    }
188
189    if let Some(channels_start) = subsystems.channels_start {
190        if has_supervised_channels(&config) {
191            let channels_cfg = config.clone();
192            let channels_start = std::sync::Arc::new(channels_start);
193            handles.push(spawn_component_supervisor(
194                "channels",
195                initial_backoff,
196                max_backoff,
197                Some(event_tx.clone()),
198                move || {
199                    let cfg = channels_cfg.clone();
200                    let start = channels_start.clone();
201                    async move { start(cfg).await }
202                },
203            ));
204        } else {
205            crate::health::mark_component_ok("channels");
206            tracing::info!("No channels configured; channel supervisor disabled");
207        }
208    } else {
209        crate::health::mark_component_ok("channels");
210        tracing::info!("Channels subsystem not wired; channel supervisor disabled");
211    }
212
213    // Wire up MQTT SOP listener if configured and enabled
214    if let Some(mqtt_start) = subsystems.mqtt_start {
215        if let Some(ref mqtt_config) = config.channels.mqtt {
216            if mqtt_config.enabled {
217                let mqtt_cfg = mqtt_config.clone();
218                let mqtt_start = std::sync::Arc::new(mqtt_start);
219                handles.push(spawn_component_supervisor(
220                    "mqtt",
221                    initial_backoff,
222                    max_backoff,
223                    Some(event_tx.clone()),
224                    move || {
225                        let cfg = mqtt_cfg.clone();
226                        let start = mqtt_start.clone();
227                        async move { start(cfg).await }
228                    },
229                ));
230            } else {
231                tracing::info!("MQTT channel configured but disabled (enabled = false)");
232                crate::health::mark_component_ok("mqtt");
233            }
234        } else {
235            crate::health::mark_component_ok("mqtt");
236        }
237    } else {
238        crate::health::mark_component_ok("mqtt");
239    }
240
241    if config.heartbeat.enabled {
242        let heartbeat_cfg = config.clone();
243        handles.push(spawn_component_supervisor(
244            "heartbeat",
245            initial_backoff,
246            max_backoff,
247            Some(event_tx.clone()),
248            move || {
249                let cfg = heartbeat_cfg.clone();
250                async move { Box::pin(run_heartbeat_worker(cfg)).await }
251            },
252        ));
253    }
254
255    if config.cron.enabled {
256        let scheduler_cfg = config.clone();
257        let scheduler_event_tx = event_tx.clone();
258        handles.push(spawn_component_supervisor(
259            "scheduler",
260            initial_backoff,
261            max_backoff,
262            Some(event_tx.clone()),
263            move || {
264                let cfg = scheduler_cfg.clone();
265                let tx = scheduler_event_tx.clone();
266                async move { Box::pin(crate::cron::scheduler::run(cfg, Some(tx))).await }
267            },
268        ));
269    } else {
270        crate::health::mark_component_ok("scheduler");
271        tracing::info!("Cron disabled; scheduler supervisor not started");
272    }
273
274    println!("🧠 ZeroClaw daemon started");
275    println!("   Gateway:  http://{host}:{port}");
276    println!("   Components: gateway, channels, heartbeat, scheduler");
277    if config.gateway.require_pairing {
278        println!("   Pairing:    enabled (code appears in gateway output above)");
279    }
280    println!("   Ctrl+C or SIGTERM to stop");
281
282    // Wait for shutdown (SIGINT/SIGTERM/Ctrl+C) or reload (in-process channel).
283    let exit = wait_for_exit_signal(reload_rx).await?;
284    // Broadcast daemon lifecycle event so SSE clients know without checking tracing output.
285    match exit {
286        DaemonExit::Reload => {
287            let _ = event_tx.send(serde_json::json!({
288                "type": "daemon_reload",
289                "timestamp": chrono::Utc::now().to_rfc3339(),
290            }));
291        }
292        DaemonExit::Shutdown => {
293            let _ = event_tx.send(serde_json::json!({
294                "type": "daemon_shutdown",
295                "timestamp": chrono::Utc::now().to_rfc3339(),
296            }));
297        }
298    }
299    crate::health::mark_component_error(
300        "daemon",
301        match exit {
302            DaemonExit::Shutdown => "shutdown requested",
303            DaemonExit::Reload => "reload requested",
304        },
305    );
306
307    for handle in &handles {
308        handle.abort();
309    }
310    for handle in handles {
311        let _ = handle.await;
312    }
313
314    Ok(exit)
315}

317pub fn state_file_path(config: &Config) -> PathBuf {
318    config
319        .config_path
320        .parent()
321        .map_or_else(|| PathBuf::from("."), PathBuf::from)
322        .join("daemon_state.json")
323}

325fn spawn_state_writer(config: Config) -> JoinHandle<()> {
326    tokio::spawn(async move {
327        let path = state_file_path(&config);
328        if let Some(parent) = path.parent() {
329            let _ = tokio::fs::create_dir_all(parent).await;
330        }
331
332        let mut interval = tokio::time::interval(Duration::from_secs(STATUS_FLUSH_SECONDS));
333        loop {
334            interval.tick().await;
335            let mut json = crate::health::snapshot_json();
336            if let Some(obj) = json.as_object_mut() {
337                obj.insert(
338                    "written_at".into(),
339                    serde_json::json!(Utc::now().to_rfc3339()),
340                );
341            }
342            let data = serde_json::to_vec_pretty(&json).unwrap_or_else(|_| b"{}".to_vec());
343            let _ = tokio::fs::write(&path, data).await;
344        }
345    })
346}

348fn spawn_component_supervisor<F, Fut>(
349    name: &'static str,
350    initial_backoff_secs: u64,
351    max_backoff_secs: u64,
352    event_tx: Option<tokio::sync::broadcast::Sender<serde_json::Value>>,
353    mut run_component: F,
354) -> JoinHandle<()>
355where
356    F: FnMut() -> Fut + Send + 'static,
357    Fut: Future<Output = Result<()>> + Send + 'static,
358{
359    tokio::spawn(async move {
360        let mut backoff = initial_backoff_secs.max(1);
361        let max_backoff = max_backoff_secs.max(backoff);
362
363        loop {
364            crate::health::mark_component_ok(name);
365            match run_component().await {
366                Ok(()) => {
367                    crate::health::mark_component_error(name, "component exited unexpectedly");
368                    tracing::warn!("Daemon component '{name}' exited unexpectedly");
369                    backoff = initial_backoff_secs.max(1);
370                }
371                Err(e) => {
372                    crate::health::mark_component_error(name, e.to_string());
373                    tracing::error!("Daemon component '{name}' failed: {e}");
374                }
375            }
376
377            crate::health::bump_component_restart(name);
378            // Broadcast component restart so SSE clients know without checking tracing output.
379            if let Some(ref tx) = event_tx {
380                let _ = tx.send(serde_json::json!({
381                    "type": "component_restart",
382                    "component": name,
383                    "timestamp": chrono::Utc::now().to_rfc3339(),
384                }));
385            }
386            tokio::time::sleep(Duration::from_secs(backoff)).await;
387            backoff = backoff.saturating_mul(2).min(max_backoff);
388        }
389    })
390}

392async fn run_heartbeat_worker(config: Config) -> Result<()> {
393    use crate::heartbeat::engine::{
394        HeartbeatEngine, HeartbeatTask, TaskPriority, TaskStatus, compute_adaptive_interval,
395    };
396    use std::sync::Arc;
397
398    // Use the full configured observer (log / Prometheus / OTel + SSE via BROADCAST_HOOK)
399    // for the engine so HeartbeatTick/Error events reach the primary backend.
400    // The agent::run() calls below pass None so loop_::run() calls create_observer()
401    // internally, preserving the configured backend alongside the gateway's SSE hook.
402    // See: crates/zeroclaw-runtime/src/observability/mod.rs — set_scoped_broadcast_hook.
403    let engine = HeartbeatEngine::new(
404        config.heartbeat.clone(),
405        config.workspace_dir.clone(),
406        std::sync::Arc::from(crate::observability::create_observer(&config.observability)),
407    );
408    let metrics = engine.metrics();
409    let delivery = resolve_heartbeat_delivery(&config)?;
410    let two_phase = config.heartbeat.two_phase;
411    let adaptive = config.heartbeat.adaptive;
412    let start_time = std::time::Instant::now();
413
414    // ── Deadman watcher ──────────────────────────────────────────
415    let deadman_timeout = config.heartbeat.deadman_timeout_minutes;
416    if deadman_timeout > 0 {
417        let dm_metrics = Arc::clone(&metrics);
418        let dm_config = config.clone();
419        let dm_delivery = delivery.clone();
420        tokio::spawn(async move {
421            let check_interval = Duration::from_secs(60);
422            let timeout = chrono::Duration::minutes(i64::from(deadman_timeout));
423            loop {
424                tokio::time::sleep(check_interval).await;
425                let last_tick = dm_metrics.lock().last_tick_at;
426                if let Some(last) = last_tick
427                    && chrono::Utc::now() - last > timeout
428                {
429                    let alert = format!(
430                        "⚠️ Heartbeat dead-man's switch: no tick in {deadman_timeout} minutes"
431                    );
432                    let (channel, target) = if let Some(ch) = &dm_config.heartbeat.deadman_channel {
433                        let to = dm_config
434                            .heartbeat
435                            .deadman_to
436                            .as_deref()
437                            .or(dm_config.heartbeat.to.as_deref())
438                            .unwrap_or_default();
439                        (ch.clone(), to.to_string())
440                    } else if let Some((ch, to)) = &dm_delivery {
441                        (ch.clone(), to.clone())
442                    } else {
443                        continue;
444                    };
445                    let delivery_fut = crate::cron::scheduler::deliver_announcement(
446                        &dm_config, &channel, &target, None, &alert,
447                    );
448                    match tokio::time::timeout(Duration::from_secs(30), delivery_fut).await {
449                        Ok(Err(e)) => {
450                            tracing::warn!("Deadman alert delivery failed: {e}");
451                        }
452                        Err(_) => {
453                            tracing::warn!("Deadman alert delivery timed out (30s)");
454                        }
455                        Ok(Ok(())) => {}
456                    }
457                }
458            }
459        });
460    }
461
462    let base_interval = config.heartbeat.interval_minutes.max(1);
463    let mut sleep_mins = base_interval;
464
465    loop {
466        tokio::time::sleep(Duration::from_secs(u64::from(sleep_mins) * 60)).await;
467
468        // Update uptime
469        {
470            let mut m = metrics.lock();
471            m.uptime_secs = start_time.elapsed().as_secs();
472        }
473
474        let tick_start = std::time::Instant::now();
475
476        // Collect runnable tasks (active only, sorted by priority)
477        let mut tasks = engine.collect_runnable_tasks().await?;
478        let has_high_priority = tasks.iter().any(|t| t.priority == TaskPriority::High);
479
480        if tasks.is_empty() {
481            if let Some(fallback) = config
482                .heartbeat
483                .message
484                .as_deref()
485                .map(str::trim)
486                .filter(|m| !m.is_empty())
487            {
488                tasks.push(HeartbeatTask {
489                    text: fallback.to_string(),
490                    priority: TaskPriority::Medium,
491                    status: TaskStatus::Active,
492                });
493            } else {
494                #[allow(clippy::cast_precision_loss)]
495                let elapsed = tick_start.elapsed().as_millis() as f64;
496                metrics.lock().record_success(elapsed);
497                continue;
498            }
499        }
500
501        // ── Phase 1: LLM decision (two-phase mode) ──────────────
502        let tasks_to_run = if two_phase {
503            let decision_prompt = format!(
504                "[Heartbeat Task | decision] {}",
505                HeartbeatEngine::build_decision_prompt(&tasks),
506            );
507            let phase1_fut = Box::pin(crate::agent::run(
508                config.clone(),
509                Some(decision_prompt),
510                None,
511                None,
512                0.0,
513                vec![],
514                false,
515                None,
516                None,
517                None,
518            ));
519            let phase1_result = if config.heartbeat.task_timeout_secs > 0 {
520                match tokio::time::timeout(
521                    Duration::from_secs(config.heartbeat.task_timeout_secs),
522                    phase1_fut,
523                )
524                .await
525                {
526                    Ok(r) => r,
527                    Err(_) => Err(anyhow::anyhow!(
528                        "Phase 1 decision timed out ({}s)",
529                        config.heartbeat.task_timeout_secs
530                    )),
531                }
532            } else {
533                phase1_fut.await
534            };
535            match phase1_result {
536                Ok(response) => {
537                    let indices = HeartbeatEngine::parse_decision_response(&response, tasks.len());
538                    if indices.is_empty() {
539                        tracing::info!("💓 Heartbeat Phase 1: skip (nothing to do)");
540                        crate::health::mark_component_ok("heartbeat");
541                        #[allow(clippy::cast_precision_loss)]
542                        let elapsed = tick_start.elapsed().as_millis() as f64;
543                        metrics.lock().record_success(elapsed);
544                        continue;
545                    }
546                    tracing::info!(
547                        "💓 Heartbeat Phase 1: run {} of {} tasks",
548                        indices.len(),
549                        tasks.len()
550                    );
551                    indices
552                        .into_iter()
553                        .filter_map(|i| tasks.get(i).cloned())
554                        .collect()
555                }
556                Err(e) => {
557                    tracing::warn!("💓 Heartbeat Phase 1 failed, running all tasks: {e}");
558                    tasks
559                }
560            }
561        } else {
562            tasks
563        };
564
565        // ── Phase 2: Execute selected tasks ─────────────────────
566        // Re-read session context on every tick so we pick up messages
567        // that arrived since the daemon started.
568        let session_context = if config.heartbeat.load_session_context {
569            load_heartbeat_session_context(&config)
570        } else {
571            None
572        };
573
574        // Create memory once per tick for recall + consolidation.
575        let heartbeat_memory: Option<Box<dyn zeroclaw_memory::Memory>> =
576            zeroclaw_memory::create_memory(
577                &config.memory,
578                &config.workspace_dir,
579                config
580                    .providers
581                    .fallback_provider()
582                    .and_then(|e| e.api_key.as_deref()),
583            )
584            .ok();
585
586        let mut tick_had_error = false;
587        for task in &tasks_to_run {
588            let task_start = std::time::Instant::now();
589            let task_prompt = format!("[Heartbeat Task | {}] {}", task.priority, task.text);
590
591            // Recall relevant memories so heartbeat tasks have context awareness.
592            // Exclude `Conversation` memories to prevent chat context from
593            // leaking into scheduled executions (see #5415).
594            let memory_context = if let Some(ref mem) = heartbeat_memory {
595                match mem.recall(&task.text, 5, None, None, None).await {
596                    Ok(entries) if !entries.is_empty() => {
597                        let ctx: String = entries
598                            .iter()
599                            .filter(|e| {
600                                !matches!(
601                                    e.category,
602                                    zeroclaw_memory::traits::MemoryCategory::Conversation
603                                )
604                            })
605                            .map(|e| format!("- {}: {}", e.key, e.content))
606                            .collect::<Vec<_>>()
607                            .join("\n");
608                        if ctx.is_empty() {
609                            None
610                        } else {
611                            Some(format!(
612                                "{MEMORY_CONTEXT_OPEN}\n{ctx}\n{MEMORY_CONTEXT_CLOSE}\n\n"
613                            ))
614                        }
615                    }
616                    _ => None,
617                }
618            } else {
619                None
620            };
621
622            let prompt = match (&session_context, &memory_context) {
623                (Some(sc), Some(mc)) => format!("{mc}\n{sc}\n\n{task_prompt}"),
624                (Some(sc), None) => format!("{sc}\n\n{task_prompt}"),
625                (None, Some(mc)) => format!("{mc}\n\n{task_prompt}"),
626                (None, None) => task_prompt,
627            };
628            let temp = config
629                .providers
630                .fallback_provider()
631                .and_then(|e| e.temperature)
632                .unwrap_or(0.7);
633            let phase2_fut = Box::pin(crate::agent::run(
634                config.clone(),
635                Some(prompt),
636                None,
637                None,
638                temp,
639                vec![],
640                false,
641                None,
642                None,
643                None,
644            ));
645            let phase2_result = if config.heartbeat.task_timeout_secs > 0 {
646                match tokio::time::timeout(
647                    Duration::from_secs(config.heartbeat.task_timeout_secs),
648                    phase2_fut,
649                )
650                .await
651                {
652                    Ok(r) => r,
653                    Err(_) => Err(anyhow::anyhow!(
654                        "Heartbeat task timed out ({}s)",
655                        config.heartbeat.task_timeout_secs
656                    )),
657                }
658            } else {
659                phase2_fut.await
660            };
661            match phase2_result {
662                Ok(output) => {
663                    crate::health::mark_component_ok("heartbeat");
664                    #[allow(clippy::cast_possible_truncation)]
665                    let duration_ms = task_start.elapsed().as_millis() as i64;
666                    let now = chrono::Utc::now();
667                    let _ = crate::heartbeat::store::record_run(
668                        &config.workspace_dir,
669                        &task.text,
670                        &task.priority.to_string(),
671                        now - chrono::Duration::milliseconds(duration_ms),
672                        now,
673                        "ok",
674                        Some(output.as_str()),
675                        duration_ms,
676                        config.heartbeat.max_run_history,
677                    );
678                    // Consolidate heartbeat output to memory for cross-session awareness.
679                    if config.memory.auto_save
680                        && output.chars().count() >= 50
681                        && let Some(ref mem) = heartbeat_memory
682                    {
683                        let key = format!("heartbeat_{}", uuid::Uuid::new_v4());
684                        let summary = if output.len() > 500 {
685                            // Find a valid UTF-8 char boundary at or before 500.
686                            let mut end = 500;
687                            while end > 0 && !output.is_char_boundary(end) {
688                                end -= 1;
689                            }
690                            &output[..end]
691                        } else {
692                            &output
693                        };
694                        let _ = mem
695                            .store(
696                                &key,
697                                &format!("Heartbeat task '{}': {}", task.text, summary),
698                                zeroclaw_memory::MemoryCategory::Daily,
699                                None,
700                            )
701                            .await;
702                    }
703
704                    let announcement = if output.trim().is_empty() {
705                        format!("💓 heartbeat task completed: {}", task.text)
706                    } else {
707                        output
708                    };
709                    if let Some((channel, target)) = &delivery {
710                        let delivery_result = tokio::time::timeout(
711                            Duration::from_secs(30),
712                            crate::cron::scheduler::deliver_announcement(
713                                &config,
714                                channel,
715                                target,
716                                None,
717                                &announcement,
718                            ),
719                        )
720                        .await;
721                        match delivery_result {
722                            Ok(Err(e)) => {
723                                crate::health::mark_component_error(
724                                    "heartbeat",
725                                    format!("delivery failed: {e}"),
726                                );
727                                tracing::warn!("Heartbeat delivery failed: {e}");
728                            }
729                            Err(_) => {
730                                crate::health::mark_component_error(
731                                    "heartbeat",
732                                    "delivery timed out (30s)".to_string(),
733                                );
734                                tracing::warn!("Heartbeat delivery timed out (30s)");
735                            }
736                            Ok(Ok(())) => {}
737                        }
738                    }
739                }
740                Err(e) => {
741                    tick_had_error = true;
742                    #[allow(clippy::cast_possible_truncation)]
743                    let duration_ms = task_start.elapsed().as_millis() as i64;
744                    let now = chrono::Utc::now();
745                    let _ = crate::heartbeat::store::record_run(
746                        &config.workspace_dir,
747                        &task.text,
748                        &task.priority.to_string(),
749                        now - chrono::Duration::milliseconds(duration_ms),
750                        now,
751                        "error",
752                        Some(&e.to_string()),
753                        duration_ms,
754                        config.heartbeat.max_run_history,
755                    );
756                    crate::health::mark_component_error("heartbeat", e.to_string());
757                    tracing::warn!("Heartbeat task failed: {e}");
758                }
759            }
760        }
761
762        // Update metrics
763        #[allow(clippy::cast_precision_loss)]
764        let tick_elapsed = tick_start.elapsed().as_millis() as f64;
765        {
766            let mut m = metrics.lock();
767            if tick_had_error {
768                m.record_failure(tick_elapsed);
769            } else {
770                m.record_success(tick_elapsed);
771            }
772        }
773
774        // Compute next sleep interval
775        if adaptive {
776            let failures = metrics.lock().consecutive_failures;
777            sleep_mins = compute_adaptive_interval(
778                base_interval,
779                config.heartbeat.min_interval_minutes,
780                config.heartbeat.max_interval_minutes,
781                failures,
782                has_high_priority,
783            );
784        } else {
785            sleep_mins = base_interval;
786        }
787    }
788}

790/// Resolve delivery target: explicit config > auto-detect first configured channel.
791fn resolve_heartbeat_delivery(config: &Config) -> Result<Option<(String, String)>> {
792    let channel = config
793        .heartbeat
794        .target
795        .as_deref()
796        .map(str::trim)
797        .filter(|value| !value.is_empty());
798    let target = config
799        .heartbeat
800        .to
801        .as_deref()
802        .map(str::trim)
803        .filter(|value| !value.is_empty());
804
805    match (channel, target) {
806        // Both explicitly set — validate and use.
807        (Some(channel), Some(target)) => {
808            validate_heartbeat_channel_config(config, channel)?;
809            Ok(Some((channel.to_string(), target.to_string())))
810        }
811        // Only one set — error.
812        (Some(_), None) => anyhow::bail!("heartbeat.to is required when heartbeat.target is set"),
813        (None, Some(_)) => anyhow::bail!("heartbeat.target is required when heartbeat.to is set"),
814        // Neither set — try auto-detect the first configured channel.
815        (None, None) => Ok(auto_detect_heartbeat_channel(config)),
816    }
817}

819/// Load recent conversation history for the heartbeat's delivery target and
820/// format it as a text preamble to inject into the task prompt.
821///
822/// Scans `{workspace}/sessions/` for JSONL files whose name starts with
823/// `{channel}_` and ends with `_{to}.jsonl` (or exactly `{channel}_{to}.jsonl`),
824/// then picks the most recently modified match. This handles session key
825/// formats such as `telegram_diskiller.jsonl` and
826/// `telegram_5673725398_diskiller.jsonl`.
827/// Returns `None` when `target`/`to` are not configured or no session exists.
828const HEARTBEAT_SESSION_CONTEXT_MESSAGES: usize = 20;
829
830fn load_heartbeat_session_context(config: &Config) -> Option<String> {
831    use zeroclaw_providers::traits::ChatMessage;
832
833    let channel = config
834        .heartbeat
835        .target
836        .as_deref()
837        .map(str::trim)
838        .filter(|v| !v.is_empty())?;
839    let to = config
840        .heartbeat
841        .to
842        .as_deref()
843        .map(str::trim)
844        .filter(|v| !v.is_empty())?;
845
846    if channel.contains('/') || channel.contains('\\') || to.contains('/') || to.contains('\\') {
847        tracing::warn!("heartbeat session context: channel/to contains path separators, skipping");
848        return None;
849    }
850
851    let sessions_dir = config.workspace_dir.join("sessions");
852
853    // Find the most recently modified JSONL file that belongs to this target.
854    // Matches both `{channel}_{to}.jsonl` and `{channel}_{anything}_{to}.jsonl`.
855    let prefix = format!("{channel}_");
856    let suffix = format!("_{to}.jsonl");
857    let exact = format!("{channel}_{to}.jsonl");
858    let mid_prefix = format!("{channel}_{to}_");
859
860    let path = std::fs::read_dir(&sessions_dir)
861        .ok()?
862        .filter_map(|e| e.ok())
863        .filter(|e| {
864            let name = e.file_name();
865            let name = name.to_string_lossy();
866            name.ends_with(".jsonl")
867                && (name == exact
868                    || (name.starts_with(&prefix) && name.ends_with(&suffix))
869                    || name.starts_with(&mid_prefix))
870        })
871        .max_by_key(|e| {
872            e.metadata()
873                .and_then(|m| m.modified())
874                .unwrap_or(std::time::SystemTime::UNIX_EPOCH)
875        })
876        .map(|e| e.path())?;
877
878    if !path.exists() {
879        tracing::debug!("💓 Heartbeat session context: no session file found for {channel}/{to}");
880        return None;
881    }
882
883    let messages = load_jsonl_messages(&path);
884    if messages.is_empty() {
885        return None;
886    }
887
888    let recent: Vec<&ChatMessage> = messages
889        .iter()
890        .filter(|m| m.role == "user" || m.role == "assistant")
891        .rev()
892        .take(HEARTBEAT_SESSION_CONTEXT_MESSAGES)
893        .collect::<Vec<_>>()
894        .into_iter()
895        .rev()
896        .collect();
897
898    // Only inject context if there is at least one real user message in the
899    // window. If the JSONL contains only assistant messages (e.g. previous
900    // heartbeat outputs with no reply yet), skip context to avoid feeding
901    // Monika's own messages back to her in a loop.
902    let has_user_message = recent.iter().any(|m| m.role == "user");
903    if !has_user_message {
904        tracing::debug!(
905            "💓 Heartbeat session context: no user messages in recent history — skipping"
906        );
907        return None;
908    }
909
910    // Use the session file's mtime as a proxy for when the last message arrived.
911    let last_message_age = std::fs::metadata(&path)
912        .ok()
913        .and_then(|m| m.modified().ok())
914        .and_then(|mtime| mtime.elapsed().ok());
915
916    let silence_note = match last_message_age {
917        Some(age) => {
918            let mins = age.as_secs() / 60;
919            if mins < 60 {
920                format!("(last message ~{mins} minutes ago)\n")
921            } else {
922                let hours = mins / 60;
923                let rem = mins % 60;
924                if rem == 0 {
925                    format!("(last message ~{hours}h ago)\n")
926                } else {
927                    format!("(last message ~{hours}h {rem}m ago)\n")
928                }
929            }
930        }
931        None => String::new(),
932    };
933
934    tracing::debug!(
935        "💓 Heartbeat session context: {} messages from {}, silence: {}",
936        recent.len(),
937        path.display(),
938        silence_note.trim(),
939    );
940
941    let mut ctx = format!(
942        "[Recent conversation history — use this for context when composing your message] {silence_note}",
943    );
944    for msg in &recent {
945        let label = if msg.role == "user" { "User" } else { "You" };
946        // Truncate very long messages to avoid bloating the prompt.
947        // Use char_indices to avoid panicking on multi-byte UTF-8 characters.
948        let content = if msg.content.len() > 500 {
949            let truncate_at = msg
950                .content
951                .char_indices()
952                .map(|(i, _)| i)
953                .take_while(|&i| i <= 500)
954                .last()
955                .unwrap_or(0);
956            format!("{}…", &msg.content[..truncate_at])
957        } else {
958            msg.content.clone()
959        };
960        ctx.push_str(label);
961        ctx.push_str(": ");
962        ctx.push_str(&content);
963        ctx.push('\n');
964    }
965
966    Some(ctx)
967}

969/// Read the last `HEARTBEAT_SESSION_CONTEXT_MESSAGES` `ChatMessage` lines from
970/// a JSONL session file using a bounded rolling window so we never hold the
971/// entire file in memory.
972fn load_jsonl_messages(path: &std::path::Path) -> Vec<zeroclaw_providers::traits::ChatMessage> {
973    use std::collections::VecDeque;
974    use std::io::BufRead;
975
976    let file = match std::fs::File::open(path) {
977        Ok(f) => f,
978        Err(_) => return Vec::new(),
979    };
980    let reader = std::io::BufReader::new(file);
981    let mut window: VecDeque<zeroclaw_providers::traits::ChatMessage> =
982        VecDeque::with_capacity(HEARTBEAT_SESSION_CONTEXT_MESSAGES + 1);
983    for line in reader.lines() {
984        let Ok(line) = line else { continue };
985        let trimmed = line.trim();
986        if trimmed.is_empty() {
987            continue;
988        }
989        if let Ok(msg) = serde_json::from_str::<zeroclaw_providers::traits::ChatMessage>(trimmed) {
990            window.push_back(msg);
991            if window.len() > HEARTBEAT_SESSION_CONTEXT_MESSAGES {
992                window.pop_front();
993            }
994        }
995    }
996    window.into_iter().collect()
997}

999/// Auto-detect the best channel for heartbeat delivery by checking which
1000/// channels are configured. Returns the first match in priority order.
1001fn auto_detect_heartbeat_channel(config: &Config) -> Option<(String, String)> {
1002    // Priority order: telegram > discord > slack > mattermost
1003    if let Some(tg) = &config.channels.telegram {
1004        // Use the first allowed_user as target, or fall back to empty (broadcast)
1005        let target = tg.allowed_users.first().cloned().unwrap_or_default();
1006        if !target.is_empty() {
1007            return Some(("telegram".to_string(), target));
1008        }
1009    }
1010    if config.channels.discord.is_some() {
1011        // Discord requires explicit target — can't auto-detect
1012        return None;
1013    }
1014    if config.channels.slack.is_some() {
1015        // Slack requires explicit target
1016        return None;
1017    }
1018    if config.channels.mattermost.is_some() {
1019        // Mattermost requires explicit target
1020        return None;
1021    }
1022    None
1023}

1025fn validate_heartbeat_channel_config(config: &Config, channel: &str) -> Result<()> {
1026    match channel.to_ascii_lowercase().as_str() {
1027        "telegram" => {
1028            if config.channels.telegram.is_none() {
1029                anyhow::bail!(
1030                    "heartbeat.target is set to telegram but channels.telegram is not configured"
1031                );
1032            }
1033        }
1034        "discord" => {
1035            if config.channels.discord.is_none() {
1036                anyhow::bail!(
1037                    "heartbeat.target is set to discord but channels.discord is not configured"
1038                );
1039            }
1040        }
1041        "slack" => {
1042            if config.channels.slack.is_none() {
1043                anyhow::bail!(
1044                    "heartbeat.target is set to slack but channels.slack is not configured"
1045                );
1046            }
1047        }
1048        "mattermost" => {
1049            if config.channels.mattermost.is_none() {
1050                anyhow::bail!(
1051                    "heartbeat.target is set to mattermost but channels.mattermost is not configured"
1052                );
1053            }
1054        }
1055        other => anyhow::bail!("unsupported heartbeat.target channel: {other}"),
1056    }
1057
1058    Ok(())
1059}

1340    /// and only terminate on SIGINT or SIGTERM.
1341    #[cfg(unix)]
1342    #[tokio::test]
1343    async fn sighup_does_not_shut_down_daemon() {
1344        use libc;
1345        use tokio::time::{Duration, timeout};
1346
1347        let (_reload_tx, reload_rx) = tokio::sync::watch::channel(false);
1348        let handle = tokio::spawn(wait_for_exit_signal(reload_rx));
1349
1350        // Give the signal handler time to register
1351        tokio::time::sleep(Duration::from_millis(50)).await;
1352
1353        // Send SIGHUP to ourselves — should be ignored by the handler
1354        unsafe { libc::raise(libc::SIGHUP) };

1371        let handle = tokio::spawn(wait_for_exit_signal(reload_rx));

1373        reload_tx.send(true).expect("send reload");
