5/// Log-based observer — uses tracing, zero external deps

20impl Observer for LogObserver {
21    fn record_event(&self, event: &ObserverEvent) {
22        match event {
23            ObserverEvent::AgentStart { provider, model } => {
24                info!(provider = %provider, model = %model, "agent.start");
25            }
26            ObserverEvent::AgentEnd {
27                provider,
28                model,
29                duration,
30                tokens_used,
31                cost_usd,
32            } => {
33                let ms = u64::try_from(duration.as_millis()).unwrap_or(u64::MAX);
34                info!(provider = %provider, model = %model, duration_ms = ms, tokens = ?tokens_used, cost_usd = ?cost_usd, "agent.end");
35            }
36            ObserverEvent::ToolCallStart { tool, .. } => {
37                info!(tool = %tool, "tool.start");
38            }
39            ObserverEvent::ToolCall {
40                tool,
41                duration,
42                success,
43            } => {
44                let ms = u64::try_from(duration.as_millis()).unwrap_or(u64::MAX);
45                info!(tool = %tool, duration_ms = ms, success = success, "tool.call");
46            }
47            ObserverEvent::TurnComplete => {
48                info!("turn.complete");
49            }
50            ObserverEvent::ChannelMessage { channel, direction } => {
51                info!(channel = %channel, direction = %direction, "channel.message");
52            }
53            ObserverEvent::HeartbeatTick => {
54                info!("heartbeat.tick");
55            }
56            ObserverEvent::CacheHit {
57                cache_type,
58                tokens_saved,
59            } => {
60                info!(cache_type = %cache_type, tokens_saved = tokens_saved, "cache.hit");
61            }
62            ObserverEvent::CacheMiss { cache_type } => {
63                info!(cache_type = %cache_type, "cache.miss");
64            }
65            ObserverEvent::Error { component, message } => {
66                info!(component = %component, error = %message, "error");
67            }
68            ObserverEvent::LlmRequest {
69                provider,
70                model,
71                messages_count,
72            } => {
73                info!(
74                    provider = %provider,
75                    model = %model,
76                    messages_count = messages_count,
77                    "llm.request"
78                );
79            }
80            ObserverEvent::LlmResponse {
81                provider,
82                model,
83                duration,
84                success,
85                error_message,
86                input_tokens,
87                output_tokens,
88            } => {
89                let ms = u64::try_from(duration.as_millis()).unwrap_or(u64::MAX);
90                info!(
91                    provider = %provider,
92                    model = %model,
93                    duration_ms = ms,
94                    success = success,
95                    error = ?error_message,
96                    input_tokens = ?input_tokens,
97                    output_tokens = ?output_tokens,
98                    "llm.response"
99                );
100            }
101            ObserverEvent::HandStarted { hand_name } => {
102                info!(hand = %hand_name, "hand.started");
103            }
104            ObserverEvent::HandCompleted {
105                hand_name,
106                duration_ms,
107                findings_count,
108            } => {
109                info!(hand = %hand_name, duration_ms = duration_ms, findings = findings_count, "hand.completed");
110            }
111            ObserverEvent::HandFailed {
112                hand_name,
113                error,
114                duration_ms,
115            } => {
116                info!(hand = %hand_name, error = %error, duration_ms = duration_ms, "hand.failed");
117            }
118            ObserverEvent::DeploymentStarted { deploy_id } => {
119                info!(deploy_id = %deploy_id, "deployment.started");
120            }
121            ObserverEvent::DeploymentCompleted {
122                deploy_id,
123                commit_sha,
124            } => {
125                info!(deploy_id = %deploy_id, commit_sha = %commit_sha, "deployment.completed");
126            }
127            ObserverEvent::DeploymentFailed { deploy_id, reason } => {
128                info!(deploy_id = %deploy_id, reason = %reason, "deployment.failed");
129            }
130            ObserverEvent::RecoveryCompleted { deploy_id } => {
131                info!(deploy_id = %deploy_id, "recovery.completed");
132            }
133        }
134    }
135
136    fn record_metric(&self, metric: &ObserverMetric) {
137        match metric {
138            ObserverMetric::RequestLatency(d) => {
139                let ms = u64::try_from(d.as_millis()).unwrap_or(u64::MAX);
140                info!(latency_ms = ms, "metric.request_latency");
141            }
142            ObserverMetric::TokensUsed(t) => {
143                info!(tokens = t, "metric.tokens_used");
144            }
145            ObserverMetric::ActiveSessions(s) => {
146                info!(sessions = s, "metric.active_sessions");
147            }
148            ObserverMetric::QueueDepth(d) => {
149                info!(depth = d, "metric.queue_depth");
150            }
151            ObserverMetric::HandRunDuration {
152                hand_name,
153                duration,
154            } => {
155                let ms = u64::try_from(duration.as_millis()).unwrap_or(u64::MAX);
156                info!(hand = %hand_name, duration_ms = ms, "metric.hand_run_duration");
157            }
158            ObserverMetric::HandFindingsCount { hand_name, count } => {
159                info!(hand = %hand_name, count = count, "metric.hand_findings_count");
160            }
161            ObserverMetric::HandSuccessRate { hand_name, success } => {
162                info!(hand = %hand_name, success = success, "metric.hand_success_rate");
163            }
164            ObserverMetric::DeploymentLeadTime(d) => {
165                let ms = u64::try_from(d.as_millis()).unwrap_or(u64::MAX);
166                info!(lead_time_ms = ms, "metric.deployment_lead_time");
167            }
168            ObserverMetric::RecoveryTime(d) => {
169                let ms = u64::try_from(d.as_millis()).unwrap_or(u64::MAX);
170                info!(recovery_time_ms = ms, "metric.recovery_time");
171            }
172        }
173    }
174
175    fn name(&self) -> &str {
176        "log"
177    }
178
179    fn as_any(&self) -> &dyn Any {
180        self
181    }
182}
