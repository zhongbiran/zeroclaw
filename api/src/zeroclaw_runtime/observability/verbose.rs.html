4/// Human-readable progress observer for interactive CLI sessions.
5///
6/// This observer prints compact `>` / `<` progress lines without exposing
7/// prompt contents. It is intended to be opt-in (e.g. `--verbose`).
8pub struct VerboseObserver;
9
10impl Default for VerboseObserver {
11    fn default() -> Self {
12        Self::new()
13    }
14}
15
16impl VerboseObserver {
17    pub fn new() -> Self {
18        Self
19    }
20}
21
22impl Observer for VerboseObserver {
23    fn record_event(&self, event: &ObserverEvent) {
24        match event {
25            ObserverEvent::LlmRequest {
26                provider,
27                model,
28                messages_count,
29            } => {
30                eprintln!("> Thinking");
31                eprintln!(
32                    "> Send (provider={}, model={}, messages={})",
33                    provider, model, messages_count
34                );
35            }
36            ObserverEvent::LlmResponse {
37                duration, success, ..
38            } => {
39                let ms = u64::try_from(duration.as_millis()).unwrap_or(u64::MAX);
40                eprintln!("< Receive (success={success}, duration_ms={ms})");
41            }
42            ObserverEvent::ToolCallStart { tool, .. } => {
43                eprintln!("> Tool {tool}");
44            }
45            ObserverEvent::ToolCall {
46                tool,
47                duration,
48                success,
49            } => {
50                let ms = u64::try_from(duration.as_millis()).unwrap_or(u64::MAX);
51                eprintln!("< Tool {tool} (success={success}, duration_ms={ms})");
52            }
53            ObserverEvent::TurnComplete => {
54                eprintln!("< Complete");
55            }
56            _ => {}
57        }
58    }
59
60    #[inline(always)]
61    fn record_metric(&self, _metric: &ObserverMetric) {}
62
63    fn name(&self) -> &str {
64        "verbose"
65    }
66
67    fn as_any(&self) -> &dyn Any {
68        self
69    }
70}

72#[cfg(test)]
73mod tests {
74    use super::*;
75    use std::time::Duration;
76
77    #[test]
78    fn verbose_name() {
79        assert_eq!(VerboseObserver::new().name(), "verbose");
80    }
81
82    #[test]
83    fn verbose_events_do_not_panic() {
84        let obs = VerboseObserver::new();
85        obs.record_event(&ObserverEvent::LlmRequest {
86            provider: "openrouter".into(),
87            model: "claude".into(),
88            messages_count: 3,
89        });
90        obs.record_event(&ObserverEvent::LlmResponse {
91            provider: "openrouter".into(),
92            model: "claude".into(),
93            duration: Duration::from_millis(12),
94            success: true,
95            error_message: None,
96            input_tokens: Some(50),
97            output_tokens: Some(25),
98        });
99        obs.record_event(&ObserverEvent::ToolCallStart {
100            tool: "shell".into(),
101            arguments: None,
102        });
103        obs.record_event(&ObserverEvent::ToolCall {
104            tool: "shell".into(),
105            duration: Duration::from_millis(2),
106            success: true,
107        });
108        obs.record_event(&ObserverEvent::TurnComplete);
109    }
110
111    #[test]
112    fn verbose_hand_events_do_not_panic() {
113        let obs = VerboseObserver::new();
114        obs.record_event(&ObserverEvent::HandStarted {
115            hand_name: "review".into(),
116        });
117        obs.record_event(&ObserverEvent::HandCompleted {
118            hand_name: "review".into(),
119            duration_ms: 1500,
120            findings_count: 3,
121        });
122        obs.record_event(&ObserverEvent::HandFailed {
123            hand_name: "review".into(),
124            error: "timeout".into(),
125            duration_ms: 5000,
126        });
127    }
