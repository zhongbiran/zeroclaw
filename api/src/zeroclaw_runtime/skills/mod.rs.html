107/// Provenance metadata emitted by the SkillForge integrator (see
108/// `crates/zeroclaw-runtime/src/skillforge/integrate.rs`). Lives at the
109/// top level of SKILL.toml under `[forge]`, kept separate from
110/// `[skill]` so the canonical skill identity stays decoupled from the
111/// integrator's emit format. Strict by design: a typo here is just as
112/// bad as a typo in `[skill]` (silent misconfiguration of provenance).
113#[derive(Debug, Clone, Serialize, Deserialize)]
114#[serde(deny_unknown_fields)]
115struct ForgeMetadata {
116    /// Upstream URL the skill was integrated from.
117    #[serde(default)]
118    source: Option<String>,
119    /// Upstream owner (GitHub user / org).
120    #[serde(default)]
121    owner: Option<String>,
122    /// Primary language reported by the source (or `"unknown"`).
123    #[serde(default)]
124    language: Option<String>,
125    /// `true` if the upstream repo carries a license file.
126    #[serde(default)]
127    license: Option<bool>,
128    /// Upstream star count at integration time.
129    #[serde(default)]
130    stars: Option<u64>,
131    /// Upstream `updated_at` timestamp formatted `YYYY-MM-DD`, or
132    /// `"unknown"` if the integrator could not resolve one.
133    #[serde(default)]
134    updated_at: Option<String>,
135    /// Runtime/version requirements declared by the integrator.
136    #[serde(default)]
137    requirements: BTreeMap<String, toml::Value>,
138    /// Free-form integrator metadata (e.g. `auto_integrated`,
139    /// `forge_timestamp`). **This is the intended extension point** for
140    /// future SkillForge metadata: prefer adding new keys under
141    /// `[forge.metadata.X]` over new top-level `[forge]` fields, which
142    /// would require a coordinated `ForgeMetadata` schema bump and break
143    /// strict parsing for anyone running an older runtime.
144    #[serde(default)]
145    metadata: BTreeMap<String, toml::Value>,

161/// Trust tier of a skill listed in the `zeroclaw-skills` registry.
162///
163/// Derived from the `tags` array in `registry.json`. `Unknown` is used as the
164/// "no recognized tier tag" fallback and is treated like `Community` for trust
165/// purposes when displaying the install banner.
166///
167/// `Featured` is intentionally kept as a distinct variant even though it
168/// renders identically to `Community` today: the registry's `Featured` tag is
169/// a separate curation signal (zeroclaw-labs hand-picked, but still authored
170/// outside zeroclaw-labs) and we expect to render it differently later — e.g.
171/// "Featured — community-curated by zeroclaw-labs but not maintained by us".
172/// Keeping the variant now avoids a churn-y enum extension once that copy
173/// lands.
174#[derive(Debug, Clone, Copy, PartialEq, Eq)]
175pub enum SkillTier {
176    Official,
177    Community,
178    Featured,
179    Unknown,
180}

197fn tier_from_tags(tags: &[String]) -> SkillTier {
198    let has = |needle: &str| tags.iter().any(|t| t.eq_ignore_ascii_case(needle));
199    if has("Official") {
200        SkillTier::Official
201    } else if has("Community") {
202        SkillTier::Community
203    } else if has("Featured") {
204        SkillTier::Featured
205    } else {
206        SkillTier::Unknown
207    }
208}

210/// Look up a skill in `<registry_dir>/registry.json` and return its trust tier
211/// and version. Returns `(SkillTier::Unknown, None)` if the index file is
212/// missing, malformed, or does not list the skill.
213pub fn lookup_registry_skill_tier(registry_dir: &Path, name: &str) -> (SkillTier, Option<String>) {
214    let path = registry_dir.join("registry.json");
215    let Ok(data) = std::fs::read_to_string(&path) else {
216        return (SkillTier::Unknown, None);
217    };
218    let Ok(index) = serde_json::from_str::<RegistryIndex>(&data) else {
219        return (SkillTier::Unknown, None);
220    };
221    let Some(entry) = index.skills.into_iter().find(|e| e.name == name) else {
222        return (SkillTier::Unknown, None);
223    };
224    (tier_from_tags(&entry.tags), entry.version)
225}

239    let mut banner = crate::i18n::get_required_cli_string_with_args(key, &args);
240    if !banner.ends_with('\n') {
241        banner.push('\n');
242    }
243    banner

251/// Emit a user-visible warning when a skill directory is skipped due to audit
252/// findings. When the findings mention blocked scripts and `allow_scripts` is
253/// `false`, the message includes actionable remediation guidance so users know
254/// how to enable their skill.
255fn warn_skipped_skill(path: &Path, summary: &str, allow_scripts: bool) {
256    let scripts_blocked = summary.contains("script-like files are blocked");
257    if scripts_blocked && !allow_scripts {
258        tracing::warn!(
259            "skipping skill directory {}: {summary}. \
260             To allow script files in skills, set `skills.allow_scripts = true` in your config.",
261            path.display(),
262        );
263        eprintln!(
264            "warning: skill '{}' was skipped because it contains script files. \
265             Set `skills.allow_scripts = true` in your zeroclaw config to enable it.",
266            path.file_name()
267                .map(|n| n.to_string_lossy().into_owned())
268                .unwrap_or_else(|| path.display().to_string()),
269        );
270    } else {
271        tracing::warn!(
272            "skipping insecure skill directory {}: {summary}",
273            path.display(),
274        );
275    }
276}

278fn warn_metadata_drift(skill_dir: &Path, toml_skill: &Skill, md_path: &Path) {
279    if !md_path.exists() {
280        return;
281    }
282    let Ok(md_content) = std::fs::read_to_string(md_path) else {
283        return;
284    };
285    let parsed = parse_skill_markdown(&md_content);
286    let dir_name = skill_dir.file_name().and_then(|n| n.to_str()).unwrap_or("");
287
288    if let Some(ref md_name) = parsed.meta.name
289        && md_name != &toml_skill.name
290    {
291        tracing::warn!(
292            "skill '{}': name mismatch between TOML ('{}') and SKILL.md ('{}')",
293            dir_name,
294            toml_skill.name,
295            md_name,
296        );
297    }
298    if let Some(ref md_desc) = parsed.meta.description {
299        let md_desc = md_desc.trim();
300        if !md_desc.is_empty() && md_desc != ">-" && md_desc != toml_skill.description.trim() {
301            tracing::warn!(
302                "skill '{}': description mismatch between TOML and SKILL.md — TOML takes precedence",
303                dir_name,
304            );
305        }
306    }
307}

372pub fn load_skills_from_directory(skills_dir: &Path, allow_scripts: bool) -> Vec<Skill> {
373    if !skills_dir.exists() {
374        return Vec::new();
375    }
376
377    let mut skills = Vec::new();
378
379    let Ok(entries) = std::fs::read_dir(skills_dir) else {
380        return skills;
381    };
382
383    for entry in entries.flatten() {
384        let path = entry.path();
385        if !path.is_dir() {
386            continue;
387        }
388
389        match audit::audit_skill_directory_with_options(
390            &path,
391            audit::SkillAuditOptions { allow_scripts },
392        ) {
393            Ok(report) if report.is_clean() => {}
394            Ok(report) => {
395                let summary = report.summary();
396                warn_skipped_skill(&path, &summary, allow_scripts);
397                continue;
398            }
399            Err(err) => {
400                tracing::warn!(
401                    "skipping unauditable skill directory {}: {err}",
402                    path.display()
403                );
404                continue;
405            }
406        }
407
408        // Try SKILL.toml first, then manifest.toml (registry format), then SKILL.md
409        let skill_toml_path = path.join("SKILL.toml");
410        let manifest_toml_path = path.join("manifest.toml");
411        let md_path = path.join("SKILL.md");
412
413        let toml_path = if skill_toml_path.exists() {
414            Some(skill_toml_path)
415        } else if manifest_toml_path.exists() {
416            Some(manifest_toml_path)
417        } else {
418            None
419        };
420
421        if let Some(toml_path) = toml_path {
422            match load_skill_toml(&toml_path) {
423                Ok(skill) => {
424                    warn_metadata_drift(&path, &skill, &md_path);
425                    skills.push(skill);
426                }
427                Err(e) => {
428                    tracing::warn!(
429                        path = %toml_path.display(),
430                        err  = %e,
431                        "failed to load SKILL.toml — skill directory skipped",
432                    );
433                }
434            }
435        } else if md_path.exists()
436            && let Ok(skill) = load_skill_md(&md_path, &path)
437        {
438            skills.push(skill);
439        }
440    }
441
442    skills
443}

527fn load_open_skills(repo_dir: &Path, allow_scripts: bool) -> Vec<Skill> {
528    // Modern open-skills layout stores skill packages in `skills/<name>/SKILL.md`.
529    // Prefer that structure to avoid treating repository docs (e.g. CONTRIBUTING.md)
530    // as executable skills.
531    let nested_skills_dir = repo_dir.join("skills");
532    if nested_skills_dir.is_dir() {
533        return load_open_skills_from_directory(&nested_skills_dir, allow_scripts);
534    }
535
536    let mut skills = Vec::new();
537
538    let Ok(entries) = std::fs::read_dir(repo_dir) else {
539        return skills;
540    };
541
542    for entry in entries.flatten() {
543        let path = entry.path();
544        if !path.is_file() {
545            continue;
546        }
547
548        let is_markdown = path
549            .extension()
550            .and_then(|ext| ext.to_str())
551            .is_some_and(|ext| ext.eq_ignore_ascii_case("md"));
552        if !is_markdown {
553            continue;
554        }
555
556        let is_readme = path
557            .file_name()
558            .and_then(|name| name.to_str())
559            .is_some_and(|name| name.eq_ignore_ascii_case("README.md"));
560        if is_readme {
561            continue;
562        }
563
564        match audit::audit_open_skill_markdown(&path, repo_dir) {
565            Ok(report) if report.is_clean() => {}
566            Ok(report) => {
567                tracing::warn!(
568                    "skipping insecure open-skill file {}: {}",
569                    path.display(),
570                    report.summary()
571                );
572                continue;
573            }
574            Err(err) => {
575                tracing::warn!(
576                    "skipping unauditable open-skill file {}: {err}",
577                    path.display()
578                );
579                continue;
580            }
581        }
582
583        if let Ok(skill) = load_open_skill_md(&path) {
584            skills.push(skill);
585        }
586    }
587
588    skills
589}

624    config_dir: Option<&str>,

626) -> Option<PathBuf> {

631        } else {

641    }

643}

655fn ensure_open_skills_repo(
656    config_open_skills_enabled: Option<bool>,
657    config_open_skills_dir: Option<&str>,
658) -> Option<PathBuf> {
659    if !open_skills_enabled(config_open_skills_enabled) {
660        return None;
661    }

664

676        } else {
677            tracing::warn!(
678                "open-skills update failed; using local copy from {}",
679                repo_dir.display()
680            );
681        }
682    }
683
684    Some(repo_dir)

711            false
712        }
713        Err(err) => {
714            tracing::warn!("failed to run git clone for open-skills: {err}");
715            false
716        }
717    }

738        }

740            tracing::warn!("failed to run git pull for open-skills: {err}");
741            false
742        }
743    }

872/// Lightweight YAML-like frontmatter parser for simple `key: value` pairs.
873/// Replaces `serde_yaml` to avoid pulling in the full YAML parser (~30KB)
874/// for a struct with only 5 optional string fields.
875fn parse_simple_frontmatter(s: &str) -> SkillMarkdownMeta {
876    let mut meta = SkillMarkdownMeta::default();
877    let mut collecting_tags = false;
878    let mut collecting_multiline: Option<String> = None;
879    let mut multiline_parts: Vec<String> = Vec::new();
880
881    let flush_multiline = |key: &str, parts: &[String], meta: &mut SkillMarkdownMeta| {
882        let joined = parts.join(" ");
883        let val = joined.trim();
884        if !val.is_empty() {
885            match key {
886                "description" => meta.description = Some(val.to_string()),
887                "name" => meta.name = Some(val.to_string()),
888                _ => {}
889            }
890        }
891    };
892
893    for line in s.lines() {
894        // Collect indented continuation lines for YAML block scalars (>- or |)
895        if let Some(ref key) = collecting_multiline {
896            if line.starts_with(' ') || line.starts_with('\t') {
897                multiline_parts.push(line.trim().to_string());
898                continue;
899            }
900            flush_multiline(key, &multiline_parts, &mut meta);
901            collecting_multiline = None;
902            multiline_parts.clear();
903        }
904
905        // Handle YAML list items under `tags:` (e.g. "  - parser")
906        if collecting_tags {
907            let trimmed = line.trim();
908            if let Some(item) = trimmed.strip_prefix("- ") {
909                let tag = item.trim().trim_matches('"').trim_matches('\'');
910                if !tag.is_empty() {
911                    meta.tags.push(tag.to_string());
912                }
913                continue;
914            }
915            // Non-list-item line → stop collecting tags
916            collecting_tags = false;
917        }
918        let Some((key, val)) = line.split_once(':') else {
919            continue;
920        };
921        let key = key.trim();
922        let val = val.trim().trim_matches('"').trim_matches('\'');
923        // YAML block scalar indicators — collect continuation lines
924        if val == ">-" || val == ">" || val == "|" || val == "|-" {
925            collecting_multiline = Some(key.to_string());
926            multiline_parts.clear();
927            continue;
928        }
929        match key {
930            "name" => meta.name = Some(val.to_string()),
931            "description" => meta.description = Some(val.to_string()),
932            "version" => meta.version = Some(val.to_string()),
933            "author" => meta.author = Some(val.to_string()),
934            "tags" => {
935                if val.is_empty() {
936                    // YAML block list follows on subsequent lines
937                    collecting_tags = true;
938                } else {
939                    // Inline: [a, b, c] or comma-separated
940                    let val = val.trim_start_matches('[').trim_end_matches(']');
941                    meta.tags = val
942                        .split(',')
943                        .map(|t| t.trim().trim_matches('"').trim_matches('\'').to_string())
944                        .filter(|t| !t.is_empty())
945                        .collect();
946                }
947            }
948            _ => {}
949        }
950    }
951    if let Some(ref key) = collecting_multiline {
952        flush_multiline(key, &multiline_parts, &mut meta);
953    }
954    meta
955}

957fn split_skill_frontmatter(content: &str) -> Option<(String, String)> {
958    let normalized = content.replace("\r\n", "\n");
959    let rest = normalized.strip_prefix("---\n")?;
960    if let Some(idx) = rest.find("\n---\n") {
961        let frontmatter = rest[..idx].to_string();
962        let body = rest[idx + 5..].to_string();
963        return Some((frontmatter, body));
964    }
965    if let Some(frontmatter) = rest.strip_suffix("\n---") {
966        return Some((frontmatter.to_string(), String::new()));
967    }
968    None
969}

980fn append_xml_escaped(out: &mut String, text: &str) {
981    for ch in text.chars() {
982        match ch {
983            '&' => out.push_str("&amp;"),
984            '<' => out.push_str("&lt;"),
985            '>' => out.push_str("&gt;"),
986            '"' => out.push_str("&quot;"),
987            '\'' => out.push_str("&apos;"),
988            _ => out.push(ch),
989        }
990    }
991}

993fn write_xml_text_element(out: &mut String, indent: usize, tag: &str, value: &str) {
994    for _ in 0..indent {
995        out.push(' ');
996    }
997    out.push('<');
998    out.push_str(tag);
999    out.push('>');
1000    append_xml_escaped(out, value);
1001    out.push_str("</");
1002    out.push_str(tag);
1003    out.push_str(">\n");
1004}

1032/// Build the "Available Skills" system prompt section with configurable verbosity.
1033pub fn skills_to_prompt_with_mode(
1034    skills: &[Skill],
1035    workspace_dir: &Path,
1036    mode: zeroclaw_config::schema::SkillsPromptInjectionMode,
1037) -> String {
1038    use std::fmt::Write;
1039
1040    if skills.is_empty() {
1041        return String::new();
1042    }
1043
1044    let mut prompt = match mode {
1045        zeroclaw_config::schema::SkillsPromptInjectionMode::Full => String::from(
1046            "## Available Skills\n\n\
1047             Skill instructions and tool metadata are preloaded below.\n\
1048             Follow these instructions directly; do not read skill files at runtime unless the user asks.\n\n\
1049             <available_skills>\n",
1050        ),
1051        zeroclaw_config::schema::SkillsPromptInjectionMode::Compact => String::from(
1052            "## Available Skills\n\n\
1053             Skill summaries are preloaded below to keep context compact.\n\
1054             Skill instructions are loaded on demand: call `read_skill(name)` with the skill's `<name>` when you need the full skill file.\n\
1055             The `location` field is included for reference.\n\n\
1056             <available_skills>\n",
1057        ),
1058    };
1059
1060    for skill in skills {
1061        let _ = writeln!(prompt, "  <skill>");
1062        write_xml_text_element(&mut prompt, 4, "name", &skill.name);
1063        write_xml_text_element(&mut prompt, 4, "description", &skill.description);
1064        let location = render_skill_location(
1065            skill,
1066            workspace_dir,
1067            matches!(
1068                mode,
1069                zeroclaw_config::schema::SkillsPromptInjectionMode::Compact
1070            ),
1071        );
1072        write_xml_text_element(&mut prompt, 4, "location", &location);
1073
1074        // In Full mode, inline both instructions and tools.
1075        // In Compact mode, skip instructions (loaded on demand) but keep tools
1076        // so the LLM knows which skill tools are available.
1077        if matches!(
1078            mode,
1079            zeroclaw_config::schema::SkillsPromptInjectionMode::Full
1080        ) && !skill.prompts.is_empty()
1081        {
1082            let _ = writeln!(prompt, "    <instructions>");
1083            for instruction in &skill.prompts {
1084                write_xml_text_element(&mut prompt, 6, "instruction", instruction);
1085            }
1086            let _ = writeln!(prompt, "    </instructions>");
1087        }
1088
1089        if !skill.tools.is_empty() {
1090            // Tools with known kinds (shell, script, http) are registered as
1091            // callable tool specs and can be invoked directly via function calling.
1092            // We note them here for context but mark them as callable.
1093            let registered: Vec<_> = skill
1094                .tools
1095                .iter()
1096                .filter(|t| matches!(t.kind.as_str(), "shell" | "script" | "http"))
1097                .collect();
1098            let unregistered: Vec<_> = skill
1099                .tools
1100                .iter()
1101                .filter(|t| !matches!(t.kind.as_str(), "shell" | "script" | "http"))
1102                .collect();
1103
1104            if !registered.is_empty() {
1105                let _ = writeln!(
1106                    prompt,
1107                    "    <callable_tools hint=\"These are registered as callable tool specs. Invoke them directly by name ({{}}__{{}}) instead of using shell.\">"
1108                );
1109                for tool in &registered {
1110                    let _ = writeln!(prompt, "      <tool>");
1111                    write_xml_text_element(
1112                        &mut prompt,
1113                        8,
1114                        "name",
1115                        &format!("{}__{}", skill.name, tool.name),
1116                    );
1117                    write_xml_text_element(&mut prompt, 8, "description", &tool.description);
1118                    let _ = writeln!(prompt, "      </tool>");
1119                }
1120                let _ = writeln!(prompt, "    </callable_tools>");
1121            }
1122
1123            if !unregistered.is_empty() {
1124                let _ = writeln!(prompt, "    <tools>");
1125                for tool in &unregistered {
1126                    let _ = writeln!(prompt, "      <tool>");
1127                    write_xml_text_element(&mut prompt, 8, "name", &tool.name);
1128                    write_xml_text_element(&mut prompt, 8, "description", &tool.description);
1129                    write_xml_text_element(&mut prompt, 8, "kind", &tool.kind);
1130                    let _ = writeln!(prompt, "      </tool>");
1131                }
1132                let _ = writeln!(prompt, "    </tools>");
1133            }
1134        }
1135
1136        let _ = writeln!(prompt, "  </skill>");
1137    }
1138
1139    prompt.push_str("</available_skills>");
1140    prompt
1141}

1143/// Convert skill tools into callable `Tool` trait objects.
1144///
1145/// Each skill's `[[tools]]` entries are converted to either `SkillShellTool`
1146/// (for `shell`/`script` kinds) or `SkillHttpTool` (for `http` kind),
1147/// enabling them to appear as first-class callable tool specs rather than
1148/// only as XML in the system prompt.
1149pub fn skills_to_tools(
1150    skills: &[Skill],
1151    security: std::sync::Arc<crate::security::SecurityPolicy>,
1152) -> Vec<Box<dyn zeroclaw_api::tool::Tool>> {
1153    let mut tools: Vec<Box<dyn zeroclaw_api::tool::Tool>> = Vec::new();
1154    for skill in skills {
1155        for tool in &skill.tools {
1156            match tool.kind.as_str() {
1157                "shell" | "script" => {
1158                    let inner = crate::skills::skill_tool::SkillShellTool::new(
1159                        &skill.name,
1160                        tool,
1161                        security.clone(),
1162                    );
1163                    tools.push(Box::new(zeroclaw_tools::wrappers::RateLimitedTool::new(
1164                        inner,
1165                        security.clone(),
1166                    )));
1167                }
1168                "http" => {
1169                    tools.push(Box::new(crate::skills::skill_http::SkillHttpTool::new(
1170                        &skill.name,
1171                        tool,
1172                    )));
1173                }
1174                other => {
1175                    tracing::warn!(
1176                        "Unknown skill tool kind '{}' for {}.{}, skipping",
1177                        other,
1178                        skill.name,
1179                        tool.name
1180                    );
1181                }
1182            }
1183        }
1184    }
1185    tools
1186}

1258fn clawhub_download_url(source: &str) -> Result<String> {
1259    // Short prefix: clawhub:<slug>
1260    if let Some(slug) = source.strip_prefix("clawhub:") {
1261        let slug = slug.trim().trim_end_matches('/');
1262        if slug.is_empty() || slug.contains('/') {
1263            anyhow::bail!(
1264                "invalid clawhub source '{}': expected 'clawhub:<slug>' (no slashes in slug)",
1265                source
1266            );
1267        }
1268        return Ok(format!("{CLAWHUB_DOWNLOAD_API}?slug={slug}"));
1269    }
1270
1271    // Profile URL: https://clawhub.ai/<owner>/<slug> or https://www.clawhub.ai/<slug>
1272    if let Some(parsed) = parse_clawhub_url(source) {
1273        let path = parsed
1274            .path_segments()
1275            .into_iter()
1276            .flatten()
1277            .collect::<Vec<_>>()
1278            .join("/");
1279
1280        if path.is_empty() {
1281            anyhow::bail!("could not extract slug from ClawhHub URL: {source}");
1282        }
1283

1328pub fn is_git_source(source: &str) -> bool {
1329    // ClawHub URLs look like https:// but are not git repos
1330    if is_clawhub_source(source) {
1331        return false;
1332    }
1333    is_git_scheme_source(source, "https://")
1334        || is_git_scheme_source(source, "http://")
1335        || is_git_scheme_source(source, "ssh://")
1336        || is_git_scheme_source(source, "git://")
1337        || is_git_scp_source(source)
1338}

1340fn is_git_scheme_source(source: &str, scheme: &str) -> bool {
1341    let Some(rest) = source.strip_prefix(scheme) else {
1342        return false;
1343    };
1344    if rest.is_empty() || rest.starts_with('/') {
1345        return false;
1346    }
1347
1348    let host = rest.split(['/', '?', '#']).next().unwrap_or_default();
1349    !host.is_empty()
1350}

1352fn is_git_scp_source(source: &str) -> bool {
1353    // SCP-like syntax accepted by git, e.g. git@host:owner/repo.git
1354    // Keep this strict enough to avoid treating local paths as git remotes.
1355    let Some((user_host, remote_path)) = source.split_once(':') else {
1356        return false;
1357    };
1358    if remote_path.is_empty() {
1359        return false;
1360    }
1361    if source.contains("://") {
1362        return false;
1363    }
1364
1365    let Some((user, host)) = user_host.split_once('@') else {
1366        return false;
1367    };
1368    !user.is_empty()
1369        && !host.is_empty()
1370        && !user.contains('/')
1371        && !user.contains('\\')
1372        && !host.contains('/')
1373        && !host.contains('\\')
1374}

1433fn copy_dir_recursive_secure(src: &Path, dest: &Path) -> Result<()> {
1434    let src_meta = std::fs::symlink_metadata(src)
1435        .with_context(|| format!("failed to read metadata for {}", src.display()))?;
1436    if src_meta.file_type().is_symlink() {
1437        anyhow::bail!(
1438            "Refusing to copy symlinked skill source path: {}",
1439            src.display()
1440        );
1441    }
1442    if !src_meta.is_dir() {
1443        anyhow::bail!("Skill source must be a directory: {}", src.display());
1444    }
1445
1446    std::fs::create_dir_all(dest)
1447        .with_context(|| format!("failed to create destination {}", dest.display()))?;
1448    for entry in std::fs::read_dir(src)? {
1449        let entry = entry?;
1450        let src_path = entry.path();
1451        let dest_path = dest.join(entry.file_name());
1452        let metadata = std::fs::symlink_metadata(&src_path)
1453            .with_context(|| format!("failed to read metadata for {}", src_path.display()))?;
1454
1455        if metadata.file_type().is_symlink() {
1456            anyhow::bail!(
1457                "Refusing to copy symlink within skill source: {}",
1458                src_path.display()
1459            );
1460        }
1461
1462        if metadata.is_dir() {
1463            copy_dir_recursive_secure(&src_path, &dest_path)?;
1464        } else if metadata.is_file() {
1465            std::fs::copy(&src_path, &dest_path).with_context(|| {
1466                format!(
1467                    "failed to copy skill file from {} to {}",
1468                    src_path.display(),
1469                    dest_path.display()
1470                )
1471            })?;
1472        }
1473    }
1474
1475    Ok(())
1476}

1478pub fn install_local_skill_source(
1479    source: &str,
1480    skills_path: &Path,
1481    allow_scripts: bool,
1482) -> Result<(PathBuf, usize)> {
1483    let source_path = PathBuf::from(source);
1484    if !source_path.exists() {
1485        anyhow::bail!("Source path does not exist: {source}");
1486    }
1487
1488    let source_path = source_path
1489        .canonicalize()
1490        .with_context(|| format!("failed to canonicalize source path {source}"))?;
1491    let _ = enforce_skill_security_audit(&source_path, allow_scripts)?;
1492
1493    let name = source_path
1494        .file_name()
1495        .context("Source path must include a directory name")?;
1496    let dest = skills_path.join(name);
1497    if dest.exists() {
1498        anyhow::bail!("Destination skill already exists: {}", dest.display());
1499    }
1500
1501    if let Err(err) = copy_dir_recursive_secure(&source_path, &dest) {
1502        let _ = std::fs::remove_dir_all(&dest);
1503        return Err(err);
1504    }
1505
1506    match enforce_skill_security_audit(&dest, allow_scripts) {
1507        Ok(report) => Ok((dest, report.files_scanned)),
1508        Err(err) => {
1509            let _ = std::fs::remove_dir_all(&dest);
1510            Err(err)
1511        }
1512    }
1513}

1541pub async fn install_clawhub_skill_source(
1542    source: &str,
1543    skills_path: &Path,
1544    allow_scripts: bool,
1545) -> Result<(PathBuf, usize)> {
1546    let download_url = clawhub_download_url(source)
1547        .with_context(|| format!("invalid ClawhHub source: {source}"))?;
1548    let skill_dir_name = clawhub_skill_dir_name(source)?;
1549    let installed_dir = skills_path.join(&skill_dir_name);
1550    if installed_dir.exists() {
1551        anyhow::bail!(
1552            "Destination skill already exists: {}",
1553            installed_dir.display()
1554        );
1555    }
1556
1557    let client = reqwest::Client::builder()
1558        .timeout(Duration::from_secs(30))
1559        .build()?;
1560
1561    let resp = client
1562        .get(&download_url)
1563        .send()
1564        .await
1565        .with_context(|| format!("failed to fetch zip from {download_url}"))?;
1566
1567    if resp.status() == reqwest::StatusCode::TOO_MANY_REQUESTS {
1568        anyhow::bail!("ClawhHub rate limit reached (HTTP 429). Wait a moment and retry.");
1569    }
1570    if !resp.status().is_success() {
1571        anyhow::bail!("ClawhHub download failed (HTTP {})", resp.status());
1572    }
1573
1574    let bytes = resp.bytes().await?.to_vec();
1575    if bytes.len() as u64 > MAX_CLAWHUB_ZIP_BYTES {
1576        anyhow::bail!(
1577            "ClawhHub zip rejected: too large ({} bytes > {})",
1578            bytes.len(),
1579            MAX_CLAWHUB_ZIP_BYTES
1580        );
1581    }
1582
1583    std::fs::create_dir_all(&installed_dir)?;
1584
1585    let cursor = Cursor::new(bytes);
1586    let mut archive = ZipArchive::new(cursor).context("downloaded content is not a valid zip")?;
1587
1588    for i in 0..archive.len() {
1589        let mut entry = archive.by_index(i)?;
1590        let raw_name = entry.name().to_string();
1591
1592        if raw_name.is_empty()
1593            || raw_name.contains("..")
1594            || raw_name.starts_with('/')
1595            || raw_name.contains('\\')
1596            || raw_name.contains(':')
1597        {
1598            let _ = std::fs::remove_dir_all(&installed_dir);
1599            anyhow::bail!("zip entry contains unsafe path: {raw_name}");
1600        }
1601
1602        let out_path = installed_dir.join(&raw_name);
1603        if entry.is_dir() {
1604            std::fs::create_dir_all(&out_path)?;
1605            continue;
1606        }
1607
1608        if let Some(parent) = out_path.parent() {
1609            std::fs::create_dir_all(parent)?;
1610        }
1611
1612        let mut out_file = std::fs::File::create(&out_path)
1613            .with_context(|| format!("failed to create extracted file: {}", out_path.display()))?;
1614        std::io::copy(&mut entry, &mut out_file)?;
1615    }
1616
1617    let has_manifest = installed_dir.join("SKILL.md").exists()
1618        || installed_dir.join("SKILL.toml").exists()
1619        || installed_dir.join("manifest.toml").exists();
1620    if !has_manifest {
1621        std::fs::write(
1622            installed_dir.join("SKILL.toml"),
1623            format!(
1624                "[skill]\nname = \"{}\"\ndescription = \"ClawhHub installed skill\"\nversion = \"0.1.0\"\n",
1625                skill_dir_name
1626            ),
1627        )?;
1628    }
1629
1630    match enforce_skill_security_audit(&installed_dir, allow_scripts) {
1631        Ok(report) => Ok((installed_dir, report.files_scanned)),
1632        Err(err) => {
1633            let _ = std::fs::remove_dir_all(&installed_dir);
1634            Err(err)
1635        }
1636    }
1637}

1800/// Load skills from skill-capable plugins discovered by the plugin host.
1801///
1802/// Each plugin's `skills/` directory is fed to the existing skill loader, and
1803/// every loaded skill is renamed to `plugin:<plugin>/<skill>` to avoid
1804/// collisions with user-authored skills and between bundles. The `plugin:<name>`
1805/// tag is also added so prompts can distinguish plugin skills.
1806#[cfg(feature = "plugins-wasm")]
1807pub fn load_plugin_skills_from_config(config: &zeroclaw_config::schema::Config) -> Vec<Skill> {
1808    if !config.plugins.enabled {
1809        return Vec::new();
1810    }
1811
1812    let plugins_dir = expand_plugins_dir(&config.plugins.plugins_dir);
1813    let parent = match plugins_dir.parent() {
1814        Some(p) => p.to_path_buf(),
1815        None => return Vec::new(),
1816    };
1817
1818    let signature_mode = zeroclaw_plugins::host::PluginHost::parse_signature_mode(
1819        &config.plugins.security.signature_mode,
1820    );
1821    let trusted_keys = config.plugins.security.trusted_publisher_keys.clone();
1822
1823    let host = match zeroclaw_plugins::host::PluginHost::with_security(
1824        &parent,
1825        signature_mode,
1826        trusted_keys,
1827    ) {
1828        Ok(host) => host,
1829        Err(err) => {
1830            tracing::warn!("failed to discover plugin skills: {err}");
1831            return Vec::new();
1832        }
1833    };
1834
1835    let allow_scripts = config.skills.allow_scripts;
1836    let mut skills = Vec::new();
1837    for (manifest, skills_dir) in host.skill_plugin_details() {
1838        for raw in load_skills_from_directory(&skills_dir, allow_scripts) {
1839            skills.push(namespace_plugin_skill(&manifest.name, raw));
1840        }
1841    }
1842    skills
1843}

1845#[cfg(feature = "plugins-wasm")]
1846fn expand_plugins_dir(plugins_dir: &str) -> PathBuf {
1847    if let Some(rest) = plugins_dir.strip_prefix("~/")
1848        && let Some(dirs) = UserDirs::new()
1849    {
1850        return dirs.home_dir().join(rest);
1851    }
1852    PathBuf::from(plugins_dir)
1853}

1855#[cfg(feature = "plugins-wasm")]
1856fn namespace_plugin_skill(plugin_name: &str, mut skill: Skill) -> Skill {
1857    let qualified = format!("plugin:{}/{}", plugin_name, skill.name);
1858    skill.name = qualified;
1859    let plugin_tag = format!("plugin:{plugin_name}");
1860    if !skill.tags.iter().any(|t| t == &plugin_tag) {
1861        skill.tags.push(plugin_tag);
1862    }
1863    skill
