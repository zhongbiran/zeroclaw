34    /// Attempt to create a skill from a successful multi-step task execution.
35    /// Returns `Ok(Some(slug))` if a skill was created, `Ok(None)` if skipped
36    /// (disabled, duplicate, or insufficient tool calls).
37    pub async fn create_from_execution(
38        &self,
39        task_description: &str,
40        tool_calls: &[ToolCallRecord],
41        embedding_provider: Option<&dyn EmbeddingProvider>,
42    ) -> Result<Option<String>> {
43        if !self.config.enabled {
44            return Ok(None);
45        }
46
47        if tool_calls.len() < 2 {
48            return Ok(None);
49        }
50
51        // Deduplicate via embeddings when an embedding provider is available.
52        if let Some(provider) = embedding_provider
53            && provider.name() != "none"
54            && self.is_duplicate(task_description, provider).await?
55        {
56            return Ok(None);
57        }
58
59        let slug = Self::generate_slug(task_description);
60        if !Self::validate_slug(&slug) {
61            return Ok(None);
62        }
63
64        // Enforce LRU limit before writing a new skill.
65        self.enforce_lru_limit().await?;
66
67        let skill_dir = self.skills_dir().join(&slug);
68        tokio::fs::create_dir_all(&skill_dir)
69            .await
70            .with_context(|| {
71                format!("Failed to create skill directory: {}", skill_dir.display())
72            })?;
73
74        let toml_content = Self::generate_skill_toml(&slug, task_description, tool_calls);
75        let toml_path = skill_dir.join("SKILL.toml");
76        tokio::fs::write(&toml_path, toml_content.as_bytes())
77            .await
78            .with_context(|| format!("Failed to write {}", toml_path.display()))?;
79
80        Ok(Some(slug))
81    }

83    /// Generate a URL-safe slug from a task description.
84    /// Alphanumeric and hyphens only, max 64 characters.
85    fn generate_slug(description: &str) -> String {
86        let slug: String = description
87            .to_lowercase()
88            .chars()
89            .map(|c| if c.is_alphanumeric() { c } else { '-' })
90            .collect();
91
92        // Collapse consecutive hyphens.
93        let mut collapsed = String::with_capacity(slug.len());
94        let mut prev_hyphen = false;
95        for c in slug.chars() {
96            if c == '-' {
97                if !prev_hyphen {
98                    collapsed.push('-');
99                }
100                prev_hyphen = true;
101            } else {
102                collapsed.push(c);
103                prev_hyphen = false;
104            }
105        }
106
107        // Trim leading/trailing hyphens, then truncate.
108        let trimmed = collapsed.trim_matches('-');
109        if trimmed.len() > 64 {
110            // Find the nearest valid character boundary at or before 64 bytes.
111            let safe_index = trimmed
112                .char_indices()
113                .map(|(i, _)| i)
114                .take_while(|&i| i <= 64)
115                .last()
116                .unwrap_or(0);
117            let truncated = &trimmed[..safe_index];
118            truncated.trim_end_matches('-').to_string()
119        } else {
120            trimmed.to_string()
121        }
122    }
123
124    /// Validate that a slug is non-empty, alphanumeric + hyphens, max 64 chars.
125    fn validate_slug(slug: &str) -> bool {
126        !slug.is_empty()
127            && slug.len() <= 64
128            && slug.chars().all(|c| c.is_ascii_alphanumeric() || c == '-')
129            && !slug.starts_with('-')
130            && !slug.ends_with('-')
131    }

133    /// Generate SKILL.toml content from task execution data.
134    fn generate_skill_toml(slug: &str, description: &str, tool_calls: &[ToolCallRecord]) -> String {
135        use std::fmt::Write;
136        let mut toml = String::new();
137        toml.push_str("[skill]\n");
138        let _ = writeln!(toml, "name = {}", toml_escape(slug));
139        let _ = writeln!(
140            toml,
141            "description = {}",
142            toml_escape(&format!("Auto-generated: {description}"))
143        );
144        toml.push_str("version = \"0.1.0\"\n");
145        toml.push_str("author = \"zeroclaw-auto\"\n");
146        toml.push_str("tags = [\"auto-generated\"]\n");
147
148        for call in tool_calls {
149            toml.push('\n');
150            toml.push_str("[[tools]]\n");
151            let _ = writeln!(toml, "name = {}", toml_escape(&call.name));
152            let _ = writeln!(
153                toml,
154                "description = {}",
155                toml_escape(&format!("Tool used in task: {}", call.name))
156            );
157            toml.push_str("kind = \"shell\"\n");
158
159            // Extract the command from args if available, otherwise use the tool name.
160            let command = call
161                .args
162                .get("command")
163                .and_then(serde_json::Value::as_str)
164                .unwrap_or(&call.name);
165            let _ = writeln!(toml, "command = {}", toml_escape(command));
166        }
167
168        toml
169    }

171    /// Check if a skill with a similar description already exists.
172    async fn is_duplicate(
173        &self,
174        description: &str,
175        embedding_provider: &dyn EmbeddingProvider,
176    ) -> Result<bool> {
177        let new_embedding = embedding_provider.embed_one(description).await?;
178        if new_embedding.is_empty() {
179            return Ok(false);
180        }
181
182        let skills_dir = self.skills_dir();
183        if !skills_dir.exists() {
184            return Ok(false);
185        }
186
187        let mut entries = tokio::fs::read_dir(&skills_dir).await?;
188        while let Some(entry) = entries.next_entry().await? {
189            let toml_path = entry.path().join("SKILL.toml");
190            if !toml_path.exists() {
191                continue;
192            }
193
194            let content = tokio::fs::read_to_string(&toml_path).await?;
195            // Extract description from the TOML to compare.
196            if let Some(desc) = extract_description_from_toml(&content) {
197                let existing_embedding = embedding_provider.embed_one(&desc).await?;
198                if !existing_embedding.is_empty() {
199                    #[allow(clippy::cast_possible_truncation)]
200                    let similarity =
201                        f64::from(cosine_similarity(&new_embedding, &existing_embedding));
202                    if similarity > self.config.similarity_threshold {
203                        return Ok(true);
204                    }
205                }
206            }
207        }
208
209        Ok(false)
210    }

212    /// Remove the oldest auto-generated skill when we exceed `max_skills`.
213    async fn enforce_lru_limit(&self) -> Result<()> {
214        let skills_dir = self.skills_dir();
215        if !skills_dir.exists() {
216            return Ok(());
217        }
218
219        let mut auto_skills: Vec<(PathBuf, std::time::SystemTime)> = Vec::new();
220
221        let mut entries = tokio::fs::read_dir(&skills_dir).await?;
222        while let Some(entry) = entries.next_entry().await? {
223            let toml_path = entry.path().join("SKILL.toml");
224            if !toml_path.exists() {
225                continue;
226            }
227
228            let content = tokio::fs::read_to_string(&toml_path).await?;
229            if content.contains("\"zeroclaw-auto\"") || content.contains("\"auto-generated\"") {
230                let modified = tokio::fs::metadata(&toml_path)
231                    .await?
232                    .modified()
233                    .unwrap_or(std::time::UNIX_EPOCH);
234                auto_skills.push((entry.path(), modified));
235            }
236        }
237
238        // If at or above the limit, remove the oldest.
239        if auto_skills.len() >= self.config.max_skills {
240            auto_skills.sort_by_key(|(_, modified)| *modified);
241            if let Some((oldest_dir, _)) = auto_skills.first() {
242                tokio::fs::remove_dir_all(oldest_dir)
243                    .await
244                    .with_context(|| {
245                        format!(
246                            "Failed to remove oldest auto-generated skill: {}",
247                            oldest_dir.display()
248                        )
249                    })?;
250            }
251        }
252
253        Ok(())
254    }
255
256    fn skills_dir(&self) -> PathBuf {
257        self.workspace_dir.join("skills")
258    }

261/// Escape a string for TOML value (double-quoted).
262fn toml_escape(s: &str) -> String {
263    let escaped = s
264        .replace('\\', "\\\\")
265        .replace('"', "\\\"")
266        .replace('\n', "\\n")
267        .replace('\r', "\\r")
268        .replace('\t', "\\t");
269    format!("\"{escaped}\"")
270}

272/// Extract the description field from a SKILL.toml string.
273fn extract_description_from_toml(content: &str) -> Option<String> {
274    #[derive(serde::Deserialize)]
275    struct Partial {
276        skill: PartialSkill,
277    }
278    #[derive(serde::Deserialize)]
279    struct PartialSkill {
280        description: Option<String>,
281    }
282    toml::from_str::<Partial>(content)
283        .ok()
284        .and_then(|p| p.skill.description)
285}

287/// Extract `ToolCallRecord`s from the agent conversation history.
288///
289/// Scans assistant messages for tool call patterns (both JSON and XML formats)
290/// and returns records for each unique tool invocation.
291pub fn extract_tool_calls_from_history(
292    history: &[zeroclaw_providers::ChatMessage],
293) -> Vec<ToolCallRecord> {
294    let mut records = Vec::new();
295
296    for msg in history {
297        if msg.role != "assistant" {
298            continue;
299        }
300
301        // Try parsing as JSON (native tool_calls format).
302        if let Ok(value) = serde_json::from_str::<serde_json::Value>(&msg.content)
303            && let Some(tool_calls) = value.get("tool_calls").and_then(|v| v.as_array())
304        {
305            for call in tool_calls {
306                if let Some(function) = call.get("function") {
307                    let name = function
308                        .get("name")
309                        .and_then(serde_json::Value::as_str)
310                        .unwrap_or("")
311                        .to_string();
312                    let args_str = function
313                        .get("arguments")
314                        .and_then(serde_json::Value::as_str)
315                        .unwrap_or("{}");
316                    let args = serde_json::from_str(args_str).unwrap_or_default();
317                    if !name.is_empty() {
318                        records.push(ToolCallRecord { name, args });
319                    }
320                }
321            }
322        }
323
324        // Also try XML tool call format: <tool_name>...</tool_name>
325        // Simple extraction for `<shell>{"command":"..."}</shell>` style tags.
326        let content = &msg.content;
327        let mut pos = 0;
328        while pos < content.len() {
329            if let Some(start) = content[pos..].find('<') {
330                let abs_start = pos + start;
331                if let Some(end) = content[abs_start..].find('>') {
332                    let tag = &content[abs_start + 1..abs_start + end];
333                    // Skip closing tags and meta tags.
334                    if tag.starts_with('/') || tag.starts_with('!') || tag.starts_with('?') {
335                        pos = abs_start + end + 1;
336                        continue;
337                    }
338                    let tag_name = tag.split_whitespace().next().unwrap_or(tag);
339                    let close_tag = format!("</{tag_name}>");
340                    if let Some(close_pos) = content[abs_start + end + 1..].find(&close_tag) {
341                        let inner = &content[abs_start + end + 1..abs_start + end + 1 + close_pos];
342                        let args: serde_json::Value =
343                            serde_json::from_str(inner.trim()).unwrap_or_default();
344                        // Only add if it looks like a tool call (not HTML/formatting tags).
345                        if tag_name != "tool_result"
346                            && tag_name != "tool_results"
347                            && !tag_name.contains(':')
348                            && args.is_object()
349                            && !args.as_object().is_none_or(|o| o.is_empty())
350                        {
351                            records.push(ToolCallRecord {
352                                name: tag_name.to_string(),
353                                args,
354                            });
355                        }
356                        pos = abs_start + end + 1 + close_pos + close_tag.len();
357                    } else {
358                        pos = abs_start + end + 1;
359                    }
360                } else {
361                    break;
362                }
363            } else {
364                break;
365            }
366        }
367    }
368
369    records
370}

554    /// A mock embedding provider that returns deterministic embeddings.
555    ///
556    /// The "new" description (first text embedded) always gets `[1, 0, 0]`.
557    /// The "existing" skill description (second text embedded) gets a vector
558    /// whose cosine similarity with `[1, 0, 0]` equals `self.similarity`.
559    struct MockEmbeddingProvider {
560        similarity: f32,
561        call_count: std::sync::atomic::AtomicUsize,
562    }
563
564    impl MockEmbeddingProvider {
565        fn new(similarity: f32) -> Self {
566            Self {
567                similarity,
568                call_count: std::sync::atomic::AtomicUsize::new(0),
569            }
570        }
571    }
572
573    #[async_trait]
574    impl EmbeddingProvider for MockEmbeddingProvider {
575        fn name(&self) -> &str {
576            "mock"
577        }
578        fn dimensions(&self) -> usize {
579            3
580        }
581        async fn embed(&self, texts: &[&str]) -> anyhow::Result<Vec<Vec<f32>>> {
582            Ok(texts
583                .iter()
584                .map(|_| {
585                    let call = self
586                        .call_count
587                        .fetch_add(1, std::sync::atomic::Ordering::Relaxed);
588                    if call == 0 {
589                        // First call: the "new" description.
590                        vec![1.0, 0.0, 0.0]
591                    } else {
592                        // Subsequent calls: existing skill descriptions.
593                        // Produce a vector with the configured cosine similarity to [1,0,0].
594                        vec![
595                            self.similarity,
596                            (1.0 - self.similarity * self.similarity).sqrt(),
597                            0.0,
598                        ]
599                    }
600                })
601                .collect())
602        }
603    }
