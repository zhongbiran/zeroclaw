24/// Parse an execution mode string into `SopExecutionMode`, falling back to
25/// `Supervised` for unknown values.
26pub fn parse_execution_mode(s: &str) -> SopExecutionMode {
27    match s.trim().to_lowercase().as_str() {
28        "auto" => SopExecutionMode::Auto,
29        "step_by_step" => SopExecutionMode::StepByStep,
30        "priority_based" => SopExecutionMode::PriorityBased,
31        "deterministic" => SopExecutionMode::Deterministic,
32        // "supervised" and any unknown value
33        _ => SopExecutionMode::Supervised,
34    }
35}

39/// Return the default SOPs directory: `<workspace>/sops`.
40fn sops_dir(workspace_dir: &Path) -> PathBuf {
41    workspace_dir.join("sops")
42}

44/// Resolve the SOPs directory from config, falling back to workspace default.
45pub fn resolve_sops_dir(workspace_dir: &Path, config_dir: Option<&str>) -> PathBuf {
46    match config_dir {
47        Some(dir) if !dir.is_empty() => {
48            let expanded = shellexpand::tilde(dir);
49            PathBuf::from(expanded.as_ref())
50        }
51        _ => sops_dir(workspace_dir),
52    }
53}

67/// Load SOPs from a specific directory. Each subdirectory may contain
68/// `SOP.toml` (metadata + triggers) and `SOP.md` (procedure steps).
69pub fn load_sops_from_directory(
70    sops_dir: &Path,
71    default_execution_mode: SopExecutionMode,
72) -> Vec<Sop> {
73    if !sops_dir.exists() {
74        return Vec::new();
75    }
76
77    let mut sops = Vec::new();
78
79    let Ok(entries) = std::fs::read_dir(sops_dir) else {
80        return sops;
81    };
82
83    for entry in entries.flatten() {
84        let path = entry.path();
85        if !path.is_dir() {
86            continue;
87        }
88
89        let toml_path = path.join("SOP.toml");
90        if !toml_path.exists() {
91            continue;
92        }
93
94        match load_sop(&path, default_execution_mode) {
95            Ok(sop) => sops.push(sop),
96            Err(e) => {
97                warn!("Failed to load SOP from {}: {e}", path.display());
98            }
99        }
100    }
101
102    sops.sort_by(|a, b| a.name.cmp(&b.name));
103    sops
104}

106/// Load a single SOP from a directory containing SOP.toml and optionally SOP.md.
107fn load_sop(sop_dir: &Path, default_execution_mode: SopExecutionMode) -> Result<Sop> {
108    let toml_path = sop_dir.join("SOP.toml");
109    let toml_content = std::fs::read_to_string(&toml_path)?;
110    let manifest: SopManifest = toml::from_str(&toml_content)?;
111
112    let md_path = sop_dir.join("SOP.md");
113    let steps = if md_path.exists() {
114        let md_content = std::fs::read_to_string(&md_path)?;
115        parse_steps(&md_content)
116    } else {
117        Vec::new()
118    };
119
120    let SopMeta {
121        name,
122        description,
123        version,
124        priority,
125        execution_mode,
126        cooldown_secs,
127        max_concurrent,
128        deterministic,
129    } = manifest.sop;
130
131    // When deterministic=true, override execution_mode to Deterministic
132    let effective_mode = if deterministic {
133        SopExecutionMode::Deterministic
134    } else {
135        execution_mode.unwrap_or(default_execution_mode)
136    };
137
138    Ok(Sop {
139        name,
140        description,
141        version,
142        priority,
143        execution_mode: effective_mode,
144        triggers: manifest.triggers,
145        steps,
146        cooldown_secs,
147        max_concurrent,
148        location: Some(sop_dir.to_path_buf()),
149        deterministic,
150    })
151}

155/// Parse procedure steps from SOP.md content.
156///
157/// Expects a `## Steps` heading followed by numbered items (`1.`, `2.`, …).
158/// Each item's first bold text (`**...**`) is the step title; the rest is body.
159/// Sub-bullets `- tools:` and `- requires_confirmation: true` are parsed.
160pub fn parse_steps(md: &str) -> Vec<SopStep> {
161    let mut steps = Vec::new();
162    let mut in_steps_section = false;
163    let mut current_number: Option<u32> = None;
164    let mut current_title = String::new();
165    let mut current_body = String::new();
166    let mut current_tools: Vec<String> = Vec::new();
167    let mut current_requires_confirmation = false;
168    let mut current_kind = SopStepKind::Execute;
169
170    for line in md.lines() {
171        let trimmed = line.trim();
172
173        // Detect ## Steps heading
174        if trimmed.starts_with("## ") {
175            if trimmed.eq_ignore_ascii_case("## steps") || trimmed.eq_ignore_ascii_case("## Steps")
176            {
177                in_steps_section = true;
178                continue;
179            }
180            // Any other ## heading ends the steps section
181            if in_steps_section {
182                // Flush pending step
183                flush_step(
184                    &mut steps,
185                    &mut current_number,
186                    &mut current_title,
187                    &mut current_body,
188                    &mut current_tools,
189                    &mut current_requires_confirmation,
190                    &mut current_kind,
191                );
192                in_steps_section = false;
193            }
194            continue;
195        }
196
197        if !in_steps_section {
198            continue;
199        }
200
201        // Check for numbered item: `1.`, `2.`, etc.
202        if let Some(rest) = parse_numbered_item(trimmed) {
203            // Flush previous step
204            flush_step(
205                &mut steps,
206                &mut current_number,
207                &mut current_title,
208                &mut current_body,
209                &mut current_tools,
210                &mut current_requires_confirmation,
211                &mut current_kind,
212            );
213
214            let step_num = u32::try_from(steps.len())
215                .unwrap_or(u32::MAX)
216                .saturating_add(1);
217            current_number = Some(step_num);
218
219            // Extract title from bold text: **title** — body
220            if let Some((title, body)) = extract_bold_title(rest) {
221                current_title = title;
222                current_body = body;
223            } else {
224                current_title = rest.to_string();
225                current_body = String::new();
226            }
227            current_tools = Vec::new();
228            current_requires_confirmation = false;
229            continue;
230        }
231
232        // Sub-bullet parsing (only when inside a step)
233        if current_number.is_some() && trimmed.starts_with("- ") {
234            let bullet = trimmed.trim_start_matches("- ").trim();
235            if let Some(tools_str) = bullet.strip_prefix("tools:") {
236                current_tools = tools_str
237                    .split(',')
238                    .map(|t| t.trim().to_string())
239                    .filter(|t| !t.is_empty())
240                    .collect();
241            } else if bullet.starts_with("requires_confirmation:") {
242                if let Some(val) = bullet.strip_prefix("requires_confirmation:") {
243                    current_requires_confirmation = val.trim().eq_ignore_ascii_case("true");
244                }
245            } else if bullet.starts_with("kind:") {
246                if let Some(val) = bullet.strip_prefix("kind:") {
247                    let val = val.trim();
248                    if val.eq_ignore_ascii_case("checkpoint") {
249                        current_kind = SopStepKind::Checkpoint;
250                    } else {
251                        current_kind = SopStepKind::Execute;
252                    }
253                }
254            } else {
255                // Continuation body line
256                if !current_body.is_empty() {
257                    current_body.push('\n');
258                }
259                current_body.push_str(trimmed);
260            }
261            continue;
262        }
263
264        // Continuation line for step body
265        if current_number.is_some() && !trimmed.is_empty() {
266            if !current_body.is_empty() {
267                current_body.push('\n');
268            }
269            current_body.push_str(trimmed);
270        }
271    }
272
273    // Flush final step
274    flush_step(
275        &mut steps,
276        &mut current_number,
277        &mut current_title,
278        &mut current_body,
279        &mut current_tools,
280        &mut current_requires_confirmation,
281        &mut current_kind,
282    );
283
284    steps
285}

287/// Flush accumulated step state into the steps vector.
288fn flush_step(
289    steps: &mut Vec<SopStep>,
290    number: &mut Option<u32>,
291    title: &mut String,
292    body: &mut String,
293    tools: &mut Vec<String>,
294    requires_confirmation: &mut bool,
295    kind: &mut SopStepKind,
296) {
297    if let Some(n) = number.take() {
298        steps.push(SopStep {
299            number: n,
300            title: std::mem::take(title),
301            body: body.trim().to_string(),
302            suggested_tools: std::mem::take(tools),
303            requires_confirmation: *requires_confirmation,
304            kind: *kind,
305            schema: None,
306        });
307        *body = String::new();
308        *requires_confirmation = false;
309        *kind = SopStepKind::Execute;
310    }
311}

313/// Try to parse `N. rest` from a line, returning `rest` if successful.
314fn parse_numbered_item(line: &str) -> Option<&str> {
315    let dot_pos = line.find(". ")?;
316    let prefix = &line[..dot_pos];
317    if prefix.chars().all(|c| c.is_ascii_digit()) && !prefix.is_empty() {
318        Some(line[dot_pos + 2..].trim())
319    } else {
320        None
321    }
322}

324/// Extract `**title**` from the beginning of text, returning (title, rest).
325pub fn extract_bold_title(text: &str) -> Option<(String, String)> {
326    let start = text.find("**")?;
327    let after_start = start + 2;
328    let end = text[after_start..].find("**")?;
329    let title = text[after_start..after_start + end].to_string();
330
331    // Rest is everything after the closing ** and any separator (— or -)
332    let rest_start = after_start + end + 2;
333    let rest = text[rest_start..].trim();
334    let rest = rest
335        .strip_prefix("—")
336        .or_else(|| rest.strip_prefix("–"))
337        .or_else(|| rest.strip_prefix("-"))
338        .unwrap_or(rest)
339        .trim();
340
341    Some((title, rest.to_string()))
342}

346/// Validate a loaded SOP and return a list of warnings.
347pub fn validate_sop(sop: &Sop) -> Vec<String> {
348    let mut warnings = Vec::new();
349
350    if sop.name.is_empty() {
351        warnings.push("SOP name is empty".into());
352    }
353    if sop.description.is_empty() {
354        warnings.push("SOP description is empty".into());
355    }
356    if sop.triggers.is_empty() {
357        warnings.push("SOP has no triggers defined".into());
358    }
359    if sop.steps.is_empty() {
360        warnings.push("SOP has no steps (missing or empty SOP.md)".into());
361    }
362
363    // Check step numbering continuity
364    for (i, step) in sop.steps.iter().enumerate() {
365        let expected = u32::try_from(i).unwrap_or(u32::MAX).saturating_add(1);
366        if step.number != expected {
367            warnings.push(format!(
368                "Step numbering gap: expected {expected}, got {}",
369                step.number
370            ));
371        }
372        if step.title.is_empty() {
373            warnings.push(format!("Step {} has an empty title", step.number));
374        }
375    }
376
377    warnings
