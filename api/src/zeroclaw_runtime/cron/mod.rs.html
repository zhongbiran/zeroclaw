26/// Validate a shell command against the full security policy (allowlist + risk gate).
27///
28/// Returns `Ok(())` if the command passes all checks, or an error describing
29/// why it was blocked.
30pub fn validate_shell_command(config: &Config, command: &str, approved: bool) -> Result<()> {
31    let security = SecurityPolicy::from_config(&config.autonomy, &config.workspace_dir);
32    validate_shell_command_with_security(&security, command, approved)
33}
34
35/// Validate a shell command using an existing `SecurityPolicy` instance.
36///
37/// Preferred when the caller already holds a `SecurityPolicy` (e.g. scheduler).
38pub fn validate_shell_command_with_security(
39    security: &SecurityPolicy,
40    command: &str,
41    approved: bool,
42) -> Result<()> {
43    security
44        .validate_command_execution(command, approved)
45        .map(|_| ())
46        .map_err(|reason| anyhow!("blocked by security policy: {reason}"))
47}

49pub fn validate_delivery_config(delivery: Option<&DeliveryConfig>) -> Result<()> {
50    let Some(delivery) = delivery else {
51        return Ok(());
52    };
53
54    if delivery.mode.eq_ignore_ascii_case("none") {
55        return Ok(());
56    }
57    if !delivery.mode.eq_ignore_ascii_case("announce") {
58        bail!("unsupported delivery mode: {}", delivery.mode);
59    }
60
61    let channel = delivery.channel.as_deref().map(str::trim);
62    let Some(channel) = channel.filter(|value| !value.is_empty()) else {
63        bail!("delivery.channel is required for announce mode");
64    };
65    match channel.to_ascii_lowercase().as_str() {
66        "telegram" | "discord" | "slack" | "mattermost" | "signal" | "matrix" | "qq"
67        | "webhook" => {}
68        other => bail!("unsupported delivery channel: {other}"),
69    }
70
71    let has_target = delivery
72        .to
73        .as_deref()
74        .map(str::trim)
75        .is_some_and(|value| !value.is_empty());
76    if !has_target {
77        bail!("delivery.to is required for announce mode");
78    }
79
80    Ok(())
81}

83/// Create a validated shell job, enforcing security policy before persistence.
84///
85/// All entrypoints that create shell cron jobs should route through this
86/// function to guarantee consistent policy enforcement.
87pub fn add_shell_job_with_approval(
88    config: &Config,
89    name: Option<String>,
90    schedule: Schedule,
91    command: &str,
92    delivery: Option<DeliveryConfig>,
93    approved: bool,
94) -> Result<CronJob> {
95    validate_shell_command(config, command, approved)?;
96    validate_delivery_config(delivery.as_ref())?;
97    store::add_shell_job(config, name, schedule, command, delivery)
98}

100/// Update a shell job's command with security validation.
101///
102/// Validates the new command (if changed) before persisting.
103pub fn update_shell_job_with_approval(
104    config: &Config,
105    job_id: &str,
106    patch: CronJobPatch,
107    approved: bool,
108) -> Result<CronJob> {
109    if let Some(command) = patch.command.as_deref() {
110        validate_shell_command(config, command, approved)?;
111    }
112    update_job(config, job_id, patch)
113}

115/// Create a one-shot validated shell job from a delay string (e.g. "30m").
116pub fn add_once_validated(
117    config: &Config,
118    delay: &str,
119    command: &str,
120    approved: bool,
121) -> Result<CronJob> {
122    let duration = parse_delay(delay)?;
123    let at = chrono::Utc::now() + duration;
124    add_once_at_validated(config, at, command, approved)
125}

192pub fn parse_delay(input: &str) -> Result<chrono::Duration> {
193    let input = input.trim();
194    if input.is_empty() {
195        anyhow::bail!("delay must not be empty");
196    }
197    let split = input
198        .find(|c: char| !c.is_ascii_digit())
199        .unwrap_or(input.len());
200    let (num, unit) = input.split_at(split);
201    let amount: i64 = num.parse()?;
202    let unit = if unit.is_empty() { "m" } else { unit };
203    let duration = match unit {
204        "s" => chrono::Duration::seconds(amount),
205        "m" => chrono::Duration::minutes(amount),
206        "h" => chrono::Duration::hours(amount),
207        "d" => chrono::Duration::days(amount),
208        _ => anyhow::bail!("unsupported delay unit '{unit}', use s/m/h/d"),
209    };
210    Ok(duration)
211}
