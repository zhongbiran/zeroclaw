31#[async_trait]
32impl Tool for SopExecuteTool {
33    fn name(&self) -> &str {
34        "sop_execute"
35    }
36
37    fn description(&self) -> &str {
38        "Manually trigger a Standard Operating Procedure (SOP) by name. Returns the run ID and first step instruction. Use sop_list to see available SOPs."
39    }
40
41    fn parameters_schema(&self) -> serde_json::Value {
42        json!({
43            "type": "object",
44            "properties": {
45                "name": {
46                    "type": "string",
47                    "description": "Name of the SOP to execute"
48                },
49                "payload": {
50                    "type": "string",
51                    "description": "Optional trigger payload (JSON string)"
52                }
53            },
54            "required": ["name"]
55        })
56    }
57
58    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
59        let sop_name = args
60            .get("name")
61            .and_then(|v| v.as_str())
62            .ok_or_else(|| anyhow::anyhow!("Missing 'name' parameter"))?;
63
64        let payload = args
65            .get("payload")
66            .and_then(|v| v.as_str())
67            .map(String::from);
68
69        let event = SopEvent {
70            source: SopTriggerSource::Manual,
71            topic: None,
72            payload,
73            timestamp: now_iso8601(),
74        };
75
76        // Lock engine, start run, snapshot run for audit, then drop lock
77        let (action, run_snapshot) = {
78            let mut engine = self
79                .engine
80                .lock()
81                .map_err(|e| anyhow::anyhow!("Engine lock poisoned: {e}"))?;
82
83            match engine.start_run(sop_name, event) {
84                Ok(action) => {
85                    let run_id = action_run_id(&action);
86                    let snapshot = run_id.and_then(|id| engine.get_run(id).cloned());
87                    (Ok(action), snapshot)
88                }
89                Err(e) => (Err(e), None),
90            }
91        };
92
93        // Audit log (engine lock dropped, safe to await)
94        if let Some(ref audit) = self.audit
95            && let Some(ref run) = run_snapshot
96            && let Err(e) = audit.log_run_start(run).await
97        {
98            warn!("SOP audit log_run_start failed: {e}");
99        }
100
101        match action {
102            Ok(action) => {
103                let output = match action {
104                    SopRunAction::ExecuteStep {
105                        run_id, context, ..
106                    } => {
107                        format!("SOP run started: {run_id}\n\n{context}")
108                    }
109                    SopRunAction::WaitApproval {
110                        run_id, context, ..
111                    } => {
112                        format!("SOP run started: {run_id} (waiting for approval)\n\n{context}")
113                    }
114                    SopRunAction::Completed { run_id, sop_name } => {
115                        format!("SOP '{sop_name}' run {run_id} completed immediately (no steps).")
116                    }
117                    SopRunAction::Failed { run_id, reason, .. } => {
118                        format!("SOP run {run_id} failed: {reason}")
119                    }
120                    SopRunAction::DeterministicStep { run_id, step, .. } => {
121                        format!(
122                            "SOP run started (deterministic): {run_id}\nFirst step: {}",
123                            step.title
124                        )
125                    }
126                    SopRunAction::CheckpointWait { run_id, step, .. } => {
127                        format!(
128                            "SOP run started: {run_id} (paused at checkpoint: {})",
129                            step.title
130                        )
131                    }
132                };
133                Ok(ToolResult {
134                    success: true,
135                    output,
136                    error: None,
137                })
138            }
139            Err(e) => Ok(ToolResult {
140                success: false,
141                output: String::new(),
142                error: Some(format!("Failed to start SOP: {e}")),
143            }),
144        }
145    }
146}
147
148/// Extract run_id from any SopRunAction variant.
149fn action_run_id(action: &SopRunAction) -> Option<&str> {
150    match action {
151        SopRunAction::ExecuteStep { run_id, .. }
152        | SopRunAction::WaitApproval { run_id, .. }
153        | SopRunAction::Completed { run_id, .. }
154        | SopRunAction::Failed { run_id, .. }
155        | SopRunAction::DeterministicStep { run_id, .. }
156        | SopRunAction::CheckpointWait { run_id, .. } => Some(run_id),
157    }
158}

169    fn test_sop(name: &str, mode: SopExecutionMode) -> Sop {
170        Sop {
171            name: name.into(),
172            description: format!("Test SOP: {name}"),
173            version: "1.0.0".into(),
174            priority: SopPriority::Normal,
175            execution_mode: mode,
176            triggers: vec![SopTrigger::Manual],
177            steps: vec![
178                SopStep {
179                    number: 1,
180                    title: "Step one".into(),
181                    body: "Do step one".into(),
182                    suggested_tools: vec!["shell".into()],
183                    requires_confirmation: false,
184                    kind: SopStepKind::default(),
185                    schema: None,
186                },
187                SopStep {
188                    number: 2,
189                    title: "Step two".into(),
190                    body: "Do step two".into(),
191                    suggested_tools: vec![],
192                    requires_confirmation: false,

204    fn engine_with_sops(sops: Vec<Sop>) -> Arc<Mutex<SopEngine>> {
205        let mut engine = SopEngine::new(SopConfig::default());
206        engine.set_sops_for_test(sops);
207        Arc::new(Mutex::new(engine))
208    }

211    async fn execute_auto_sop() {
