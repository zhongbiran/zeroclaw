10/// Lists all loaded SOPs with their triggers, priority, step count, and active runs.
11pub struct SopListTool {
12    engine: std::sync::Arc<Mutex<SopEngine>>,
13}
14
15impl SopListTool {
16    pub fn new(engine: std::sync::Arc<Mutex<SopEngine>>) -> Self {
17        Self { engine }
18    }
19}
20
21#[async_trait]
22impl Tool for SopListTool {
23    fn name(&self) -> &str {
24        "sop_list"
25    }
26
27    fn description(&self) -> &str {
28        "List all loaded Standard Operating Procedures (SOPs) with their triggers, priority, step count, and active run count. Optionally filter by name or priority."
29    }
30
31    fn parameters_schema(&self) -> serde_json::Value {
32        json!({
33            "type": "object",
34            "properties": {
35                "filter": {
36                    "type": "string",
37                    "description": "Filter SOPs by name substring or priority (low/normal/high/critical)"
38                }
39            }
40        })
41    }
42
43    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
44        let filter = args.get("filter").and_then(|v| v.as_str()).unwrap_or("");
45        let filter_lower = filter.to_lowercase();
46
47        let engine = self
48            .engine
49            .lock()
50            .map_err(|e| anyhow::anyhow!("Engine lock poisoned: {e}"))?;
51        let sops = engine.sops();
52
53        if sops.is_empty() {
54            return Ok(ToolResult {
55                success: true,
56                output: "No SOPs loaded.".into(),
57                error: None,
58            });
59        }
60
61        let filtered: Vec<_> = if filter_lower.is_empty() {
62            sops.iter().collect()
63        } else {
64            sops.iter()
65                .filter(|s| {
66                    s.name.to_lowercase().contains(&filter_lower)
67                        || s.priority.to_string() == filter_lower
68                })
69                .collect()
70        };
71
72        if filtered.is_empty() {
73            return Ok(ToolResult {
74                success: true,
75                output: format!("No SOPs match filter '{filter}'."),
76                error: None,
77            });
78        }
79
80        let active_runs = engine.active_runs();
81        let mut output = format!(
82            "Loaded SOPs ({} total, {} shown):\n\n",
83            sops.len(),
84            filtered.len()
85        );
86
87        for sop in &filtered {
88            let active_count = active_runs
89                .values()
90                .filter(|r| r.sop_name == sop.name)
91                .count();
92            let triggers: Vec<String> = sop.triggers.iter().map(|t| t.to_string()).collect();
93
94            let _ = writeln!(
95                output,
96                "- **{}** [{}] — {} steps, {} trigger(s): {}{}",
97                sop.name,
98                sop.priority,
99                sop.steps.len(),
100                sop.triggers.len(),
101                triggers.join(", "),
102                if active_count > 0 {
103                    format!(" (active runs: {active_count})")
104                } else {
105                    String::new()
106                }
107            );
108        }
109
110        Ok(ToolResult {
111            success: true,
112            output,
113            error: None,
114        })
115    }
116}

134            steps: vec![SopStep {

136                title: "Step one".into(),

139                requires_confirmation: false,
140                kind: SopStepKind::default(),

150    fn engine_with_sops(sops: Vec<Sop>) -> Arc<Mutex<SopEngine>> {
151        let mut engine = SopEngine::new(SopConfig::default());
152        engine.set_sops_for_test(sops);
153        Arc::new(Mutex::new(engine))
154    }

156    #[tokio::test]
157    async fn list_all_sops() {
158        let engine = engine_with_sops(vec![
159            test_sop("pump-shutdown", SopPriority::Critical),
160            test_sop("daily-check", SopPriority::Normal),
161        ]);
162        let tool = SopListTool::new(engine);
163        let result = tool.execute(json!({})).await.unwrap();
164        assert!(result.success);
165        assert!(result.output.contains("pump-shutdown"));
166        assert!(result.output.contains("daily-check"));
167        assert!(result.output.contains("2 total"));
