21/// Thread-safe ring buffer that retains recent events for history replay.
22pub struct EventBuffer {
23    inner: Mutex<VecDeque<serde_json::Value>>,
24    capacity: usize,
25}
26
27impl EventBuffer {
28    pub fn new(capacity: usize) -> Self {
29        Self {
30            inner: Mutex::new(VecDeque::with_capacity(capacity)),
31            capacity,
32        }
33    }
34
35    /// Push an event into the buffer, evicting the oldest if at capacity.
36    pub fn push(&self, event: serde_json::Value) {
37        let mut buf = self.inner.lock().unwrap();
38        if buf.len() == self.capacity {
39            buf.pop_front();
40        }
41        buf.push_back(event);
42    }
43
44    /// Return a snapshot of all buffered events (oldest first).
45    pub fn snapshot(&self) -> Vec<serde_json::Value> {
46        self.inner.lock().unwrap().iter().cloned().collect()
47    }
48}

50/// GET /api/events — SSE event stream
51pub async fn handle_sse_events(
52    State(state): State<AppState>,
53    headers: HeaderMap,
54) -> impl IntoResponse {
55    // Auth check
56    if state.pairing.require_pairing() {
57        let token = headers
58            .get(header::AUTHORIZATION)
59            .and_then(|v| v.to_str().ok())
60            .and_then(|auth| auth.strip_prefix("Bearer "))
61            .unwrap_or("");
62
63        if !state.pairing.is_authenticated(token) {
64            return (
65                StatusCode::UNAUTHORIZED,
66                "Unauthorized — provide Authorization: Bearer <token>",
67            )
68                .into_response();
69        }
70    }
71
72    let rx = state.event_tx.subscribe();
73    let stream = BroadcastStream::new(rx).filter_map(
74        |result: Result<
75            serde_json::Value,
76            tokio_stream::wrappers::errors::BroadcastStreamRecvError,
77        >| {
78            match result {
79                Ok(value) if is_public_sse_event(&value) => Some(Ok::<_, Infallible>(
80                    Event::default().data(value.to_string()),
81                )),
82                Ok(_) => None,
83                Err(_) => None, // Skip lagged messages
84            }
85        },
86    );
87
88    Sse::new(stream)
89        .keep_alive(KeepAlive::default())
90        .into_response()
91}

93/// GET /api/events/history — return buffered recent events as JSON.
94pub async fn handle_events_history(
95    State(state): State<AppState>,
96    headers: HeaderMap,
97) -> impl IntoResponse {
98    if let Err(e) = super::api::require_auth(&state, &headers) {
99        return e.into_response();
100    }
101    let events: Vec<_> = state
102        .event_buffer
103        .snapshot()
104        .into_iter()
105        .filter(is_public_sse_event)
106        .collect();
107    Json(serde_json::json!({ "events": events })).into_response()
108}

110/// Returns true for events that should be visible on the global SSE stream.
111///
112/// Contract: broadcast events must not include `session_id` unless they are
113/// intentionally scoped to that session and hidden from global `/api/events`.
114fn is_public_sse_event(event: &serde_json::Value) -> bool {
115    event
116        .get("session_id")
117        .and_then(serde_json::Value::as_str)
118        .is_none()
119}

121/// Broadcast observer that fans events out to SSE subscribers.
122///
123/// Installed as the process-wide broadcast hook by [`crate::run_gateway`] so
124/// that events recorded by *any* observer built through
125/// `observability::create_observer` — including the per-call observer the
126/// agent loop creates inside `process_message` — also reach `/api/events`
127/// clients.
128///
129/// Crate-private: the constructor signature is intentionally not part of any
130/// stable surface, since it is wired directly into `run_gateway`.
131pub(crate) struct BroadcastObserver {
132    tx: tokio::sync::broadcast::Sender<serde_json::Value>,
133    buffer: Arc<EventBuffer>,
134}
135
136impl BroadcastObserver {
137    pub(crate) fn new(
138        tx: tokio::sync::broadcast::Sender<serde_json::Value>,
139        buffer: Arc<EventBuffer>,
140    ) -> Self {
141        Self { tx, buffer }
142    }
143}

145impl zeroclaw_runtime::observability::Observer for BroadcastObserver {
146    fn record_event(&self, event: &zeroclaw_runtime::observability::ObserverEvent) {
147        // Recording into the primary observer (logs / Prometheus) is the
148        // responsibility of whoever built the event source; `TeeObserver`
149        // takes care of that fan-out. Here we only translate to JSON and
150        // ship to SSE subscribers.
151        let json = match event {
152            zeroclaw_runtime::observability::ObserverEvent::LlmRequest {
153                provider, model, ..
154            } => serde_json::json!({
155                "type": "llm_request",
156                "provider": provider,
157                "model": model,
158                "timestamp": chrono::Utc::now().to_rfc3339(),
159            }),
160            zeroclaw_runtime::observability::ObserverEvent::LlmResponse {
161                provider,
162                model,
163                duration,
164                success,
165                error_message,
166                input_tokens,
167                output_tokens,
168            } => serde_json::json!({
169                "type": "llm_response",
170                "provider": provider,
171                "model": model,
172                "duration_ms": duration.as_millis(),
173                "success": success,
174                "error_message": error_message,
175                "input_tokens": input_tokens,
176                "output_tokens": output_tokens,
177                "timestamp": chrono::Utc::now().to_rfc3339(),
178            }),
179            zeroclaw_runtime::observability::ObserverEvent::ToolCall {
180                tool,
181                duration,
182                success,
183            } => serde_json::json!({
184                "type": "tool_call",
185                "tool": tool,
186                "duration_ms": duration.as_millis(),
187                "success": success,
188                "timestamp": chrono::Utc::now().to_rfc3339(),
189            }),
190            zeroclaw_runtime::observability::ObserverEvent::ToolCallStart { tool, .. } => {
191                serde_json::json!({
192                    "type": "tool_call_start",
193                    "tool": tool,
194                    "timestamp": chrono::Utc::now().to_rfc3339(),
195                })
196            }
197            zeroclaw_runtime::observability::ObserverEvent::TurnComplete => {
198                serde_json::json!({
199                    "type": "turn_complete",
200                    "timestamp": chrono::Utc::now().to_rfc3339(),
201                })
202            }
203            zeroclaw_runtime::observability::ObserverEvent::ChannelMessage {
204                channel,
205                direction,
206            } => serde_json::json!({
207                "type": "channel_message",
208                "channel": channel,
209                "direction": direction,
210                "timestamp": chrono::Utc::now().to_rfc3339(),
211            }),
212            zeroclaw_runtime::observability::ObserverEvent::HeartbeatTick => {
213                serde_json::json!({
214                    "type": "heartbeat_tick",
215                    "timestamp": chrono::Utc::now().to_rfc3339(),
216                })
217            }
218            zeroclaw_runtime::observability::ObserverEvent::Error { component, message } => {
219                serde_json::json!({
220                    "type": "error",
221                    "component": component,
222                    "message": message,
223                    "timestamp": chrono::Utc::now().to_rfc3339(),
224                })
225            }
226            zeroclaw_runtime::observability::ObserverEvent::AgentStart { provider, model } => {
227                serde_json::json!({
228                    "type": "agent_start",
229                    "provider": provider,
230                    "model": model,
231                    "timestamp": chrono::Utc::now().to_rfc3339(),
232                })
233            }
234            zeroclaw_runtime::observability::ObserverEvent::AgentEnd {
235                provider,
236                model,
237                duration,
238                tokens_used,
239                cost_usd,
240            } => serde_json::json!({
241                "type": "agent_end",
242                "provider": provider,
243                "model": model,
244                "duration_ms": duration.as_millis(),
245                "tokens_used": tokens_used,
246                "cost_usd": cost_usd,
247                "timestamp": chrono::Utc::now().to_rfc3339(),
248            }),
249            zeroclaw_runtime::observability::ObserverEvent::CacheHit {
250                cache_type,
251                tokens_saved,
252            } => serde_json::json!({
253                "type": "cache_hit",
254                "cache_type": cache_type,
255                "tokens_saved": tokens_saved,
256                "timestamp": chrono::Utc::now().to_rfc3339(),
257            }),
258            zeroclaw_runtime::observability::ObserverEvent::CacheMiss { cache_type } => {
259                serde_json::json!({
260                    "type": "cache_miss",
261                    "cache_type": cache_type,
262                    "timestamp": chrono::Utc::now().to_rfc3339(),
263                })
264            }
265            _ => return, // Skip events we don't broadcast
266        };
267
268        self.buffer.push(json.clone());
269        let _ = self.tx.send(json);
270    }

272    fn record_metric(&self, _metric: &zeroclaw_runtime::observability::traits::ObserverMetric) {
273        // Metrics are not broadcast over SSE; the primary observer records them.
274    }

276    fn name(&self) -> &str {
277        "broadcast"
278    }

281        self
282    }

290    fn make_broadcast() -> (
291        Arc<BroadcastObserver>,
292        tokio::sync::broadcast::Receiver<serde_json::Value>,
293        Arc<EventBuffer>,
294    ) {
295        let (tx, rx) = tokio::sync::broadcast::channel(16);
296        let buffer = Arc::new(EventBuffer::new(16));
297        let obs = Arc::new(BroadcastObserver::new(tx, buffer.clone()));
298        (obs, rx, buffer)
299    }

301    #[test]
302    fn tool_call_event_is_broadcast_and_buffered() {
303        let (obs, mut rx, buffer) = make_broadcast();
304
305        obs.record_event(&ObserverEvent::ToolCall {
306            tool: "shell".into(),
307            duration: std::time::Duration::from_millis(42),
308            success: true,
309        });
310
311        let value = rx.try_recv().expect("event should be broadcast");
312        assert_eq!(value["type"], "tool_call");
313        assert_eq!(value["tool"], "shell");
314        assert_eq!(value["success"], true);
315
316        let snap = buffer.snapshot();
317        assert_eq!(snap.len(), 1);
318        assert_eq!(snap[0]["type"], "tool_call");
319    }

365    /// End-to-end coverage of the wiring `run_gateway` performs at startup:
366    /// installing `BroadcastObserver` as the process-wide broadcast hook and
367    /// then building an observer through `create_observer` (the path the
368    /// agent loop takes inside `process_message`) must surface events on the
369    /// SSE broadcast channel. Codifies the load-bearing ordering so that
370    /// reordering or dropping `set_scoped_broadcast_hook` in `run_gateway` is caught
371    /// by `cargo test`, not by a silent regression in production.
372    #[test]
373    fn factory_observer_events_reach_broadcast_hook() {
374        // The broadcast hook is process-wide; serialize hook-touching tests
375        // within this test binary so they don't observe each other's state.
376        static HOOK_TEST_LOCK: parking_lot::Mutex<()> = parking_lot::Mutex::new(());
377        let _guard = HOOK_TEST_LOCK.lock();
378
379        zeroclaw_runtime::observability::clear_broadcast_hook();
380
381        let (tx, mut rx) = tokio::sync::broadcast::channel(16);
382        let buffer = Arc::new(EventBuffer::new(16));
383        let bo: Arc<dyn Observer> = Arc::new(BroadcastObserver::new(tx, buffer.clone()));
384        zeroclaw_runtime::observability::set_broadcast_hook(bo);
385
386        // Same factory call site as `process_message` in the agent loop.
387        let cfg = zeroclaw_config::schema::ObservabilityConfig {
388            backend: "noop".into(),
389            ..Default::default()
390        };
391        let observer = zeroclaw_runtime::observability::create_observer(&cfg);
392
393        observer.record_event(&ObserverEvent::ToolCall {
394            tool: "shell".into(),
395            duration: std::time::Duration::from_millis(7),
396            success: true,
397        });
398
399        let value = rx
400            .try_recv()
401            .expect("factory-built observer event must reach the SSE broadcast channel");
402        assert_eq!(value["type"], "tool_call");
403        assert_eq!(value["tool"], "shell");
404        assert_eq!(value["success"], true);
405
406        let snap = buffer.snapshot();
407        assert_eq!(
408            snap.len(),
409            1,
410            "broadcast events must also land in the buffer"
411        );
412
413        zeroclaw_runtime::observability::clear_broadcast_hook();
414    }
