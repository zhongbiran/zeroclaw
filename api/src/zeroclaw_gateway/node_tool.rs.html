1//! Wraps a node capability as a zeroclaw [`Tool`] so it can be dispatched
2//! through the existing tool registry and agent loop.
3//!
4//! Tool names are prefixed with the node ID: `node:<node_id>:<capability_name>`.
5
6use std::sync::Arc;
7
8use async_trait::async_trait;
9use tokio::time::Duration;
10
11use crate::nodes::{NodeInvocation, NodeRegistry};
12use zeroclaw_api::tool::{Tool, ToolResult};
13use zeroclaw_tools::node_capabilities::requires_approval;
14
15/// Default timeout for node invocations (30 seconds).
16const NODE_INVOKE_TIMEOUT_SECS: u64 = 30;
17
18/// A zeroclaw [`Tool`] backed by a node capability.
19///
20/// The `prefixed_name` (e.g. `node:phone-1:camera.snap`) is what the agent
21/// loop sees. Invocations are routed to the connected node via WebSocket.
22pub struct NodeTool {
23    /// Prefixed name: `node:<node_id>:<capability_name>`.
24    prefixed_name: String,
25    /// The node ID this tool belongs to.
26    node_id: String,
27    /// The original capability name.
28    capability_name: String,
29    /// Human-readable description.
30    description: String,
31    /// JSON schema for parameters.
32    parameters: serde_json::Value,
33    /// Node registry for routing invocations.
34    registry: Arc<NodeRegistry>,
35}
36
37impl NodeTool {
38    /// Create a new node tool wrapper.
39    pub fn new(
40        node_id: String,
41        capability_name: String,
42        description: String,
43        parameters: serde_json::Value,
44        registry: Arc<NodeRegistry>,
45    ) -> Self {
46        let prefixed_name = format!("node:{node_id}:{capability_name}");
47        Self {
48            prefixed_name,
49            node_id,
50            capability_name,
51            description,
52            parameters,
53            registry,
54        }
55    }
56
57    /// Build the prefixed tool name for a node capability.
58    pub fn tool_name(node_id: &str, capability_name: &str) -> String {
59        format!("node:{node_id}:{capability_name}")
60    }
61}
62
63#[async_trait]
64impl Tool for NodeTool {
65    fn name(&self) -> &str {
66        &self.prefixed_name
67    }
68
69    fn description(&self) -> &str {
70        &self.description
71    }
72
73    fn parameters_schema(&self) -> serde_json::Value {
74        self.parameters.clone()
75    }
76
77    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
78        // Check if this capability requires approval
79        if requires_approval(&self.capability_name) {
80            let approved = args
81                .get("approved")
82                .and_then(|v| v.as_bool())
83                .unwrap_or(false);
84            if !approved {
85                return Ok(ToolResult {
86                    success: false,
87                    output: String::new(),
88                    error: Some(format!(
89                        "Capability '{}' requires approval. Set approved=true to proceed.",
90                        self.capability_name
91                    )),
92                });
93            }
94        }
95
96        // Strip the `approved` field (same as MCP tools)
97        let args = match args {
98            serde_json::Value::Object(mut map) => {
99                map.remove("approved");
100                serde_json::Value::Object(map)
101            }
102            other => other,
103        };
104
105        let invoke_tx: tokio::sync::mpsc::Sender<NodeInvocation> =
106            match self.registry.invoke_tx(&self.node_id) {
107                Some(tx) => tx,
108                None => {
109                    return Ok(ToolResult {
110                        success: false,
111                        output: String::new(),
112                        error: Some(format!("Node '{}' is not connected", self.node_id)),
113                    });
114                }
115            };
116
117        let call_id = uuid::Uuid::new_v4().to_string();
118        let (response_tx, response_rx) = tokio::sync::oneshot::channel();
119
120        let invocation = NodeInvocation {
121            call_id,
122            capability: self.capability_name.clone(),
123            args,
124            response_tx,
125        };
126
127        if invoke_tx.send(invocation).await.is_err() {
128            return Ok(ToolResult {
129                success: false,
130                output: String::new(),
131                error: Some(format!(
132                    "Failed to send invocation to node '{}'",
133                    self.node_id
134                )),
135            });
136        }
137
138        // Wait for response with timeout
139        match tokio::time::timeout(Duration::from_secs(NODE_INVOKE_TIMEOUT_SECS), response_rx).await
140        {
141            Ok(Ok(result)) => Ok(ToolResult {
142                success: result.success,
143                output: result.output,
144                error: result.error,
145            }),
146            Ok(Err(_)) => Ok(ToolResult {
147                success: false,
148                output: String::new(),
149                error: Some(format!(
150                    "Node '{}' dropped the invocation channel",
151                    self.node_id
152                )),
153            }),
154            Err(_) => Ok(ToolResult {
155                success: false,
156                output: String::new(),
157                error: Some(format!(
158                    "Node '{}' invocation timed out after {NODE_INVOKE_TIMEOUT_SECS}s",
159                    self.node_id
160                )),
161            }),
162        }
163    }
164}

171    #[test]
172    fn node_tool_name_format() {
173        assert_eq!(
174            NodeTool::tool_name("phone-1", "camera.snap"),
175            "node:phone-1:camera.snap"
176        );
177    }

179    #[test]
180    fn node_tool_metadata() {
181        let registry = Arc::new(NodeRegistry::new(10));
182        let tool = NodeTool::new(
183            "phone-1".to_string(),
184            "camera.snap".to_string(),
185            "Take a photo".to_string(),
186            serde_json::json!({"type": "object", "properties": {"resolution": {"type": "string"}}}),
187            registry,
188        );
189
190        assert_eq!(tool.name(), "node:phone-1:camera.snap");
191        assert_eq!(tool.description(), "Take a photo");
192        assert_eq!(tool.parameters_schema()["type"], "object");
193    }

195    #[tokio::test]
196    async fn node_tool_execute_node_not_connected() {
197        let registry = Arc::new(NodeRegistry::new(10));
198        let tool = NodeTool::new(

211    #[tokio::test]
212    async fn node_tool_execute_success() {
213        let registry = Arc::new(NodeRegistry::new(10));
214        let (invoke_tx, mut invoke_rx) = tokio::sync::mpsc::channel(32);
215
216        registry.register(NodeInfo {
217            node_id: "test-node".to_string(),
218            capabilities: vec![NodeCapability {
219                name: "echo".to_string(),
220                description: "Echo back".to_string(),
221                parameters: serde_json::json!({"type": "object", "properties": {}}),
222            }],
223            invoke_tx,
224        });
225
226        let tool = NodeTool::new(
227            "test-node".to_string(),
228            "echo".to_string(),
229            "Echo back".to_string(),
230            serde_json::json!({"type": "object", "properties": {}}),
231            Arc::clone(&registry),
232        );
233
234        // Spawn a task that simulates the node responding
235        tokio::spawn(async move {
236            if let Some(invocation) = invoke_rx.recv().await {
237                let _ = invocation
238                    .response_tx
239                    .send(crate::nodes::NodeInvocationResult {
240                        success: true,
241                        output: "echoed".to_string(),
242                        error: None,
243                    });
244            }
245        });
246
247        let result = tool
248            .execute(serde_json::json!({"msg": "hello"}))
249            .await
250            .unwrap();
251        assert!(result.success);
252        assert_eq!(result.output, "echoed");
253        assert!(result.error.is_none());
254    }

273    #[tokio::test]
274    async fn node_tool_rejects_unapproved_sensitive_operation() {
275        let registry = Arc::new(NodeRegistry::new(10));
276        let tool = NodeTool::new(
277            "phone-1".to_string(),
278            "camera.snap".to_string(),
279            "Take a photo".to_string(),
280            serde_json::json!({
281                "type": "object",
282                "properties": {
283                    "approved": { "type": "boolean" }
284                },
285                "required": ["approved"]
286            }),
287            registry,
288        );
289
290        // Without approved field
291        let result = tool.execute(serde_json::json!({})).await.unwrap();
292        assert!(!result.success);
293        assert!(result.error.as_ref().unwrap().contains("requires approval"));
294
295        // With approved=false
296        let result = tool
297            .execute(serde_json::json!({"approved": false}))
298            .await
299            .unwrap();
300        assert!(!result.success);
301        assert!(result.error.as_ref().unwrap().contains("requires approval"));
302    }
