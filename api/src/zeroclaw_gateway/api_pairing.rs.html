16/// Metadata about a paired device.
17#[derive(Debug, Clone, Serialize, Deserialize)]
18pub struct DeviceInfo {
19    pub id: String,
20    pub name: Option<String>,
21    pub device_type: Option<String>,
22    pub paired_at: DateTime<Utc>,
23    pub last_seen: DateTime<Utc>,
24    pub ip_address: Option<String>,
25    /// macOS TCC permissions (and equivalent on other OSes) the device reports as granted.
26    /// Pushed by the desktop app via POST /api/devices/me/capabilities.
27    #[serde(default, skip_serializing_if = "Option::is_none")]
28    pub capabilities: Option<Vec<String>>,
29}

31/// Registry of paired devices backed by SQLite.
32#[derive(Debug)]
33pub struct DeviceRegistry {
34    cache: Mutex<HashMap<String, DeviceInfo>>,
35    db_path: PathBuf,
36}
37
38impl DeviceRegistry {
39    pub fn new(workspace_dir: &Path) -> Self {
40        let db_path = workspace_dir.join("devices.db");
41        let conn = Connection::open(&db_path).expect("Failed to open device registry database");
42        conn.execute_batch(
43            "CREATE TABLE IF NOT EXISTS devices (
44                token_hash TEXT PRIMARY KEY,
45                id TEXT NOT NULL,
46                name TEXT,
47                device_type TEXT,
48                paired_at TEXT NOT NULL,
49                last_seen TEXT NOT NULL,
50                ip_address TEXT,
51                capabilities TEXT
52            )",
53        )
54        .expect("Failed to create devices table");
55
56        // Additive migration for DBs created before the capabilities column existed.
57        // SQLite has no IF NOT EXISTS for columns; the duplicate-column error here is benign.
58        let _ = conn.execute("ALTER TABLE devices ADD COLUMN capabilities TEXT", []);
59
60        // Warm the in-memory cache from DB
61        let mut cache = HashMap::new();
62        let mut stmt = conn
63            .prepare("SELECT token_hash, id, name, device_type, paired_at, last_seen, ip_address, capabilities FROM devices")
64            .expect("Failed to prepare device select");
65        let rows = stmt
66            .query_map([], |row| {
67                let token_hash: String = row.get(0)?;
68                let id: String = row.get(1)?;
69                let name: Option<String> = row.get(2)?;
70                let device_type: Option<String> = row.get(3)?;
71                let paired_at_str: String = row.get(4)?;
72                let last_seen_str: String = row.get(5)?;
73                let ip_address: Option<String> = row.get(6)?;
74                let capabilities_json: Option<String> = row.get(7)?;
75                let paired_at = DateTime::parse_from_rfc3339(&paired_at_str)
76                    .map(|dt| dt.with_timezone(&Utc))
77                    .unwrap_or_else(|_| Utc::now());
78                let last_seen = DateTime::parse_from_rfc3339(&last_seen_str)
79                    .map(|dt| dt.with_timezone(&Utc))
80                    .unwrap_or_else(|_| Utc::now());
81                let capabilities = capabilities_json
82                    .as_deref()
83                    .and_then(|s| serde_json::from_str::<Vec<String>>(s).ok());
84                Ok((
85                    token_hash,
86                    DeviceInfo {
87                        id,
88                        name,
89                        device_type,
90                        paired_at,
91                        last_seen,
92                        ip_address,
93                        capabilities,
94                    },
95                ))
96            })
97            .expect("Failed to query devices");
98        for (hash, info) in rows.flatten() {
99            cache.insert(hash, info);
100        }
101
102        Self {
103            cache: Mutex::new(cache),
104            db_path,
105        }
106    }
107
108    fn open_db(&self) -> Connection {
109        Connection::open(&self.db_path).expect("Failed to open device registry database")
110    }
111
112    pub fn register(&self, token_hash: String, info: DeviceInfo) {
113        let capabilities_json = info
114            .capabilities
115            .as_ref()
116            .and_then(|c| serde_json::to_string(c).ok());
117        let conn = self.open_db();
118        conn.execute(
119            "INSERT OR REPLACE INTO devices (token_hash, id, name, device_type, paired_at, last_seen, ip_address, capabilities) VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8)",
120            rusqlite::params![
121                token_hash,
122                info.id,
123                info.name,
124                info.device_type,
125                info.paired_at.to_rfc3339(),
126                info.last_seen.to_rfc3339(),
127                info.ip_address,
128                capabilities_json,
129            ],
130        )
131        .expect("Failed to insert device");
132        self.cache.lock().insert(token_hash, info);
133    }
134
135    pub fn list(&self) -> Vec<DeviceInfo> {
136        let conn = self.open_db();
137        let mut stmt = conn
138            .prepare("SELECT token_hash, id, name, device_type, paired_at, last_seen, ip_address, capabilities FROM devices")
139            .expect("Failed to prepare device select");
140        let rows = stmt
141            .query_map([], |row| {
142                let id: String = row.get(1)?;
143                let name: Option<String> = row.get(2)?;
144                let device_type: Option<String> = row.get(3)?;
145                let paired_at_str: String = row.get(4)?;
146                let last_seen_str: String = row.get(5)?;
147                let ip_address: Option<String> = row.get(6)?;
148                let capabilities_json: Option<String> = row.get(7)?;
149                let paired_at = DateTime::parse_from_rfc3339(&paired_at_str)
150                    .map(|dt| dt.with_timezone(&Utc))
151                    .unwrap_or_else(|_| Utc::now());
152                let last_seen = DateTime::parse_from_rfc3339(&last_seen_str)
153                    .map(|dt| dt.with_timezone(&Utc))
154                    .unwrap_or_else(|_| Utc::now());
155                let capabilities = capabilities_json
156                    .as_deref()
157                    .and_then(|s| serde_json::from_str::<Vec<String>>(s).ok());
158                Ok(DeviceInfo {
159                    id,
160                    name,
161                    device_type,
162                    paired_at,
163                    last_seen,
164                    ip_address,
165                    capabilities,
166                })
167            })
168            .expect("Failed to query devices");
169        rows.filter_map(|r| r.ok()).collect()
170    }

172    pub fn revoke(&self, device_id: &str) -> bool {
173        let conn = self.open_db();
174        let deleted = conn
175            .execute(
176                "DELETE FROM devices WHERE id = ?1",
177                rusqlite::params![device_id],
178            )
179            .unwrap_or(0);
180        if deleted > 0 {
181            let mut cache = self.cache.lock();
182            let key = cache
183                .iter()
184                .find(|(_, v)| v.id == device_id)
185                .map(|(k, _)| k.clone());
186            if let Some(key) = key {
187                cache.remove(&key);
188            }
189            true
190        } else {
191            false
192        }
193    }
194
195    pub fn update_last_seen(&self, token_hash: &str) {
196        let now = Utc::now();
197        let conn = self.open_db();
198        conn.execute(
199            "UPDATE devices SET last_seen = ?1 WHERE token_hash = ?2",
200            rusqlite::params![now.to_rfc3339(), token_hash],
201        )
202        .ok();
203        if let Some(device) = self.cache.lock().get_mut(token_hash) {
204            device.last_seen = now;
205        }
206    }
207
208    /// Replace the capability list for the device identified by `token_hash`.
209    /// Returns true if a row was updated.
210    pub fn update_capabilities(&self, token_hash: &str, capabilities: Vec<String>) -> bool {
211        let json = serde_json::to_string(&capabilities).unwrap_or_else(|_| "[]".into());
212        let conn = self.open_db();
213        let updated = conn
214            .execute(
215                "UPDATE devices SET capabilities = ?1, last_seen = ?2 WHERE token_hash = ?3",
216                rusqlite::params![json, Utc::now().to_rfc3339(), token_hash],
217            )
218            .unwrap_or(0);
219        if updated > 0
220            && let Some(device) = self.cache.lock().get_mut(token_hash)
221        {
222            device.capabilities = Some(capabilities);
223            device.last_seen = Utc::now();
224        }
225        updated > 0
226    }
227
228    pub fn device_count(&self) -> usize {
229        self.cache.lock().len()
230    }
231}

233/// Store for pending pairing requests.
234#[derive(Debug)]
235pub struct PairingStore {
236    pending: Mutex<Vec<PendingPairing>>,
237    #[allow(dead_code)] // WIP: will be used to cap pending pairing requests
238    max_pending: usize,
239}
240
241#[derive(Debug, Clone, Serialize)]
242struct PendingPairing {
243    code: String,
244    created_at: DateTime<Utc>,
245    expires_at: DateTime<Utc>,
246    client_ip: Option<String>,
247    attempts: u32,
248}
249
250impl PairingStore {
251    pub fn new(max_pending: usize) -> Self {
252        Self {
253            pending: Mutex::new(Vec::new()),
254            max_pending,
255        }
256    }
257
258    pub fn pending_count(&self) -> usize {
259        let mut pending = self.pending.lock();
260        pending.retain(|p| p.expires_at > Utc::now());
261        pending.len()
262    }
263}

265fn extract_bearer(headers: &HeaderMap) -> Option<&str> {
266    headers
267        .get(header::AUTHORIZATION)
268        .and_then(|v| v.to_str().ok())
269        .and_then(|auth| auth.strip_prefix("Bearer "))
270}
271
272fn require_auth(state: &AppState, headers: &HeaderMap) -> Result<(), (StatusCode, &'static str)> {
273    if state.pairing.require_pairing() {
274        let token = extract_bearer(headers).unwrap_or("");
275        if !state.pairing.is_authenticated(token) {
276            return Err((StatusCode::UNAUTHORIZED, "Unauthorized"));
277        }
278    }
279    Ok(())
280}

282/// POST /api/pairing/initiate — initiate a new pairing session
283pub async fn initiate_pairing(

302    }

305/// POST /api/pair — submit pairing code (for new device pairing)
306pub async fn submit_pairing_enhanced(
307    State(state): State<AppState>,
308    headers: HeaderMap,
309    Json(body): Json<serde_json::Value>,
310) -> impl IntoResponse {
311    let code = body["code"].as_str().unwrap_or("");
312    let device_name = body["device_name"].as_str().map(String::from);
313    let device_type = body["device_type"].as_str().map(String::from);
314
315    let client_id = headers
316        .get("X-Forwarded-For")
317        .and_then(|v| v.to_str().ok())
318        .unwrap_or("unknown")
319        .to_string();
320
321    match state.pairing.try_pair(code, &client_id).await {
322        Ok(Some(token)) => {
323            // Register the new device
324            let token_hash = {
325                use sha2::{Digest, Sha256};
326                let hash = Sha256::digest(token.as_bytes());
327                hex::encode(hash)
328            };
329            if let Some(ref registry) = state.device_registry {
330                registry.register(
331                    token_hash,
332                    DeviceInfo {
333                        id: uuid::Uuid::new_v4().to_string(),
334                        name: device_name,
335                        device_type,
336                        paired_at: Utc::now(),
337                        last_seen: Utc::now(),
338                        ip_address: Some(client_id),
339                        capabilities: None,
340                    },
341                );
342            }
343            Json(serde_json::json!({
344                "token": token,
345                "message": "Pairing successful"
346            }))
347            .into_response()
348        }
349        Ok(None) => (StatusCode::BAD_REQUEST, "Invalid or expired pairing code").into_response(),
350        Err(lockout_secs) => (
351            StatusCode::TOO_MANY_REQUESTS,
352            format!("Too many attempts. Locked out for {lockout_secs}s"),
353        )
354            .into_response(),
355    }
356}

358/// GET /api/devices — list paired devices
359pub async fn list_devices(State(state): State<AppState>, headers: HeaderMap) -> impl IntoResponse {
360    if let Err(e) = require_auth(&state, &headers) {
361        return e.into_response();
362    }
363
364    let devices = state
365        .device_registry
366        .as_ref()
367        .map(|r| r.list())
368        .unwrap_or_default();
369
370    let count = devices.len();
371    Json(serde_json::json!({
372        "devices": devices,
373        "count": count
374    }))
375    .into_response()
376}

405/// POST /api/devices/me/capabilities — the calling device replaces its capability list.
406///
407/// The "me" path means there's no separate device id in the URL — the bearer token in
408/// Authorization identifies which row gets updated. Body: `{ "capabilities": ["..."] }`.
409pub async fn update_my_capabilities(
410    State(state): State<AppState>,
411    headers: HeaderMap,
412    Json(body): Json<serde_json::Value>,
413) -> impl IntoResponse {
414    if let Err(e) = require_auth(&state, &headers) {
415        return e.into_response();
416    }
417
418    let token = match extract_bearer(&headers) {
419        Some(t) => t,
420        None => return (StatusCode::UNAUTHORIZED, "Missing bearer token").into_response(),
421    };
422    let token_hash = {
423        use sha2::{Digest, Sha256};
424        let hash = Sha256::digest(token.as_bytes());
425        hex::encode(hash)
426    };
427
428    let capabilities: Vec<String> = body
429        .get("capabilities")
430        .and_then(|v| v.as_array())
431        .map(|arr| {
432            arr.iter()
433                .filter_map(|v| v.as_str().map(String::from))
434                .collect()
435        })
436        .unwrap_or_default();
437
438    let registry = match state.device_registry.as_ref() {
439        Some(r) => r,
440        None => {
441            return (
442                StatusCode::SERVICE_UNAVAILABLE,
443                "Device registry is disabled",
444            )
445                .into_response();
446        }
447    };
448
449    if registry.update_capabilities(&token_hash, capabilities.clone()) {
450        Json(serde_json::json!({
451            "message": "Capabilities updated",
452            "capabilities": capabilities,
453        }))
454        .into_response()
455    } else {
456        (StatusCode::NOT_FOUND, "Device not found for this token").into_response()
457    }
458}
