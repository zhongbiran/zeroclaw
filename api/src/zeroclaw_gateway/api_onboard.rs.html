1//! Onboard catalog endpoint — exposes the provider + model catalog the CLI
2//! wizard already uses, so the dashboard's "+ Add provider" affordance and
3//! model-picker dropdown share the same source of truth as the CLI.
4//!
5//! No catalog data is hand-maintained at this layer. `list_providers()` lives
6//! in `zeroclaw-providers` and is the canonical list; `list_models()` per
7//! provider fetches from models.dev (cached) or the provider's own /models
8//! endpoint. Same code paths as the CLI wizard.
9//!
10//! Issue #6175.

42/// `GET /api/onboard/catalog` — list every provider the CLI wizard knows
43/// about. The dashboard shows these in the "+ Add provider" picker so
44/// CLI / web stay in sync.
45pub async fn handle_catalog(State(state): State<AppState>, headers: HeaderMap) -> Response {
46    if let Err(e) = require_auth(&state, &headers) {
47        return e.into_response();
48    }
49    let _ = state;
50
51    let providers: Vec<CatalogProvider> = zeroclaw_providers::list_providers()
52        .into_iter()
53        .map(|p| CatalogProvider {
54            name: p.name.to_string(),
55            display_name: p.display_name.to_string(),
56            local: p.local,
57            aliases: p.aliases.iter().map(|s| s.to_string()).collect(),
58        })
59        .collect();
60
61    axum::Json(CatalogResponse { providers }).into_response()
62}

82/// `GET /api/onboard/catalog/models?provider=<name>` — fetch the model list
83/// for one provider. Same code path the CLI wizard uses
84/// (`zeroclaw_providers::create_provider(...).list_models()`), which goes
85/// through the models.dev cached catalog for OpenAI / Anthropic / Gemini,
86/// the live `/v1/models` endpoint for OpenRouter, etc.
87///
88/// Lazy: the dashboard hits this only when the user picks a provider, so
89/// initial catalog load stays fast. Fetch failures return an empty list
90/// with `live: false` so the form falls back to a free-text input.
91pub async fn handle_catalog_models(
92    State(state): State<AppState>,
93    headers: HeaderMap,
94    Query(q): Query<ModelsQuery>,
95) -> Response {
96    if let Err(e) = require_auth(&state, &headers) {
97        return e.into_response();
98    }
99    let _ = state;
100
101    let handle = match zeroclaw_providers::create_provider(&q.provider, None) {
102        Ok(h) => h,
103        Err(e) => {
104            return error_response(
105                ConfigApiError::new(
106                    ConfigApiCode::PathNotFound,
107                    format!("unknown provider `{}`: {e}", q.provider),
108                )
109                .with_path(&q.provider),
110            );
111        }
112    };
113
114    let (models, live) = match handle.list_models().await {
115        Ok(m) => (m, true),
116        Err(e) => {
117            tracing::debug!(provider = %q.provider, error = ?e, "model catalog fetch failed");
118            (Vec::new(), false)
119        }
120    };
121
122    axum::Json(ModelsResponse {
123        provider: q.provider,
124        models,
125        live,
126    })
127    .into_response()
128}

130fn error_response(err: ConfigApiError) -> Response {
131    let status = axum::http::StatusCode::from_u16(err.code.http_status())
132        .unwrap_or(axum::http::StatusCode::INTERNAL_SERVER_ERROR);
133    (status, axum::Json(err)).into_response()
134}

170pub struct OnboardStatusResponse {
171    /// `true` when the user hasn't started onboarding yet — no completed
172    /// section markers AND no usable provider configured. The dashboard
173    /// uses this signal to redirect first-load visits from `/` to
174    /// `/onboard`.
175    pub needs_onboarding: bool,
176    /// Short machine-readable reason for the value of `needs_onboarding`,
177    /// for logs / debugging. Stable: `fresh_install` / `has_provider` /
178    /// `has_completed_sections`.
179    pub reason: &'static str,
180}

182/// `GET /api/onboard/status` — boolean signal for the dashboard's
183/// fresh-install redirect. The daemon writes a default `config.toml` on
184/// first init, so file existence isn't a useful "is the user new?" check.
185/// Instead we look at two explicit user-driven markers: any
186/// `onboard_state.completed_sections` entry (set when the wizard finishes
187/// a section) OR any usable provider (`providers.fallback` set, or any
188/// entry under `providers.models`). When neither is present, the user is
189/// fresh and should land at `/onboard` instead of the empty Dashboard.
190pub async fn handle_onboard_status(State(state): State<AppState>, headers: HeaderMap) -> Response {
191    if let Err(e) = require_auth(&state, &headers) {
192        return e.into_response();
193    }
194    let cfg = state.config.lock().clone();
195
196    let has_completed = !cfg.onboard_state.completed_sections.is_empty();
197    let has_provider = cfg.providers.fallback.is_some() || !cfg.providers.models.is_empty();
198
199    let (needs_onboarding, reason) = if has_completed {
200        (false, "has_completed_sections")
201    } else if has_provider {
202        (false, "has_provider")
203    } else {
204        (true, "fresh_install")
205    };
206
207    axum::Json(OnboardStatusResponse {
208        needs_onboarding,
209        reason,
210    })
211    .into_response()
212}

214/// `GET /api/onboard/sections` — list every top-level config section.
215///
216/// Schema-driven: walks `Config::prop_fields()` and collects unique first
217/// segments, then asks `Config::map_key_sections()` for which ones have
218/// pickers. The 4 onboarding sections (`providers`, `channels`, `memory`,
219/// `tunnel`) keep their existing per-section dispatch in
220/// `handle_section_picker`; everything else (`gateway`, `observability`,
221/// `scheduler`, ...) renders as a direct form. Adding a new top-level
222/// field to `Config` makes it appear here automatically.
223pub async fn handle_sections(State(state): State<AppState>, headers: HeaderMap) -> Response {
224    if let Err(e) = require_auth(&state, &headers) {
225        return e.into_response();
226    }
227    let cfg = state.config.lock().clone();
228    let completed: std::collections::HashSet<String> = cfg
229        .onboard_state
230        .completed_sections
231        .iter()
232        .cloned()
233        .collect();
234
235    // First segment of every reachable prop path. BTreeSet for stable
236    // alphabetical order and dedup.
237    let mut roots: std::collections::BTreeSet<String> = cfg
238        .prop_fields()
239        .iter()
240        .filter_map(|f| f.name.split('.').next().map(str::to_string))
241        .collect();
242
243    // System / housekeeping fields the user never edits via the dashboard.
244    for hidden in HIDDEN_TOP_LEVEL {
245        roots.remove(*hidden);
246    }
247
248    // Map-keyed roots get pickers automatically (the picker shows existing
249    // keys / catalog entries; selecting an item opens its form).
250    let map_keyed_roots: std::collections::HashSet<&'static str> =
251        zeroclaw_config::schema::Config::map_key_sections()
252            .iter()
253            .filter_map(|s| s.path.split('.').next())
254            .collect();
255
256    let sections: Vec<SectionInfo> = roots
257        .into_iter()
258        .map(|key| {
259            let has_picker = SECTIONS_WITH_PICKER.contains(&key.as_str())
260                || map_keyed_roots.contains(key.as_str());
261            SectionInfo {
262                completed: completed.contains(&key),
263                label: humanize_section(&key),
264                help: section_help(&key).to_string(),
265                has_picker,
266                group: section_group(&key).to_string(),
267                key,
268            }
269        })
270        .collect();
271
272    axum::Json(SectionsResponse { sections }).into_response()
273}

284/// Humanize a section key for display (`google_workspace` → `Google workspace`).
285/// Keeps things simple and predictable; specific wording overrides go in
286/// the section-help table or per-section labels if/when we add them.
287fn humanize_section(key: &str) -> String {
288    let mut s = key.replace(['_', '-'], " ");
289    if let Some(c) = s.get_mut(0..1) {
290        c.make_ascii_uppercase();
291    }
292    s
293}

406/// `GET /api/onboard/sections/<section>` — picker items for that section.
407///
408/// Per-section dispatch:
409/// * `providers` → `zeroclaw_providers::list_providers()` (CLI's catalog).
410/// * `memory` → `zeroclaw_memory::selectable_memory_backends()`.
411/// * `channels` / `tunnel` → schema-walk: clone config, `init_defaults` the
412///   section, then strip the section prefix from `prop_fields()` and dedupe
413///   by first segment. Same trick the TUI uses; new channels appear
414///   automatically when a `#[nested] Option<...>` field is added.
415/// * Anything else returns 404 (workspace/hardware have no picker).
416pub async fn handle_section_picker(
417    State(state): State<AppState>,
418    headers: HeaderMap,
419    axum::extract::Path(SectionPath { section }): axum::extract::Path<SectionPath>,
420) -> Response {
421    if let Err(e) = require_auth(&state, &headers) {
422        return e.into_response();
423    }
424    let cfg = state.config.lock().clone();
425
426    let (items, help) = match section.as_str() {
427        "providers" => (
428            providers_picker(&cfg),
429            section_help("providers").to_string(),
430        ),
431        "memory" => (memory_picker(&cfg), section_help("memory").to_string()),
432        "channels" => (
433            schema_walk_picker(&cfg, "channels"),
434            section_help("channels").to_string(),
435        ),
436        "tunnel" => (
437            schema_walk_picker_with_none(&cfg, "tunnel", "tunnel.provider"),
438            section_help("tunnel").to_string(),
439        ),
440        other => {
441            return error_response(
442                ConfigApiError::new(
443                    ConfigApiCode::PathNotFound,
444                    format!(
445                        "no picker for section `{other}`; \
446                         workspace and hardware are direct-form sections \
447                         (use GET /api/config/list?prefix=<section>)"
448                    ),
449                )
450                .with_path(other),
451            );
452        }
453    };
454
455    axum::Json(PickerResponse {
456        section,
457        items,
458        help,
459    })
460    .into_response()
461}

503/// Generic schema-walk picker for sections like `channels` whose subsections
504/// are `#[nested] Option<T>` fields. Discovery: clone the config,
505/// init_defaults the section to materialize every Option<T> as Some(default),
506/// then read prop_fields() on the probe — every reachable subsection's name
507/// surfaces as a path segment under `<section>.`.
508fn schema_walk_picker(cfg: &zeroclaw_config::schema::Config, section: &str) -> Vec<PickerItem> {
509    let mut probe = cfg.clone();
510    probe.init_defaults(Some(section));
511    let prefix_with_dot = format!("{section}.");
512
513    let configured: std::collections::BTreeSet<String> = cfg
514        .prop_fields()
515        .iter()
516        .filter_map(|f| f.name.strip_prefix(&prefix_with_dot))
517        .filter_map(|suffix| suffix.split_once('.').map(|(head, _)| head.to_string()))
518        .collect();
519
520    let all: std::collections::BTreeSet<String> = probe
521        .prop_fields()
522        .iter()
523        .filter_map(|f| f.name.strip_prefix(&prefix_with_dot))
524        .filter_map(|suffix| suffix.split_once('.').map(|(head, _)| head.to_string()))
525        .collect();
526
527    all.into_iter()
528        .map(|name| {
529            // Two-tier badge: `configured` = a block exists on disk for this
530            // item (auto-created on click + persisted). `active` = the block
531            // exists AND its `enabled` field is currently `true`. Items
532            // without an `enabled` field stay at `configured`. Surfaces the
533            // "is this actually doing anything?" distinction the contributor
534            // feedback on PR #6179 asked for, without changing init-on-click
535            // semantics (the macro-level fix lands in schema v3 / #5947).
536            let enabled_path = format!("{prefix_with_dot}{name}.enabled");
537            let is_active = cfg.get_prop(&enabled_path).ok().as_deref() == Some("true");
538            let badge = if is_active {
539                Some("active".to_string())
540            } else if configured.contains(&name) {
541                Some("configured".to_string())
542            } else {
543                None
544            };
545            PickerItem {
546                key: name.clone(),
547                label: name.clone(),
548                description: None,
549                badge,
550            }
551        })
552        .collect()
553}

555/// `tunnel`-flavored picker: same as `schema_walk_picker` plus a synthetic
556/// `none` entry at the top, marked active when the current `tunnel.provider`
557/// matches. Mirrors the TUI's tunnel section.
558fn schema_walk_picker_with_none(
559    cfg: &zeroclaw_config::schema::Config,
560    section: &str,
561    active_prop_path: &str,
562) -> Vec<PickerItem> {
563    let active = cfg.get_prop(active_prop_path).unwrap_or_default();
564    let mut items = vec![PickerItem {
565        key: "none".to_string(),
566        label: "none".to_string(),
567        description: Some("Localhost only — no public tunnel.".to_string()),
568        badge: if active == "none" || active.is_empty() {
569            Some("active".to_string())
570        } else {
571            None
572        },
573    }];
574    let mut rest = schema_walk_picker(cfg, section);
575    // Re-mark the active one in the schema-walk results.
576    for item in &mut rest {
577        if item.key == active {
578            item.badge = Some("active".to_string());
579        }
580    }
581    items.extend(rest);
582    items
583}

603/// `POST /api/onboard/sections/<section>/items/<key>` — instantiate the
604/// selected item in the live config (idempotent) and return the dotted
605/// prefix the frontend should fetch fields under.
606///
607/// Per-section dispatch:
608/// * `providers` → POST equivalent of `/api/config/map-key?path=providers.models&key=<key>`,
609///   then return `providers.models.<key>`.
610/// * `channels` → init_defaults under `channels.<key>`, return `channels.<key>`.
611/// * `memory` → set_prop `memory.backend = <key>`, return `memory`.
612/// * `tunnel` → set_prop `tunnel.provider = <key>` (and init_defaults the
613///   subsection if `<key>` is not "none"), return `tunnel.<key>` (or `tunnel`
614///   for the `none` case).
615pub async fn handle_section_select(
616    State(state): State<AppState>,
617    headers: HeaderMap,
618    axum::extract::Path(SectionItemPath { section, key }): axum::extract::Path<SectionItemPath>,
619) -> Response {
620    if let Err(e) = require_auth(&state, &headers) {
621        return e.into_response();
622    }
623
624    let mut working = state.config.lock().clone();
625
626    let (fields_prefix, created) = match section.as_str() {
627        "providers" => {
628            let created = working
629                .create_map_key("providers.models", &key)
630                .map_err(|msg| {
631                    error_response(
632                        ConfigApiError::new(
633                            ConfigApiCode::PathNotFound,
634                            format!("could not select provider `{key}`: {msg}"),
635                        )
636                        .with_path("providers.models"),
637                    )
638                });
639            let created = match created {
640                Ok(c) => c,
641                Err(resp) => return resp,
642            };
643            // Pre-populate the provider's trait-level defaults (base_url for
644            // Ollama, default temperature / max_tokens / etc.) so the form
645            // opens with sensible values instead of a sea of empty inputs.
646            // Idempotent — only fills paths that are still unset, so a user
647            // who's already overridden a field doesn't get clobbered on
648            // re-select.
649            let prefix = format!("providers.models.{key}");
650            if let Err(e) =
651                zeroclaw_runtime::onboard::field_visibility::apply_provider_trait_defaults(
652                    &mut working,
653                    &key,
654                    &prefix,
655                )
656            {
657                tracing::warn!(provider = %key, error = ?e, "failed to apply trait defaults; form will start blank");
658            }
659            // Make the picked provider the runtime fallback so chat actually
660            // routes to it. Without this, picking Ollama in onboarding would
661            // create the entry but the chat path keeps using the prior
662            // fallback (or fails because none is set), and the user lands
663            // on "OpenRouter API key not set" trying to chat with a model
664            // they thought they'd configured. Mirrors `zeroclaw onboard`'s
665            // post-pick semantics.
666            if let Err(e) = working.set_prop("providers.fallback", &key) {
667                tracing::warn!(provider = %key, error = %e, "failed to set providers.fallback after pick");
668            }
669            (prefix, created)
670        }
671        "channels" => {
672            let prefix = format!("channels.{key}");
673            // init_defaults instantiates the subsection if it doesn't exist.
674            // The set returned tells us whether something was newly created.
675            let initialized = working.init_defaults(Some(&prefix));
676            (prefix, !initialized.is_empty())
677        }
678        "memory" => {
679            // Set memory.backend to the picked key. Fields_prefix points at
680            // `memory` so the form renders the whole memory section
681            // (the active backend's specific fields show up there).
682            if let Err(e) = working.set_prop("memory.backend", &key) {
683                return error_response(
684                    ConfigApiError::new(
685                        ConfigApiCode::ValidationFailed,
686                        format!("could not set memory.backend = `{key}`: {e}"),
687                    )
688                    .with_path("memory.backend"),
689                );
690            }
691            ("memory".to_string(), true)
692        }
693        "tunnel" => {
694            if let Err(e) = working.set_prop("tunnel.provider", &key) {
695                return error_response(
696                    ConfigApiError::new(
697                        ConfigApiCode::ValidationFailed,
698                        format!("could not set tunnel.provider = `{key}`: {e}"),
699                    )
700                    .with_path("tunnel.provider"),
701                );
702            }
703            let prefix = if key == "none" {
704                "tunnel".to_string()
705            } else {
706                let p = format!("tunnel.{key}");
707                working.init_defaults(Some(&p));
708                p
709            };
710            (prefix, true)
711        }
712        other => {
713            return error_response(
714                ConfigApiError::new(
715                    ConfigApiCode::PathNotFound,
716                    format!("no picker semantics defined for section `{other}`"),
717                )
718                .with_path(other),
719            );
720        }
721    };
722
723    if let Err(e) = working.save().await {
724        return error_response(ConfigApiError::new(
725            ConfigApiCode::ReloadFailed,
726            format!("save after select failed: {e}"),
727        ));
728    }
729    *state.config.lock() = working;
730
731    axum::Json(SelectItemResponse {
732        fields_prefix,
733        created,
734    })
735    .into_response()
736}
