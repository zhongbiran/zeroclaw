1//! Read/write endpoints for the per-workspace personality markdown files
2//! (`SOUL.md`, `IDENTITY.md`, `USER.md`, `AGENTS.md`, `TOOLS.md`,
3//! `HEARTBEAT.md`, `BOOTSTRAP.md`, `MEMORY.md`).
4//!
5//! The runtime injects these into the system prompt at request time
6//! (see `zeroclaw_runtime::agent::personality::load_personality`). This
7//! module is the dashboard's authoring surface for them.
8//!
9//! Sandbox: filenames are matched against the static `EDITABLE_PERSONALITY_FILES`
10//! allowlist re-exported from the runtime crate. The on-disk path is
11//! built from a `&'static str` taken from that allowlist plus the
12//! current `workspace_dir`, so user-supplied path components cannot
13//! escape the workspace.
14//!
15//! The `agent` query parameter is reserved for #5890 (multi-agent
16//! workspaces). It is accepted on every endpoint and ignored today;
17//! when #5890 lands it will validate an alias and append a subdir.
18

127fn validate_filename(
128    filename: &str,
129) -> Result<&'static str, (StatusCode, Json<serde_json::Value>)> {
130    EDITABLE_PERSONALITY_FILES
131        .iter()
132        .copied()
133        .find(|allowed| *allowed == filename)
134        .ok_or_else(|| {
135            (
136                StatusCode::BAD_REQUEST,
137                Json(serde_json::json!({
138                    "error": "filename not in personality allowlist",
139                    "filename": filename,
140                    "allowed": EDITABLE_PERSONALITY_FILES,
141                })),
142            )
143        })
144}
145
146fn personality_path(workspace_dir: &Path, _agent: Option<&str>, filename: &'static str) -> PathBuf {
147    // `_agent` is reserved for #5890. Today every personality file
148    // lives at the workspace root.
149    workspace_dir.join(filename)
150}
151
152fn mtime_ms_of(meta: &std::fs::Metadata) -> Option<i64> {
153    meta.modified()
154        .ok()
155        .and_then(|t| t.duration_since(SystemTime::UNIX_EPOCH).ok())
156        .and_then(|d| i64::try_from(d.as_millis()).ok())
157}
158
159fn truncate_to_chars(content: &str, max: usize) -> (String, bool) {
160    if content.chars().count() <= max {
161        return (content.to_string(), false);
162    }
163    let cut = content
164        .char_indices()
165        .nth(max)
166        .map(|(idx, _)| &content[..idx])
167        .unwrap_or(content);
168    (cut.to_string(), true)
169}

173/// GET /api/personality — index of all allowlist files in the active workspace.
174pub async fn handle_index(
175    State(state): State<AppState>,
176    headers: HeaderMap,
177    Query(_q): Query<AgentQuery>,
178) -> impl IntoResponse {
179    if let Err(e) = require_auth(&state, &headers) {
180        return e.into_response();
181    }
182
183    let workspace_dir = {
184        let cfg = state.config.lock();
185        cfg.workspace_dir.clone()
186    };
187
188    let files: Vec<PersonalityIndexEntry> = EDITABLE_PERSONALITY_FILES
189        .iter()
190        .copied()
191        .map(|filename| {
192            let path = workspace_dir.join(filename);
193            match std::fs::metadata(&path) {
194                Ok(meta) => PersonalityIndexEntry {
195                    filename,
196                    exists: meta.is_file(),
197                    size: meta.len(),
198                    mtime_ms: mtime_ms_of(&meta),
199                },
200                Err(_) => PersonalityIndexEntry {
201                    filename,
202                    exists: false,
203                    size: 0,
204                    mtime_ms: None,
205                },
206            }
207        })
208        .collect();
209
210    Json(PersonalityIndex {
211        files,
212        max_chars: MAX_FILE_CHARS,
213    })
214    .into_response()
215}

217/// GET /api/personality/{filename} — read one file's full content.
218pub async fn handle_get(
219    State(state): State<AppState>,
220    headers: HeaderMap,
221    axum::extract::Path(filename): axum::extract::Path<String>,
222    Query(q): Query<AgentQuery>,
223) -> impl IntoResponse {
224    if let Err(e) = require_auth(&state, &headers) {
225        return e.into_response();
226    }
227
228    let allowed = match validate_filename(&filename) {
229        Ok(f) => f,
230        Err(e) => return e.into_response(),
231    };
232
233    let workspace_dir = {
234        let cfg = state.config.lock();
235        cfg.workspace_dir.clone()
236    };
237    let path = personality_path(&workspace_dir, q.agent.as_deref(), allowed);
238
239    match std::fs::read_to_string(&path) {
240        Ok(raw) => {
241            let (content, truncated) = truncate_to_chars(&raw, MAX_FILE_CHARS);
242            let mtime_ms = std::fs::metadata(&path).ok().and_then(|m| mtime_ms_of(&m));
243            Json(PersonalityFileResponse {
244                filename: allowed.to_string(),
245                content,
246                exists: true,
247                truncated,
248                mtime_ms,
249            })
250            .into_response()
251        }
252        Err(err) if err.kind() == std::io::ErrorKind::NotFound => Json(PersonalityFileResponse {
253            filename: allowed.to_string(),
254            content: String::new(),
255            exists: false,
256            truncated: false,
257            mtime_ms: None,
258        })
259        .into_response(),
260        Err(err) => (
261            StatusCode::INTERNAL_SERVER_ERROR,
262            Json(serde_json::json!({
263                "error": "failed to read personality file",
264                "filename": allowed,
265                "detail": err.to_string(),
266            })),
267        )
268            .into_response(),
269    }
270}

272/// PUT /api/personality/{filename} — overwrite the file.
273pub async fn handle_put(
274    State(state): State<AppState>,
275    headers: HeaderMap,
276    axum::extract::Path(filename): axum::extract::Path<String>,
277    Query(q): Query<AgentQuery>,
278    Json(body): Json<PersonalityPutBody>,
279) -> impl IntoResponse {
280    if let Err(e) = require_auth(&state, &headers) {
281        return e.into_response();
282    }
283
284    let allowed = match validate_filename(&filename) {
285        Ok(f) => f,
286        Err(e) => return e.into_response(),
287    };
288
289    if body.content.chars().count() > MAX_FILE_CHARS {
290        return (
291            StatusCode::PAYLOAD_TOO_LARGE,
292            Json(serde_json::json!({
293                "error": "content exceeds MAX_FILE_CHARS",
294                "max_chars": MAX_FILE_CHARS,
295            })),
296        )
297            .into_response();
298    }
299
300    let workspace_dir = {
301        let cfg = state.config.lock();
302        cfg.workspace_dir.clone()
303    };
304    let path = personality_path(&workspace_dir, q.agent.as_deref(), allowed);
305
306    // Disk-drift guard: if the editor told us what mtime it saw, reject
307    // the write when disk has moved since.
308    if let Some(expected) = body.expected_mtime_ms {
309        let current = std::fs::metadata(&path).ok().and_then(|m| mtime_ms_of(&m));
310        if current != Some(expected) {
311            let current_content = std::fs::read_to_string(&path).unwrap_or_default();
312            return (
313                StatusCode::CONFLICT,
314                Json(PersonalityConflict {
315                    error: "personality_disk_drift",
316                    filename: allowed.to_string(),
317                    current_content,
318                    current_mtime_ms: current,
319                }),
320            )
321                .into_response();
322        }
323    }
324
325    if let Some(parent) = path.parent()
326        && let Err(err) = std::fs::create_dir_all(parent)
327    {
328        return (
329            StatusCode::INTERNAL_SERVER_ERROR,
330            Json(serde_json::json!({
331                "error": "failed to create workspace dir",
332                "detail": err.to_string(),
333            })),
334        )
335            .into_response();
336    }
337
338    if let Err(err) = std::fs::write(&path, body.content.as_bytes()) {
339        return (
340            StatusCode::INTERNAL_SERVER_ERROR,
341            Json(serde_json::json!({
342                "error": "failed to write personality file",
343                "filename": allowed,
344                "detail": err.to_string(),
345            })),
346        )
347            .into_response();
348    }
349
350    let meta = std::fs::metadata(&path).ok();
351    let bytes_written = meta.as_ref().map(|m| m.len()).unwrap_or(0);
352    let mtime_ms = meta.as_ref().and_then(mtime_ms_of);
353
354    Json(PersonalityPutResponse {
355        bytes_written,
356        mtime_ms,
357    })
358    .into_response()
359}

361/// GET /api/personality/templates — render the default starter set.
362///
363/// Reuses `TemplateContext::default()` for any field the caller didn't
364/// override. The `memory.backend` config is consulted as a sensible
365/// default for `include_memory` when the query parameter is absent, so
366/// onboarding picks the right MEMORY.md behaviour without the user
367/// having to repeat themselves.
368pub async fn handle_templates(
369    State(state): State<AppState>,
370    headers: HeaderMap,
371    Query(q): Query<TemplateQuery>,
372) -> impl IntoResponse {
373    if let Err(e) = require_auth(&state, &headers) {
374        return e.into_response();
375    }
376
377    let memory_default_enabled = {
378        let cfg = state.config.lock();
379        cfg.memory.backend.as_str() != "none"
380    };
381
382    let defaults = TemplateContext::default();
383    let ctx = TemplateContext {
384        agent: q.agent_name.unwrap_or(defaults.agent),
385        user: q.user_name.unwrap_or(defaults.user),
386        timezone: q.timezone.unwrap_or(defaults.timezone),
387        communication_style: q
388            .communication_style
389            .unwrap_or(defaults.communication_style),
390        include_memory: q.include_memory.unwrap_or(memory_default_enabled),
391    };
392
393    let files = render_preset_default(&ctx)
394        .into_iter()
395        .map(|(filename, content)| TemplateFile { filename, content })
396        .collect();
397
398    Json(TemplateResponse {
399        preset: "default",
400        files,
401    })
402    .into_response()
403}

446        assert!(trunc);
