1//! Per-property CRUD endpoints for `/api/config/*`.
2//!
3//! These endpoints expose the same `Config::get_prop` / `set_prop` core that
4//! `zeroclaw config get/set/list/init/migrate` uses on the CLI. Both are thin
5//! frontends over the same mutation primitive.
6//!
7//! Returns structured `ConfigApiError` responses with stable codes the
8//! dashboard / scripts can match programmatically. Secret fields are
9//! write-only over HTTP per the secrets-handling boundary defined in
10//! the issue body.
11//!
12//! See #6175 for the full surface and acceptance checklist.

55/// One JSON Patch (RFC 6902) operation. We support a strict subset:

68    pub path: String,

110fn parse_patch_ops(value: serde_json::Value) -> Result<Vec<PatchOp>, ConfigApiError> {
111    let ops = value.as_array().ok_or_else(|| {
112        ConfigApiError::new(
113            ConfigApiCode::ValueTypeMismatch,
114            "JSON Patch body must be a JSON array of operations",
115        )
116    })?;
117
118    let mut parsed = Vec::with_capacity(ops.len());
119    for (idx, op) in ops.iter().enumerate() {
120        let object = op.as_object().ok_or_else(|| {
121            ConfigApiError::new(
122                ConfigApiCode::ValueTypeMismatch,
123                format!("JSON Patch op[{idx}] must be an object"),
124            )
125            .with_op_index(idx)
126        })?;
127        let op_name = object.get("op").and_then(|v| v.as_str()).ok_or_else(|| {
128            ConfigApiError::new(
129                ConfigApiCode::ValueTypeMismatch,
130                format!("JSON Patch op[{idx}] requires string `op` field"),
131            )
132            .with_op_index(idx)
133        })?;
134        let path = object.get("path").and_then(|v| v.as_str()).ok_or_else(|| {
135            ConfigApiError::new(
136                ConfigApiCode::ValueTypeMismatch,
137                format!("JSON Patch op[{idx}] requires string `path` field"),
138            )
139            .with_op_index(idx)
140        })?;
141        let comment = match object.get("comment") {
142            Some(value) => Some(
143                value
144                    .as_str()
145                    .ok_or_else(|| {
146                        ConfigApiError::new(
147                            ConfigApiCode::ValueTypeMismatch,
148                            format!("JSON Patch op[{idx}] `comment` field must be a string"),
149                        )
150                        .with_path(json_pointer_to_dotted(path))
151                        .with_op_index(idx)
152                    })?
153                    .to_string(),
154            ),
155            None => None,
156        };
157
158        parsed.push(PatchOp {
159            op: op_name.to_string(),
160            path: path.to_string(),
161            value: object.get("value").cloned(),
162            comment,
163        });
164    }
165
166    Ok(parsed)
167}

286/// Convert a `ConfigApiError` into an axum `Response` with the correct status.
287fn error_response(err: ConfigApiError) -> Response {
288    let status =
289        StatusCode::from_u16(err.code.http_status()).unwrap_or(StatusCode::INTERNAL_SERVER_ERROR);
290    (status, axum::Json(err)).into_response()
291}

293/// Wrap an `anyhow::Error` from `Config::set_prop` / `get_prop` into a
294/// `ConfigApiError`. Path-not-found errors get the specific code; everything
295/// else falls through to ValidationFailed.
296fn map_prop_error(err: anyhow::Error, path: &str) -> ConfigApiError {
297    let msg = err.to_string();
298    if msg.starts_with("Unknown property") {
299        ConfigApiError::path_not_found(path)
300    } else {
301        ConfigApiError::from_validation(err).with_path(path)
302    }
303}

325/// Save the config and refresh in-memory state. Captures a snapshot of the
326/// pre-write disk state and reverts to it if the save itself fails, so that
327/// on-disk and in-memory state stay consistent under any failure mode.
328///
329/// On the happy path: validate (caller's responsibility) → save to disk →
330/// swap in-memory → respond OK.
331///
332/// On save failure: best-effort restore the pre-write disk content (when
333/// readable), keep in-memory state untouched, return `reload_failed`.
334async fn persist_and_swap(
335    state: &AppState,
336    new_config: zeroclaw_config::schema::Config,
337) -> Result<(), ConfigApiError> {
338    let config_path = new_config.config_path.clone();
339
340    // Snapshot pre-write disk state (used for revert on save failure). When
341    // the file doesn't exist yet, snapshot is None — we'll remove the file
342    // again on rollback so a failed first-write doesn't leak partial state.
343    let snapshot = if config_path.exists() {
344        // best-effort; if we can't read, we can't revert
345        tokio::fs::read(&config_path).await.ok()
346    } else {
347        None
348    };
349
350    if let Err(e) = new_config.save().await {
351        // Save failed — try to restore the pre-write snapshot. This isn't
352        // strictly necessary (Config::save uses an atomic-replace via tmp
353        // file) but defends against the rare case where save partially
354        // wrote then errored (e.g. fsync mid-write).
355        if let Some(prev) = snapshot {
356            let _ = tokio::fs::write(&config_path, prev).await;
357        } else if config_path.exists() {
358            let _ = tokio::fs::remove_file(&config_path).await;
359        }
360        return Err(ConfigApiError::new(
361            ConfigApiCode::ReloadFailed,
362            format!("save failed: {e}"),
363        ));
364    }
365
366    *state.config.lock() = new_config;
367    Ok(())
368}

370/// Compute drift between the in-memory config and what's on disk right now.
371/// Returns one entry per drifted property; empty when in-memory and disk
372/// agree (or when the on-disk file can't be parsed).
373///
374/// **Secrets:** never surface values. We compare in-memory and on-disk
375/// representations server-side — for secret paths, the comparison happens
376/// over the raw display strings (which include the encrypted form on disk
377/// vs. the decrypted form in memory, so most secret drift is false-positive
378/// against `Configurable`'s display layer). To stay honest about that, the
379/// on-disk side is round-tripped through the full deserializer + decrypt
380/// pass before comparison, so we only surface drift the daemon would
381/// actually pick up on its next read of the file.
382pub async fn compute_drift(in_memory: &zeroclaw_config::schema::Config) -> Vec<DriftEntry> {
383    let path = &in_memory.config_path;
384    if !path.exists() {
385        return Vec::new();
386    }
387
388    let raw = match tokio::fs::read_to_string(path).await {
389        Ok(s) => s,
390        Err(_) => return Vec::new(),
391    };
392
393    // Re-parse the on-disk form into a fresh Config for value-by-value comparison.
394    let on_disk: zeroclaw_config::schema::Config =
395        match toml::from_str::<zeroclaw_config::schema::Config>(&raw) {
396            Ok(mut cfg) => {
397                cfg.config_path = path.clone();
398                cfg
399            }
400            Err(_) => return Vec::new(),
401        };
402
403    let in_memory_props: std::collections::HashMap<String, zeroclaw_config::traits::PropFieldInfo> =
404        in_memory
405            .prop_fields()
406            .into_iter()
407            .map(|p| (p.name.clone(), p))
408            .collect();
409    let on_disk_props: std::collections::HashMap<String, zeroclaw_config::traits::PropFieldInfo> =
410        on_disk
411            .prop_fields()
412            .into_iter()
413            .map(|p| (p.name.clone(), p))
414            .collect();
415
416    let mut drift: Vec<DriftEntry> = Vec::new();
417    for (name, mem) in &in_memory_props {
418        let disk = match on_disk_props.get(name) {
419            Some(d) => d,
420            None => continue,
421        };
422        if mem.display_value == disk.display_value {
423            continue;
424        }
425        let is_sensitive = mem.is_secret || mem.derived_from_secret;
426        if is_sensitive {
427            // Hash-compare server-side so we don't conflate ciphertext-vs-
428            // plaintext display drift with real value drift. If the SHA-256
429            // hashes match, the underlying secret is the same and we hide
430            // the entry.
431            use sha2::{Digest, Sha256};
432            let mem_hash = Sha256::digest(mem.display_value.as_bytes());
433            let disk_hash = Sha256::digest(disk.display_value.as_bytes());
434            if mem_hash == disk_hash {
435                continue;
436            }
437            drift.push(DriftEntry {
438                path: name.clone(),
439                secret: true,
440                drifted: true,
441                in_memory_value: None,
442                on_disk_value: None,
443            });
444        } else {
445            drift.push(DriftEntry {
446                path: name.clone(),
447                secret: false,
448                drifted: true,
449                in_memory_value: Some(serde_json::Value::String(mem.display_value.clone())),
450                on_disk_value: Some(serde_json::Value::String(disk.display_value.clone())),
451            });
452        }
453    }
454
455    // Stable order so callers can diff snapshots.
456    drift.sort_by(|a, b| a.path.cmp(&b.path));
457    drift
458}

462/// GET /api/config/prop?path=providers.fallback
463///
464/// Returns the user's current value for non-secret fields. For secret fields,
465/// returns `{path, populated}` only — the value, length, and any encoded form
466/// are deliberately withheld per the secrets-handling boundary.
467pub async fn handle_prop_get(
468    State(state): State<AppState>,
469    headers: HeaderMap,
470    Query(q): Query<PropQuery>,
471) -> Response {
472    if let Err(e) = require_auth(&state, &headers) {
473        return e.into_response();
474    }
475
476    let config = state.config.lock().clone();
477    let info = match lookup_prop_field(&config, &q.path) {
478        Some(info) => info,
479        None => return error_response(ConfigApiError::path_not_found(&q.path)),
480    };
481
482    if info.is_secret || info.derived_from_secret {
483        let populated = info.display_value != "<unset>";
484        return axum::Json(SecretResponse {
485            path: q.path,
486            populated,
487        })
488        .into_response();
489    }
490
491    match config.get_prop(&q.path) {
492        Ok(value_str) => {
493            // get_prop returns the display string; surface it as JSON.
494            // For typed-value fidelity, callers should hit OPTIONS to learn
495            // the type and parse client-side. Future iterations can route
496            // typed values through serde directly.
497            let warnings = config.collect_warnings();
498            axum::Json(PropResponse {
499                path: q.path,
500                value: serde_json::Value::String(value_str),
501                warnings,
502            })
503            .into_response()
504        }
505        Err(e) => error_response(map_prop_error(e, &q.path)),
506    }
507}

509/// PUT /api/config/prop with body `{path, value, comment?}`
510///
511/// Sets the value via `Config::set_prop`, validates the resulting whole-config
512/// state, persists, and swaps in-memory. For secret fields, response carries
513/// only `{path, populated: true}` — never echoes the value back.
514pub async fn handle_prop_put(
515    State(state): State<AppState>,
516    headers: HeaderMap,
517    axum::Json(body): axum::Json<PropPutBody>,
518) -> Response {
519    if let Err(e) = require_auth(&state, &headers) {
520        return e.into_response();
521    }
522
523    let mut new_config = state.config.lock().clone();
524    let info = match lookup_prop_field(&new_config, &body.path) {
525        Some(info) => info,
526        None => return error_response(ConfigApiError::path_not_found(&body.path)),
527    };
528
529    let value_str = match json_to_setprop_string(&body.value, Some(info.kind)) {
530        Ok(s) => s,
531        Err(e) => return error_response(e.with_path(&body.path)),
532    };
533
534    if let Err(e) = new_config.set_prop(&body.path, &value_str) {
535        return error_response(map_prop_error(e, &body.path));
536    }
537
538    if let Err(e) = new_config.validate() {
539        return error_response(ConfigApiError::from_validation(e).with_path(&body.path));
540    }
541
542    let config_path = new_config.config_path.clone();
543    // Collect non-fatal validation warnings against the post-save state
544    // before the config is moved into persist_and_swap. Warnings here are
545    // the same signal `validate()` emits via `tracing::warn!` — surfaced
546    // structured so dashboard callers see what CLI users see on stderr.
547    let warnings = new_config.collect_warnings();
548    if let Err(e) = persist_and_swap(&state, new_config).await {
549        return error_response(e);
550    }
551    if let Some(comment) = body.comment.as_ref() {
552        let annotations = [(body.path.clone(), comment.clone())];
553        if let Err(e) =
554            zeroclaw_config::comment_writer::apply_comments(&config_path, &annotations).await
555        {
556            tracing::warn!(error = %e, "failed to apply PUT comment to config.toml");
557        }
558    }
559
560    if info.is_secret || info.derived_from_secret {
561        axum::Json(SecretResponse {
562            path: body.path,
563            populated: !value_str.is_empty(),
564        })
565        .into_response()
566    } else {
567        axum::Json(PropResponse {
568            path: body.path,
569            value: serde_json::Value::String(value_str),
570            warnings,
571        })
572        .into_response()
573    }
574}

576/// DELETE /api/config/prop?path=channels.matrix.allowed-users
577///
578/// Resets the field to its declared default. For `Option<T>` fields, this
579/// sets to `None`. For secrets, response carries only `{path, populated: false}`.
580///
581/// The current implementation routes through `set_prop` with an empty string,
582/// which exercises the same validator path. A more semantically pure reset
583/// (re-deriving the field's literal default) is a refinement for a later step.
584pub async fn handle_prop_delete(
585    State(state): State<AppState>,
586    headers: HeaderMap,
587    Query(q): Query<PropQuery>,
588) -> Response {
589    if let Err(e) = require_auth(&state, &headers) {
590        return e.into_response();
591    }
592
593    let mut new_config = state.config.lock().clone();
594    let info = match lookup_prop_field(&new_config, &q.path) {
595        Some(info) => info,
596        None => return error_response(ConfigApiError::path_not_found(&q.path)),
597    };
598
599    if let Err(e) = new_config.set_prop(&q.path, "") {
600        return error_response(map_prop_error(e, &q.path));
601    }
602
603    if let Err(e) = new_config.validate() {
604        return error_response(ConfigApiError::from_validation(e).with_path(&q.path));
605    }
606
607    let warnings = new_config.collect_warnings();
608    if let Err(e) = persist_and_swap(&state, new_config).await {
609        return error_response(e);
610    }
611
612    if info.is_secret || info.derived_from_secret {
613        axum::Json(SecretResponse {
614            path: q.path,
615            populated: false,
616        })
617        .into_response()
618    } else {
619        axum::Json(PropResponse {
620            path: q.path,
621            value: serde_json::Value::Null,
622            warnings,
623        })
624        .into_response()
625    }
626}

805/// PATCH /api/config — apply a JSON Patch document atomically.
806///
807/// Body is an array of operations executed in order against an in-memory
808/// copy of the config. After all ops apply, `Config::validate()` runs once;
809/// if it passes the snapshot is persisted and swapped in. If any op fails or
810/// validation fails, on-disk + in-memory state are unchanged and the response
811/// carries the offending op's index.
812///
813/// Supported ops: `add`, `remove`, `replace`, `test`.
814/// `move` and `copy` return `op_not_supported` (no reference-graph in this PR).
815/// `test` against a `#[secret]` or `#[derived_from_secret]` path is rejected
816/// with `secret_test_forbidden` (would leak the value via differential outcome).
817pub async fn handle_patch(
818    State(state): State<AppState>,
819    headers: HeaderMap,
820    axum::Json(body): axum::Json<serde_json::Value>,
821) -> Response {
822    if let Err(e) = require_auth(&state, &headers) {
823        return e.into_response();
824    }
825
826    let ops = match parse_patch_ops(body) {
827        Ok(ops) => ops,
828        Err(e) => return error_response(e),
829    };
830
831    let working = state.config.lock().clone();
832
833    // Drift guard: if the on-disk file diverges from in-memory state on any
834    // path the PATCH would touch, refuse with 409 ConfigChangedExternally
835    // unless the client explicitly opts in to overwrite via the
836    // `X-ZeroClaw-Override-Drift: true` header. The opt-in surface keeps
837    // the contract loud: the only way to silently overwrite a hand-edit is
838    // a deliberate header, never an accident.
839    let override_drift = headers
840        .get("x-zeroclaw-override-drift")
841        .and_then(|v| v.to_str().ok())
842        .map(|v| v.eq_ignore_ascii_case("true"))
843        .unwrap_or(false);
844    if !override_drift {
845        let drifted = compute_drift(&working).await;
846        if !drifted.is_empty() {
847            let touched: std::collections::HashSet<String> = ops
848                .iter()
849                .map(|op| json_pointer_to_dotted(&op.path))
850                .collect();
851            let conflicts: Vec<&DriftEntry> = drifted
852                .iter()
853                .filter(|d| touched.contains(&d.path))
854                .collect();
855            if !conflicts.is_empty() {
856                let conflict_paths: Vec<String> =
857                    conflicts.iter().map(|d| d.path.clone()).collect();
858                return error_response(ConfigApiError::new(
859                    ConfigApiCode::ConfigChangedExternally,
860                    format!(
861                        "on-disk config has drifted from in-memory state on \
862                         {} path(s) being patched: {}. Send `X-ZeroClaw-Override-Drift: true` \
863                         to overwrite, or GET /api/config/drift to inspect first.",
864                        conflicts.len(),
865                        conflict_paths.join(", "),
866                    ),
867                ));
868            }
869        }
870    }
871
872    let mut working = working;
873    let mut results = Vec::with_capacity(ops.len());
874
875    for (idx, op) in ops.iter().enumerate() {
876        let path = json_pointer_to_dotted(&op.path);
877        let info = lookup_prop_field(&working, &path);
878        let is_sensitive = info
879            .as_ref()
880            .map(|i| i.is_secret || i.derived_from_secret)
881            .unwrap_or(false);
882
883        match op.op.as_str() {
884            "test" => {
885                // Secret values can't leave the server, so a differential
886                // test response would be the only signal — ban the op.
887                if is_sensitive {
888                    return error_response(
889                        ConfigApiError::secret_test_forbidden(&path).with_op_index(idx),
890                    );
891                }
892                let want = match op.value.as_ref() {
893                    Some(v) => v.clone(),
894                    None => {
895                        return error_response(
896                            ConfigApiError::new(
897                                ConfigApiCode::ValueTypeMismatch,
898                                "JSON Patch `test` op requires `value` field",
899                            )
900                            .with_path(&path)
901                            .with_op_index(idx),
902                        );
903                    }
904                };
905                let actual_str = match working.get_prop(&path) {
906                    Ok(v) => v,
907                    Err(e) => return error_response(map_prop_error(e, &path).with_op_index(idx)),
908                };
909                let want_str = match json_to_setprop_string(&want, info.as_ref().map(|i| i.kind)) {
910                    Ok(s) => s,
911                    Err(e) => return error_response(e.with_path(&path).with_op_index(idx)),
912                };
913                if actual_str != want_str {
914                    return error_response(
915                        ConfigApiError::new(
916                            ConfigApiCode::ValidationFailed,
917                            format!("`test` op failed: expected {want_str:?}, got {actual_str:?}"),
918                        )
919                        .with_path(&path)
920                        .with_op_index(idx),
921                    );
922                }
923                results.push(PatchOpResult {
924                    op: op.op.clone(),
925                    path,
926                    value: Some(serde_json::Value::String(actual_str)),
927                    populated: None,
928                    comment: None, // `test` ops don't write
929                });
930            }
931            "add" | "replace" => {
932                let value = match op.value.as_ref() {
933                    Some(v) => v.clone(),
934                    None => {
935                        return error_response(
936                            ConfigApiError::new(
937                                ConfigApiCode::ValueTypeMismatch,
938                                format!("JSON Patch `{}` op requires `value` field", op.op),
939                            )
940                            .with_path(&path)
941                            .with_op_index(idx),
942                        );
943                    }
944                };
945                let value_str = match json_to_setprop_string(&value, info.as_ref().map(|i| i.kind))
946                {
947                    Ok(s) => s,
948                    Err(e) => {
949                        return error_response(e.with_path(&path).with_op_index(idx));
950                    }
951                };
952                if let Err(e) = working.set_prop(&path, &value_str) {
953                    return error_response(map_prop_error(e, &path).with_op_index(idx));
954                }
955                if is_sensitive {
956                    results.push(PatchOpResult {
957                        op: op.op.clone(),
958                        path,
959                        value: None,
960                        populated: Some(!value_str.is_empty()),
961                        comment: op.comment.clone(),
962                    });
963                } else {
964                    results.push(PatchOpResult {
965                        op: op.op.clone(),
966                        path,
967                        value: Some(serde_json::Value::String(value_str)),
968                        populated: None,
969                        comment: op.comment.clone(),
970                    });
971                }
972            }
973            "remove" => {
974                if let Err(e) = working.set_prop(&path, "") {
975                    return error_response(map_prop_error(e, &path).with_op_index(idx));
976                }
977                if is_sensitive {
978                    results.push(PatchOpResult {
979                        op: op.op.clone(),
980                        path,
981                        value: None,
982                        populated: Some(false),
983                        comment: op.comment.clone(),
984                    });
985                } else {
986                    results.push(PatchOpResult {
987                        op: op.op.clone(),
988                        path,
989                        value: Some(serde_json::Value::Null),
990                        populated: None,
991                        comment: op.comment.clone(),
992                    });
993                }
994            }
995            "move" | "copy" => {
996                return error_response(
997                    ConfigApiError::op_not_supported(&op.op)
998                        .with_path(&path)
999                        .with_op_index(idx),
1000                );
1001            }
1002            other => {
1003                return error_response(
1004                    ConfigApiError::new(
1005                        ConfigApiCode::OpNotSupported,
1006                        format!("unknown JSON Patch operation `{other}`"),
1007                    )
1008                    .with_path(&path)
1009                    .with_op_index(idx),
1010                );
1011            }
1012        }
1013    }
1014
1015    if let Err(e) = working.validate() {
1016        return error_response(ConfigApiError::from_validation(e));
1017    }
1018
1019    // Collect (path, comment) pairs from any op that supplied a non-None
1020    // comment. Applied after save() so the comment-preserving sync_table
1021    // pass doesn't strip them.
1022    let annotations: Vec<(String, String)> = ops
1023        .iter()
1024        .zip(results.iter())
1025        .filter_map(|(op, res)| op.comment.as_ref().map(|c| (res.path.clone(), c.clone())))
1026        .collect();
1027
1028    let config_path = working.config_path.clone();
1029    // Collect non-fatal validation warnings against the post-save state
1030    // before working is moved into persist_and_swap. Same signal as
1031    // `tracing::warn!` from `validate()`, surfaced structured so dashboard
1032    // callers see it.
1033    let warnings = working.collect_warnings();
1034    if let Err(e) = persist_and_swap(&state, working).await {
1035        return error_response(e);
1036    }
1037    if !annotations.is_empty()
1038        && let Err(e) =
1039            zeroclaw_config::comment_writer::apply_comments(&config_path, &annotations).await
1040    {
1041        // Comments are best-effort decoration; surface as a non-fatal warn.
1042        // The patch itself succeeded — return success but log the failure.
1043        tracing::warn!(error = %e, "failed to apply PATCH op comments to config.toml");
1044    }
1045
1046    axum::Json(PatchResponse {
1047        saved: true,
1048        results,
1049        warnings,
1050    })
1051    .into_response()
1052}

1125/// POST /api/config/migrate — apply V1→V2 migration to the on-disk
1126/// config file in place. Mirrors `zeroclaw config migrate`. Backs up the
1127/// previous content alongside the original (`config.toml.bak`) before
1128/// writing the migrated form. Returns `{migrated: false}` when the config
1129/// is already at the current schema version.
1130pub async fn handle_migrate(State(state): State<AppState>, headers: HeaderMap) -> Response {
1131    if let Err(e) = require_auth(&state, &headers) {
1132        return e.into_response();
1133    }
1134
1135    let config_path = state.config.lock().config_path.clone();
1136
1137    let raw = match tokio::fs::read_to_string(&config_path).await {
1138        Ok(s) => s,
1139        Err(e) => {
1140            return error_response(ConfigApiError::new(
1141                ConfigApiCode::InternalError,
1142                format!("failed to read config file: {e}"),
1143            ));
1144        }
1145    };
1146
1147    let migrated = match zeroclaw_config::migration::migrate_file(&raw) {
1148        Ok(out) => out,
1149        Err(e) => {
1150            return error_response(ConfigApiError::new(
1151                ConfigApiCode::ValidationFailed,
1152                format!("migration failed: {e}"),
1153            ));
1154        }
1155    };
1156
1157    match migrated {
1158        Some(new_content) => {
1159            let backup_path = config_path.with_extension("toml.bak");
1160            if let Err(e) = tokio::fs::copy(&config_path, &backup_path).await {
1161                return error_response(ConfigApiError::new(
1162                    ConfigApiCode::InternalError,
1163                    format!("failed to write backup: {e}"),
1164                ));
1165            }
1166            if let Err(e) = tokio::fs::write(&config_path, &new_content).await {
1167                return error_response(ConfigApiError::new(
1168                    ConfigApiCode::InternalError,
1169                    format!("failed to write migrated config: {e}"),
1170                ));
1171            }
1172
1173            // Re-read into memory so subsequent requests see the migrated state.
1174            let new_cfg: zeroclaw_config::schema::Config = match toml::from_str(&new_content) {
1175                Ok(c) => c,
1176                Err(e) => {
1177                    return error_response(ConfigApiError::new(
1178                        ConfigApiCode::ReloadFailed,
1179                        format!("re-parse after migration failed: {e}"),
1180                    ));
1181                }
1182            };
1183            *state.config.lock() = new_cfg;
1184
1185            axum::Json(MigrateResponse {
1186                migrated: true,
1187                backup_path: Some(backup_path.display().to_string()),
1188                schema_version: zeroclaw_config::migration::CURRENT_SCHEMA_VERSION,
1189            })
1190            .into_response()
1191        }
1192        None => axum::Json(MigrateResponse {
1193            migrated: false,
1194            backup_path: None,
1195            schema_version: zeroclaw_config::migration::CURRENT_SCHEMA_VERSION,
1196        })
1197        .into_response(),

1230/// in-memory config — same contract as `GET /api/config/prop`. Previously
1231/// returned the whole-config schema regardless, which silently masked typos.
1232///
1233/// Per-path subtree extraction (walking the JSON Schema tree by JSON Pointer
1234/// to return just the relevant subtree) is a follow-up; today we still return
1235/// the full schema with a `x-zeroclaw-requested-path` + per-field metadata
1236/// (kind, type_hint, is_secret) so the frontend has everything it needs to
1237/// render the input without a separate round-trip.
1238pub async fn handle_options_prop(
1239    State(state): State<AppState>,
1240    headers: HeaderMap,
1241    Query(q): Query<PropQuery>,
1242) -> Response {
1243    if headers.contains_key("access-control-request-method") {
1244        let mut response = StatusCode::NO_CONTENT.into_response();
1245        let h = response.headers_mut();
1246        h.insert(
1247            "Access-Control-Allow-Methods",
1248            HeaderValue::from_static("GET, PUT, DELETE, OPTIONS"),
1249        );
1250        h.insert(
1251            "Access-Control-Allow-Headers",
1252            HeaderValue::from_static("Authorization, Content-Type, If-None-Match"),
1253        );
1254        return response;
1255    }
1256
1257    // Resolve the path against the in-memory config; 404 if it doesn't
1258    // exist. (No auth required for shape discovery — same as OPTIONS /api/config.)
1259    let config = state.config.lock().clone();
1260    let info = match lookup_prop_field(&config, &q.path) {
1261        Some(info) => info,
1262        None => return error_response(ConfigApiError::path_not_found(&q.path)),
1263    };
1264
1265    let (whole_body, etag) = cached_schema();
1266    let mut body = whole_body.clone();
1267    if let serde_json::Value::Object(ref mut map) = body {
1268        map.insert(
1269            "x-zeroclaw-requested-path".into(),
1270            serde_json::Value::String(q.path.clone()),
1271        );
1272        map.insert(
1273            "x-zeroclaw-prop".into(),
1274            serde_json::json!({
1275                "path": q.path,
1276                "kind": prop_kind_wire(info.kind),
1277                "type_hint": info.type_hint,
1278                "is_secret": info.is_secret || info.derived_from_secret,
1279                "enum_variants": info.enum_variants.map(|f| f()).unwrap_or_default(),
1280                "category": info.category,
1281            }),
1282        );
1283    }
1284    let mut response = (StatusCode::OK, axum::Json(body)).into_response();
1285    response.headers_mut().insert(
1286        header::ALLOW,
1287        HeaderValue::from_static("GET, PUT, DELETE, OPTIONS"),
1288    );
1289    response
1290        .headers_mut()
1291        .insert(header::ETAG, HeaderValue::from_str(etag).unwrap());
1292    response
1293}

1295fn schema_response(_label: &'static str) -> Response {
1296    let (body, etag) = cached_schema();
1297    let mut response = (StatusCode::OK, axum::Json(body.clone())).into_response();
1298    response.headers_mut().insert(
1299        header::ALLOW,
1300        HeaderValue::from_static("GET, PUT, PATCH, OPTIONS"),
1301    );

1303        .headers_mut()
1304        .insert(header::ETAG, HeaderValue::from_str(etag).unwrap());

1308/// Compute the OPTIONS schema body + ETag once and cache them. The schema is
1309/// static per build (schemars output is deterministic for a given Config
1310/// type), so re-rendering on every request is pure waste — we'd send the
1311/// same bytes back every time and re-hash them too. The previous
1312/// implementation re-rendered + re-hashed on every OPTIONS hit; this caches
1313/// both behind a `OnceLock`.
1314fn cached_schema() -> (&'static serde_json::Value, &'static str) {
1315    use std::sync::OnceLock;
1316    static CACHE: OnceLock<(serde_json::Value, String)> = OnceLock::new();
1317    let entry = CACHE.get_or_init(|| {
1318        let body = schema_body_value();
1319        let etag = build_etag_for(&body);
1320        (body, etag)
1321    });
1322    (&entry.0, entry.1.as_str())
1323}

1327    let schema = schemars::schema_for!(zeroclaw_config::schema::Config);

1338/// Stable ETag derived from the rendered schema bytes. Computed once via
1339/// `cached_schema()`; this helper is kept separate so tests can verify
1340/// determinism.
1341fn build_etag_for(body: &serde_json::Value) -> String {
1342    use std::hash::{Hash, Hasher};
1343    let bytes = body.to_string();
1344    let mut hasher = std::collections::hash_map::DefaultHasher::new();
1345    bytes.hash(&mut hasher);
1346    format!("\"{:016x}\"", hasher.finish())
1347}
