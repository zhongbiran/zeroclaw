25/// HTTP request sent from plugin to host via `zc_http_request`.
26#[derive(Debug, Serialize, Deserialize)]
27struct HttpRequest {
28    method: String,
29    url: String,
30    #[serde(default)]
31    headers: std::collections::HashMap<String, String>,
32    #[serde(default)]
33    body: Option<String>,
34}
35
36/// HTTP response returned from host to plugin.
37#[derive(Debug, Serialize, Deserialize)]
38struct HttpResponse {
39    status: u16,
40    body: String,
41    #[serde(default)]
42    headers: std::collections::HashMap<String, String>,
43}

62// ── Host function implementations ─────────────────────────────────
63
64fn handle_http_request(
65    plugin: &mut CurrentPlugin,
66    inputs: &[Val],
67    outputs: &mut [Val],
68    user_data: UserData<HostContext>,
69) -> Result<(), Error> {
70    let ctx = user_data.get()?;
71    let ctx = ctx.lock().unwrap();
72
73    if !ctx.permissions.contains(&PluginPermission::HttpClient) {
74        return Err(Error::msg(
75            "permission denied: plugin does not have 'http_client' permission",
76        ));
77    }
78
79    // Read input string from WASM memory
80    let request_json: String = plugin.memory_get_val(&inputs[0])?;
81
82    let req: HttpRequest = serde_json::from_str(&request_json)
83        .map_err(|e| Error::msg(format!("invalid HTTP request JSON: {e}")))?;
84
85    // 120s ceiling covers legitimate slow cases: large file downloads and slow
86    // model-inference endpoints (fal.ai image generation routinely takes 20-60s
87    // on cold models). A per-plugin override or tighter default is a candidate
88    // follow-up — see ADR-003 §"Known gaps". Note: this runs inside
89    // spawn_blocking, so a stalled request holds a blocking-pool thread for
90    // the full duration.
91    let client = reqwest::blocking::Client::builder()
92        .timeout(std::time::Duration::from_secs(120))
93        .build()
94        .map_err(|e| Error::msg(format!("failed to create HTTP client: {e}")))?;
95
96    let mut builder = match req.method.to_uppercase().as_str() {
97        "GET" => client.get(&req.url),
98        "POST" => client.post(&req.url),
99        "PUT" => client.put(&req.url),
100        "DELETE" => client.delete(&req.url),
101        "PATCH" => client.patch(&req.url),
102        "HEAD" => client.head(&req.url),
103        other => {
104            return Err(Error::msg(format!("unsupported HTTP method: {other}")));
105        }
106    };
107
108    for (k, v) in &req.headers {
109        builder = builder.header(k.as_str(), v.as_str());
110    }
111
112    if let Some(body) = req.body {
113        builder = builder.body(body);
114    }
115
116    let resp = builder
117        .send()
118        .map_err(|e| Error::msg(format!("HTTP request failed: {e}")))?;
119
120    let status = resp.status().as_u16();
121    let headers: std::collections::HashMap<String, String> = resp
122        .headers()
123        .iter()
124        .map(|(k, v)| (k.to_string(), v.to_str().unwrap_or("").to_string()))
125        .collect();
126    let body = resp
127        .text()
128        .map_err(|e| Error::msg(format!("failed to read response body: {e}")))?;
129
130    let response = HttpResponse {
131        status,
132        body,
133        headers,
134    };
135
136    let response_json = serde_json::to_string(&response)
137        .map_err(|e| Error::msg(format!("failed to serialize response: {e}")))?;
138
139    plugin.memory_set_val(&mut outputs[0], response_json)?;
140
141    Ok(())
142}

144fn handle_env_read(
145    plugin: &mut CurrentPlugin,
146    inputs: &[Val],
147    outputs: &mut [Val],
148    user_data: UserData<HostContext>,
149) -> Result<(), Error> {
150    let ctx = user_data.get()?;
151    let ctx = ctx.lock().unwrap();
152
153    if !ctx.permissions.contains(&PluginPermission::EnvRead) {
154        return Err(Error::msg(
155            "permission denied: plugin does not have 'env_read' permission",
156        ));
157    }
158
159    let var_name: String = plugin.memory_get_val(&inputs[0])?;
160
161    let value = std::env::var(&var_name)
162        .map_err(|_| Error::msg(format!("environment variable '{var_name}' not set")))?;
163
164    plugin.memory_set_val(&mut outputs[0], value)?;
165
166    Ok(())
167}

171/// Create an Extism plugin from a WASM file with the given permissions.
172pub fn create_plugin(wasm_path: &Path, permissions: &[PluginPermission]) -> Result<extism::Plugin> {
173    let perm_set: HashSet<PluginPermission> = permissions.iter().cloned().collect();
174    let ctx = UserData::new(HostContext {
175        permissions: perm_set,
176    });
177
178    let http_fn = Function::new(
179        "zc_http_request",
180        [PTR],
181        [PTR],
182        ctx.clone(),
183        handle_http_request,
184    );
185
186    let env_fn = Function::new("zc_env_read", [PTR], [PTR], ctx, handle_env_read);
187
188    let manifest = Manifest::new([Wasm::file(wasm_path)]);
189
190    Plugin::new(manifest, [http_fn, env_fn], true)
191        .with_context(|| format!("failed to load WASM plugin from {}", wasm_path.display()))
192}

194/// Call the `tool_metadata` export and parse the result.
195pub fn call_tool_metadata(plugin: &mut extism::Plugin) -> Result<ToolMetadata> {
196    let output = plugin
197        .call::<&str, String>("tool_metadata", "")
198        .context("failed to call tool_metadata export")?;
199
200    serde_json::from_str(&output).context("failed to parse tool_metadata JSON")
201}
202
203/// Call the `execute` export with the given args JSON and return a `ToolResult`.
204pub fn call_execute(plugin: &mut extism::Plugin, args_json: &[u8]) -> Result<ToolResult> {
205    let input = std::str::from_utf8(args_json).context("plugin args are not valid UTF-8")?;
206
207    let output = plugin
208        .call::<&str, String>("execute", input)
209        .context("failed to call plugin execute export")?;
210
211    let result: PluginToolResult =
212        serde_json::from_str(&output).context("failed to parse plugin execute result")?;
213
214    Ok(ToolResult {
215        success: result.success,
216        output: result.output,
217        error: result.error,
218    })
219}
