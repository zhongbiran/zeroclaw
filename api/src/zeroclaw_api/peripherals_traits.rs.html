1//! Peripheral trait — hardware boards (STM32, RPi GPIO) that expose tools.
2//!
3//! Peripherals are the agent's "arms and legs": remote devices that run minimal
4//! firmware and expose capabilities (GPIO, sensors, actuators) as tools.
5//! See `docs/hardware-peripherals-design.md` for the communication protocol
6//! and firmware integration guide.
7
8use async_trait::async_trait;
9
10use crate::tool::Tool;
11
12/// A hardware peripheral that exposes capabilities as agent tools.
13///
14/// Implement this trait for each supported board type (e.g., Nucleo-F401RE
15/// over serial, Raspberry Pi GPIO via sysfs/gpiod). When the agent connects
16/// to a peripheral, the tools returned by [`tools`](Peripheral::tools) are
17/// merged into the agent's tool registry, making hardware capabilities
18/// available to the LLM as callable functions.
19///
20/// The lifecycle follows a connect → use → disconnect pattern. Implementations
21/// must be `Send + Sync` because the peripheral may be accessed from multiple
22/// async tasks after connection.
23#[async_trait]
24pub trait Peripheral: Send + Sync {
25    /// Return the human-readable instance name of this peripheral.
26    ///
27    /// Should uniquely identify a specific device instance, including an index
28    /// or serial number when multiple boards of the same type are connected
29    /// (e.g., `"nucleo-f401re-0"`, `"rpi-gpio-hat-1"`).
30    fn name(&self) -> &str;
31
32    /// Return the board type identifier for this peripheral.
33    ///
34    /// A stable, lowercase string used in configuration and factory registration
35    /// (e.g., `"nucleo-f401re"`, `"rpi-gpio"`). Must match the key used in
36    /// the config schema's peripheral section.
37    fn board_type(&self) -> &str;
38
39    /// Establish a connection to the peripheral hardware.
40    ///
41    /// Opens the underlying transport (serial port, GPIO bus, I²C, etc.) and
42    /// performs any initialization handshake required by the firmware.
43    ///
44    /// # Errors
45    ///
46    /// Returns an error if the device is unreachable, the transport cannot be
47    /// opened, or the firmware handshake fails.
48    async fn connect(&mut self) -> anyhow::Result<()>;
49
50    /// Disconnect from the peripheral and release all held resources.
51    ///
52    /// Closes serial ports, unexports GPIO pins, and performs any cleanup
53    /// required for a safe shutdown. After this call, [`health_check`](Peripheral::health_check)
54    /// should return `false` until [`connect`](Peripheral::connect) is called again.
55    ///
56    /// # Errors
57    ///
58    /// Returns an error if resource cleanup fails (e.g., serial port busy).
59    async fn disconnect(&mut self) -> anyhow::Result<()>;
60
61    /// Check whether the peripheral is reachable and responsive.
62    ///
63    /// Performs a lightweight probe (e.g., a ping command over serial) without
64    /// altering device state. Returns `true` if the device responds within an
65    /// implementation-defined timeout.
66    async fn health_check(&self) -> bool;
67
68    /// Return the tools this peripheral exposes to the agent.
69    ///
70    /// Each returned [`Tool`] delegates execution to the underlying hardware
71    /// (e.g., `gpio_read`, `gpio_write`, `sensor_read`). The agent merges
72    /// these into its tool registry after a successful
73    /// [`connect`](Peripheral::connect).
74    fn tools(&self) -> Vec<Box<dyn Tool>>;
75}
