16/// The operator's response to a channel-presented approval prompt.
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
18#[serde(rename_all = "lowercase")]
19pub enum ChannelApprovalResponse {
20    /// Execute this one call.
21    Approve,
22    /// Deny this call.
23    Deny,
24    /// Execute and add tool to session-scoped allowlist.

26    AlwaysApprove,

29/// A message received from or sent to a channel
30#[derive(Debug, Clone)]
31pub struct ChannelMessage {
32    pub id: String,
33    pub sender: String,
34    pub reply_target: String,
35    pub content: String,
36    pub channel: String,
37    pub timestamp: u64,
38    /// Platform thread identifier (e.g. Slack `ts`, Discord thread ID).
39    /// When set, replies should be posted as threaded responses.
40    pub thread_ts: Option<String>,
41    /// Thread scope identifier for interruption/cancellation grouping.
42    /// Distinct from `thread_ts` (reply anchor): this is `Some` only when the message
43    /// is genuinely inside a reply thread and should be isolated from other threads.
44    /// `None` means top-level — scope is sender+channel only.
45    pub interruption_scope_id: Option<String>,
46    /// Media attachments (audio, images, video) for the media pipeline.
47    /// Channels populate this when they receive media alongside a text message.
48    /// Defaults to empty — existing channels are unaffected.
49    pub attachments: Vec<MediaAttachment>,
50}

52/// Message to send through a channel
53#[derive(Debug, Clone)]
54pub struct SendMessage {
55    pub content: String,
56    pub recipient: String,
57    pub subject: Option<String>,
58    /// Platform thread identifier for threaded replies (e.g. Slack `thread_ts`).
59    pub thread_ts: Option<String>,
60    /// Optional cancellation token for interruptible delivery (e.g. multi-message mode).
61    pub cancellation_token: Option<CancellationToken>,
62    /// File attachments to send with the message.
63    /// Channels that don't support attachments ignore this field.
64    pub attachments: Vec<MediaAttachment>,
65}
66
67impl SendMessage {
68    /// Create a new message with content and recipient
69    pub fn new(content: impl Into<String>, recipient: impl Into<String>) -> Self {
70        Self {
71            content: content.into(),
72            recipient: recipient.into(),
73            subject: None,
74            thread_ts: None,
75            cancellation_token: None,
76            attachments: vec![],
77        }
78    }
79
80    /// Create a new message with content, recipient, and subject
81    pub fn with_subject(
82        content: impl Into<String>,
83        recipient: impl Into<String>,
84        subject: impl Into<String>,
85    ) -> Self {
86        Self {
87            content: content.into(),
88            recipient: recipient.into(),
89            subject: Some(subject.into()),
90            thread_ts: None,
91            cancellation_token: None,
92            attachments: vec![],
93        }
94    }
95
96    /// Set the thread identifier for threaded replies.
97    pub fn in_thread(mut self, thread_ts: Option<String>) -> Self {
98        self.thread_ts = thread_ts;
99        self
100    }
101
102    /// Attach a cancellation token for interruptible delivery.
103    pub fn with_cancellation(mut self, token: CancellationToken) -> Self {
104        self.cancellation_token = Some(token);
105        self
106    }
107
108    /// Attach files to this message.
109    pub fn with_attachments(mut self, attachments: Vec<MediaAttachment>) -> Self {
110        self.attachments = attachments;
111        self
112    }
113}

237    /// Request interactive tool-call approval from the channel operator.
238    ///
239    /// Returns `Ok(Some(response))` when the operator answers within the
240    /// channel's configured `approval_timeout_secs`; timeouts are surfaced
241    /// as `Deny`. Returns `Ok(None)` only for channels that do not implement
242    /// the prompt at all — the caller should fall back to its default policy
243    /// (typically auto-deny).
244    ///
245    /// Surface varies by channel:
246    /// - **Telegram** uses inline keyboard buttons.
247    /// - **Slack** Socket Mode uses Block Kit buttons; webhook fallback and
248    ///   non–Socket Mode deployments use a token text reply.
249    /// - **Discord, Signal, Matrix, WhatsApp** embed a 6-character
250    ///   alphanumeric token in the prompt and wait for a
251    ///   `<token> approve|deny|always` reply on the same conversation.
252    async fn request_approval(
253        &self,
254        _recipient: &str,
255        _request: &ChannelApprovalRequest,
256    ) -> anyhow::Result<Option<ChannelApprovalResponse>> {
257        Ok(None)
258    }

260    /// Ask the user a multiple-choice question and return the chosen option's text.
261    ///
262    /// Returns `Ok(Some(answer))` if the channel handled the question natively
263    /// (e.g. ACP `session/request_permission`, Telegram inline keyboard).
264    /// Returns `Ok(None)` to signal the caller should fall back to the

270        &self,
271        _question: &str,
272        _choices: &[String],
273        _timeout: std::time::Duration,
274    ) -> anyhow::Result<Option<String>> {
275        Ok(None)
276    }

278    /// Whether this channel can answer free-form (no-choices) `ask_user`
279    /// questions via the standard `send` + `listen` flow.
280    ///
281    /// Channels that can only handle structured choices (e.g. ACP today, until
282    /// the elicitation RFD lands) should return `false` so callers can fail
283    /// fast with a useful error instead of timing out on `listen`.
284    fn supports_free_form_ask(&self) -> bool {
285        true
