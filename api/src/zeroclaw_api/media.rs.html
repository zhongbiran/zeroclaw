1/// Classifies an attachment by MIME type or file extension.
2#[derive(Debug, Clone, Copy, PartialEq, Eq)]
3pub enum MediaKind {
4    Audio,
5    Image,
6    Video,
7    Unknown,
8}
9
10/// A single media attachment on an inbound message.
11#[derive(Debug, Clone)]
12pub struct MediaAttachment {
13    /// Original file name (e.g. `voice.ogg`, `photo.jpg`).
14    pub file_name: String,
15    /// Raw bytes of the attachment.
16    pub data: Vec<u8>,
17    /// MIME type if known (e.g. `audio/ogg`, `image/jpeg`).
18    pub mime_type: Option<String>,
19}
20
21impl MediaAttachment {
22    /// Classify this attachment into a [`MediaKind`].
23    pub fn kind(&self) -> MediaKind {
24        // Try MIME type first.
25        if let Some(ref mime) = self.mime_type {
26            let lower = mime.to_ascii_lowercase();
27            if lower.starts_with("audio/") {
28                return MediaKind::Audio;
29            }
30            if lower.starts_with("image/") {
31                return MediaKind::Image;
32            }
33            if lower.starts_with("video/") {
34                return MediaKind::Video;
35            }
36        }
37
38        // Fall back to file extension.
39        let ext = self
40            .file_name
41            .rsplit_once('.')
42            .map(|(_, e)| e.to_ascii_lowercase())
43            .unwrap_or_default();
44
45        match ext.as_str() {
46            "flac" | "mp3" | "mpeg" | "mpga" | "m4a" | "ogg" | "oga" | "opus" | "wav" | "webm" => {
47                MediaKind::Audio
48            }
49            "png" | "jpg" | "jpeg" | "gif" | "bmp" | "webp" | "heic" | "tiff" | "svg" => {
50                MediaKind::Image
51            }
52            "mp4" | "mkv" | "avi" | "mov" | "wmv" | "flv" => MediaKind::Video,
53            _ => MediaKind::Unknown,
54        }
55    }
56}
