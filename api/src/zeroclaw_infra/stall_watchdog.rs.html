1//! Generic transport stall watchdog for WebSocket-based channels.
2//!
3//! [`StallWatchdog`] detects when a channel transport goes idle beyond a
4//! configurable threshold.  Channels call [`StallWatchdog::touch`] on every
5//! received event; the watchdog fires a caller-supplied callback when the
6//! elapsed silence exceeds `timeout_secs`.
7//!
8//! The timestamp is stored in an [`AtomicU64`] so `touch()` is lock-free and
9//! safe to call from any async context.
10
11use std::sync::Arc;
12use std::sync::atomic::{AtomicU64, Ordering};
13use tokio::sync::Mutex;
14use tokio::task::JoinHandle;
15
16/// Returns the current Unix timestamp in seconds.
17fn now_secs() -> u64 {
18    std::time::SystemTime::now()
19        .duration_since(std::time::UNIX_EPOCH)
20        .unwrap_or_default()
21        .as_secs()
22}
23
24/// A reusable watchdog that detects stalled (idle) WebSocket transports.
25///
26/// Create one per channel, call [`touch`](Self::touch) on every received
27/// message or event, and [`start`](Self::start) with a callback that triggers
28/// reconnection.
29pub struct StallWatchdog {
30    /// Unix timestamp (seconds) of the last received event.
31    last_event: Arc<AtomicU64>,
32    /// Stall threshold in seconds.
33    timeout_secs: u64,
34    /// Handle to the background polling task (if running).
35    task: Mutex<Option<JoinHandle<()>>>,
36}
37
38impl StallWatchdog {
39    /// Create a new watchdog with the given stall threshold.
40    ///
41    /// The watchdog is **not** started — call [`start`](Self::start) to begin
42    /// monitoring.
43    pub fn new(timeout_secs: u64) -> Self {
44        Self {
45            last_event: Arc::new(AtomicU64::new(now_secs())),
46            timeout_secs,
47            task: Mutex::new(None),
48        }
49    }
50
51    /// Record that an event was received **right now**.
52    ///
53    /// This is lock-free (atomic store) and can be called from any async
54    /// context without contention.
55    pub fn touch(&self) {
56        self.last_event.store(now_secs(), Ordering::Relaxed);
57    }
58
59    /// Returns `true` if the time since the last event exceeds the configured
60    /// timeout.
61    pub fn is_stalled(&self) -> bool {
62        let last = self.last_event.load(Ordering::Relaxed);
63        now_secs().saturating_sub(last) > self.timeout_secs
64    }
65
66    /// Start the background polling task.
67    ///
68    /// The task wakes every `timeout_secs / 2` seconds and checks whether the
69    /// transport has stalled.  When a stall is detected `on_stall` is invoked
70    /// (typically to log a warning and break out of the listen loop).
71    ///
72    /// Calling `start` while a task is already running replaces the previous
73    /// task (the old one is aborted).
74    pub async fn start(&self, on_stall: impl Fn() + Send + 'static) {
75        // Reset timestamp so the freshly-started watchdog doesn't immediately
76        // fire.
77        self.touch();
78
79        let last_event = Arc::clone(&self.last_event);
80        let timeout = self.timeout_secs;
81        let poll_interval = std::time::Duration::from_secs((timeout / 2).max(1));
82
83        let handle = tokio::spawn(async move {
84            let mut interval = tokio::time::interval(poll_interval);
85            // The first tick completes immediately — skip it so we wait a full
86            // interval before the first check.
87            interval.tick().await;
88
89            loop {
90                interval.tick().await;
91                let last = last_event.load(Ordering::Relaxed);
92                if now_secs().saturating_sub(last) > timeout {
93                    on_stall();
94                    break;
95                }
96            }
97        });
98
99        let mut task = self.task.lock().await;
100        if let Some(old) = task.take() {
101            old.abort();
102        }
103        *task = Some(handle);
104    }
105
106    /// Stop the background polling task (if running).
107    pub async fn stop(&self) {
108        let mut task = self.task.lock().await;
109        if let Some(handle) = task.take() {
110            handle.abort();
111        }
112    }
113}
114
115impl Drop for StallWatchdog {
116    fn drop(&mut self) {
117        // Best-effort cleanup — abort the task synchronously if it exists.
118        if let Ok(mut guard) = self.task.try_lock()
119            && let Some(handle) = guard.take()
120        {
121            handle.abort();
122        }
123    }
124}

126#[cfg(test)]
127mod tests {
128    use super::*;
129    use std::sync::atomic::AtomicBool;
130
131    #[tokio::test]
132    async fn touch_prevents_stall() {
133        let wd = StallWatchdog::new(2);
134        wd.touch();
135        assert!(!wd.is_stalled());
136    }
137
138    #[tokio::test]
139    async fn is_stalled_after_timeout() {
140        let wd = StallWatchdog::new(0); // 0-second timeout → always stalled
141        // Force last_event into the past
142        wd.last_event.store(0, Ordering::Relaxed);
143        assert!(wd.is_stalled());
144    }
145
146    #[tokio::test]
147    async fn callback_fires_on_stall() {
148        let fired = Arc::new(AtomicBool::new(false));
149        let fired_clone = Arc::clone(&fired);
150
151        let wd = StallWatchdog::new(1);
152
153        wd.start(move || {
154            fired_clone.store(true, Ordering::Relaxed);
155        })
156        .await;
157
158        // Force last_event far into the past *after* start() so the next poll
159        // detects a stall (start() calls touch() which would overwrite an
160        // earlier store).
161        wd.last_event.store(0, Ordering::Relaxed);
162
163        // Wait long enough for the poll interval (1 / 2 = clamped to 1s) + margin.
164        tokio::time::sleep(std::time::Duration::from_secs(2)).await;
165        assert!(fired.load(Ordering::Relaxed));
166    }
167
168    #[tokio::test]
169    async fn stop_prevents_callback() {
170        let fired = Arc::new(AtomicBool::new(false));
171        let fired_clone = Arc::clone(&fired);
172
173        let wd = StallWatchdog::new(1);
174
175        wd.start(move || {
176            fired_clone.store(true, Ordering::Relaxed);
177        })
178        .await;
179
180        wd.last_event.store(0, Ordering::Relaxed);
181
182        // Stop immediately before the poll task can fire.
183        wd.stop().await;
184
185        tokio::time::sleep(std::time::Duration::from_secs(2)).await;
186        assert!(!fired.load(Ordering::Relaxed));
187    }
188}
