1//! SQLite-backed session persistence with FTS5 search.
2//!
3//! Stores sessions in `{workspace}/sessions/sessions.db` using WAL mode.
4//! Provides full-text search via FTS5 and automatic TTL-based cleanup.
5//! Designed as the default backend, replacing JSONL for new installations.
6
7use crate::session_backend::{SessionBackend, SessionMetadata, SessionQuery, SessionState};
8use anyhow::{Context, Result};
9use chrono::{DateTime, Duration, Utc};
10use parking_lot::Mutex;
11use rusqlite::{Connection, params};
12use std::path::{Path, PathBuf};
13use zeroclaw_api::provider::ChatMessage;

15/// SQLite-backed session store with FTS5 and WAL mode.
16pub struct SqliteSessionBackend {
17    conn: Mutex<Connection>,
18    #[allow(dead_code)]
19    db_path: PathBuf,
20}

22impl SqliteSessionBackend {
23    /// Open or create the sessions database.
24    pub fn new(workspace_dir: &Path) -> Result<Self> {
25        let sessions_dir = workspace_dir.join("sessions");
26        std::fs::create_dir_all(&sessions_dir).context("Failed to create sessions directory")?;
27        let db_path = sessions_dir.join("sessions.db");
28
29        let conn = Connection::open(&db_path)
30            .with_context(|| format!("Failed to open session DB: {}", db_path.display()))?;
31
32        conn.execute_batch(
33            "PRAGMA journal_mode = WAL;
34             PRAGMA synchronous = NORMAL;
35             PRAGMA temp_store = MEMORY;
36             PRAGMA mmap_size = 4194304;",
37        )?;
38
39        conn.execute_batch(
40            "CREATE TABLE IF NOT EXISTS sessions (
41                id          INTEGER PRIMARY KEY AUTOINCREMENT,
42                session_key TEXT NOT NULL,
43                role        TEXT NOT NULL,
44                content     TEXT NOT NULL,
45                created_at  TEXT NOT NULL
46             );
47             CREATE INDEX IF NOT EXISTS idx_sessions_key ON sessions(session_key);
48             CREATE INDEX IF NOT EXISTS idx_sessions_key_id ON sessions(session_key, id);
49
50             CREATE TABLE IF NOT EXISTS session_metadata (
51                session_key  TEXT PRIMARY KEY,
52                created_at   TEXT NOT NULL,
53                last_activity TEXT NOT NULL,
54                message_count INTEGER NOT NULL DEFAULT 0,
55                name         TEXT
56             );
57
58             CREATE VIRTUAL TABLE IF NOT EXISTS sessions_fts USING fts5(
59                session_key, content, content=sessions, content_rowid=id
60             );
61
62             CREATE TRIGGER IF NOT EXISTS sessions_ai AFTER INSERT ON sessions BEGIN
63                INSERT INTO sessions_fts(rowid, session_key, content)
64                VALUES (new.id, new.session_key, new.content);
65             END;
66             CREATE TRIGGER IF NOT EXISTS sessions_ad AFTER DELETE ON sessions BEGIN
67                INSERT INTO sessions_fts(sessions_fts, rowid, session_key, content)
68                VALUES ('delete', old.id, old.session_key, old.content);
69             END;
70             CREATE TRIGGER IF NOT EXISTS sessions_au AFTER UPDATE ON sessions BEGIN
71                INSERT INTO sessions_fts(sessions_fts, rowid, session_key, content)
72                VALUES ('delete', old.id, old.session_key, old.content);
73                INSERT INTO sessions_fts(rowid, session_key, content)
74                VALUES (new.id, new.session_key, new.content);
75             END;",
76        )
77        .context("Failed to initialize session schema")?;
78
79        // Migration: add name column to existing databases
80        let has_name: bool = conn
81            .query_row(
82                "SELECT COUNT(*) > 0 FROM pragma_table_info('session_metadata') WHERE name = 'name'",
83                [],
84                |row| row.get(0),
85            )
86            .unwrap_or(false);
87        if !has_name {
88            let _ = conn.execute("ALTER TABLE session_metadata ADD COLUMN name TEXT", []);
89        }
90
91        // Migration: add state tracking columns
92        let has_state: bool = conn
93            .query_row(
94                "SELECT COUNT(*) > 0 FROM pragma_table_info('session_metadata') WHERE name = 'state'",
95                [],
96                |row| row.get(0),
97            )
98            .unwrap_or(false);
99        if !has_state {
100            let _ = conn.execute(
101                "ALTER TABLE session_metadata ADD COLUMN state TEXT NOT NULL DEFAULT 'idle'",
102                [],
103            );
104            let _ = conn.execute("ALTER TABLE session_metadata ADD COLUMN turn_id TEXT", []);
105            let _ = conn.execute(
106                "ALTER TABLE session_metadata ADD COLUMN turn_started_at TEXT",
107                [],
108            );
109        }
110
111        Ok(Self {
112            conn: Mutex::new(conn),
113            db_path,
114        })
115    }

117    /// Migrate JSONL session files into SQLite. Renames migrated files to `.jsonl.migrated`.
118    pub fn migrate_from_jsonl(&self, workspace_dir: &Path) -> Result<usize> {
119        let sessions_dir = workspace_dir.join("sessions");
120        let entries = match std::fs::read_dir(&sessions_dir) {
121            Ok(e) => e,
122            Err(_) => return Ok(0),
123        };
124
125        let mut migrated = 0;
126        for entry in entries {
127            let entry = match entry {
128                Ok(e) => e,
129                Err(_) => continue,
130            };
131            let name = match entry.file_name().into_string() {
132                Ok(n) => n,
133                Err(_) => continue,
134            };
135            let Some(key) = name.strip_suffix(".jsonl") else {
136                continue;
137            };
138
139            let path = entry.path();
140            let file = match std::fs::File::open(&path) {
141                Ok(f) => f,
142                Err(_) => continue,
143            };
144
145            let reader = std::io::BufReader::new(file);
146            let mut count = 0;
147            for line in std::io::BufRead::lines(reader) {
148                let Ok(line) = line else { continue };
149                let trimmed = line.trim();
150                if trimmed.is_empty() {
151                    continue;
152                }
153                if let Ok(msg) = serde_json::from_str::<ChatMessage>(trimmed)
154                    && self.append(key, &msg).is_ok()
155                {
156                    count += 1;
157                }
158            }
159
160            if count > 0 {
161                let migrated_path = path.with_extension("jsonl.migrated");
162                let _ = std::fs::rename(&path, &migrated_path);
163                migrated += 1;
164            }
165        }
166
167        Ok(migrated)
168    }
169}

172    fn load(&self, session_key: &str) -> Vec<ChatMessage> {
173        let conn = self.conn.lock();
174        let mut stmt = match conn
175            .prepare("SELECT role, content FROM sessions WHERE session_key = ?1 ORDER BY id ASC")
176        {
177            Ok(s) => s,
178            Err(_) => return Vec::new(),
179        };
180
181        let rows = match stmt.query_map(params![session_key], |row| {
182            Ok(ChatMessage {
183                role: row.get(0)?,
184                content: row.get(1)?,
185            })
186        }) {
187            Ok(r) => r,
188            Err(_) => return Vec::new(),
189        };
190
191        rows.filter_map(|r| r.ok()).collect()
192    }

194    fn append(&self, session_key: &str, message: &ChatMessage) -> std::io::Result<()> {
195        let conn = self.conn.lock();
196        let now = Utc::now().to_rfc3339();
197
198        conn.execute(
199            "INSERT INTO sessions (session_key, role, content, created_at)
200             VALUES (?1, ?2, ?3, ?4)",
201            params![session_key, message.role, message.content, now],
202        )
203        .map_err(std::io::Error::other)?;
204
205        // Upsert metadata
206        conn.execute(
207            "INSERT INTO session_metadata (session_key, created_at, last_activity, message_count)
208             VALUES (?1, ?2, ?3, 1)
209             ON CONFLICT(session_key) DO UPDATE SET
210                last_activity = excluded.last_activity,
211                message_count = message_count + 1",
212            params![session_key, now, now],
213        )
214        .map_err(std::io::Error::other)?;
215
216        Ok(())
217    }

219    fn remove_last(&self, session_key: &str) -> std::io::Result<bool> {
220        let conn = self.conn.lock();
221
222        let last_id: Option<i64> = conn
223            .query_row(
224                "SELECT id FROM sessions WHERE session_key = ?1 ORDER BY id DESC LIMIT 1",
225                params![session_key],
226                |row| row.get(0),
227            )
228            .ok();
229
230        let Some(id) = last_id else {
231            return Ok(false);
232        };
233
234        conn.execute("DELETE FROM sessions WHERE id = ?1", params![id])
235            .map_err(std::io::Error::other)?;
236
237        // Update metadata count
238        conn.execute(
239            "UPDATE session_metadata SET message_count = MAX(0, message_count - 1)
240             WHERE session_key = ?1",
241            params![session_key],
242        )
243        .map_err(std::io::Error::other)?;
244
245        Ok(true)
246    }

248    /// Efficiently update the last message in-place (single UPDATE instead of
249    /// DELETE + INSERT). Used for incremental persistence during streaming.
250    fn update_last(&self, session_key: &str, message: &ChatMessage) -> std::io::Result<bool> {
251        let conn = self.conn.lock();
252
253        let last_id: Option<i64> = conn
254            .query_row(
255                "SELECT id FROM sessions WHERE session_key = ?1 ORDER BY id DESC LIMIT 1",
256                params![session_key],
257                |row| row.get(0),
258            )
259            .ok();
260
261        let Some(id) = last_id else {
262            return Ok(false);
263        };
264
265        conn.execute(
266            "UPDATE sessions SET role = ?1, content = ?2 WHERE id = ?3",
267            params![message.role, message.content, id],
268        )
269        .map_err(std::io::Error::other)?;
270
271        // NOTE: FTS index becomes stale here (no UPDATE trigger, only
272        // INSERT/DELETE triggers). This is acceptable — update_last is
273        // used for transient streaming snapshots. The final content will
274        // be correct in the sessions table for load().
275
276        let now = chrono::Utc::now().to_rfc3339();
277        conn.execute(
278            "UPDATE session_metadata SET last_activity = ?1 WHERE session_key = ?2",
279            params![now, session_key],
280        )
281        .map_err(std::io::Error::other)?;
282
283        Ok(true)
284    }

343    fn cleanup_stale(&self, ttl_hours: u32) -> std::io::Result<usize> {
344        let conn = self.conn.lock();
345        let cutoff = (Utc::now() - Duration::hours(i64::from(ttl_hours))).to_rfc3339();
346
347        // Find stale sessions
348        let stale_keys: Vec<String> = {
349            let mut stmt = conn
350                .prepare("SELECT session_key FROM session_metadata WHERE last_activity < ?1")
351                .map_err(std::io::Error::other)?;
352            let rows = stmt
353                .query_map(params![cutoff], |row| row.get(0))
354                .map_err(std::io::Error::other)?;
355            rows.filter_map(|r| r.ok()).collect()
356        };
357
358        let count = stale_keys.len();
359        for key in &stale_keys {
360            let _ = conn.execute("DELETE FROM sessions WHERE session_key = ?1", params![key]);
361            let _ = conn.execute(
362                "DELETE FROM session_metadata WHERE session_key = ?1",
363                params![key],
364            );
365        }
366
367        Ok(count)
368    }

370    fn clear_messages(&self, session_key: &str) -> std::io::Result<usize> {
371        let conn = self.conn.lock();
372
373        conn.execute(
374            "DELETE FROM sessions WHERE session_key = ?1",
375            params![session_key],
376        )
377        .map_err(std::io::Error::other)?;
378
379        let count = conn.changes() as usize;
380
381        if count > 0 {
382            conn.execute(
383                "UPDATE session_metadata SET message_count = 0, last_activity = ?1 WHERE session_key = ?2",
384                params![Utc::now().to_rfc3339(), session_key],
385            )
386            .map_err(std::io::Error::other)?;
387        }
388
389        Ok(count)
390    }

392    fn delete_session(&self, session_key: &str) -> std::io::Result<bool> {
393        let conn = self.conn.lock();
394
395        // Check if session exists
396        let exists: bool = conn
397            .query_row(
398                "SELECT COUNT(*) > 0 FROM session_metadata WHERE session_key = ?1",
399                params![session_key],
400                |row| row.get(0),
401            )
402            .unwrap_or(false);
403
404        if !exists {
405            return Ok(false);
406        }
407
408        // Delete messages (FTS5 trigger handles sessions_fts cleanup)
409        conn.execute(
410            "DELETE FROM sessions WHERE session_key = ?1",
411            params![session_key],
412        )
413        .map_err(std::io::Error::other)?;
414
415        // Delete metadata
416        conn.execute(
417            "DELETE FROM session_metadata WHERE session_key = ?1",
418            params![session_key],
419        )
420        .map_err(std::io::Error::other)?;
421
422        Ok(true)
423    }

609    fn search(&self, query: &SessionQuery) -> Vec<SessionMetadata> {
610        let Some(keyword) = &query.keyword else {
611            return self.list_sessions_with_metadata();
612        };
613
614        let conn = self.conn.lock();
615        #[allow(clippy::cast_possible_wrap)]
616        let limit = query.limit.unwrap_or(50) as i64;
617
618        // FTS5 search
619        let mut stmt = match conn.prepare(
620            "SELECT DISTINCT f.session_key
621             FROM sessions_fts f
622             WHERE sessions_fts MATCH ?1
623             LIMIT ?2",
624        ) {
625            Ok(s) => s,
626            Err(_) => return Vec::new(),
627        };
628
629        // Quote each word for FTS5
630        let fts_query: String = keyword
631            .split_whitespace()
632            .map(|w| format!("\"{w}\""))
633            .collect::<Vec<_>>()
634            .join(" OR ");
635
636        let keys: Vec<String> = match stmt.query_map(params![fts_query, limit], |row| row.get(0)) {
637            Ok(r) => r.filter_map(|r| r.ok()).collect(),
638            Err(_) => return Vec::new(),
639        };
640
641        // Look up metadata for matched sessions
642        keys.iter()
643            .filter_map(|key| {
644                conn.query_row(
645                    "SELECT created_at, last_activity, message_count, name FROM session_metadata WHERE session_key = ?1",
646                    params![key],
647                    |row| {
648                        let created_str: String = row.get(0)?;
649                        let activity_str: String = row.get(1)?;
650                        let count: i64 = row.get(2)?;
651                        let name: Option<String> = row.get(3)?;
652                        Ok(SessionMetadata {
653                            key: key.clone(),
654                            name,
655                            created_at: DateTime::parse_from_rfc3339(&created_str)
656                                .map(|dt| dt.with_timezone(&Utc))
657                                .unwrap_or_else(|_| Utc::now()),
658                            last_activity: DateTime::parse_from_rfc3339(&activity_str)
659                                .map(|dt| dt.with_timezone(&Utc))
660                                .unwrap_or_else(|_| Utc::now()),
661                            #[allow(clippy::cast_sign_loss, clippy::cast_possible_truncation)]
662                            message_count: count as usize,
663                        })
664                    },
665                )
666                .ok()
667            })
668            .collect()
669    }
