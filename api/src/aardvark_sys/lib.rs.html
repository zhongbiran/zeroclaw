1//! Bindings for the Total Phase Aardvark I2C/SPI/GPIO USB adapter.
2//!
3//! Uses [`libloading`] to load `aardvark.so` at runtime — the same pattern
4//! the official Total Phase C stub (`aardvark.c`) uses internally.
5//!
6//! # Library search order
7//!
8//! 1. `ZEROCLAW_AARDVARK_LIB` environment variable (full path to `aardvark.so`)
9//! 2. `<workspace>/crates/aardvark-sys/vendor/aardvark.so` (development default)
10//! 3. `./aardvark.so` (next to the binary, for deployment)
11//!
12//! If none resolve, every method returns
13//! [`Err(AardvarkError::LibraryNotFound)`](AardvarkError::LibraryNotFound).
14//!
15//! # Safety
16//!
17//! This crate is the **only** place in ZeroClaw where `unsafe` is permitted.
18//! All `unsafe` is confined to `extern "C"` call sites inside this file.
19//! The public API is fully safe Rust.
20

29/// Bit set on a port returned by `aa_find_devices` when that port is in use.
30const AA_PORT_NOT_FREE: u16 = 0x8000;
31/// Configure adapter for I2C + GPIO (I2C master mode, SPI disabled).
32const AA_CONFIG_GPIO_I2C: i32 = 0x02;
33/// Configure adapter for SPI + GPIO (SPI master mode, I2C disabled).
34const AA_CONFIG_SPI_GPIO: i32 = 0x01;
35/// No I2C flags (standard 7-bit addressing, normal stop condition).
36const AA_I2C_NO_FLAGS: i32 = 0x00;
37/// Enable both onboard I2C pullup resistors (hardware v2+ only).
38const AA_I2C_PULLUP_BOTH: u8 = 0x03;

40// ── Library loading ───────────────────────────────────────────────────────
41
42static AARDVARK_LIB: OnceLock<Option<Library>> = OnceLock::new();
43
44fn lib() -> Option<&'static Library> {
45    AARDVARK_LIB
46        .get_or_init(|| {
47            let candidates: Vec<PathBuf> = vec![
48                // 1. Explicit env-var override (full path)
49                std::env::var("ZEROCLAW_AARDVARK_LIB")
50                    .ok()
51                    .map(PathBuf::from)
52                    .unwrap_or_default(),
53                // 2. Vendor directory shipped with this crate (dev default)
54                {
55                    let mut p = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
56                    p.push("vendor/aardvark.so");
57                    p
58                },
59                // 3. Next to the running binary (deployment)
60                std::env::current_exe()
61                    .ok()
62                    .and_then(|e| e.parent().map(|d| d.join("aardvark.so")))
63                    .unwrap_or_default(),
64                // 4. Current working directory
65                PathBuf::from("aardvark.so"),
66            ];
67            let mut tried_any = false;
68            for path in &candidates {
69                if path.as_os_str().is_empty() {
70                    continue;
71                }
72                tried_any = true;
73                match unsafe { Library::new(path) } {
74                    Ok(lib) => {
75                        // Verify the .so exports aa_c_version (Total Phase version gate).
76                        // The .so exports c_aa_* symbols (not aa_*); aa_c_version is the
77                        // one non-prefixed symbol used to confirm library identity.
78                        let version_ok = unsafe {
79                            lib.get::<unsafe extern "C" fn() -> u32>(b"aa_c_version\0").is_ok()
80                        };
81                        if !version_ok {
82                            eprintln!(
83                                "[aardvark-sys] {} loaded but aa_c_version not found — \
84                                 not a valid Aardvark library, skipping",
85                                path.display()
86                            );
87                            continue;
88                        }
89                        eprintln!("[aardvark-sys] loaded library from {}", path.display());
90                        return Some(lib);
91                    }
92                    Err(e) => {
93                        let msg = e.to_string();
94                        // Surface architecture mismatch explicitly — the most common
95                        // failure on Apple Silicon machines with an x86_64 SDK.
96                        if msg.contains("incompatible architecture") || msg.contains("mach-o file") {
97                            eprintln!(
98                                "[aardvark-sys] ARCHITECTURE MISMATCH loading {}: {}\n\
99                                 [aardvark-sys] The vendored aardvark.so is x86_64 but this \
100                                 binary is {}.\n\
101                                 [aardvark-sys] Download the arm64 SDK from https://www.totalphase.com/downloads/ \
102                                 or build with --target x86_64-apple-darwin.",
103                                path.display(),
104                                msg,
105                                std::env::consts::ARCH,
106                            );
107                        } else {
108                            eprintln!(
109                                "[aardvark-sys] could not load {}: {}",
110                                path.display(),
111                                msg
112                            );
113                        }
114                    }
115                }
116            }
117            if !tried_any {
118                eprintln!("[aardvark-sys] no library candidates found; set ZEROCLAW_AARDVARK_LIB or place aardvark.so next to the binary");
119            }
120            None
121        })
122        .as_ref()
123}

125/// Errors returned by Aardvark hardware operations.
126#[derive(Debug, Error)]
127pub enum AardvarkError {
128    /// No Aardvark adapter found — adapter not plugged in.
129    #[error("Aardvark adapter not found — is it plugged in?")]
130    NotFound,
131    /// `aa_open` returned a non-positive handle.
132    #[error("Aardvark open failed (code {0})")]
133    OpenFailed(i32),
134    /// `aa_i2c_write` returned a negative status code.
135    #[error("I2C write failed (code {0})")]
136    I2cWriteFailed(i32),
137    /// `aa_i2c_read` returned a negative status code.
138    #[error("I2C read failed (code {0})")]
139    I2cReadFailed(i32),
140    /// `aa_spi_write` returned a negative status code.
141    #[error("SPI transfer failed (code {0})")]
142    SpiTransferFailed(i32),
143    /// GPIO operation returned a negative status code.
144    #[error("GPIO error (code {0})")]
145    GpioError(i32),
146    /// `aardvark.so` could not be found or loaded.
147    #[error("aardvark.so not found — set ZEROCLAW_AARDVARK_LIB or place it next to the binary")]
148    LibraryNotFound,
149}

156/// Safe RAII handle over the Aardvark C library handle.
157///
158/// Automatically closes the adapter on `Drop`.
159///
160/// **Usage pattern:** open a fresh handle per command and let it drop at the
161/// end of each operation (lazy-open / eager-close).
162pub struct AardvarkHandle {
163    handle: i32,
164}
165
166impl AardvarkHandle {
167    // ── Lifecycle ─────────────────────────────────────────────────────────
168
169    /// Open the first available (free) Aardvark adapter.
170    pub fn open() -> Result<Self> {
171        let ports = Self::find_devices();
172        let port = ports.first().copied().ok_or(AardvarkError::NotFound)?;
173        Self::open_port(i32::from(port))
174    }
175
176    /// Open a specific Aardvark adapter by port index.
177    pub fn open_port(port: i32) -> Result<Self> {
178        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
179        let handle: i32 = unsafe {
180            let f: Symbol<unsafe extern "C" fn(i32) -> i32> = lib
181                .get(b"c_aa_open\0")
182                .map_err(|_| AardvarkError::LibraryNotFound)?;
183            f(port)
184        };
185        if handle <= 0 {
186            Err(AardvarkError::OpenFailed(handle))
187        } else {
188            Ok(Self { handle })
189        }
190    }
191
192    /// Return the port numbers of all **free** connected adapters.
193    ///
194    /// Ports in-use by another process are filtered out.
195    /// Returns an empty `Vec` when `aardvark.so` cannot be loaded.
196    pub fn find_devices() -> Vec<u16> {
197        let Some(lib) = lib() else {
198            eprintln!("[aardvark-sys] find_devices: library not loaded");
199            return Vec::new();
200        };
201        let mut ports = [0u16; 16];
202        let n: i32 = unsafe {
203            let f: std::result::Result<Symbol<unsafe extern "C" fn(i32, *mut u16) -> i32>, _> =
204                lib.get(b"c_aa_find_devices\0");
205            match f {
206                Ok(f) => f(16, ports.as_mut_ptr()),
207                Err(e) => {
208                    eprintln!("[aardvark-sys] find_devices: symbol lookup failed: {e}");
209                    return Vec::new();
210                }
211            }
212        };
213        eprintln!(
214            "[aardvark-sys] find_devices: c_aa_find_devices returned {n}, ports={:?}",
215            &ports[..n.max(0) as usize]
216        );
217        if n <= 0 {
218            return Vec::new();
219        }
220        let free: Vec<u16> = ports[..n as usize]
221            .iter()
222            .filter(|&&p| (p & AA_PORT_NOT_FREE) == 0)
223            .copied()
224            .collect();
225        eprintln!("[aardvark-sys] find_devices: free ports={free:?}");
226        free
227    }

231    /// Enable I2C mode and set the bitrate (kHz).
232    pub fn i2c_enable(&self, bitrate_khz: u32) -> Result<()> {
233        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
234        unsafe {
235            let configure: Symbol<unsafe extern "C" fn(i32, i32) -> i32> = lib
236                .get(b"c_aa_configure\0")
237                .map_err(|_| AardvarkError::LibraryNotFound)?;
238            configure(self.handle, AA_CONFIG_GPIO_I2C);
239            let pullup: Symbol<unsafe extern "C" fn(i32, u8) -> i32> = lib
240                .get(b"c_aa_i2c_pullup\0")
241                .map_err(|_| AardvarkError::LibraryNotFound)?;
242            pullup(self.handle, AA_I2C_PULLUP_BOTH);
243            let bitrate: Symbol<unsafe extern "C" fn(i32, i32) -> i32> = lib
244                .get(b"c_aa_i2c_bitrate\0")
245                .map_err(|_| AardvarkError::LibraryNotFound)?;
246            bitrate(self.handle, bitrate_khz as i32);
247        }
248        Ok(())
249    }

251    /// Write `data` bytes to the I2C device at `addr`.
252    pub fn i2c_write(&self, addr: u8, data: &[u8]) -> Result<()> {
253        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
254        let ret: i32 = unsafe {
255            let f: Symbol<unsafe extern "C" fn(i32, u16, i32, u16, *const u8) -> i32> = lib
256                .get(b"c_aa_i2c_write\0")
257                .map_err(|_| AardvarkError::LibraryNotFound)?;
258            f(
259                self.handle,
260                u16::from(addr),
261                AA_I2C_NO_FLAGS,
262                data.len() as u16,
263                data.as_ptr(),
264            )
265        };
266        if ret < 0 {
267            Err(AardvarkError::I2cWriteFailed(ret))
268        } else {
269            Ok(())
270        }
271    }
272
273    /// Read `len` bytes from the I2C device at `addr`.
274    pub fn i2c_read(&self, addr: u8, len: usize) -> Result<Vec<u8>> {
275        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
276        let mut buf = vec![0u8; len];
277        let ret: i32 = unsafe {
278            let f: Symbol<unsafe extern "C" fn(i32, u16, i32, u16, *mut u8) -> i32> = lib
279                .get(b"c_aa_i2c_read\0")
280                .map_err(|_| AardvarkError::LibraryNotFound)?;
281            f(
282                self.handle,
283                u16::from(addr),
284                AA_I2C_NO_FLAGS,
285                len as u16,
286                buf.as_mut_ptr(),
287            )
288        };
289        if ret < 0 {
290            Err(AardvarkError::I2cReadFailed(ret))
291        } else {
292            Ok(buf)
293        }
294    }

296    /// Write then read — standard I2C register-read pattern.
297    pub fn i2c_write_read(&self, addr: u8, write_data: &[u8], read_len: usize) -> Result<Vec<u8>> {
298        self.i2c_write(addr, write_data)?;
299        self.i2c_read(addr, read_len)
300    }

302    /// Scan the I2C bus, returning addresses of all responding devices.
303    ///
304    /// Probes `0x08–0x77` with a 1-byte read; returns addresses that ACK.
305    pub fn i2c_scan(&self) -> Vec<u8> {
306        let Some(lib) = lib() else {
307            return Vec::new();
308        };
309        let Ok(f): std::result::Result<
310            Symbol<unsafe extern "C" fn(i32, u16, i32, u16, *mut u8) -> i32>,
311            _,
312        > = (unsafe { lib.get(b"c_aa_i2c_read\0") }) else {
313            return Vec::new();
314        };
315        let mut found = Vec::new();
316        let mut buf = [0u8; 1];
317        for addr in 0x08u16..=0x77 {
318            let ret = unsafe { f(self.handle, addr, AA_I2C_NO_FLAGS, 1, buf.as_mut_ptr()) };
319            // ret > 0: bytes received → device ACKed
320            // ret == 0: NACK → no device at this address
321            // ret < 0: error code → skip
322            if ret > 0 {
323                found.push(addr as u8);
324            }
325        }
326        found
327    }

331    /// Enable SPI mode and set the bitrate (kHz).
332    pub fn spi_enable(&self, bitrate_khz: u32) -> Result<()> {
333        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
334        unsafe {
335            let configure: Symbol<unsafe extern "C" fn(i32, i32) -> i32> = lib
336                .get(b"c_aa_configure\0")
337                .map_err(|_| AardvarkError::LibraryNotFound)?;
338            configure(self.handle, AA_CONFIG_SPI_GPIO);
339            // SPI mode 0: polarity=rising/falling(0), phase=sample/setup(0), MSB first(0)
340            let spi_cfg: Symbol<unsafe extern "C" fn(i32, i32, i32, i32) -> i32> = lib
341                .get(b"c_aa_spi_configure\0")
342                .map_err(|_| AardvarkError::LibraryNotFound)?;
343            spi_cfg(self.handle, 0, 0, 0);
344            let bitrate: Symbol<unsafe extern "C" fn(i32, i32) -> i32> = lib
345                .get(b"c_aa_spi_bitrate\0")
346                .map_err(|_| AardvarkError::LibraryNotFound)?;
347            bitrate(self.handle, bitrate_khz as i32);
348        }
349        Ok(())
350    }

352    /// Full-duplex SPI transfer.
353    ///
354    /// Sends `send` bytes; returns the simultaneously received bytes (same length).
355    pub fn spi_transfer(&self, send: &[u8]) -> Result<Vec<u8>> {
356        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
357        let mut recv = vec![0u8; send.len()];
358        // aa_spi_write(aardvark, out_num_bytes, data_out, in_num_bytes, data_in)
359        let ret: i32 = unsafe {
360            let f: Symbol<unsafe extern "C" fn(i32, u16, *const u8, u16, *mut u8) -> i32> = lib
361                .get(b"c_aa_spi_write\0")
362                .map_err(|_| AardvarkError::LibraryNotFound)?;
363            f(
364                self.handle,
365                send.len() as u16,
366                send.as_ptr(),
367                recv.len() as u16,
368                recv.as_mut_ptr(),
369            )
370        };
371        if ret < 0 {
372            Err(AardvarkError::SpiTransferFailed(ret))
373        } else {
374            Ok(recv)
375        }
376    }

380    /// Set GPIO pin directions and output values.
381    ///
382    /// `direction`: bitmask — `1` = output, `0` = input.
383    /// `value`: output state bitmask.
384    pub fn gpio_set(&self, direction: u8, value: u8) -> Result<()> {
385        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
386        unsafe {
387            let dir_f: Symbol<unsafe extern "C" fn(i32, u8) -> i32> = lib
388                .get(b"c_aa_gpio_direction\0")
389                .map_err(|_| AardvarkError::LibraryNotFound)?;
390            let d = dir_f(self.handle, direction);
391            if d < 0 {
392                return Err(AardvarkError::GpioError(d));
393            }
394            let set_f: Symbol<unsafe extern "C" fn(i32, u8) -> i32> =
395                lib.get(b"c_aa_gpio_set\0")
396                    .map_err(|_| AardvarkError::LibraryNotFound)?;
397            let r = set_f(self.handle, value);
398            if r < 0 {
399                return Err(AardvarkError::GpioError(r));
400            }
401        }
402        Ok(())
403    }

405    /// Read the current GPIO pin states as a bitmask.
406    pub fn gpio_get(&self) -> Result<u8> {
407        let lib = lib().ok_or(AardvarkError::LibraryNotFound)?;
408        let ret: i32 = unsafe {
409            let f: Symbol<unsafe extern "C" fn(i32) -> i32> = lib
410                .get(b"c_aa_gpio_get\0")
411                .map_err(|_| AardvarkError::LibraryNotFound)?;
412            f(self.handle)
413        };
414        if ret < 0 {
415            Err(AardvarkError::GpioError(ret))
416        } else {
417            Ok(ret as u8)
418        }
419    }

422impl Drop for AardvarkHandle {
423    fn drop(&mut self) {
424        if let Some(lib) = lib() {
425            unsafe {
426                if let Ok(f) = lib.get::<unsafe extern "C" fn(i32) -> i32>(b"c_aa_close\0") {
427                    f(self.handle);
428                }
429            }
430        }
431    }
