1//! MCP (Model Context Protocol) JSON-RPC 2.0 protocol types.
2//! Protocol version: 2024-11-05
3//! Adapted from ops-mcp-server/src/protocol.rs for client use.
4//! Both Serialize and Deserialize are derived — the client both sends (Serialize)
5//! and receives (Deserialize) JSON-RPC messages.
6
7use serde::{Deserialize, Serialize};
8
9pub const JSONRPC_VERSION: &str = "2.0";
10pub const MCP_PROTOCOL_VERSION: &str = "2024-11-05";
11
12// Standard JSON-RPC 2.0 error codes
13pub const PARSE_ERROR: i32 = -32700;
14pub const INVALID_REQUEST: i32 = -32600;
15pub const METHOD_NOT_FOUND: i32 = -32601;
16pub const INVALID_PARAMS: i32 = -32602;
17pub const INTERNAL_ERROR: i32 = -32603;
18
19/// Outbound JSON-RPC request (client → MCP server).
20/// Used for both method calls (with id) and notifications (id = None).
21#[derive(Debug, Clone, Serialize, Deserialize)]
22pub struct JsonRpcRequest {
23    pub jsonrpc: String,
24    #[serde(skip_serializing_if = "Option::is_none")]
25    pub id: Option<serde_json::Value>,
26    pub method: String,
27    #[serde(skip_serializing_if = "Option::is_none")]
28    pub params: Option<serde_json::Value>,
29}
30
31impl JsonRpcRequest {
32    /// Create a method call request with a numeric id.
33    pub fn new(id: u64, method: impl Into<String>, params: serde_json::Value) -> Self {
34        Self {
35            jsonrpc: JSONRPC_VERSION.to_string(),
36            id: Some(serde_json::Value::Number(id.into())),
37            method: method.into(),
38            params: Some(params),
39        }
40    }
41
42    /// Create a notification — no id, no response expected from server.
43    pub fn notification(method: impl Into<String>, params: serde_json::Value) -> Self {
44        Self {
45            jsonrpc: JSONRPC_VERSION.to_string(),
46            id: None,
47            method: method.into(),
48            params: Some(params),
49        }
50    }
51}
52
53/// Inbound JSON-RPC response (MCP server → client).
54#[derive(Debug, Clone, Serialize, Deserialize)]
55pub struct JsonRpcResponse {
56    pub jsonrpc: String,
57    #[serde(skip_serializing_if = "Option::is_none")]
58    pub id: Option<serde_json::Value>,
59    #[serde(skip_serializing_if = "Option::is_none")]
60    pub result: Option<serde_json::Value>,
61    #[serde(skip_serializing_if = "Option::is_none")]
62    pub error: Option<JsonRpcError>,
63}
64
65/// JSON-RPC error object embedded in a response.
66#[derive(Debug, Clone, Serialize, Deserialize)]
67pub struct JsonRpcError {
68    pub code: i32,
69    pub message: String,
70    #[serde(skip_serializing_if = "Option::is_none")]
71    pub data: Option<serde_json::Value>,
72}
73
74/// A tool advertised by an MCP server (from `tools/list` response).
75#[derive(Debug, Clone, Serialize, Deserialize)]
76pub struct McpToolDef {
77    pub name: String,
78    #[serde(skip_serializing_if = "Option::is_none")]
79    pub description: Option<String>,
80    #[serde(rename = "inputSchema")]
81    pub input_schema: serde_json::Value,
82}
83
84/// Expected shape of the `tools/list` result payload.
85#[derive(Debug, Deserialize)]
86pub struct McpToolsListResult {
87    pub tools: Vec<McpToolDef>,
88}

94    #[test]
95    fn request_serializes_with_id() {
96        let req = JsonRpcRequest::new(1, "tools/list", serde_json::json!({}));
97        let s = serde_json::to_string(&req).unwrap();
98        assert!(s.contains("\"id\":1"));
99        assert!(s.contains("\"method\":\"tools/list\""));
100        assert!(s.contains("\"jsonrpc\":\"2.0\""));
101    }
102
103    #[test]
104    fn notification_omits_id() {
105        let notif =
106            JsonRpcRequest::notification("notifications/initialized", serde_json::json!({}));
107        let s = serde_json::to_string(&notif).unwrap();
108        assert!(!s.contains("\"id\""));
109    }
110
111    #[test]
112    fn response_deserializes() {
113        let json = r#"{"jsonrpc":"2.0","id":1,"result":{"tools":[]}}"#;
114        let resp: JsonRpcResponse = serde_json::from_str(json).unwrap();
115        assert!(resp.result.is_some());
116        assert!(resp.error.is_none());
117    }

120    fn tool_def_deserializes_input_schema() {
121        let json = r#"{"name":"read_file","description":"Read a file","inputSchema":{"type":"object","properties":{"path":{"type":"string"}}}}"#;
122        let def: McpToolDef = serde_json::from_str(json).unwrap();
123        assert_eq!(def.name, "read_file");
124        assert!(def.input_schema.is_object());
125    }

203            jsonrpc: JSONRPC_VERSION.to_string(),

205            result: Some(serde_json::json!({"answer": 42})),
206            error: None,

208        let serialized = serde_json::to_string(&original).unwrap();
209        let deserialized: JsonRpcResponse = serde_json::from_str(&serialized).unwrap();
