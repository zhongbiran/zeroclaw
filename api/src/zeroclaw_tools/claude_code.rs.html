12const SAFE_ENV_VARS: &[&str] = &[

16/// Delegates coding tasks to the Claude Code CLI (`claude -p`).
17///
18/// This creates a two-tier agent architecture: ZeroClaw orchestrates high-level
19/// tasks and delegates complex coding work to Claude Code, which has its own
20/// agent loop with Read/Edit/Bash tools.
21///
22/// Authentication uses the `claude` binary's own OAuth session (Max subscription)
23/// by default. No API key is needed unless `env_passthrough` includes
24/// `ANTHROPIC_API_KEY` for API-key billing.
25pub struct ClaudeCodeTool {
26    security: Arc<SecurityPolicy>,
27    config: ClaudeCodeConfig,
28}
29
30impl ClaudeCodeTool {
31    pub fn new(security: Arc<SecurityPolicy>, config: ClaudeCodeConfig) -> Self {
32        Self { security, config }
33    }
34}

80    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
81        // Rate limiting is applied by the RateLimitedTool wrapper at
82        // registration time (see zeroclaw-runtime::tools::mod).
83
84        // Enforce act policy
85        if let Err(error) = self
86            .security
87            .enforce_tool_operation(ToolOperation::Act, "claude_code")
88        {
89            return Ok(ToolResult {
90                success: false,
91                output: String::new(),
92                error: Some(error),
93            });
94        }
95
96        // Extract prompt (required)
97        let prompt = args
98            .get("prompt")
99            .and_then(|v| v.as_str())
100            .ok_or_else(|| anyhow::anyhow!("Missing 'prompt' parameter"))?;
101
102        // Extract optional params
103        let allowed_tools: Vec<String> = args
104            .get("allowed_tools")
105            .and_then(|v| v.as_array())
106            .map(|arr| {
107                arr.iter()
108                    .filter_map(|v| v.as_str().map(String::from))
109                    .collect()
110            })
111            .unwrap_or_else(|| self.config.allowed_tools.clone());
112
113        let system_prompt = args
114            .get("system_prompt")
115            .and_then(|v| v.as_str())
116            .map(String::from)
117            .or_else(|| self.config.system_prompt.clone());
118
119        let session_id = args.get("session_id").and_then(|v| v.as_str());
120
121        let json_schema = args.get("json_schema").filter(|v| v.is_object());
122
123        // Validate working directory — require both paths to exist (reject
124        // non-existent paths instead of falling back to the raw value, which
125        // could bypass the workspace containment check via symlinks or
126        // specially-crafted path components).
127        let work_dir = if let Some(wd) = args.get("working_directory").and_then(|v| v.as_str()) {
128            let wd_path = std::path::PathBuf::from(wd);
129            let workspace = &self.security.workspace_dir;
130            let canonical_wd = match wd_path.canonicalize() {
131                Ok(p) => p,
132                Err(_) => {
133                    return Ok(ToolResult {
134                        success: false,
135                        output: String::new(),
136                        error: Some(format!(
137                            "working_directory '{}' does not exist or is not accessible",
138                            wd
139                        )),
140                    });
141                }
142            };
143            let canonical_ws = match workspace.canonicalize() {
144                Ok(p) => p,
145                Err(_) => {
146                    return Ok(ToolResult {
147                        success: false,
148                        output: String::new(),
149                        error: Some(format!(
150                            "workspace directory '{}' does not exist or is not accessible",
151                            workspace.display()
152                        )),
153                    });
154                }
155            };
156            if !canonical_wd.starts_with(&canonical_ws) {
157                return Ok(ToolResult {
158                    success: false,
159                    output: String::new(),
160                    error: Some(format!(
161                        "working_directory '{}' is outside the workspace '{}'",
162                        wd,
163                        workspace.display()
164                    )),
165                });
166            }
167            canonical_wd
168        } else {
169            self.security.workspace_dir.clone()
170        };
171
172        // Build CLI command
173        let claude_bin = if cfg!(target_os = "windows") {
174            "claude.cmd"
175        } else {
176            "claude"
177        };
178        let mut cmd = Command::new(claude_bin);
179        cmd.arg("-p").arg(prompt);
180        cmd.arg("--output-format").arg("json");
181
182        if !allowed_tools.is_empty() {
183            for tool in &allowed_tools {
184                cmd.arg("--allowedTools").arg(tool);
185            }
186        }
187
188        if let Some(ref sp) = system_prompt {
189            cmd.arg("--append-system-prompt").arg(sp);
190        }
191
192        if let Some(sid) = session_id {
193            cmd.arg("--resume").arg(sid);
194        }
195
196        if let Some(schema) = json_schema {
197            let schema_str = serde_json::to_string(schema).unwrap_or_else(|_| "{}".to_string());
198            cmd.arg("--json-schema").arg(schema_str);
199        }
200
201        // Environment: clear everything, pass only safe vars + configured passthrough.
202        // HOME is critical so `claude` finds its OAuth session in ~/.claude/
203        cmd.env_clear();
204        for var in SAFE_ENV_VARS {
205            if let Ok(val) = std::env::var(var) {
206                cmd.env(var, val);
207            }
208        }
209        for var in &self.config.env_passthrough {
210            let trimmed = var.trim();
211            if !trimmed.is_empty()
212                && let Ok(val) = std::env::var(trimmed)
213            {
214                cmd.env(trimmed, val);
215            }
216        }
217
218        cmd.current_dir(&work_dir);
219        // Execute with timeout — use kill_on_drop(true) so the child process
220        // is automatically killed when the future is dropped on timeout,
221        // preventing zombie processes.
222        let timeout = Duration::from_secs(self.config.timeout_secs);
223        cmd.kill_on_drop(true);
224
225        let result = tokio::time::timeout(timeout, cmd.output()).await;
226
227        match result {
228            Ok(Ok(output)) => {
229                let mut stdout = String::from_utf8_lossy(&output.stdout).to_string();
230                let stderr = String::from_utf8_lossy(&output.stderr).to_string();
231
232                // Truncate to max_output_bytes with char-boundary safety
233                if stdout.len() > self.config.max_output_bytes {
234                    let mut b = self.config.max_output_bytes.min(stdout.len());
235                    while b > 0 && !stdout.is_char_boundary(b) {
236                        b -= 1;
237                    }
238                    stdout.truncate(b);
239                    stdout.push_str("\n... [output truncated]");
240                }
241
242                // Try to parse JSON response and extract result + session_id
243                if let Ok(json_resp) = serde_json::from_str::<serde_json::Value>(&stdout) {
244                    let result_text = json_resp
245                        .get("result")
246                        .and_then(|v| v.as_str())
247                        .unwrap_or("");
248                    let resp_session_id = json_resp
249                        .get("session_id")
250                        .and_then(|v| v.as_str())
251                        .unwrap_or("");
252
253                    let mut formatted = String::new();
254                    if result_text.is_empty() {
255                        // Fall back to full JSON if no "result" key
256                        formatted.push_str(&stdout);
257                    } else {
258                        formatted.push_str(result_text);
259                    }
260                    if !resp_session_id.is_empty() {
261                        use std::fmt::Write;
262                        let _ = write!(formatted, "\n\n[session_id: {}]", resp_session_id);
263                    }
264
265                    Ok(ToolResult {
266                        success: output.status.success(),
267                        output: formatted,
268                        error: if stderr.is_empty() {
269                            None
270                        } else {
271                            Some(stderr)
272                        },
273                    })
274                } else {
275                    // JSON parse failed — return raw stdout (defensive)
276                    Ok(ToolResult {
277                        success: output.status.success(),
278                        output: stdout,
279                        error: if stderr.is_empty() {
280                            None
281                        } else {
282                            Some(stderr)
283                        },
284                    })
285                }
286            }
287            Ok(Err(e)) => {
288                let err_msg = e.to_string();
289                let msg = if err_msg.contains("No such file or directory")
290                    || err_msg.contains("not found")
291                    || err_msg.contains("cannot find")
292                {
293                    "Claude Code CLI ('claude') not found in PATH. Install with: npm install -g @anthropic-ai/claude-code".into()
294                } else {
295                    format!("Failed to execute claude: {e}")
296                };
297                Ok(ToolResult {
298                    success: false,
299                    output: String::new(),
300                    error: Some(msg),
301                })
302            }
303            Err(_) => {
304                // Timeout — kill_on_drop(true) ensures the child is killed
305                // when the future is dropped.
306                Ok(ToolResult {
307                    success: false,
308                    output: String::new(),
309                    error: Some(format!(
310                        "Claude Code timed out after {}s and was killed",
311                        self.config.timeout_secs
312                    )),
313                })
314            }
315        }
316    }
317}
