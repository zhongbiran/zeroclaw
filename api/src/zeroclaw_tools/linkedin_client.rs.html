57    fn parse_env_value(raw: &str) -> String {
58        let raw = raw.trim();
59
60        let unquoted = if raw.len() >= 2
61            && ((raw.starts_with('"') && raw.ends_with('"'))
62                || (raw.starts_with('\'') && raw.ends_with('\'')))
63        {
64            &raw[1..raw.len() - 1]
65        } else {
66            raw
67        };
68
69        // Strip inline comments in unquoted values: KEY=value # comment
70        unquoted.split_once(" #").map_or_else(
71            || unquoted.trim().to_string(),
72            |(value, _)| value.trim().to_string(),
73        )
74    }

76    pub async fn get_credentials(&self) -> anyhow::Result<LinkedInCredentials> {
77        let env_path = self.workspace_dir.join(".env");
78        let content = tokio::fs::read_to_string(&env_path)
79            .await
80            .with_context(|| format!("Failed to read {}", env_path.display()))?;
81
82        let mut client_id = None;
83        let mut client_secret = None;
84        let mut access_token = None;
85        let mut refresh_token = None;
86        let mut person_id = None;
87
88        for line in content.lines() {
89            let line = line.trim();
90            if line.starts_with('#') || line.is_empty() {
91                continue;
92            }
93            let line = line.strip_prefix("export ").map(str::trim).unwrap_or(line);
94            if let Some((key, value)) = line.split_once('=') {
95                let key = key.trim();
96                let value = Self::parse_env_value(value);
97
98                match key {
99                    "LINKEDIN_CLIENT_ID" => client_id = Some(value),
100                    "LINKEDIN_CLIENT_SECRET" => client_secret = Some(value),
101                    "LINKEDIN_ACCESS_TOKEN" => access_token = Some(value),
102                    "LINKEDIN_REFRESH_TOKEN" if !value.is_empty() => refresh_token = Some(value),
103                    "LINKEDIN_PERSON_ID" => person_id = Some(value),
104                    _ => {}
105                }
106            }
107        }
108
109        let client_id =
110            client_id.ok_or_else(|| anyhow::anyhow!("LINKEDIN_CLIENT_ID not found in .env"))?;
111        let client_secret = client_secret
112            .ok_or_else(|| anyhow::anyhow!("LINKEDIN_CLIENT_SECRET not found in .env"))?;
113        let access_token = access_token
114            .ok_or_else(|| anyhow::anyhow!("LINKEDIN_ACCESS_TOKEN not found in .env"))?;
115        let person_id =
116            person_id.ok_or_else(|| anyhow::anyhow!("LINKEDIN_PERSON_ID not found in .env"))?;
117
118        Ok(LinkedInCredentials {
119            client_id,
120            client_secret,
121            access_token,
122            refresh_token,
123            person_id,
124        })
125    }

153    async fn api_request(
154        &self,
155        method: Method,
156        url: &str,
157        token: &str,
158        body: Option<serde_json::Value>,
159    ) -> anyhow::Result<reqwest::Response> {
160        let client = Self::client();
161        let headers = self.api_headers(token);
162
163        let mut req = client.request(method.clone(), url).headers(headers);
164        if let Some(ref json_body) = body {
165            req = req.json(json_body);
166        }
167
168        let response = req.send().await.context("LinkedIn API request failed")?;
169
170        if response.status() == reqwest::StatusCode::UNAUTHORIZED {
171            // Attempt token refresh and retry once
172            let creds = self.get_credentials().await?;
173            let new_token = self.refresh_token(&creds).await?;
174            self.update_env_token(&new_token).await?;
175
176            let retry_headers = self.api_headers(&new_token);
177            let mut retry_req = Self::client().request(method, url).headers(retry_headers);
178            if let Some(json_body) = body {
179                retry_req = retry_req.json(&json_body);
180            }
181
182            let retry_response = retry_req
183                .send()
184                .await
185                .context("LinkedIn API retry request failed")?;
186
187            return Ok(retry_response);
188        }
189
190        Ok(response)
191    }

193    pub async fn create_post(
194        &self,
195        text: &str,
196        visibility: &str,
197        article_url: Option<&str>,
198        article_title: Option<&str>,
199        scheduled_at: Option<&str>,
200    ) -> anyhow::Result<String> {
201        let creds = self.get_credentials().await?;
202        let author_urn = format!("urn:li:person:{}", creds.person_id);
203
204        let lifecycle = if scheduled_at.is_some() {
205            "DRAFT"
206        } else {
207            "PUBLISHED"
208        };
209
210        let mut body = json!({
211            "author": author_urn,
212            "lifecycleState": lifecycle,
213            "visibility": visibility,
214            "commentary": text,
215            "distribution": {
216                "feedDistribution": "MAIN_FEED",
217                "targetEntities": [],
218                "thirdPartyDistributionChannels": []
219            }
220        });
221
222        // Add scheduled publish options if a future timestamp is provided.
223        // The timestamp must be ISO 8601 / RFC 3339, e.g. "2026-03-17T08:00:00Z".
224        if let Some(ts) = scheduled_at
225            && let Ok(dt) = chrono::DateTime::parse_from_rfc3339(ts)
226        {
227            let epoch_ms = dt.timestamp_millis();
228            body.as_object_mut().unwrap().insert(
229                "scheduledPublishOptions".to_string(),
230                json!({ "scheduledPublishTime": epoch_ms }),
231            );
232            // Scheduled posts use DRAFT lifecycle
233            body["lifecycleState"] = json!("DRAFT");
234        }
235
236        if let Some(url) = article_url {
237            let mut article = json!({
238                "source": url,
239                "title": article_title.unwrap_or(""),
240            });
241            if article_title.is_none() || article_title.is_some_and(|t| t.is_empty()) {
242                article.as_object_mut().unwrap().remove("title");
243            }
244            body.as_object_mut().unwrap().insert(
245                "content".to_string(),
246                json!({
247                    "article": {
248                        "source": url,
249                        "title": article_title.unwrap_or("")
250                    }
251                }),
252            );
253        }
254
255        let url = format!("{}/rest/posts", LINKEDIN_API_BASE);
256        let response = self
257            .api_request(Method::POST, &url, &creds.access_token, Some(body))
258            .await?;
259
260        let status = response.status();
261        if !status.is_success() {
262            let body_text = response.text().await.unwrap_or_default();
263            anyhow::bail!("LinkedIn create_post failed ({}): {}", status, body_text);
264        }
265
266        // The post URN is returned in the x-restli-id header
267        let post_urn = response
268            .headers()
269            .get("x-restli-id")
270            .and_then(|v| v.to_str().ok())
271            .map(String::from)
272            .unwrap_or_default();
273
274        Ok(post_urn)
275    }

544    /// Register an image asset with LinkedIn, upload binary data, and return the asset URN.
545    ///
546    /// LinkedIn's image post flow is three steps:
547    /// 1. Register the upload → get an upload URL + asset URN
548    /// 2. PUT the binary image to the upload URL
549    /// 3. Reference the asset URN when creating the post
550    pub async fn upload_image(
551        &self,
552        image_bytes: &[u8],
553        token: &str,
554        person_id: &str,
555    ) -> anyhow::Result<String> {
556        let owner_urn = format!("urn:li:person:{person_id}");
557
558        // Step 1: Register upload
559        let register_body = json!({
560            "initializeUploadRequest": {
561                "owner": owner_urn
562            }
563        });
564        let register_url = format!("{LINKEDIN_API_BASE}/rest/images?action=initializeUpload");
565        let register_resp = self
566            .api_request(Method::POST, &register_url, token, Some(register_body))
567            .await?;
568
569        let status = register_resp.status();
570        if !status.is_success() {
571            let body_text = register_resp.text().await.unwrap_or_default();
572            anyhow::bail!("LinkedIn image register failed ({status}): {body_text}");
573        }
574
575        let register_json: serde_json::Value = register_resp
576            .json()
577            .await
578            .context("Failed to parse image register response")?;
579
580        let upload_url = register_json
581            .pointer("/value/uploadUrl")
582            .and_then(|v| v.as_str())
583            .ok_or_else(|| anyhow::anyhow!("Missing uploadUrl in register response"))?
584            .to_string();
585
586        let image_urn = register_json
587            .pointer("/value/image")
588            .and_then(|v| v.as_str())
589            .ok_or_else(|| anyhow::anyhow!("Missing image URN in register response"))?
590            .to_string();
591
592        // Step 2: Upload binary
593        let client = Self::client();
594        let mut upload_headers = HeaderMap::new();
595        upload_headers.insert(
596            reqwest::header::AUTHORIZATION,
597            HeaderValue::from_str(&format!("Bearer {token}")).expect("valid bearer token header"),
598        );
599
600        let upload_resp = client
601            .put(&upload_url)
602            .headers(upload_headers)
603            .header("Content-Type", "image/png")
604            .body(image_bytes.to_vec())
605            .send()
606            .await
607            .context("LinkedIn image upload failed")?;
608
609        let upload_status = upload_resp.status();
610        if !upload_status.is_success() {
611            let body_text = upload_resp.text().await.unwrap_or_default();
612            anyhow::bail!("LinkedIn image upload failed ({upload_status}): {body_text}");
613        }
614
615        Ok(image_urn)
616    }

688
689        let mut updated_lines: Vec<String> = Vec::new();
690        let mut found = false;
691
692        for line in content.lines() {
693            let trimmed = line.trim();
694
695            // Detect the LINKEDIN_ACCESS_TOKEN line (with or without export prefix)
696            let is_token_line = if trimmed.starts_with('#') || trimmed.is_empty() {
697                false
698            } else {
699                let check = trimmed
700                    .strip_prefix("export ")
701                    .map(str::trim)
702                    .unwrap_or(trimmed);
703                check
704                    .split_once('=')
705                    .is_some_and(|(key, _)| key.trim() == "LINKEDIN_ACCESS_TOKEN")
706            };
707
708            if is_token_line {
709                // Preserve the export prefix and quoting style
710                let has_export = trimmed.starts_with("export ");
711                let after_key = trimmed.strip_prefix("export ").unwrap_or(trimmed).trim();
712                let (_key, old_val) = after_key
713                    .split_once('=')
714                    .unwrap_or(("LINKEDIN_ACCESS_TOKEN", ""));
715                let old_val = old_val.trim();
716
717                let new_val = if old_val.starts_with('"') {
718                    format!("\"{}\"", new_token)
719                } else if old_val.starts_with('\'') {
720                    format!("'{}'", new_token)
721                } else {
722                    new_token.to_string()
723                };
724
725                let new_line = if has_export {
726                    format!("export LINKEDIN_ACCESS_TOKEN={}", new_val)
727                } else {
728                    format!("LINKEDIN_ACCESS_TOKEN={}", new_val)
729                };
730
731                updated_lines.push(new_line);
732                found = true;
733            } else {
734                updated_lines.push(line.to_string());
735            }
736        }
737
738        if !found {
739            anyhow::bail!("LINKEDIN_ACCESS_TOKEN not found in .env for update");
740        }
741
742        // Preserve trailing newline if original had one
743        let mut output = updated_lines.join("\n");
744        if content.ends_with('\n') {
745            output.push('\n');
746        }
747
748        tokio::fs::write(&env_path, &output)
749            .await
750            .with_context(|| format!("Failed to write {}", env_path.display()))?;
751
752        Ok(())
753    }

775    /// Generate an image for the given prompt text. Returns the path to the saved PNG/SVG file.
776    pub async fn generate(&self, prompt: &str) -> anyhow::Result<PathBuf> {
777        let image_dir = self.workspace_dir.join(&self.config.temp_dir);
778        tokio::fs::create_dir_all(&image_dir).await?;
779
780        let timestamp = std::time::SystemTime::now()
781            .duration_since(std::time::UNIX_EPOCH)
782            .unwrap_or_default()
783            .as_secs();
784        let base_name = format!("post_{timestamp}");
785
786        // Try each configured provider in order
787        for provider_name in &self.config.providers {
788            let result = match provider_name.as_str() {
789                "stability" => self.try_stability(prompt, &image_dir, &base_name).await,
790                "imagen" => self.try_imagen(prompt, &image_dir, &base_name).await,
791                "dalle" => self.try_dalle(prompt, &image_dir, &base_name).await,
792                "flux" => self.try_flux(prompt, &image_dir, &base_name).await,
793                other => {
794                    tracing::warn!("Unknown image provider '{other}', skipping");
795                    continue;
796                }
797            };
798
799            match result {
800                Ok(path) => {
801                    tracing::info!("Image generated via {provider_name}: {}", path.display());
802                    return Ok(path);
803                }
804                Err(e) => {
805                    tracing::warn!("Image provider '{provider_name}' failed: {e}");
806                }
807            }
808        }
809
810        // All AI providers failed — try SVG fallback
811        if self.config.fallback_card {
812            let svg_path = image_dir.join(format!("{base_name}.svg"));
813            let svg_content = Self::generate_fallback_card(prompt, &self.config.card_accent_color);
814            tokio::fs::write(&svg_path, &svg_content).await?;
815            tracing::info!("Fallback SVG card generated: {}", svg_path.display());
816            return Ok(svg_path);
817        }
818
819        anyhow::bail!("All image generation providers failed and fallback_card is disabled")
820    }

1067    /// Generate a branded SVG text card with the post title on a gradient background.
1068    pub fn generate_fallback_card(title: &str, accent_color: &str) -> String {
1069        // Truncate title to ~80 chars for clean display
1070        let display_title = if title.len() > 80 {
1071            format!("{}...", &title[..77])
1072        } else {
1073            title.to_string()
1074        };
1075
1076        // Word-wrap at ~35 chars per line, max 3 lines
1077        let lines = word_wrap(&display_title, 35, 3);
1078        let line_height: i32 = 48;
1079        // lines.len() is capped at max_lines=3, so this cast is safe
1080        #[allow(clippy::cast_possible_truncation)]
1081        let line_count: i32 = lines.len() as i32;
1082        let total_text_height = line_count * line_height;
1083        let start_y = (1024 - total_text_height) / 2 + 24;
1084
1085        let font = "system-ui, sans-serif";
1086        let text_elements: String = lines
1087            .iter()
1088            .enumerate()
1089            .map(|(i, line)| {
1090                #[allow(clippy::cast_possible_truncation)]
1091                let y = start_y + (i as i32 * line_height); // i is max 2, safe
1092                format!(
1093                    "    <text x=\"512\" y=\"{y}\" text-anchor=\"middle\" fill=\"white\" \
1094                     font-family=\"{font}\" font-size=\"36\" font-weight=\"600\">{}</text>",
1095                    xml_escape(line)
1096                )
1097            })
1098            .collect::<Vec<_>>()
1099            .join("\n");
1100
1101        format!(
1102            "<svg xmlns=\"http://www.w3.org/2000/svg\" width=\"1024\" height=\"1024\" \
1103             viewBox=\"0 0 1024 1024\">\n\
1104             \x20 <defs>\n\
1105             \x20   <linearGradient id=\"bg\" x1=\"0\" y1=\"0\" x2=\"1\" y2=\"1\">\n\
1106             \x20     <stop offset=\"0%\" stop-color=\"{accent_color}\"/>\n\
1107             \x20     <stop offset=\"100%\" stop-color=\"#1a1a2e\"/>\n\
1108             \x20   </linearGradient>\n\
1109             \x20 </defs>\n\
1110             \x20 <rect width=\"1024\" height=\"1024\" fill=\"url(#bg)\" rx=\"0\"/>\n\
1111             \x20 <rect x=\"60\" y=\"60\" width=\"904\" height=\"904\" rx=\"24\" \
1112             fill=\"none\" stroke=\"rgba(255,255,255,0.15)\" stroke-width=\"2\"/>\n\
1113             {text_elements}\n\
1114             \x20 <text x=\"512\" y=\"920\" text-anchor=\"middle\" \
1115             fill=\"rgba(255,255,255,0.5)\" font-family=\"{font}\" \
1116             font-size=\"18\">ZeroClaw</text>\n\
1117             </svg>"
1118        )

1138/// Simple word-wrap: break text into lines of at most `max_width` chars, capped at `max_lines`.
1139fn word_wrap(text: &str, max_width: usize, max_lines: usize) -> Vec<String> {
1140    let mut lines = Vec::new();
1141    let mut current_line = String::new();
1142
1143    for word in text.split_whitespace() {
1144        if current_line.is_empty() {
1145            current_line = word.to_string();
1146        } else if current_line.len() + 1 + word.len() <= max_width {
1147            current_line.push(' ');
1148            current_line.push_str(word);
1149        } else {
1150            lines.push(current_line);
1151            current_line = word.to_string();
1152            if lines.len() >= max_lines {
1153                break;
1154            }
1155        }
1156    }
1157
1158    if !current_line.is_empty() && lines.len() < max_lines {
1159        lines.push(current_line);
1160    }
1161
1162    lines
1163}

1165/// Escape XML special characters for SVG text content.

1167    text.replace('&', "&amp;")
1168        .replace('<', "&lt;")
1169        .replace('>', "&gt;")
1170        .replace('"', "&quot;")
1171        .replace('\'', "&apos;")
1172}
