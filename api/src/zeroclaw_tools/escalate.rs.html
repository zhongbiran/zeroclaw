1//! Human escalation tool with urgency-aware routing.
2//!
3//! Exposes `escalate_to_human` as an agent-callable tool that sends a structured
4//! escalation message to a messaging channel. High/critical urgency escalations
5//! additionally notify any channels listed in `[escalation] alert_channels`.
6//! Supports optional blocking mode to wait for a human response.
7

19const DEFAULT_TIMEOUT_SECS: u64 = 600;
20
21const VALID_URGENCY_LEVELS: &[&str] = &["low", "medium", "high", "critical"];
22
23/// Agent-callable tool for escalating situations to a human operator with urgency routing.
24pub struct EscalateToHumanTool {
25    security: Arc<SecurityPolicy>,
26    channel_map: ChannelMapHandle,
27    alert_channels: Vec<String>,
28}
29
30impl EscalateToHumanTool {
31    pub fn new(security: Arc<SecurityPolicy>, alert_channels: Vec<String>) -> Self {
32        Self {
33            security,
34            channel_map: Arc::new(RwLock::new(HashMap::new())),
35            alert_channels,
36        }
37    }
38
39    /// Return the shared handle so callers can populate it after channel init.
40    pub fn channel_map_handle(&self) -> ChannelMapHandle {
41        Arc::clone(&self.channel_map)
42    }
43
44    /// Format the escalation message with urgency prefix.
45    fn format_message(urgency: &str, summary: &str, context: Option<&str>) -> String {
46        let prefix = match urgency {
47            "low" => "\u{2139}\u{fe0f} [LOW]",
48            "high" => "\u{1f534} [HIGH]",
49            "critical" => "\u{1f6a8} [CRITICAL]",
50            // "medium" and any other value
51            _ => "\u{26a0}\u{fe0f} [MEDIUM]",
52        };
53
54        let mut lines = vec![
55            format!("{prefix} Agent Escalation"),
56            format!("Summary: {summary}"),
57        ];
58
59        if let Some(ctx) = context {
60            lines.push(format!("Context: {ctx}"));
61        }
62
63        lines.push("---".to_string());
64        lines.push("Reply to this message to respond.".to_string());
65
66        lines.join("\n")
67    }
68
69    /// Send best-effort alerts to configured alert channels for high/critical urgency.
70    async fn send_alerts(&self, text: &str) {
71        // Collect Arc clones while holding the lock, then drop the guard before awaiting.
72        let targets: Vec<(String, Arc<dyn Channel>)> = {
73            let channels = self.channel_map.read();
74            self.alert_channels
75                .iter()
76                .filter_map(|name| {
77                    if let Some(ch) = channels.get(name) {
78                        Some((name.clone(), Arc::clone(ch)))
79                    } else {
80                        tracing::warn!(
81                            "escalate_to_human: alert channel '{name}' not found in channel map"
82                        );
83                        None
84                    }
85                })
86                .collect()
87        };
88        for (name, ch) in targets {
89            let msg = SendMessage::new(text, "");
90            if let Err(e) = ch.send(&msg).await {
91                tracing::warn!("escalate_to_human: alert to channel '{name}' failed: {e}");
92            }
93        }
94    }
95}

97#[async_trait]
98impl Tool for EscalateToHumanTool {
99    fn name(&self) -> &str {
100        "escalate_to_human"
101    }
102
103    fn description(&self) -> &str {
104        "Escalate a situation to a human operator with urgency routing. \
105         Sends a structured message to the active channel. High/critical urgency \
106         also notifies any channels listed in `[escalation] alert_channels`. \
107         Optionally blocks to wait for a human response."
108    }
109
110    fn parameters_schema(&self) -> serde_json::Value {
111        json!({
112            "type": "object",
113            "properties": {
114                "summary": {
115                    "type": "string",
116                    "description": "One-line escalation summary"
117                },
118                "context": {
119                    "type": "string",
120                    "description": "Detailed context for the human"
121                },
122                "urgency": {
123                    "type": "string",
124                    "enum": ["low", "medium", "high", "critical"],
125                    "description": "Urgency level (default: medium). high/critical also notifies alert_channels."
126                },
127                "wait_for_response": {
128                    "type": "boolean",
129                    "description": "Block and return the human's reply (default: false)"
130                },
131                "timeout_secs": {
132                    "type": "integer",
133                    "description": "Seconds to wait for a response when wait_for_response is true (default: 600)"
134                }
135            },
136            "required": ["summary"]
137        })
138    }
139
140    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
141        // Security gate
142        if let Err(e) = self
143            .security
144            .enforce_tool_operation(ToolOperation::Act, "escalate_to_human")
145        {
146            return Ok(ToolResult {
147                success: false,
148                output: String::new(),
149                error: Some(format!("Action blocked: {e}")),
150            });
151        }
152
153        // Parse required params
154        let summary = args
155            .get("summary")
156            .and_then(|v| v.as_str())
157            .map(|s| s.trim())
158            .filter(|s| !s.is_empty())
159            .ok_or_else(|| anyhow::anyhow!("Missing 'summary' parameter"))?
160            .to_string();
161
162        let context = args
163            .get("context")
164            .and_then(|v| v.as_str())
165            .map(|s| s.trim().to_string())
166            .filter(|s| !s.is_empty());
167
168        let urgency = args
169            .get("urgency")
170            .and_then(|v| v.as_str())
171            .unwrap_or("medium");
172
173        if !VALID_URGENCY_LEVELS.contains(&urgency) {
174            return Ok(ToolResult {
175                success: false,
176                output: String::new(),
177                error: Some(format!(
178                    "Invalid urgency '{}'. Must be one of: {}",
179                    urgency,
180                    VALID_URGENCY_LEVELS.join(", ")
181                )),
182            });
183        }
184
185        let wait_for_response = args
186            .get("wait_for_response")
187            .and_then(|v| v.as_bool())
188            .unwrap_or(false);
189
190        let timeout_secs = args
191            .get("timeout_secs")
192            .and_then(|v| v.as_u64())
193            .unwrap_or(DEFAULT_TIMEOUT_SECS);
194
195        // Format the message
196        let text = Self::format_message(urgency, &summary, context.as_deref());
197
198        // Resolve channel — block-scoped to drop the RwLock guard before any .await
199        let (channel_name, channel): (String, Arc<dyn Channel>) = {
200            let channels = self.channel_map.read();
201            if channels.is_empty() {
202                return Ok(ToolResult {
203                    success: false,
204                    output: String::new(),
205                    error: Some("No channels available yet (channels not initialized)".to_string()),
206                });
207            }
208            let (name, ch) = channels.iter().next().ok_or_else(|| {
209                anyhow::anyhow!("No channels available. Configure at least one channel.")
210            })?;
211            (name.clone(), ch.clone())
212        };
213
214        // Channels without free-form `listen` support (e.g. ACP today, until
215        // the elicitation RFD lands) can't deliver the human's reply. Fail
216        // fast so the agent can route the escalation differently or proceed
217        // without blocking — the alternative is silently timing out for
218        // `timeout_secs` seconds.
219        // RFD: https://github.com/zed-industries/agent-client-protocol/blob/main/docs/rfds/elicitation.mdx
220        if wait_for_response && !channel.supports_free_form_ask() {
221            return Ok(ToolResult {
222                success: false,
223                output: String::new(),
224                error: Some(format!(
225                    "Channel '{channel_name}' cannot receive a free-form reply, \
226                     so `wait_for_response` is unsupported (awaits ACP elicitation RFD). \
227                     Retry with `wait_for_response: false`."
228                )),
229            });
230        }
231
232        // Send the escalation message
233        let msg = SendMessage::new(&text, "");
234        if let Err(e) = channel.send(&msg).await {
235            return Ok(ToolResult {
236                success: false,
237                output: String::new(),
238                error: Some(format!(
239                    "Failed to send escalation to channel '{channel_name}': {e}"
240                )),
241            });
242        }
243
244        // Notify alert channels for high/critical urgency (non-blocking, best-effort)
245        if (urgency == "high" || urgency == "critical") && !self.alert_channels.is_empty() {
246            self.send_alerts(&text).await;
247        }
248
249        if wait_for_response {
250            // Block and wait for human response (same pattern as ask_user)
251            let (tx, mut rx) = tokio::sync::mpsc::channel::<ChannelMessage>(1);
252            let timeout = std::time::Duration::from_secs(timeout_secs);
253
254            let listen_channel = Arc::clone(&channel);
255            let listen_handle = tokio::spawn(async move { listen_channel.listen(tx).await });
256
257            let response = tokio::time::timeout(timeout, rx.recv()).await;
258            listen_handle.abort();
259
260            match response {
261                Ok(Some(msg)) => Ok(ToolResult {
262                    success: true,
263                    output: msg.content,
264                    error: None,
265                }),
266                Ok(None) => Ok(ToolResult {
267                    success: false,
268                    output: "TIMEOUT".to_string(),
269                    error: Some("Channel closed before receiving a response".to_string()),
270                }),
271                Err(_) => Ok(ToolResult {
272                    success: false,
273                    output: "TIMEOUT".to_string(),
274                    error: Some(format!(
275                        "No response received within {timeout_secs} seconds"
276                    )),
277                }),
278            }
279        } else {
280            // Non-blocking: return confirmation
281            Ok(ToolResult {
282                success: true,
283                output: json!({
284                    "status": "escalated",
285                    "urgency": urgency,
286                    "channel": channel_name,
287                })
288                .to_string(),
289                error: None,
290            })
291        }
292    }

383    fn make_tool_with_channels(channels: Vec<(&str, Arc<dyn Channel>)>) -> EscalateToHumanTool {
384        let tool = EscalateToHumanTool::new(Arc::new(SecurityPolicy::default()), vec![]);
385        let map: HashMap<String, Arc<dyn Channel>> = channels
386            .into_iter()
387            .map(|(name, ch)| (name.to_string(), ch))
388            .collect();
389        *tool.channel_map.write() = map;
390        tool

597    #[tokio::test]
598    async fn wait_for_response_fails_fast_on_structured_only_channel() {
599        // ACP-shaped channel: can't listen, so wait_for_response must fail
600        // immediately rather than timing out silently.
601        let stub = Arc::new(StructuredOnlyChannel::new("acp"));
602        let stub_clone: Arc<dyn Channel> = stub.clone();
603        let tool = make_tool_with_channels(vec![("acp", stub_clone)]);
604
605        let started = std::time::Instant::now();
606        let result = tool
607            .execute(json!({
608                "summary": "Need confirmation",
609                "wait_for_response": true,
610                "timeout_secs": 30,
611            }))
612            .await
613            .unwrap();
614        let elapsed = started.elapsed();
615
616        assert!(!result.success, "expected failure, got: {:?}", result);
617        let err = result.error.unwrap_or_default();
618        assert!(
619            err.contains("wait_for_response"),
620            "error should mention wait_for_response: {err}"
621        );
622        // Must fail fast — well under the 30s timeout.
623        assert!(
624            elapsed < std::time::Duration::from_secs(2),
625            "expected fast-fail; took {elapsed:?}"
626        );
627        // No message should have been sent — gate fires before send.
628        assert!(stub.sent.read().is_empty());
629    }
