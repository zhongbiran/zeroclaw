17/// Google Workspace CLI (`gws`) integration tool.
18///
19/// Wraps the `gws` CLI binary to give the agent structured access to
20/// Google Workspace services (Drive, Gmail, Calendar, Sheets, etc.).
21/// Requires `gws` to be installed and authenticated (`gws auth login`).
22pub struct GoogleWorkspaceTool {
23    security: Arc<SecurityPolicy>,
24    allowed_services: Vec<String>,
25    allowed_operations: Vec<GoogleWorkspaceAllowedOperation>,
26    credentials_path: Option<String>,
27    default_account: Option<String>,
28    #[allow(dead_code)] // Config field for future rate-limiting
29    rate_limit_per_minute: u32,
30    timeout_secs: u64,
31    audit_log: bool,
32}
33
34impl GoogleWorkspaceTool {
35    /// Create a new `GoogleWorkspaceTool`.
36    ///
37    /// If `allowed_services` is empty, the default service set is used.
38    pub fn new(
39        security: Arc<SecurityPolicy>,
40        allowed_services: Vec<String>,
41        allowed_operations: Vec<GoogleWorkspaceAllowedOperation>,
42        credentials_path: Option<String>,
43        default_account: Option<String>,
44        rate_limit_per_minute: u32,
45        timeout_secs: u64,
46        audit_log: bool,
47    ) -> Self {
48        let services = if allowed_services.is_empty() {
49            DEFAULT_GWS_SERVICES
50                .iter()
51                .map(|s| (*s).to_string())
52                .collect()
53        } else {
54            allowed_services
55                .into_iter()
56                .map(|s| s.trim().to_string())
57                .collect()
58        };
59        // Normalize stored operation fields at construction time so runtime
60        // comparisons can use plain equality without repeated .trim() calls.
61        let operations = allowed_operations
62            .into_iter()
63            .map(|op| GoogleWorkspaceAllowedOperation {
64                service: op.service.trim().to_string(),
65                resource: op.resource.trim().to_string(),
66                sub_resource: op.sub_resource.as_deref().map(|s| s.trim().to_string()),
67                methods: op.methods.iter().map(|m| m.trim().to_string()).collect(),
68            })
69            .collect();
70        Self {
71            security,
72            allowed_services: services,
73            allowed_operations: operations,
74            credentials_path,
75            default_account,
76            rate_limit_per_minute,
77            timeout_secs,
78            audit_log,
79        }
80    }
81
82    /// Build the positional `gws` arguments: `[service, resource, (sub_resource,)? method]`.
83    fn positional_cmd_args(
84        service: &str,
85        resource: &str,
86        sub_resource: Option<&str>,
87        method: &str,
88    ) -> Vec<String> {
89        let mut args = vec![service.to_string(), resource.to_string()];
90        if let Some(sub) = sub_resource {
91            args.push(sub.to_string());
92        }
93        args.push(method.to_string());
94        args
95    }
96
97    /// Build the `--page-all` and `--page-limit` flags from validated pagination inputs.
98    /// `page_limit` alone (without `page_all`) caps page count; both together fetch all pages
99    /// up to the limit.
100    fn build_pagination_args(page_all: bool, page_limit: Option<u64>) -> Vec<String> {
101        let mut args = Vec::new();
102        if page_all {
103            args.push("--page-all".into());
104        }
105        if page_all || page_limit.is_some() {
106            args.push("--page-limit".into());
107            args.push(page_limit.unwrap_or(10).to_string());
108        }
109        args
110    }
111
112    fn is_operation_allowed(
113        &self,
114        service: &str,
115        resource: &str,
116        sub_resource: Option<&str>,
117        method: &str,
118    ) -> bool {
119        if self.allowed_operations.is_empty() {
120            return true;
121        }
122        self.allowed_operations.iter().any(|operation| {
123            operation.service == service
124                && operation.resource == resource
125                && operation.sub_resource.as_deref() == sub_resource
126                && operation.methods.iter().any(|allowed| allowed == method)
127        })
128    }
129}

193    /// Execute a Google Workspace CLI command with input validation and security enforcement.
194    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
195        let service = args
196            .get("service")
197            .and_then(|v| v.as_str())
198            .ok_or_else(|| anyhow::anyhow!("Missing 'service' parameter"))?;
199        let resource = args
200            .get("resource")
201            .and_then(|v| v.as_str())
202            .ok_or_else(|| anyhow::anyhow!("Missing 'resource' parameter"))?;
203        let method = args
204            .get("method")
205            .and_then(|v| v.as_str())
206            .ok_or_else(|| anyhow::anyhow!("Missing 'method' parameter"))?;
207
208        // Extract and validate sub_resource early so the allowlist check can account for it.
209        let sub_resource: Option<&str> = if let Some(sub_resource_value) = args.get("sub_resource")
210        {
211            let s = match sub_resource_value.as_str() {
212                Some(s) => s,
213                None => {
214                    return Ok(ToolResult {
215                        success: false,
216                        output: String::new(),
217                        error: Some("'sub_resource' must be a string".into()),
218                    });
219                }
220            };
221            if !s
222                .chars()
223                .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '_' || c == '-')
224            {
225                return Ok(ToolResult {
226                    success: false,
227                    output: String::new(),
228                    error: Some(
229                        "Invalid characters in 'sub_resource': only lowercase alphanumeric, underscore, and hyphen are allowed"
230                            .into(),
231                    ),
232                });
233            }
234            Some(s)
235        } else {
236            None
237        };
238
239        // Security checks
240        if self.security.is_rate_limited() {
241            return Ok(ToolResult {
242                success: false,
243                output: String::new(),
244                error: Some("Rate limit exceeded: too many actions in the last hour".into()),
245            });
246        }
247
248        // Validate service is in the allowlist
249        if !self.allowed_services.iter().any(|s| s == service) {
250            return Ok(ToolResult {
251                success: false,
252                output: String::new(),
253                error: Some(format!(
254                    "Service '{service}' is not in the allowed services list. \
255                     Allowed: {}",
256                    self.allowed_services.join(", ")
257                )),
258            });
259        }
260
261        if !self.is_operation_allowed(service, resource, sub_resource, method) {
262            let op_path = match sub_resource {
263                Some(sub) => format!("{service}/{resource}/{sub}/{method}"),
264                None => format!("{service}/{resource}/{method}"),
265            };
266            return Ok(ToolResult {
267                success: false,
268                output: String::new(),
269                error: Some(format!(
270                    "Operation '{op_path}' is not in the allowed operations list"
271                )),
272            });
273        }
274
275        // Validate inputs contain no shell metacharacters
276        for (label, value) in [
277            ("service", service),
278            ("resource", resource),
279            ("method", method),
280        ] {
281            if !value
282                .chars()
283                .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '_' || c == '-')
284            {
285                return Ok(ToolResult {
286                    success: false,
287                    output: String::new(),
288                    error: Some(format!(
289                        "Invalid characters in '{label}': only lowercase alphanumeric, underscore, and hyphen are allowed"
290                    )),
291                });
292            }
293        }
294
295        // Build the gws command — validate all optional fields before consuming budget
296        let mut cmd_args = Self::positional_cmd_args(service, resource, sub_resource, method);
297
298        if let Some(params) = args.get("params") {
299            if !params.is_object() {
300                return Ok(ToolResult {
301                    success: false,
302                    output: String::new(),
303                    error: Some("'params' must be an object".into()),
304                });
305            }
306            cmd_args.push("--params".into());
307            cmd_args.push(params.to_string());
308        }
309
310        if let Some(body) = args.get("body") {
311            if !body.is_object() {
312                return Ok(ToolResult {
313                    success: false,
314                    output: String::new(),
315                    error: Some("'body' must be an object".into()),
316                });
317            }
318            cmd_args.push("--json".into());
319            cmd_args.push(body.to_string());
320        }
321
322        if let Some(format_value) = args.get("format") {
323            let format = match format_value.as_str() {
324                Some(s) => s,
325                None => {
326                    return Ok(ToolResult {
327                        success: false,
328                        output: String::new(),
329                        error: Some("'format' must be a string".into()),
330                    });
331                }
332            };
333            match format {
334                "json" | "table" | "yaml" | "csv" => {
335                    cmd_args.push("--format".into());
336                    cmd_args.push(format.to_string());
337                }
338                _ => {
339                    return Ok(ToolResult {
340                        success: false,
341                        output: String::new(),
342                        error: Some(format!(
343                            "Invalid format '{format}': must be json, table, yaml, or csv"
344                        )),
345                    });
346                }
347            }
348        }
349
350        let page_all = match args.get("page_all") {
351            Some(v) => match v.as_bool() {
352                Some(b) => b,
353                None => {
354                    return Ok(ToolResult {
355                        success: false,
356                        output: String::new(),
357                        error: Some("'page_all' must be a boolean".into()),
358                    });
359                }
360            },
361            None => false,
362        };
363        let page_limit = match args.get("page_limit") {
364            Some(v) => match v.as_u64() {
365                Some(n) => Some(n),
366                None => {
367                    return Ok(ToolResult {
368                        success: false,
369                        output: String::new(),
370                        error: Some("'page_limit' must be a non-negative integer".into()),
371                    });
372                }
373            },
374            None => None,
375        };
376        cmd_args.extend(Self::build_pagination_args(page_all, page_limit));
377
378        // Charge action budget only after all validation passes
379        if !self.security.record_action() {
380            return Ok(ToolResult {
381                success: false,
382                output: String::new(),
383                error: Some("Rate limit exceeded: action budget exhausted".into()),
384            });
385        }
386
387        // Resolve `gws` via PATH so Windows `.cmd` shims (e.g. npm-installed
388        // `gws.cmd`) are picked up — `Command::new` on Windows does not append
389        // PATHEXT itself. Falls back to bare "gws" so the not-found error path
390        // below still fires when the binary is genuinely missing.
391        let gws_path: std::path::PathBuf = which::which("gws").unwrap_or_else(|_| "gws".into());
392        let mut cmd = tokio::process::Command::new(gws_path);
393        cmd.args(&cmd_args);
394        cmd.env_clear();
395        // gws needs PATH to find itself and HOME/APPDATA for credential storage
396        for key in &["PATH", "HOME", "APPDATA", "USERPROFILE", "LANG", "TERM"] {
397            if let Ok(val) = std::env::var(key) {
398                cmd.env(key, val);
399            }
400        }
401
402        // Apply credential path if configured
403        if let Some(ref creds) = self.credentials_path {
404            cmd.env("GOOGLE_APPLICATION_CREDENTIALS", creds);
405        }
406
407        // Apply default account if configured
408        if let Some(ref account) = self.default_account {
409            cmd.args(["--account", account]);
410        }
411
412        if self.audit_log {
413            tracing::info!(
414                tool = "google_workspace",
415                service = service,
416                resource = resource,
417                sub_resource = sub_resource.unwrap_or(""),
418                method = method,
419                "gws audit: executing API call"
420            );
421        }
422
423        let result =
424            tokio::time::timeout(Duration::from_secs(self.timeout_secs), cmd.output()).await;
425
426        match result {
427            Ok(Ok(output)) => {
428                let mut stdout = String::from_utf8_lossy(&output.stdout).to_string();
429                let mut stderr = String::from_utf8_lossy(&output.stderr).to_string();
430
431                if stdout.len() > MAX_OUTPUT_BYTES {
432                    // Find a valid char boundary at or before MAX_OUTPUT_BYTES
433                    let mut boundary = MAX_OUTPUT_BYTES;
434                    while boundary > 0 && !stdout.is_char_boundary(boundary) {
435                        boundary -= 1;
436                    }
437                    stdout.truncate(boundary);
438                    stdout.push_str("\n... [output truncated at 1MB]");
439                }
440                if stderr.len() > MAX_OUTPUT_BYTES {
441                    let mut boundary = MAX_OUTPUT_BYTES;
442                    while boundary > 0 && !stderr.is_char_boundary(boundary) {
443                        boundary -= 1;
444                    }
445                    stderr.truncate(boundary);
446                    stderr.push_str("\n... [stderr truncated at 1MB]");
447                }
448
449                Ok(ToolResult {
450                    success: output.status.success(),
451                    output: stdout,
452                    error: if stderr.is_empty() {
453                        None
454                    } else {
455                        Some(stderr)
456                    },
457                })
458            }
459            Ok(Err(e)) => Ok(ToolResult {
460                success: false,
461                output: String::new(),
462                error: Some(format!(
463                    "Failed to execute gws: {e}. Is gws installed? Run: npm install -g @googleworkspace/cli"
464                )),
465            }),
466            Err(_) => Ok(ToolResult {
467                success: false,
468                output: String::new(),
469                error: Some(format!(
470                    "gws command timed out after {}s and was killed",
471                    self.timeout_secs
472                )),
473            }),
474        }
475    }
