9/// Let the agent store memories — its own brain writes
10pub struct MemoryStoreTool {
11    memory: Arc<dyn Memory>,
12    security: Arc<SecurityPolicy>,
13}
14
15impl MemoryStoreTool {
16    pub fn new(memory: Arc<dyn Memory>, security: Arc<SecurityPolicy>) -> Self {
17        Self { memory, security }
18    }
19}
20
21#[async_trait]
22impl Tool for MemoryStoreTool {
23    fn name(&self) -> &str {
24        "memory_store"
25    }
26
27    fn description(&self) -> &str {
28        "Store a fact, preference, or note in long-term memory. Use category 'core' for permanent facts, 'daily' for session notes, 'conversation' for chat context, or a custom category name."
29    }
30
31    fn parameters_schema(&self) -> serde_json::Value {
32        json!({
33            "type": "object",
34            "properties": {
35                "key": {
36                    "type": "string",
37                    "description": "Unique key for this memory (e.g. 'user_lang', 'project_stack')"
38                },
39                "content": {
40                    "type": "string",
41                    "description": "The information to remember"
42                },
43                "category": {
44                    "type": "string",
45                    "description": "Memory category: 'core' (permanent), 'daily' (session), 'conversation' (chat), or a custom category name. Defaults to 'core'."
46                }
47            },
48            "required": ["key", "content"]
49        })
50    }
51
52    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
53        let key = args
54            .get("key")
55            .and_then(|v| v.as_str())
56            .ok_or_else(|| anyhow::anyhow!("Missing 'key' parameter"))?;
57
58        let content = args
59            .get("content")
60            .and_then(|v| v.as_str())
61            .ok_or_else(|| anyhow::anyhow!("Missing 'content' parameter"))?;
62
63        let category = match args.get("category").and_then(|v| v.as_str()) {
64            Some("core") | None => MemoryCategory::Core,
65            Some("daily") => MemoryCategory::Daily,
66            Some("conversation") => MemoryCategory::Conversation,
67            Some(other) => MemoryCategory::Custom(other.to_string()),
68        };
69
70        if let Err(error) = self
71            .security
72            .enforce_tool_operation(ToolOperation::Act, "memory_store")
73        {
74            return Ok(ToolResult {
75                success: false,
76                output: String::new(),
77                error: Some(error),
78            });
79        }
80
81        match self.memory.store(key, content, category, None).await {
82            Ok(()) => Ok(ToolResult {
83                success: true,
84                output: format!("Stored memory: {key}"),
85                error: None,
86            }),
87            Err(e) => Ok(ToolResult {
88                success: false,
89                output: String::new(),
90                error: Some(format!("Failed to store memory: {e}")),
91            }),
92        }
93    }
94}

97mod tests {
98    use super::*;
99    use tempfile::TempDir;

104    fn test_security() -> Arc<SecurityPolicy> {
105        Arc::new(SecurityPolicy::default())

108    fn test_mem() -> (TempDir, Arc<dyn Memory>) {
109        let tmp = TempDir::new().unwrap();
110        let mem = SqliteMemory::new(tmp.path()).unwrap();
111        (tmp, Arc::new(mem))
112    }

114    #[test]
115    fn name_and_schema() {
116        let (_tmp, mem) = test_mem();
117        let tool = MemoryStoreTool::new(mem, test_security());
118        assert_eq!(tool.name(), "memory_store");
119        let schema = tool.parameters_schema();
120        assert!(schema["properties"]["key"].is_object());
121        assert!(schema["properties"]["content"].is_object());
122    }

124    #[tokio::test]
125    async fn store_core() {
126        let (_tmp, mem) = test_mem();
127        let tool = MemoryStoreTool::new(mem.clone(), test_security());
128        let result = tool
129            .execute(json!({"key": "lang", "content": "Prefers Rust"}))
130            .await
131            .unwrap();
132        assert!(result.success);
133        assert!(result.output.contains("lang"));
134
135        let entry = mem.get("lang").await.unwrap();
136        assert!(entry.is_some());
137        assert_eq!(entry.unwrap().content, "Prefers Rust");
138    }
