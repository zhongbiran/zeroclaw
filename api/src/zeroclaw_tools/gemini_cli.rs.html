11/// Environment variables safe to pass through to the `gemini` subprocess.
12const SAFE_ENV_VARS: &[&str] = &[
13    "PATH", "HOME", "TERM", "LANG", "LC_ALL", "LC_CTYPE", "USER", "SHELL", "TMPDIR",
14];

16/// Delegates coding tasks to the Gemini CLI (`gemini -p`).
17///
18/// This creates a two-tier agent architecture: ZeroClaw orchestrates high-level
19/// tasks and delegates complex coding work to Gemini CLI, which has its own
20/// agent loop with file editing and shell tools.
21///
22/// Authentication uses the `gemini` binary's own session by default. No API key
23/// is needed unless `env_passthrough` includes `GOOGLE_API_KEY`.
24pub struct GeminiCliTool {
25    security: Arc<SecurityPolicy>,
26    config: GeminiCliConfig,
27}
28
29impl GeminiCliTool {
30    pub fn new(security: Arc<SecurityPolicy>, config: GeminiCliConfig) -> Self {
31        Self { security, config }
32    }
33}

35#[async_trait]
36impl Tool for GeminiCliTool {
37    fn name(&self) -> &str {
38        "gemini_cli"
39    }
40
41    fn description(&self) -> &str {
42        "Delegate a coding task to Gemini CLI (gemini -p). Supports file editing and shell execution. Use for complex coding work that benefits from Gemini CLI's full agent loop."
43    }
44
45    fn parameters_schema(&self) -> serde_json::Value {
46        json!({
47            "type": "object",
48            "properties": {
49                "prompt": {
50                    "type": "string",
51                    "description": "The coding task to delegate to Gemini CLI"
52                },
53                "working_directory": {
54                    "type": "string",
55                    "description": "Working directory within the workspace (must be inside workspace_dir)"
56                }
57            },
58            "required": ["prompt"]
59        })
60    }
61
62    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
63        // Rate limiting is applied by the RateLimitedTool wrapper at
64        // registration time (see zeroclaw-runtime::tools::mod).
65
66        // Enforce act policy
67        if let Err(error) = self
68            .security
69            .enforce_tool_operation(ToolOperation::Act, "gemini_cli")
70        {
71            return Ok(ToolResult {
72                success: false,
73                output: String::new(),
74                error: Some(error),
75            });
76        }
77
78        // Extract prompt (required)
79        let prompt = args
80            .get("prompt")
81            .and_then(|v| v.as_str())
82            .ok_or_else(|| anyhow::anyhow!("Missing 'prompt' parameter"))?;
83
84        // Validate working directory — require both paths to exist (reject
85        // non-existent paths instead of falling back to the raw value, which
86        // could bypass the workspace containment check via symlinks or
87        // specially-crafted path components).
88        let work_dir = if let Some(wd) = args.get("working_directory").and_then(|v| v.as_str()) {
89            let wd_path = std::path::PathBuf::from(wd);
90            let workspace = &self.security.workspace_dir;
91            let canonical_wd = match wd_path.canonicalize() {
92                Ok(p) => p,
93                Err(_) => {
94                    return Ok(ToolResult {
95                        success: false,
96                        output: String::new(),
97                        error: Some(format!(
98                            "working_directory '{}' does not exist or is not accessible",
99                            wd
100                        )),
101                    });
102                }
103            };
104            let canonical_ws = match workspace.canonicalize() {
105                Ok(p) => p,
106                Err(_) => {
107                    return Ok(ToolResult {
108                        success: false,
109                        output: String::new(),
110                        error: Some(format!(
111                            "workspace directory '{}' does not exist or is not accessible",
112                            workspace.display()
113                        )),
114                    });
115                }
116            };
117            if !canonical_wd.starts_with(&canonical_ws) {
118                return Ok(ToolResult {
119                    success: false,
120                    output: String::new(),
121                    error: Some(format!(
122                        "working_directory '{}' is outside the workspace '{}'",
123                        wd,
124                        workspace.display()
125                    )),
126                });
127            }
128            canonical_wd
129        } else {
130            self.security.workspace_dir.clone()
131        };
132
133        // Build CLI command
134        let gemini_bin = if cfg!(target_os = "windows") {
135            "gemini.cmd"
136        } else {
137            "gemini"
138        };
139        let mut cmd = Command::new(gemini_bin);
140        cmd.arg("-p").arg(prompt);
141
142        // Environment: clear everything, pass only safe vars + configured passthrough.
143        cmd.env_clear();
144        for var in SAFE_ENV_VARS {
145            if let Ok(val) = std::env::var(var) {
146                cmd.env(var, val);
147            }
148        }
149        for var in &self.config.env_passthrough {
150            let trimmed = var.trim();
151            if !trimmed.is_empty()
152                && let Ok(val) = std::env::var(trimmed)
153            {
154                cmd.env(trimmed, val);
155            }
156        }
157
158        cmd.current_dir(&work_dir);
159        // Execute with timeout — use kill_on_drop(true) so the child process
160        // is automatically killed when the future is dropped on timeout,
161        // preventing zombie processes.
162        let timeout = Duration::from_secs(self.config.timeout_secs);
163        cmd.kill_on_drop(true);
164
165        let result = tokio::time::timeout(timeout, cmd.output()).await;
166
167        match result {
168            Ok(Ok(output)) => {
169                let mut stdout = String::from_utf8_lossy(&output.stdout).to_string();
170                let stderr = String::from_utf8_lossy(&output.stderr).to_string();
171
172                // Truncate to max_output_bytes with char-boundary safety
173                if stdout.len() > self.config.max_output_bytes {
174                    let mut b = self.config.max_output_bytes.min(stdout.len());
175                    while b > 0 && !stdout.is_char_boundary(b) {
176                        b -= 1;
177                    }
178                    stdout.truncate(b);
179                    stdout.push_str("\n... [output truncated]");
180                }
181
182                Ok(ToolResult {
183                    success: output.status.success(),
184                    output: stdout,
185                    error: if stderr.is_empty() {
186                        None
187                    } else {
188                        Some(stderr)
189                    },
190                })
191            }
192            Ok(Err(e)) => {
193                let err_msg = e.to_string();
194                let msg = if err_msg.contains("No such file or directory")
195                    || err_msg.contains("not found")
196                    || err_msg.contains("cannot find")
197                {
198                    "Gemini CLI ('gemini') not found in PATH. Install with: npm install -g @google/gemini-cli or see https://github.com/google-gemini/gemini-cli".into()
199                } else {
200                    format!("Failed to execute gemini: {e}")
201                };
202                Ok(ToolResult {
203                    success: false,
204                    output: String::new(),
205                    error: Some(msg),
206                })
207            }
208            Err(_) => {
209                // Timeout — kill_on_drop(true) ensures the child is killed
210                // when the future is dropped.
211                Ok(ToolResult {
212                    success: false,
213                    output: String::new(),
214                    error: Some(format!(
215                        "Gemini CLI timed out after {}s and was killed",
216                        self.config.timeout_secs
217                    )),
218                })
219            }
220        }
221    }
222}
