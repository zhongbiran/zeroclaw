7/// Edit a file by replacing an exact string match with new content.
8///
9/// Uses `old_string` → `new_string` precise replacement within the workspace.
10/// The `old_string` must appear exactly once in the file (zero matches = not
11/// found, multiple matches = ambiguous). `new_string` may be empty to delete
12/// the matched text. Security checks mirror [`super::file_write::FileWriteTool`].
13pub struct FileEditTool {
14    security: Arc<SecurityPolicy>,
15}
16
17impl FileEditTool {
18    pub fn new(security: Arc<SecurityPolicy>) -> Self {
19        Self { security }
20    }
21}
22
23#[async_trait]
24impl Tool for FileEditTool {
25    fn name(&self) -> &str {
26        "file_edit"
27    }
28
29    fn description(&self) -> &str {
30        "Edit a file by replacing an exact string match with new content"
31    }
32
33    fn parameters_schema(&self) -> serde_json::Value {
34        json!({
35            "type": "object",
36            "properties": {
37                "path": {
38                    "type": "string",
39                    "description": "Path to the file. Relative paths resolve from workspace; outside paths require policy allowlist."
40                },
41                "old_string": {
42                    "type": "string",
43                    "description": "The exact text to find and replace (must appear exactly once in the file)"
44                },
45                "new_string": {
46                    "type": "string",
47                    "description": "The replacement text (empty string to delete the matched text)"
48                }
49            },
50            "required": ["path", "old_string", "new_string"]
51        })
52    }
53
54    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
55        // ── 1. Extract parameters ──────────────────────────────────
56        let path = args
57            .get("path")
58            .and_then(|v| v.as_str())
59            .ok_or_else(|| anyhow::anyhow!("Missing 'path' parameter"))?;
60
61        let old_string = args
62            .get("old_string")
63            .and_then(|v| v.as_str())
64            .ok_or_else(|| anyhow::anyhow!("Missing 'old_string' parameter"))?;
65
66        let new_string = args
67            .get("new_string")
68            .and_then(|v| v.as_str())
69            .ok_or_else(|| anyhow::anyhow!("Missing 'new_string' parameter"))?;
70
71        if old_string.is_empty() {
72            return Ok(ToolResult {
73                success: false,
74                output: String::new(),
75                error: Some("old_string must not be empty".into()),
76            });
77        }
78
79        // ── 2. Autonomy check ──────────────────────────────────────
80        if !self.security.can_act() {
81            return Ok(ToolResult {
82                success: false,
83                output: String::new(),
84                error: Some("Action blocked: autonomy is read-only".into()),
85            });
86        }
87
88        // Rate limiting and path-allowlist checks are applied by the
89        // RateLimitedTool + PathGuardedTool wrappers at registration time
90        // (see zeroclaw-runtime::tools::mod).
91
92        let full_path = self.security.resolve_tool_path(path);
93
94        // ── 5. Canonicalise parent ─────────────────────────────────
95        let Some(parent) = full_path.parent() else {
96            return Ok(ToolResult {
97                success: false,
98                output: String::new(),
99                error: Some("Invalid path: missing parent directory".into()),
100            });
101        };
102
103        let resolved_parent = match tokio::fs::canonicalize(parent).await {
104            Ok(p) => p,
105            Err(e) => {
106                return Ok(ToolResult {
107                    success: false,
108                    output: String::new(),
109                    error: Some(format!("Failed to resolve file path: {e}")),
110                });
111            }
112        };
113
114        // ── 6. Resolved path post-validation ───────────────────────
115        if !self.security.is_resolved_path_allowed(&resolved_parent) {
116            return Ok(ToolResult {
117                success: false,
118                output: String::new(),
119                error: Some(
120                    self.security
121                        .resolved_path_violation_message(&resolved_parent),
122                ),
123            });
124        }
125
126        let Some(file_name) = full_path.file_name() else {
127            return Ok(ToolResult {
128                success: false,
129                output: String::new(),
130                error: Some("Invalid path: missing file name".into()),
131            });
132        };
133
134        let resolved_target = resolved_parent.join(file_name);
135
136        if self.security.is_runtime_config_path(&resolved_target) {
137            return Ok(ToolResult {
138                success: false,
139                output: String::new(),
140                error: Some(
141                    self.security
142                        .runtime_config_violation_message(&resolved_target),
143                ),
144            });
145        }
146
147        // ── 7. Symlink check ───────────────────────────────────────
148        if let Ok(meta) = tokio::fs::symlink_metadata(&resolved_target).await
149            && meta.file_type().is_symlink()
150        {
151            return Ok(ToolResult {
152                success: false,
153                output: String::new(),
154                error: Some(format!(
155                    "Refusing to edit through symlink: {}",
156                    resolved_target.display()
157                )),
158            });
159        }
160
161        // ── 9. Read → match → replace → write ─────────────────────
162        let content = match tokio::fs::read_to_string(&resolved_target).await {
163            Ok(c) => c,
164            Err(e) => {
165                return Ok(ToolResult {
166                    success: false,
167                    output: String::new(),
168                    error: Some(format!("Failed to read file: {e}")),
169                });
170            }
171        };
172
173        let match_count = content.matches(old_string).count();
174
175        if match_count == 0 {
176            return Ok(ToolResult {
177                success: false,
178                output: String::new(),
179                error: Some("old_string not found in file".into()),
180            });
181        }
182
183        if match_count > 1 {
184            return Ok(ToolResult {
185                success: false,
186                output: String::new(),
187                error: Some(format!(
188                    "old_string matches {match_count} times; must match exactly once"
189                )),
190            });
191        }
192
193        let new_content = content.replacen(old_string, new_string, 1);
194
195        match tokio::fs::write(&resolved_target, &new_content).await {
196            Ok(()) => Ok(ToolResult {
197                success: true,
198                output: format!(
199                    "Edited {path}: replaced 1 occurrence ({} bytes)",
200                    new_content.len()
201                ),
202                error: None,
203            }),
204            Err(e) => Ok(ToolResult {
205                success: false,
206                output: String::new(),
207                error: Some(format!("Failed to write file: {e}")),
208            }),
209        }
210    }
211}

229    /// Wraps `FileEditTool` with the production `PathGuardedTool` + `RateLimitedTool`
230    /// stack, mirroring the registration in `zeroclaw-runtime::tools::mod`. Use this
231    /// in tests that exercise path-allowlist or rate-limit behavior.
232    fn wrapped_tool(workspace: std::path::PathBuf) -> Box<dyn Tool> {
233        let security = Arc::new(SecurityPolicy {
234            autonomy: AutonomyLevel::Supervised,
235            workspace_dir: workspace,
236            ..SecurityPolicy::default()
237        });
238        Box::new(RateLimitedTool::new(
239            PathGuardedTool::new(FileEditTool::new(security.clone()), security.clone()),
240            security,
241        ))
242    }
