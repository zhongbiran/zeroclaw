1//! Built-in `tool_search` tool for on-demand MCP tool schema loading.
2//!
3//! When `mcp.deferred_loading` is enabled, this tool lets the LLM discover and
4//! activate deferred MCP tools. Supports two query modes:
5//! - `select:name1,name2` — fetch exact tools by prefixed name.
6//! - Free-text keyword search — returns the best-matching stubs.
7
8use std::fmt::Write;
9use std::sync::{Arc, Mutex};
10
11use async_trait::async_trait;
12
13use crate::mcp_deferred::{ActivatedToolSet, DeferredMcpToolSet};
14use zeroclaw_api::tool::{Tool, ToolResult};

16/// Default maximum number of search results.
17const DEFAULT_MAX_RESULTS: usize = 5;
18
19/// Built-in tool that fetches full schemas for deferred MCP tools.
20pub struct ToolSearchTool {
21    deferred: DeferredMcpToolSet,
22    activated: Arc<Mutex<ActivatedToolSet>>,
23}
24
25impl ToolSearchTool {
26    pub fn new(deferred: DeferredMcpToolSet, activated: Arc<Mutex<ActivatedToolSet>>) -> Self {
27        Self {
28            deferred,
29            activated,
30        }
31    }
32}
33
34#[async_trait]
35impl Tool for ToolSearchTool {
36    fn name(&self) -> &str {
37        "tool_search"
38    }
39
40    fn description(&self) -> &str {
41        "Fetch full schema definitions for deferred MCP tools so they can be called. \
42         Use \"select:name1,name2\" for exact match or keywords to search."
43    }
44
45    fn parameters_schema(&self) -> serde_json::Value {
46        serde_json::json!({
47            "type": "object",
48            "properties": {
49                "query": {
50                    "description": "Query to find deferred tools. Use \"select:<tool_name>\" for direct selection, or keywords to search.",
51                    "type": "string"
52                },
53                "max_results": {
54                    "description": "Maximum number of results to return (default: 5)",
55                    "type": "number",
56                    "default": DEFAULT_MAX_RESULTS
57                }
58            },
59            "required": ["query"]
60        })
61    }
62
63    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
64        let query = args
65            .get("query")
66            .and_then(|v| v.as_str())
67            .unwrap_or_default()
68            .trim();
69
70        let max_results = args
71            .get("max_results")
72            .and_then(|v| v.as_u64())
73            .map(|v| usize::try_from(v).unwrap_or(DEFAULT_MAX_RESULTS))
74            .unwrap_or(DEFAULT_MAX_RESULTS);
75
76        if query.is_empty() {
77            return Ok(ToolResult {
78                success: false,
79                output: String::new(),
80                error: Some("query parameter is required".into()),
81            });
82        }
83
84        // Parse query mode
85        if let Some(names_str) = query.strip_prefix("select:") {
86            // Exact selection mode
87            let names: Vec<&str> = names_str.split(',').map(str::trim).collect();
88            return self.select_tools(&names);
89        }
90
91        // Keyword search mode
92        let results = self.deferred.search(query, max_results);
93        if results.is_empty() {
94            return Ok(ToolResult {
95                success: true,
96                output: "No matching deferred tools found.".into(),
97                error: None,
98            });
99        }
100
101        // Activate and return full specs
102        let mut output = String::from("<functions>\n");
103        let mut activated_count = 0;
104        let mut guard = self.activated.lock().unwrap();
105
106        for stub in &results {
107            if let Some(spec) = self.deferred.tool_spec(&stub.prefixed_name) {
108                if !guard.is_activated(&stub.prefixed_name)
109                    && let Some(tool) = self.deferred.activate(&stub.prefixed_name)
110                {
111                    guard.activate(stub.prefixed_name.clone(), Arc::from(tool));
112                    activated_count += 1;
113                }
114                let _ = writeln!(
115                    output,
116                    "<function>{{\"name\": \"{}\", \"description\": \"{}\", \"parameters\": {}}}</function>",
117                    spec.name,
118                    spec.description.replace('"', "\\\""),
119                    spec.parameters
120                );
121            }
122        }
123
124        output.push_str("</functions>\n");
125        drop(guard);
126
127        tracing::debug!(
128            "tool_search: query={query:?}, matched={}, activated={activated_count}",
129            results.len()
130        );
131
132        Ok(ToolResult {
133            success: true,
134            output,
135            error: None,
136        })
137    }
138}

140impl ToolSearchTool {
141    fn select_tools(&self, names: &[&str]) -> anyhow::Result<ToolResult> {
142        let mut output = String::from("<functions>\n");
143        let mut not_found = Vec::new();
144        let mut activated_count = 0;
145        let mut guard = self.activated.lock().unwrap();
146
147        for name in names {
148            if name.is_empty() {
149                continue;
150            }
151            match self.deferred.tool_spec(name) {
152                Some(spec) => {
153                    if !guard.is_activated(name)
154                        && let Some(tool) = self.deferred.activate(name)
155                    {
156                        guard.activate(String::from(*name), Arc::from(tool));
157                        activated_count += 1;
158                    }
159                    let _ = writeln!(
160                        output,
161                        "<function>{{\"name\": \"{}\", \"description\": \"{}\", \"parameters\": {}}}</function>",
162                        spec.name,
163                        spec.description.replace('"', "\\\""),
164                        spec.parameters
165                    );
166                }
167                None => {
168                    not_found.push(*name);
169                }
170            }
171        }
172
173        output.push_str("</functions>\n");
174        drop(guard);
175
176        if !not_found.is_empty() {
177            let _ = write!(output, "\nNot found: {}", not_found.join(", "));
178        }
179
180        tracing::debug!(
181            "tool_search select: requested={}, activated={activated_count}, not_found={}",
182            names.len(),
183            not_found.len()
184        );
185
186        Ok(ToolResult {
187            success: true,
188            output,
189            error: None,
190        })
191    }
192}

201    async fn make_deferred_set(stubs: Vec<DeferredMcpToolStub>) -> DeferredMcpToolSet {

210            input_schema: serde_json::json!({"type": "object", "properties": {}}),

316    /// Verify select mode activates tools and they stay activated across calls,
317    /// matching the daemon-mode pattern where a single ActivatedToolSet persists.
318    #[tokio::test]
319    async fn select_activates_and_persists_across_calls() {
320        let activated = Arc::new(Mutex::new(ActivatedToolSet::new()));
321        let stubs = vec![
322            make_stub("srv__tool_a", "Tool A"),
323            make_stub("srv__tool_b", "Tool B"),
324        ];
325        let tool = ToolSearchTool::new(make_deferred_set(stubs).await, Arc::clone(&activated));
326
327        // Activate tool_a
328        let result = tool
329            .execute(serde_json::json!({"query": "select:srv__tool_a"}))
330            .await
331            .unwrap();
332        assert!(result.success);
333        assert!(activated.lock().unwrap().is_activated("srv__tool_a"));
334        assert!(!activated.lock().unwrap().is_activated("srv__tool_b"));
335
336        // Activate tool_b in a separate call
337        let result = tool
338            .execute(serde_json::json!({"query": "select:srv__tool_b"}))
339            .await
340            .unwrap();
341        assert!(result.success);
342
343        // Both should remain activated
344        let guard = activated.lock().unwrap();
345        assert!(guard.is_activated("srv__tool_a"));
346        assert!(guard.is_activated("srv__tool_b"));
347        assert_eq!(guard.tool_specs().len(), 2);
348    }
