171    /// Resolve whether `channel_id` is a Discord thread (ANNOUNCEMENT,
172    /// PUBLIC, or PRIVATE thread) via `GET /channels/{id}`. Results are
173    /// cached for the channel instance's lifetime: thread-ness is stable
174    /// for a given channel ID, so one lookup per ID per process. Failures
175    /// (network, 429, missing `type` field) fall through to `false` so a
176    /// transient API hiccup never blocks inbound delivery.
177    async fn is_thread_channel(&self, client: &reqwest::Client, channel_id: &str) -> bool {
178        {
179            let cache = self.thread_channels.lock().await;
180            if let Some(&value) = cache.get(channel_id) {
181                return value;
182            }
183        }
184
185        // Only a successful API response is cached. A transient network blip
186        // or 429 must not poison the cache for the channel's lifetime; the
187        // next message should retry the lookup. Failure paths return `false`
188        // (the safe default) without writing to the cache. The whole request
189        // is wrapped in an explicit timeout so a hung Discord API call can
190        // never stall the listener; the shared channel HTTP client may not
191        // carry a request-level timeout.
192        let url = format!("https://discord.com/api/v10/channels/{channel_id}");
193        let lookup = async {
194            let resp = client
195                .get(&url)
196                .header("Authorization", format!("Bot {}", self.bot_token))
197                .send()
198                .await
199                .map_err(|e| anyhow!("request failed: {e}"))?;
200            if !resp.status().is_success() {
201                anyhow::bail!("non-success status {}", resp.status());
202            }
203            let body: serde_json::Value = resp
204                .json()
205                .await
206                .map_err(|e| anyhow!("body parse failed: {e}"))?;
207            Ok::<bool, anyhow::Error>(
208                body.get("type")
209                    .and_then(serde_json::Value::as_u64)
210                    .map(is_thread_channel_type)
211                    .unwrap_or(false),
212            )
213        };
214        let is_thread = match tokio::time::timeout(THREAD_LOOKUP_TIMEOUT, lookup).await {
215            Ok(Ok(value)) => value,
216            Ok(Err(e)) => {
217                tracing::debug!(channel_id, error = %e, "discord: channel lookup failed");
218                return false;
219            }
220            Err(_) => {
221                tracing::debug!(
222                    channel_id,
223                    timeout_secs = THREAD_LOOKUP_TIMEOUT.as_secs(),
224                    "discord: channel lookup timed out"
225                );
226                return false;
227            }
228        };
229
230        self.thread_channels
231            .lock()
232            .await
233            .insert(channel_id.to_string(), is_thread);
234        is_thread
235    }

274/// Process Discord message attachments in a single pass.
275///
276/// Returns the text block appended to the agent's prompt and the structured
277/// `MediaAttachment` list consumed by the media pipeline. Each attachment is
278/// downloaded at most once: text/* is inlined as text, audio is transcribed
279/// inline when a transcription manager is configured (otherwise it goes
280/// through the media pipeline), and image/video/document attachments are
281/// saved to the workspace and emitted as `[KIND:<path>]` markers plus a
282/// `MediaAttachment` for vision-capable providers.
283async fn process_attachments(
284    attachments: &[serde_json::Value],
285    client: &reqwest::Client,
286    workspace_dir: Option<&Path>,
287    transcription_manager: Option<&super::transcription::TranscriptionManager>,
288) -> (String, Vec<MediaAttachment>) {
289    let mut text_parts: Vec<String> = Vec::new();
290    let mut media: Vec<MediaAttachment> = Vec::new();
291
292    for att in attachments {
293        let ct = att
294            .get("content_type")
295            .and_then(|v| v.as_str())
296            .unwrap_or("");
297        let name = att
298            .get("filename")
299            .and_then(|v| v.as_str())
300            .unwrap_or("file");
301        let Some(url) = att.get("url").and_then(|v| v.as_str()) else {
302            tracing::warn!(name, "discord: attachment has no url, skipping");
303            continue;
304        };
305
306        if ct.starts_with("text/") {
307            match client.get(url).send().await {
308                Ok(resp) if resp.status().is_success() => {
309                    if let Ok(text) = resp.text().await {
310                        text_parts.push(format!("[{name}]\n{text}"));
311                    }
312                }
313                Ok(resp) => {
314                    tracing::warn!(name, status = %resp.status(), "discord attachment fetch failed");
315                }
316                Err(e) => {
317                    tracing::warn!(name, error = %e, "discord attachment fetch error");
318                }
319            }
320            continue;
321        }
322
323        let is_audio = is_discord_audio_attachment(ct, name);
324
325        // Audio with channel-level transcription configured: transcribe
326        // inline so the agent receives `[Voice] <transcript>` text rather
327        // than opaque bytes through the media pipeline.
328        if is_audio && let Some(manager) = transcription_manager {
329            let bytes = match download_attachment_bytes(client, url, name).await {
330                Some(b) => b,
331                None => continue,
332            };
333            match manager.transcribe(&bytes, name).await {
334                Ok(text) => {
335                    let trimmed = text.trim();
336                    if !trimmed.is_empty() {
337                        tracing::info!(
338                            "Discord: transcribed audio attachment {} ({} chars)",
339                            name,
340                            trimmed.len()
341                        );
342                        text_parts.push(format!("[Voice] {trimmed}"));
343                    }
344                }
345                Err(e) => {
346                    tracing::warn!(name, error = %e, "discord: voice transcription failed");
347                }
348            }
349            continue;
350        }
351
352        let marker_kind = marker_kind_for(ct, is_audio);
353
354        let bytes = match download_attachment_bytes(client, url, name).await {
355            Some(b) => b,
356            None => continue,
357        };
358
359        let marker_target = match workspace_dir {
360            Some(dir) => match save_attachment_bytes_to_workspace(dir, name, &bytes).await {
361                Ok(local_path) => local_path.display().to_string(),
362                Err(e) => {
363                    tracing::warn!(name, kind = marker_kind, error = %e, "discord: attachment save failed, falling back to url");
364                    url.to_string()
365                }
366            },
367            None => url.to_string(),
368        };
369        text_parts.push(format!("[{marker_kind}:{marker_target}]"));
370
371        media.push(MediaAttachment {
372            file_name: name.to_string(),
373            data: bytes,
374            mime_type: if ct.is_empty() {
375                None
376            } else {
377                Some(ct.to_string())
378            },
379        });
380    }
381
382    (text_parts.join("\n---\n"), media)
383}

402            None
403        }
404        Err(e) => {
405            tracing::warn!(name, error = %e, "discord: attachment fetch error");
406            None
407        }
408    }

411async fn save_attachment_bytes_to_workspace(
412    workspace_dir: &Path,
413    filename: &str,
414    bytes: &[u8],
415) -> anyhow::Result<PathBuf> {
416    let save_dir = workspace_dir.join("discord_files");
417    tokio::fs::create_dir_all(&save_dir).await?;
418
419    let safe_name = Path::new(filename)
420        .file_name()
421        .and_then(|name| name.to_str())
422        .filter(|name| !name.is_empty())
423        .unwrap_or("attachment");
424    let local_name = format!("{}_{}", Uuid::new_v4(), safe_name);
425    let local_path = save_dir.join(local_name);
426
427    tokio::fs::write(&local_path, bytes).await?;
428    Ok(local_path)
429}

501fn parse_attachment_markers(message: &str) -> (String, Vec<DiscordAttachment>) {
502    let mut cleaned = String::with_capacity(message.len());
503    let mut attachments = Vec::new();
504    let mut cursor = 0usize;
505
506    while let Some(rel_start) = message[cursor..].find('[') {
507        let start = cursor + rel_start;
508        cleaned.push_str(&message[cursor..start]);
509
510        let Some(rel_end) = message[start..].find(']') else {
511            cleaned.push_str(&message[start..]);
512            cursor = message.len();
513            break;
514        };
515        let end = start + rel_end;
516        let marker_text = &message[start + 1..end];
517
518        let parsed = marker_text.split_once(':').and_then(|(kind, target)| {
519            let kind = DiscordAttachmentKind::from_marker(kind)?;
520            let target = target.trim();
521            if target.is_empty() {
522                return None;
523            }
524            Some(DiscordAttachment {
525                kind,
526                target: target.to_string(),
527            })
528        });
529
530        if let Some(attachment) = parsed {
531            attachments.push(attachment);
532        } else {
533            cleaned.push_str(&message[start..=end]);
534        }
535
536        cursor = end + 1;
537    }
538
539    if cursor < message.len() {
540        cleaned.push_str(&message[cursor..]);
541    }
542
543    (cleaned.trim().to_string(), attachments)
544}

553/// Why a marker target was rejected. Drives the user-facing emoji reaction
554/// on the bot's outgoing message: `Refused` (trust-boundary rejection) maps
555/// to 🚫, `NotFound` (path didn't resolve on disk) maps to ⚠️. The
556/// distinction matters because a chatter should see at a glance that the
557/// bot deliberately declined a target rather than tried and failed.
558#[derive(Debug, Clone, Copy, PartialEq, Eq)]
559enum DiscordMarkerFailure {
560    /// Trust-boundary refusal: disallowed scheme, relative path, missing
561    /// workspace_dir, or canonicalised path outside the workspace.
562    Refused,
563    /// Path passed scheme/absolute/workspace checks but did not resolve
564    /// to anything on disk.
565    NotFound,
566}
567
568#[derive(Debug)]
569enum DiscordMarkerError {
570    Refused(anyhow::Error),
571    NotFound(anyhow::Error),
572}
573
574impl DiscordMarkerError {
575    fn kind(&self) -> DiscordMarkerFailure {
576        match self {
577            Self::Refused(_) => DiscordMarkerFailure::Refused,
578            Self::NotFound(_) => DiscordMarkerFailure::NotFound,
579        }
580    }
581}

591/// Validate an outbound marker target against Discord's trust-boundary policy.
592///
593/// The orchestrator system prompt mandates absolute paths for media markers,
594/// and the workspace is the only directory the agent is authorised to
595/// expose to chatters:
596///
597/// * `http`/`https` URLs are accepted and inlined as links.
598/// * Any other URL scheme (`file:`, `data:`, custom `://`) is refused.
599/// * Local paths must be absolute. Relative paths are agent
600///   misconfiguration and dropped, not silently resolved against cwd.
601/// * Absolute paths are canonicalised and must resolve inside
602///   `workspace_dir`. Anything outside or any traversal escape is
603///   refused; a path that simply doesn't exist on disk returns
604///   `NotFound`, which the caller renders differently from a refusal.
605/// * When `workspace_dir` is not configured, no local path can be safely
606///   bounded, so all local targets are refused.
607fn validate_marker_target(
608    target: &str,
609    workspace_dir: Option<&Path>,
610) -> Result<DiscordMarkerTarget, DiscordMarkerError> {
611    if target.starts_with("http://") || target.starts_with("https://") {
612        return Ok(DiscordMarkerTarget::Http(target.to_string()));
613    }
614    if target.contains("://") {
615        let scheme = target.split("://").next().unwrap_or("?");
616        return Err(DiscordMarkerError::Refused(anyhow!(
617            "discord: marker target uses disallowed scheme {scheme:?}; only http/https and absolute workspace paths are accepted"
618        )));
619    }
620    if target.starts_with("data:") || target.starts_with("file:") {
621        return Err(DiscordMarkerError::Refused(anyhow!(
622            "discord: marker target uses disallowed scheme; only http/https and absolute workspace paths are accepted"
623        )));
624    }
625
626    let target_path = Path::new(target);
627    if !target_path.is_absolute() {
628        return Err(DiscordMarkerError::Refused(anyhow!(
629            "discord: marker target {target} is not an absolute path; the agent must emit absolute paths inside workspace_dir"
630        )));
631    }
632
633    let workspace = workspace_dir.ok_or_else(|| {
634        DiscordMarkerError::Refused(anyhow!(
635            "discord: marker target {target} is a local path but the channel was started without a workspace_dir, refusing for safety"
636        ))
637    })?;
638    let workspace_canon = std::fs::canonicalize(workspace)
639        .with_context(|| format!("canonicalize workspace {}", workspace.display()))
640        .map_err(DiscordMarkerError::Refused)?;
641    let target_canon = match std::fs::canonicalize(target_path) {
642        Ok(p) => p,
643        Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
644            return Err(DiscordMarkerError::NotFound(anyhow!(
645                "discord: marker target {target} not found on disk"
646            )));
647        }
648        Err(e) => {
649            return Err(DiscordMarkerError::Refused(
650                anyhow::Error::from(e).context(format!("canonicalize marker target {target}")),
651            ));
652        }
653    };
654
655    if !target_canon.starts_with(&workspace_canon) {
656        return Err(DiscordMarkerError::Refused(anyhow!(
657            "discord: marker target {target} resolves to {} which is outside workspace_dir {}; refusing",
658            target_canon.display(),
659            workspace_canon.display(),
660        )));
661    }
662    Ok(DiscordMarkerTarget::Local(target_canon))
663}

665fn classify_outgoing_attachments(
666    attachments: &[DiscordAttachment],
667    workspace_dir: Option<&Path>,
668) -> (
669    Vec<PathBuf>,
670    Vec<String>,
671    Vec<(String, DiscordMarkerFailure)>,
672) {
673    let mut local_files = Vec::new();
674    let mut remote_urls = Vec::new();
675    let mut failures = Vec::new();
676
677    for attachment in attachments {
678        match validate_marker_target(&attachment.target, workspace_dir) {
679            Ok(DiscordMarkerTarget::Local(path)) => local_files.push(path),
680            Ok(DiscordMarkerTarget::Http(url)) => remote_urls.push(url),
681            Err(e) => {
682                let kind_label = match e.kind() {
683                    DiscordMarkerFailure::Refused => "trust boundary",
684                    DiscordMarkerFailure::NotFound => "not found",
685                };
686                tracing::warn!(
687                    kind = attachment.kind.marker_name(),
688                    target = %attachment.target,
689                    reason = kind_label,
690                    error = %e,
691                    "discord: dropping unresolved outbound attachment marker"
692                );
693                failures.push((attachment.target.clone(), e.kind()));
694            }
695        }
696    }
697
698    (local_files, remote_urls, failures)
699}

926/// Discord's maximum message length for regular messages.
927///
928/// Discord rejects longer payloads with `50035 Invalid Form Body`.
929const DISCORD_MAX_MESSAGE_LENGTH: usize = 2000;
930const DISCORD_ACK_REACTIONS: &[&str] = &["⚡️", "🦀", "🙌", "💪", "👌", "👀", "👣"];
931
932/// Split a message into chunks that respect Discord's 2000-character limit.
933/// Tries to split at word boundaries when possible.
934fn split_message_for_discord(message: &str) -> Vec<String> {
935    if message.chars().count() <= DISCORD_MAX_MESSAGE_LENGTH {
936        return vec![message.to_string()];
937    }
938
939    let mut chunks = Vec::new();
940    let mut remaining = message;
941
942    while !remaining.is_empty() {
943        // Find the byte offset for the 2000th character boundary.
944        // If there are fewer than 2000 chars left, we can emit the tail directly.
945        let hard_split = remaining
946            .char_indices()
947            .nth(DISCORD_MAX_MESSAGE_LENGTH)
948            .map_or(remaining.len(), |(idx, _)| idx);
949
950        let chunk_end = if hard_split == remaining.len() {
951            hard_split
952        } else {
953            // Try to find a good break point (newline, then space)
954            let search_area = &remaining[..hard_split];
955
956            // Prefer splitting at newline
957            if let Some(pos) = search_area.rfind('\n') {
958                // Don't split if the newline is too close to the end
959                if search_area[..pos].chars().count() >= DISCORD_MAX_MESSAGE_LENGTH / 2 {
960                    pos + 1
961                } else {
962                    // Try space as fallback
963                    search_area.rfind(' ').map_or(hard_split, |space| space + 1)
964                }
965            } else if let Some(pos) = search_area.rfind(' ') {
966                pos + 1
967            } else {
968                // Hard split at the limit
969                hard_split
970            }
971        };
972
973        chunks.push(remaining[..chunk_end].to_string());
974        remaining = &remaining[chunk_end..];
975    }
976
977    chunks
978}

980/// Split a message into multiple logical chunks at paragraph boundaries for
981/// multi-message delivery. Respects code fences — never splits inside a
982/// fenced code block. Falls back to [`split_message_for_discord`] for any
983/// segment that exceeds `max_len`.
984fn split_message_for_discord_multi(content: &str, max_len: usize) -> Vec<String> {
985    if content.is_empty() {
986        return vec![];
987    }
988
989    // Gather paragraph-level segments, respecting code fences.
990    let mut segments: Vec<String> = Vec::new();
991    let mut current = String::new();
992    let mut in_fence = false;
993
994    for line in content.lines() {
995        let trimmed = line.trim_start();
996        if trimmed.starts_with("```") {
997            in_fence = !in_fence;
998        }
999
1000        // If we hit a blank line outside a fence, that's a paragraph break.
1001        if line.is_empty() && !in_fence && !current.is_empty() {
1002            segments.push(current.trim_end().to_string());
1003            current.clear();
1004            continue;
1005        }
1006
1007        if !current.is_empty() {
1008            current.push('\n');
1009        }
1010        current.push_str(line);
1011    }
1012    if !current.is_empty() {
1013        segments.push(current.trim_end().to_string());
1014    }
1015
1016    // Now coalesce small segments and split oversized ones.
1017    let mut chunks: Vec<String> = Vec::new();
1018
1019    for segment in segments {
1020        if segment.chars().count() > max_len {
1021            // This segment (possibly a large code fence) exceeds the limit.
1022            // Fall back to the word-boundary splitter.
1023            let sub_chunks = split_message_for_discord(&segment);
1024            chunks.extend(sub_chunks);
1025        } else {
1026            chunks.push(segment);
1027        }
1028    }
1029
1030    if chunks.is_empty() {
1031        vec![content.to_string()]
1032    } else {
1033        chunks
1034    }
1035}

1037/// Choose the chunks to deliver for an outbound Discord message.
1038///
1039/// `split_message_for_discord_multi` returns an empty vec for empty input
1040/// (its paragraph splitter has no segments to emit); the non-multi
1041/// splitter returns `vec![""]`. When MultiMessage stream mode hands
1042/// `send()` a paragraph that collapses to empty text after marker strip,
1043/// the chunk loop would iterate zero times and silently skip an attached
1044/// file upload. Force a single empty chunk in exactly that case so the
1045/// multipart POST fires.
1046fn chunks_for_send(
1047    content: &str,
1048    stream_mode: zeroclaw_config::schema::StreamMode,
1049    max_len: usize,
1050    has_local_files: bool,
1051) -> Vec<String> {
1052    let mut chunks = match stream_mode {
1053        zeroclaw_config::schema::StreamMode::MultiMessage => {
1054            split_message_for_discord_multi(content, max_len)
1055        }
1056        _ => split_message_for_discord(content),
1057    };
1058    if chunks.is_empty() && has_local_files {
1059        chunks.push(String::new());
1060    }
1061    chunks
1062}

1064fn pick_uniform_index(len: usize) -> usize {
1065    debug_assert!(len > 0);
1066    let upper = len as u64;
1067    let reject_threshold = (u64::MAX / upper) * upper;
1068
1069    loop {
1070        let value = rand::random::<u64>();
1071        if value < reject_threshold {
1072            #[allow(clippy::cast_possible_truncation)]
1073            return (value % upper) as usize;
1074        }
1075    }
1076}

1148/// Minimal base64 decode (no extra dep) — only needs to decode the user ID portion
1149#[allow(clippy::cast_possible_truncation)]
1150fn base64_decode(input: &str) -> Option<String> {
1151    let padded = match input.len() % 4 {
1152        2 => format!("{input}=="),
1153        3 => format!("{input}="),
1154        _ => input.to_string(),
1155    };
1156
1157    let mut bytes = Vec::new();
1158    let chars: Vec<u8> = padded.bytes().collect();
1159
1160    for chunk in chars.chunks(4) {
1161        if chunk.len() < 4 {
1162            break;
1163        }
1164
1165        let mut v = [0usize; 4];
1166        for (i, &b) in chunk.iter().enumerate() {
1167            if b == b'=' {
1168                v[i] = 0;
1169            } else {
1170                v[i] = BASE64_ALPHABET.iter().position(|&a| a == b)?;
1171            }
1172        }
1173
1174        bytes.push(((v[0] << 2) | (v[1] >> 4)) as u8);
1175        if chunk[2] != b'=' {
1176            bytes.push((((v[1] & 0xF) << 4) | (v[2] >> 2)) as u8);
1177        }
1178        if chunk[3] != b'=' {
1179            bytes.push((((v[2] & 0x3) << 6) | v[3]) as u8);
1180        }
1181    }
1182
1183    String::from_utf8(bytes).ok()
1184}

1192    async fn send(&self, message: &SendMessage) -> anyhow::Result<()> {

1194        let (cleaned_content, parsed_attachments) = parse_attachment_markers(&raw_content);
1195        let (mut local_files, remote_urls, failures) =
1196            classify_outgoing_attachments(&parsed_attachments, self.workspace_dir.as_deref());
1197
1198        // Discord accepts max 10 files per message.

1201                count = local_files.len(),

1205        }
1206
1207        let body = with_inline_attachment_urls(&cleaned_content, &remote_urls);
1208        let note = delivery_failure_note(&failures);
1209        let content = compose_body_with_failure_note(&body, note.as_deref());
1210        let reactions = decide_failure_reactions(&failures);
1211

1213        let chunks = chunks_for_send(
1214            &content,
1215            self.stream_mode,
1216            DISCORD_MAX_MESSAGE_LENGTH,
1217            !local_files.is_empty(),
1218        );
1219        let inter_chunk_delay_ms =
1220            if self.stream_mode == zeroclaw_config::schema::StreamMode::MultiMessage {
1221                self.multi_message_delay_ms
1222            } else {
1223                500
1224            };
1225
1226        let mut first_message_id: Option<String> = None;
1227        for (i, chunk) in chunks.iter().enumerate() {
1228            let message_id = if i == 0 && !local_files.is_empty() {
1229                send_discord_message_with_files(
1230                    &client,
1231                    &self.bot_token,
1232                    &message.recipient,
1233                    chunk,

1237            } else {

1239                    .await?
1240            };

1242                first_message_id = Some(message_id);
1243            }
1244
1245            if i < chunks.len() - 1 {
1246                if message
1247                    .cancellation_token
1248                    .as_ref()
1249                    .is_some_and(|t| t.is_cancelled())
1250                {
1251                    tracing::debug!(
1252                        "Discord delivery interrupted after chunk {}/{}",
1253                        i + 1,
1254                        chunks.len()
1255                    );
1256                    break;
1257                }
1258                tokio::time::sleep(std::time::Duration::from_millis(inter_chunk_delay_ms)).await;
1259            }
1260        }
1261
1262        self.apply_failure_reactions(&message.recipient, first_message_id.as_deref(), &reactions)
1263            .await;

1266    }

1283            .get("url")
1284            .and_then(|u| u.as_str())
1285            .unwrap_or("wss://gateway.discord.gg");
1286
1287        let ws_url = format!("{gw_url}/?v=10&encoding=json");
1288        tracing::info!("Discord: connecting to gateway...");
1289
1290        let (ws_stream, _) = zeroclaw_config::schema::ws_connect_with_proxy(
1291            &ws_url,
1292            "channel.discord",
1293            self.proxy_url.as_deref(),
1294        )
1295        .await?;
1296        let (mut write, mut read) = ws_stream.split();
1297
1298        // Read Hello (opcode 10)
1299        let hello = read.next().await.ok_or(anyhow::anyhow!("No hello"))??;
1300        let hello_data: serde_json::Value = serde_json::from_str(&hello.to_string())?;
1301        let heartbeat_interval = hello_data
1302            .get("d")
1303            .and_then(|d| d.get("heartbeat_interval"))
1304            .and_then(serde_json::Value::as_u64)
1305            .unwrap_or(41250);
1306
1307        // Send Identify (opcode 2)
1308        let identify = json!({
1309            "op": 2,
1310            "d": {
1311                "token": self.bot_token,
1312                "intents": 37377, // GUILDS | GUILD_MESSAGES | MESSAGE_CONTENT | DIRECT_MESSAGES
1313                "properties": {
1314                    "os": "linux",
1315                    "browser": "zeroclaw",
1316                    "device": "zeroclaw"
1317                }
1318            }
1319        });
1320        write
1321            .send(Message::Text(identify.to_string().into()))
1322            .await?;
1323
1324        tracing::info!("Discord: connected and identified");
1325
1326        // Track the last sequence number for heartbeats and resume.
1327        // Only accessed in the select! loop below, so a plain i64 suffices.
1328        let mut sequence: i64 = -1;
1329
1330        // Spawn heartbeat timer — sends a tick signal, actual heartbeat
1331        // is assembled in the select! loop where `sequence` lives.
1332        let (hb_tx, mut hb_rx) = tokio::sync::mpsc::channel::<()>(1);
1333        let hb_interval = heartbeat_interval;
1334        tokio::spawn(async move {
1335            let mut interval = tokio::time::interval(std::time::Duration::from_millis(hb_interval));
1336            loop {
1337                interval.tick().await;
1338                if hb_tx.send(()).await.is_err() {
1339                    break;
1340                }
1341            }
1342        });
1343
1344        let guild_filter = self.guild_id.clone();
1345
1346        // --- Stall watchdog --------------------------------------------------
1347        let watchdog = if self.stall_timeout_secs > 0 {
1348            Some(zeroclaw_infra::stall_watchdog::StallWatchdog::new(
1349                self.stall_timeout_secs,
1350            ))
1351        } else {
1352            None
1353        };
1354
1355        let (stall_tx, mut stall_rx) = tokio::sync::mpsc::channel::<()>(1);
1356        if let Some(ref wd) = watchdog {
1357            let stall_signal = stall_tx.clone();
1358            wd.start(move || {
1359                tracing::warn!("Discord: stall watchdog fired — no events for configured timeout, triggering reconnect");
1360                let _ = stall_signal.try_send(());
1361            })
1362            .await;
1363        }
1364        // Keep stall_tx alive so the receiver doesn't close prematurely when
1365        // the watchdog is disabled (recv will just pend forever).
1366        let _stall_tx_guard = stall_tx;
1367
1368        loop {
1369            tokio::select! {
1370                _ = stall_rx.recv() => {
1371                    tracing::info!("Discord: breaking listen loop due to stall watchdog");
1372                    break;
1373                }
1374                _ = hb_rx.recv() => {
1375                    let d = if sequence >= 0 { json!(sequence) } else { json!(null) };
1376                    let hb = json!({"op": 1, "d": d});
1377                    if write.send(Message::Text(hb.to_string().into())).await.is_err() {
1378                        break;
1379                    }
1380                }
1381                msg = read.next() => {
1382                    let msg = match msg {
1383                        Some(Ok(Message::Text(t))) => t,
1384                        Some(Ok(Message::Ping(payload))) => {
1385                            if write.send(Message::Pong(payload)).await.is_err() {
1386                                tracing::warn!("Discord: pong send failed, reconnecting");
1387                                break;
1388                            }
1389                            continue;
1390                        }
1391                        Some(Ok(Message::Close(_))) | None => break,
1392                        Some(Err(e)) => {
1393                            tracing::warn!("Discord: websocket read error: {e}, reconnecting");
1394                            break;
1395                        }
1396                        _ => continue,
1397                    };
1398
1399                    let event: serde_json::Value = match serde_json::from_str(msg.as_ref()) {
1400                        Ok(e) => e,
1401                        Err(_) => continue,
1402                    };
1403
1404                    // Mark activity for the stall watchdog on every
1405                    // successfully parsed gateway event.
1406                    if let Some(ref wd) = watchdog {
1407                        wd.touch();
1408                    }
1409
1410                    // Track sequence number from all dispatch events
1411                    if let Some(s) = event.get("s").and_then(serde_json::Value::as_i64) {
1412                        sequence = s;
1413                    }
1414
1415                    let op = event.get("op").and_then(serde_json::Value::as_u64).unwrap_or(0);
1416
1417                    match op {
1418                        // Op 1: Server requests an immediate heartbeat
1419                        1 => {
1420                            let d = if sequence >= 0 { json!(sequence) } else { json!(null) };
1421                            let hb = json!({"op": 1, "d": d});
1422                            if write.send(Message::Text(hb.to_string().into())).await.is_err() {
1423                                break;
1424                            }
1425                            continue;
1426                        }
1427                        // Op 7: Reconnect
1428                        7 => {
1429                            tracing::warn!("Discord: received Reconnect (op 7), closing for restart");
1430                            break;
1431                        }
1432                        // Op 9: Invalid Session
1433                        9 => {
1434                            tracing::warn!("Discord: received Invalid Session (op 9), closing for restart");
1435                            break;
1436                        }
1437                        _ => {}
1438                    }
1439
1440                    // Only handle MESSAGE_CREATE (opcode 0, type "MESSAGE_CREATE")
1441                    let event_type = event.get("t").and_then(|t| t.as_str()).unwrap_or("");
1442                    if event_type != "MESSAGE_CREATE" {
1443                        continue;
1444                    }
1445
1446                    let Some(d) = event.get("d") else {
1447                        continue;
1448                    };
1449
1450                    // Skip messages from the bot itself
1451                    let author_id = d.get("author").and_then(|a| a.get("id")).and_then(|i| i.as_str()).unwrap_or("");
1452                    if author_id == bot_user_id {
1453                        continue;
1454                    }
1455
1456                    // Skip bot messages (unless listen_to_bots is enabled)
1457                    if !self.listen_to_bots && d.get("author").and_then(|a| a.get("bot")).and_then(serde_json::Value::as_bool).unwrap_or(false) {
1458                        continue;
1459                    }
1460
1461                    // Sender validation
1462                    if !self.is_user_allowed(author_id) {
1463                        tracing::warn!("Discord: ignoring message from unauthorized user: {author_id}");
1464                        continue;
1465                    }
1466
1467                    // Guild filter
1468                    if let Some(ref gid) = guild_filter {
1469                        let msg_guild = d.get("guild_id").and_then(serde_json::Value::as_str);
1470                        // DMs have no guild_id — let them through; for guild messages, enforce the filter
1471                        if let Some(g) = msg_guild
1472                            && g != gid {
1473                                continue;
1474                            }
1475                    }
1476
1477                    let content = d.get("content").and_then(|c| c.as_str()).unwrap_or("");
1478                    // DMs carry no guild_id in the Discord gateway payload. They are
1479                    // inherently private and implicitly addressed to the bot, so bypass
1480                    // the mention gate — requiring a @mention in a DM is never correct.
1481                    let is_dm = d.get("guild_id").is_none();
1482                    let effective_mention_only = self.mention_only && !is_dm;
1483                    let atts = d
1484                        .get("attachments")
1485                        .and_then(|a| a.as_array())
1486                        .cloned()
1487                        .unwrap_or_default();
1488                    let has_attachments = !atts.is_empty();
1489                    let Some(clean_content) = admit_discord_message(
1490                        content,
1491                        has_attachments,
1492                        effective_mention_only,
1493                        &bot_user_id,
1494                    ) else {
1495                        continue;
1496                    };
1497
1498                    let client = self.http_client();
1499                    let (attachment_text, media_attachments) = process_attachments(
1500                        &atts,
1501                        &client,
1502                        self.workspace_dir.as_deref(),
1503                        self.transcription_manager.as_deref(),
1504                    )
1505                    .await;
1506                    let final_content = if attachment_text.is_empty() {
1507                        clean_content
1508                    } else {
1509                        format!("{clean_content}\n\n[Attachments]\n{attachment_text}")
1510                    };
1511
1512                    // Intercept approval replies before forwarding to the agent.
1513                    if let Some((token, response)) =
1514                        crate::util::parse_approval_reply(&final_content)
1515                    {
1516                        let mut map = self.pending_approvals.lock().await;
1517                        if let Some(sender) = map.remove(&token) {
1518                            let _ = sender.send(response);
1519                            continue;
1520                        }
1521                    }
1522
1523                    let message_id = d.get("id").and_then(|i| i.as_str()).unwrap_or("");
1524                    let channel_id = d
1525                        .get("channel_id")
1526                        .and_then(|c| c.as_str())
1527                        .unwrap_or("")
1528                        .to_string();
1529
1530                    if !message_id.is_empty() && !channel_id.is_empty() {
1531                        let reaction_channel = DiscordChannel::new(
1532                            self.bot_token.clone(),
1533                            self.guild_id.clone(),
1534                            self.allowed_users.clone(),
1535                            self.listen_to_bots,
1536                            self.mention_only,
1537                        );
1538                        let reaction_channel_id = channel_id.clone();
1539                        let reaction_message_id = message_id.to_string();
1540                        let reaction_emoji = random_discord_ack_reaction().to_string();
1541                        tokio::spawn(async move {
1542                            if let Err(err) = reaction_channel
1543                                .add_reaction(
1544                                    &reaction_channel_id,
1545                                    &reaction_message_id,
1546                                    &reaction_emoji,
1547                                )
1548                                .await
1549                            {
1550                                tracing::debug!(
1551                                    "Discord: failed to add ACK reaction for message {reaction_message_id}: {err}"
1552                                );
1553                            }
1554                        });
1555                    }
1556
1557                    // Thread context decides `thread_ts` plus `interruption_scope_id`,
1558                    // which the orchestrator uses as part of the conversation-history
1559                    // key and the cancellation scope. When the lookup fails it falls
1560                    // back to `None` and the failure is not cached, so the next
1561                    // message in the same Discord thread will retry. The trade-off:
1562                    // the first message after a transient lookup miss is keyed
1563                    // without the thread suffix; once the cache warms, subsequent
1564                    // messages are keyed with it. History for that thread can split
1565                    // across two scopes until the warm-up completes. Acceptable
1566                    // because the lookup is bounded by `THREAD_LOOKUP_TIMEOUT` and
1567                    // the alternative (stalling the listener on a hung Discord call)
1568                    // is worse.
1569                    let thread_ts = if channel_id.is_empty() {
1570                        None
1571                    } else if self.is_thread_channel(&client, &channel_id).await {
1572                        Some(channel_id.clone())
1573                    } else {
1574                        None
1575                    };
1576
1577                    let channel_msg = ChannelMessage {
1578                        id: if message_id.is_empty() {
1579                            Uuid::new_v4().to_string()
1580                        } else {
1581                            format!("discord_{message_id}")
1582                        },
1583                        sender: author_id.to_string(),
1584                        reply_target: if channel_id.is_empty() {
1585                            author_id.to_string()
1586                        } else {
1587                            channel_id.clone()
1588                        },
1589                        content: final_content,
1590                        channel: "discord".to_string(),
1591                        timestamp: std::time::SystemTime::now()
1592                            .duration_since(std::time::UNIX_EPOCH)
1593                            .unwrap_or_default()
1594                            .as_secs(),
1595                        interruption_scope_id: thread_ts.clone(),
1596                        thread_ts,
1597                        attachments: media_attachments,
1598                    };
1599
1600                    if tx.send(channel_msg).await.is_err() {
1601                        break;
1602                    }
1603                }
1604            }
1605        }
1606
1607        // Clean up the watchdog task before returning so the outer
1608        // reconnection loop can start fresh.
1609        if let Some(ref wd) = watchdog {
1610            wd.stop().await;
1611        }
1612
1613        Ok(())
1614    }

1709    async fn update_draft(
1710        &self,
1711        recipient: &str,
1712        message_id: &str,
1713        text: &str,
1714    ) -> anyhow::Result<()> {
1715        use zeroclaw_config::schema::StreamMode;
1716        match self.stream_mode {
1717            StreamMode::Off => Ok(()),
1718            StreamMode::Partial => {
1719                // Rate-limit edits per channel.
1720                {
1721                    let last_edits = self.last_draft_edit.lock();
1722                    if let Some(last_time) = last_edits.get(recipient) {
1723                        let elapsed_ms =
1724                            u64::try_from(last_time.elapsed().as_millis()).unwrap_or(u64::MAX);
1725                        if elapsed_ms < self.draft_update_interval_ms {
1726                            return Ok(());
1727                        }
1728                    }
1729                }
1730
1731                // UTF-8 safe truncation to Discord limit.
1732                let display_text = if text.len() > DISCORD_MAX_MESSAGE_LENGTH {
1733                    let mut end = 0;
1734                    for (idx, ch) in text.char_indices() {
1735                        let next = idx + ch.len_utf8();
1736                        if next > DISCORD_MAX_MESSAGE_LENGTH {
1737                            break;
1738                        }
1739                        end = next;
1740                    }
1741                    &text[..end]
1742                } else {
1743                    text
1744                };
1745
1746                let client = self.http_client();
1747                match edit_discord_message(
1748                    &client,
1749                    &self.bot_token,
1750                    recipient,
1751                    message_id,
1752                    display_text,
1753                )
1754                .await
1755                {
1756                    Ok(()) => {
1757                        self.last_draft_edit
1758                            .lock()
1759                            .insert(recipient.to_string(), std::time::Instant::now());
1760                    }
1761                    Err(e) => {
1762                        tracing::debug!("Discord draft update failed: {e}");
1763                    }
1764                }
1765
1766                Ok(())
1767            }
1768            StreamMode::MultiMessage => {
1769                // Track accumulated text and send new paragraphs at \n\n boundaries.
1770                // Extract paragraph (if any) under the lock, then drop it before async work.
1771                let (paragraph, thread_ts) = {
1772                    let thread_ts = self
1773                        .multi_message_thread_ts
1774                        .lock()
1775                        .get(recipient)
1776                        .cloned()
1777                        .flatten();
1778                    let mut sent_map = self.multi_message_sent_len.lock();
1779                    let sent_so_far = sent_map.get(recipient).copied().unwrap_or(0);
1780
1781                    // DraftEvent::Clear resets accumulated text — reset our counter.
1782                    if text.len() < sent_so_far {
1783                        sent_map.insert(recipient.to_string(), 0);
1784                        return Ok(());
1785                    }
1786                    if text.len() == sent_so_far {
1787                        return Ok(());
1788                    }
1789
1790                    let new_text = &text[sent_so_far..];
1791                    let mut scan_pos = 0;
1792                    let mut in_fence = false;
1793                    let bytes = new_text.as_bytes();
1794                    let mut found_paragraph = None;
1795
1796                    while scan_pos < bytes.len() {
1797                        let ch = bytes[scan_pos];
1798
1799                        if ch == b'`'
1800                            && scan_pos + 2 < bytes.len()
1801                            && bytes[scan_pos + 1] == b'`'
1802                            && bytes[scan_pos + 2] == b'`'
1803                            && (scan_pos == 0 || bytes[scan_pos - 1] == b'\n')
1804                        {
1805                            in_fence = !in_fence;
1806                        }
1807
1808                        if !in_fence
1809                            && ch == b'\n'
1810                            && scan_pos + 1 < bytes.len()
1811                            && bytes[scan_pos + 1] == b'\n'
1812                        {
1813                            let paragraph = new_text[..scan_pos].trim().to_string();
1814                            let consumed = scan_pos + 2;
1815                            *sent_map.entry(recipient.to_string()).or_insert(0) += consumed;
1816                            if !paragraph.is_empty() {
1817                                found_paragraph = Some(paragraph);
1818                            }
1819                            break;
1820                        }
1821
1822                        scan_pos += 1;
1823                    }
1824                    // Lock is dropped here at end of block.
1825                    (found_paragraph, thread_ts)
1826                };
1827
1828                if let Some(paragraph) = paragraph {
1829                    let msg = SendMessage::new(&paragraph, recipient).in_thread(thread_ts.clone());
1830                    if let Err(e) = self.send(&msg).await {
1831                        tracing::debug!("Discord multi-message paragraph send failed: {e}");
1832                    }
1833                    if self.multi_message_delay_ms > 0 {
1834                        tokio::time::sleep(std::time::Duration::from_millis(
1835                            self.multi_message_delay_ms,
1836                        ))
1837                        .await;
1838                    }
1839                    // Recurse to handle remaining text.
1840                    return self.update_draft(recipient, message_id, text).await;
1841                }
1842
1843                Ok(())
1844            }
1845        }
1846    }

1848    async fn finalize_draft(
1849        &self,
1850        recipient: &str,
1851        message_id: &str,
1852        text: &str,
1853    ) -> anyhow::Result<()> {
1854        if self.stream_mode == zeroclaw_config::schema::StreamMode::MultiMessage {
1855            // Flush remaining buffered text.
1856            let thread_ts = self
1857                .multi_message_thread_ts
1858                .lock()
1859                .remove(recipient)
1860                .flatten();
1861            let sent_so_far = self
1862                .multi_message_sent_len
1863                .lock()
1864                .remove(recipient)
1865                .unwrap_or(0);
1866            if text.len() > sent_so_far {
1867                let remaining = text[sent_so_far..].trim().to_string();
1868                if !remaining.is_empty() {
1869                    let msg = SendMessage::new(&remaining, recipient).in_thread(thread_ts);
1870                    if let Err(e) = self.send(&msg).await {
1871                        tracing::debug!("Discord multi-message final flush failed: {e}");
1872                    }
1873                }
1874            }
1875            return Ok(());
1876        }
1877
1878        // Belt-and-suspenders: kill any typing handles for this channel.
1879        let _ = self.stop_typing(recipient).await;
1880        self.last_draft_edit.lock().remove(recipient);
1881
1882        let text = &crate::util::strip_tool_call_tags(text);
1883        let (cleaned_content, parsed_attachments) = parse_attachment_markers(text);
1884        let (mut local_files, remote_urls, failures) =
1885            classify_outgoing_attachments(&parsed_attachments, self.workspace_dir.as_deref());
1886        let body = with_inline_attachment_urls(&cleaned_content, &remote_urls);
1887        let note = delivery_failure_note(&failures);
1888        let content = compose_body_with_failure_note(&body, note.as_deref());
1889        let reactions = decide_failure_reactions(&failures);
1890
1891        let client = self.http_client();
1892
1893        // Path 1: file attachments — delete draft and POST fresh message with files.
1894        if !local_files.is_empty() {
1895            let _ = delete_discord_message(&client, &self.bot_token, recipient, message_id).await;
1896
1897            if local_files.len() > 10 {
1898                local_files.truncate(10);
1899            }
1900            let chunks = split_message_for_discord(&content);
1901            let mut first_message_id: Option<String> = None;
1902            for (i, chunk) in chunks.iter().enumerate() {
1903                let new_id = if i == 0 {
1904                    send_discord_message_with_files(
1905                        &client,
1906                        &self.bot_token,
1907                        recipient,
1908                        chunk,
1909                        &local_files,
1910                    )
1911                    .await?
1912                } else {
1913                    send_discord_message_json(&client, &self.bot_token, recipient, chunk).await?
1914                };
1915                if first_message_id.is_none() {
1916                    first_message_id = Some(new_id);
1917                }
1918                if i < chunks.len() - 1 {
1919                    tokio::time::sleep(std::time::Duration::from_millis(500)).await;
1920                }
1921            }
1922            self.apply_failure_reactions(recipient, first_message_id.as_deref(), &reactions)
1923                .await;
1924            return Ok(());
1925        }
1926
1927        // Path 2: text exceeds limit — delete draft and POST as chunked messages.
1928        if content.chars().count() > DISCORD_MAX_MESSAGE_LENGTH {
1929            let _ = delete_discord_message(&client, &self.bot_token, recipient, message_id).await;
1930
1931            let chunks = split_message_for_discord(&content);
1932            let mut first_message_id: Option<String> = None;
1933            for (i, chunk) in chunks.iter().enumerate() {
1934                let new_id =
1935                    send_discord_message_json(&client, &self.bot_token, recipient, chunk).await?;
1936                if first_message_id.is_none() {
1937                    first_message_id = Some(new_id);
1938                }
1939                if i < chunks.len() - 1 {
1940                    tokio::time::sleep(std::time::Duration::from_millis(500)).await;
1941                }
1942            }
1943            self.apply_failure_reactions(recipient, first_message_id.as_deref(), &reactions)
1944                .await;
1945            return Ok(());
1946        }
1947
1948        // Path 3: simple case — edit in-place; fall back to delete + POST on failure.
1949        // The reaction target is the draft message_id when the edit lands;
1950        // when the fallback fires it's the freshly posted message instead.
1951        let reaction_target =
1952            match edit_discord_message(&client, &self.bot_token, recipient, message_id, &content)
1953                .await
1954            {
1955                Ok(()) => message_id.to_string(),
1956                Err(e) => {
1957                    tracing::warn!(
1958                        "Discord finalize_draft edit failed: {e}; falling back to delete+send"
1959                    );
1960                    let _ = delete_discord_message(&client, &self.bot_token, recipient, message_id)
1961                        .await;
1962                    send_discord_message_json(&client, &self.bot_token, recipient, &content).await?
1963                }
1964            };
1965        self.apply_failure_reactions(recipient, Some(&reaction_target), &reactions)
1966            .await;
1967
1968        Ok(())
1969    }

2046    async fn request_approval(
2047        &self,
2048        recipient: &str,
2049        request: &ChannelApprovalRequest,
2050    ) -> anyhow::Result<Option<ChannelApprovalResponse>> {
2051        let token = crate::util::new_approval_token();
2052        let text = format!(
2053            "APPROVAL REQUIRED [{}]\nTool: {}\nArgs: {}\n\nReply: \"{} yes\", \"{} no\", or \"{} always\"",
2054            token, request.tool_name, request.arguments_summary, token, token, token,
2055        );
2056
2057        let (tx, rx) = oneshot::channel();
2058        self.pending_approvals
2059            .lock()
2060            .await
2061            .insert(token.clone(), tx);
2062
2063        // Strip thread suffix — approval message goes to the channel root.
2064        let channel_id = recipient.split(':').next().unwrap_or(recipient);
2065        if let Err(err) = self.send(&SendMessage::new(text, channel_id)).await {
2066            self.pending_approvals.lock().await.remove(&token);
2067            return Err(err);
2068        }
2069
2070        let response =
2071            match tokio::time::timeout(Duration::from_secs(self.approval_timeout_secs), rx).await {
2072                Ok(Ok(resp)) => resp,
2073                _ => {
2074                    self.pending_approvals.lock().await.remove(&token);
2075                    ChannelApprovalResponse::Deny
2076                }
2077            };
2078        Ok(Some(response))
2079    }
