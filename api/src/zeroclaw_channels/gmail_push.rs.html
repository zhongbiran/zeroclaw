1//! Gmail Pub/Sub push notification channel.
2//!
3//! Instead of polling via IMAP, this channel uses Google's Gmail Pub/Sub push
4//! notifications.  Google sends a POST to our webhook endpoint whenever the
5//! user's mailbox changes.  The notification body contains a base64-encoded
6//! JSON payload with `emailAddress` and `historyId`; we then call the Gmail
7//! History API to fetch newly arrived messages.
8//!
9//! ## Setup
10//!
11//! 1. Create a Google Cloud Pub/Sub topic and grant `gmail-api-push@system.gserviceaccount.com`
12//!    the **Pub/Sub Publisher** role on that topic.
13//! 2. Create a push subscription pointing to `https://<your-domain>/webhook/gmail`.
14//! 3. Configure `[channels_config.gmail_push]` in `config.toml` with `topic` and
15//!    `oauth_token` (or set `GMAIL_PUSH_OAUTH_TOKEN` env var).
16//!
17//! The channel automatically calls `users.watch` to register the subscription
18//! and renews it before the 7-day expiry.
19

167/// Gmail Pub/Sub push notification channel.
168///
169/// Incoming messages arrive via webhook (`POST /webhook/gmail`) and are
170/// dispatched to the agent.  The `listen` method registers the Gmail watch
171/// subscription and periodically renews it.
172pub struct GmailPushChannel {
173    pub config: GmailPushConfig,
174    http: Client,
175    last_history_id: Arc<Mutex<u64>>,
176    /// Sender half injected by the gateway to forward webhook-received messages.
177    pub tx: Arc<Mutex<Option<mpsc::Sender<ChannelMessage>>>>,
178}

180impl GmailPushChannel {
181    pub fn new(config: GmailPushConfig) -> Self {
182        let http = Client::builder()
183            .timeout(Duration::from_secs(30))
184            .build()
185            .expect("failed to build HTTP client");
186        Self {
187            config,
188            http,
189            last_history_id: Arc::new(Mutex::new(0)),
190            tx: Arc::new(Mutex::new(None)),
191        }
192    }

204        if !self.config.oauth_token.is_empty() {

206        }
207        std::env::var("GMAIL_PUSH_OAUTH_TOKEN").unwrap_or_default()

210    /// Register a Gmail watch subscription via `POST /gmail/v1/users/me/watch`.
211    pub async fn register_watch(&self) -> Result<WatchResponse> {
212        let token = self.resolve_oauth_token();
213        if token.is_empty() {
214            return Err(anyhow!("Gmail OAuth token is not configured"));
215        }
216
217        let body = serde_json::json!({
218            "topicName": self.config.topic,
219            "labelIds": self.config.label_filter,
220        });
221
222        let resp = self
223            .http
224            .post("https://gmail.googleapis.com/gmail/v1/users/me/watch")
225            .bearer_auth(&token)
226            .json(&body)
227            .send()
228            .await?;
229
230        if !resp.status().is_success() {
231            let status = resp.status();
232            let text = resp.text().await.unwrap_or_default();
233            return Err(anyhow!(
234                "Gmail watch registration failed ({}): {}",
235                status,
236                text
237            ));
238        }
239
240        let watch: WatchResponse = resp.json().await?;
241        let mut last_id = self.last_history_id.lock().await;
242        if *last_id == 0 {
243            *last_id = watch.history_id;
244        }
245        info!(
246            "Gmail watch registered — historyId={}, expiration={}",
247            watch.history_id, watch.expiration
248        );
249        Ok(watch)
250    }
251
252    /// Fetch new messages since the given `start_history_id` using the History API.
253    pub async fn fetch_history(&self, start_history_id: u64) -> Result<Vec<String>> {
254        let mut last_id = self.last_history_id.lock().await;
255        self.fetch_history_inner(start_history_id, &mut last_id)
256            .await
257    }
258
259    /// Inner history fetch that takes an already-locked history ID reference.
260    /// This allows callers that already hold the lock to avoid deadlock.
261    async fn fetch_history_inner(
262        &self,
263        start_history_id: u64,
264        last_id: &mut u64,
265    ) -> Result<Vec<String>> {
266        let token = self.resolve_oauth_token();
267        if token.is_empty() {
268            return Err(anyhow!("Gmail OAuth token is not configured"));
269        }
270
271        let mut message_ids = Vec::new();
272        let mut page_token: Option<String> = None;
273
274        loop {
275            let mut url = format!(
276                "https://gmail.googleapis.com/gmail/v1/users/me/history?startHistoryId={}&historyTypes=messageAdded",
277                start_history_id
278            );
279            if let Some(ref pt) = page_token {
280                let _ = write!(url, "&pageToken={pt}");
281            }
282
283            let resp = self.http.get(&url).bearer_auth(&token).send().await?;
284
285            if !resp.status().is_success() {
286                let status = resp.status();
287                let text = resp.text().await.unwrap_or_default();
288                return Err(anyhow!("Gmail history fetch failed ({}): {}", status, text));
289            }
290
291            let history_resp: HistoryResponse = resp.json().await?;
292
293            if let Some(records) = history_resp.history {
294                for record in records {
295                    for added in record.messages_added {
296                        message_ids.push(added.message.id);
297                    }
298                }
299            }
300
301            // Update tracked history ID
302            if history_resp.history_id > 0 && history_resp.history_id > *last_id {
303                *last_id = history_resp.history_id;
304            }
305
306            match history_resp.next_page_token {
307                Some(token) => page_token = Some(token),
308                None => break,
309            }
310        }
311
312        Ok(message_ids)
313    }

315    /// Fetch a full message by ID from the Gmail API.
316    pub async fn fetch_message(&self, message_id: &str) -> Result<GmailMessage> {
317        let token = self.resolve_oauth_token();
318        let url = format!(
319            "https://gmail.googleapis.com/gmail/v1/users/me/messages/{}?format=full",
320            message_id
321        );
322
323        let resp = self.http.get(&url).bearer_auth(&token).send().await?;
324
325        if !resp.status().is_success() {
326            let status = resp.status();
327            let text = resp.text().await.unwrap_or_default();
328            return Err(anyhow!("Gmail message fetch failed ({}): {}", status, text));
329        }
330
331        Ok(resp.json().await?)
332    }

334    /// Check if a sender email is in the allowlist.
335    pub fn is_sender_allowed(&self, email: &str) -> bool {
336        if self.config.allowed_senders.is_empty() {
337            return false;
338        }
339        if self.config.allowed_senders.iter().any(|a| a == "*") {
340            return true;
341        }
342        let email_lower = email.to_lowercase();
343        self.config.allowed_senders.iter().any(|allowed| {
344            if allowed.starts_with('@') {
345                email_lower.ends_with(&allowed.to_lowercase())
346            } else if allowed.contains('@') {
347                allowed.eq_ignore_ascii_case(email)
348            } else {
349                email_lower.ends_with(&format!("@{}", allowed.to_lowercase()))
350            }
351        })
352    }

354    /// Process a Pub/Sub push notification and dispatch new messages to the agent.
355    pub async fn handle_notification(&self, envelope: &PubSubEnvelope) -> Result<()> {
356        let notification = parse_notification(&envelope.message)?;
357        debug!(
358            "Gmail push notification: email={}, historyId={}",
359            notification.email_address, notification.history_id
360        );
361
362        // Hold the lock across read-fetch-update to prevent duplicate
363        // processing when concurrent webhook notifications arrive.
364        let mut last_id = self.last_history_id.lock().await;
365
366        if *last_id == 0 {
367            // First notification — just record the history ID.
368            *last_id = notification.history_id;
369            info!(
370                "Gmail push: first notification, seeding historyId={}",
371                notification.history_id
372            );
373            return Ok(());
374        }
375
376        let start_id = *last_id;
377        let message_ids = self.fetch_history_inner(start_id, &mut last_id).await?;
378        // Explicitly drop the lock before doing network-heavy message fetching.
379        drop(last_id);
380
381        if message_ids.is_empty() {
382            debug!("Gmail push: no new messages in history");
383            return Ok(());
384        }
385
386        info!(
387            "Gmail push: {} new message(s) to process",
388            message_ids.len()
389        );
390
391        // Clone the sender and drop the mutex immediately to avoid holding it
392        // across network calls.
393        let tx = {
394            let tx_guard = self.tx.lock().await;
395            match tx_guard.clone() {
396                Some(tx) => tx,
397                None => {
398                    warn!("Gmail push: no listener registered, dropping messages");
399                    return Ok(());
400                }
401            }
402        };
403
404        for msg_id in message_ids {
405            match self.fetch_message(&msg_id).await {
406                Ok(gmail_msg) => {
407                    let sender = extract_header(&gmail_msg, "From").unwrap_or_default();
408                    let sender_email = extract_email_from_header(&sender);
409
410                    if !self.is_sender_allowed(&sender_email) {
411                        warn!("Gmail push: blocked message from {}", sender_email);
412                        continue;
413                    }
414
415                    let subject = extract_header(&gmail_msg, "Subject").unwrap_or_default();
416                    let body_text = extract_body_text(&gmail_msg);
417
418                    let content = format!("Subject: {subject}\n\n{body_text}");
419                    let timestamp = gmail_msg
420                        .internal_date
421                        .parse::<u64>()
422                        .map(|ms| ms / 1000)
423                        .unwrap_or_else(|_| {
424                            SystemTime::now()
425                                .duration_since(UNIX_EPOCH)
426                                .map(|d| d.as_secs())
427                                .unwrap_or(0)
428                        });
429
430                    let channel_msg = ChannelMessage {
431                        id: format!("gmail_{}", gmail_msg.id),
432                        reply_target: sender_email.clone(),
433                        sender: sender_email,
434                        content,
435                        channel: "gmail_push".to_string(),
436                        timestamp,
437                        thread_ts: Some(gmail_msg.thread_id),
438                        interruption_scope_id: None,
439                        attachments: Vec::new(),
440                    };
441
442                    if tx.send(channel_msg).await.is_err() {
443                        debug!("Gmail push: listener channel closed");
444                        return Ok(());
445                    }
446                }
447                Err(e) => {
448                    error!("Gmail push: failed to fetch message {}: {}", msg_id, e);
449                }
450            }
451        }
452
453        Ok(())
454    }
455}

457#[async_trait]
458impl Channel for GmailPushChannel {
459    fn name(&self) -> &str {
460        "gmail_push"
461    }
462
463    async fn send(&self, message: &SendMessage) -> Result<()> {
464        // Send via Gmail API (drafts.send or messages.send)
465        let token = self.resolve_oauth_token();
466        if token.is_empty() {
467            return Err(anyhow!("Gmail OAuth token is not configured for sending"));
468        }
469
470        let subject = message.subject.as_deref().unwrap_or("ZeroClaw Message");
471        // Sanitize headers to prevent CRLF injection attacks.
472        let safe_recipient = sanitize_header_value(&message.recipient);
473        let safe_subject = sanitize_header_value(subject);
474        let rfc2822 = format!(
475            "To: {}\r\nSubject: {}\r\nContent-Type: text/plain; charset=utf-8\r\n\r\n{}",
476            safe_recipient, safe_subject, message.content
477        );
478        let encoded = BASE64.encode(rfc2822.as_bytes());
479        // Gmail API uses URL-safe base64 with no padding
480        let url_safe = encoded.replace('+', "-").replace('/', "_").replace('=', "");
481
482        let body = serde_json::json!({
483            "raw": url_safe,
484        });
485
486        let resp = self
487            .http
488            .post("https://gmail.googleapis.com/gmail/v1/users/me/messages/send")
489            .bearer_auth(&token)
490            .json(&body)
491            .send()
492            .await?;
493
494        if !resp.status().is_success() {
495            let status = resp.status();
496            let text = resp.text().await.unwrap_or_default();
497            return Err(anyhow!("Gmail send failed ({}): {}", status, text));
498        }
499
500        info!("Gmail message sent to {}", message.recipient);
501        Ok(())
502    }
503
504    async fn listen(&self, tx: mpsc::Sender<ChannelMessage>) -> Result<()> {
505        // Store the sender for webhook-driven message dispatch
506        {
507            let mut tx_guard = self.tx.lock().await;
508            *tx_guard = Some(tx);
509        }
510
511        info!("Gmail push channel started — registering watch subscription");
512
513        // Register initial watch
514        if !self.config.webhook_url.is_empty()
515            && let Err(e) = self.register_watch().await
516        {
517            error!("Gmail watch registration failed: {e:#}");
518            // Non-fatal — external subscription management may be in use
519        }
520
521        // Renewal loop: Gmail watch subscriptions expire after 7 days.
522        // Re-register every 6 days to maintain continuous coverage.
523        let renewal_interval = Duration::from_secs(6 * 24 * 60 * 60); // 6 days
524        loop {
525            tokio::time::sleep(renewal_interval).await;
526            info!("Gmail push: renewing watch subscription");
527            if let Err(e) = self.register_watch().await {
528                error!("Gmail watch renewal failed: {e:#}");
529            }
530        }
531    }
532
533    async fn health_check(&self) -> bool {
534        let token = self.resolve_oauth_token();
535        if token.is_empty() {
536            return false;
537        }
538
539        match self
540            .http
541            .get("https://gmail.googleapis.com/gmail/v1/users/me/profile")
542            .bearer_auth(&token)
543            .timeout(Duration::from_secs(10))
544            .send()
545            .await
546        {
547            Ok(resp) => resp.status().is_success(),
548            Err(_) => false,
549        }
550    }
551}

555/// Parse and decode the Gmail notification from a Pub/Sub message.
556pub fn parse_notification(msg: &PubSubMessage) -> Result<GmailNotification> {
557    let decoded = BASE64
558        .decode(&msg.data)
559        .map_err(|e| anyhow!("Invalid base64 in Pub/Sub message: {e}"))?;
560    let notification: GmailNotification = serde_json::from_slice(&decoded)
561        .map_err(|e| anyhow!("Invalid JSON in Gmail notification: {e}"))?;
562    Ok(notification)
563}
564
565/// Extract a header value from a Gmail message by name.
566pub fn extract_header(msg: &GmailMessage, name: &str) -> Option<String> {
567    msg.payload.as_ref().and_then(|p| {
568        p.headers
569            .iter()
570            .find(|h| h.name.eq_ignore_ascii_case(name))
571            .map(|h| h.value.clone())
572    })
573}
574
575/// Extract the plain email address from a `From` header value like `"Name <email@example.com>"`.
576pub fn extract_email_from_header(from: &str) -> String {
577    if let Some(start) = from.find('<') {
578        // Use rfind to find the matching '>' after '<', preventing panic
579        // when malformed headers have '>' before '<'.
580        if let Some(end) = from.rfind('>')
581            && end > start + 1
582        {
583            return from[start + 1..end].to_string();
584        }
585    }
586    from.trim().to_string()
587}

589/// Sanitize a string for use in an RFC 2822 header value.
590/// Removes CR and LF characters to prevent header injection attacks.
591pub fn sanitize_header_value(value: &str) -> String {
592    value.chars().filter(|c| *c != '\r' && *c != '\n').collect()
593}
594
595/// Extract the plain-text body from a Gmail message.
596///
597/// Walks MIME parts looking for `text/plain`; falls back to `text/html`
598/// with basic tag stripping; finally falls back to the `snippet`.
599pub fn extract_body_text(msg: &GmailMessage) -> String {
600    if let Some(ref payload) = msg.payload {
601        // Single-part message
602        if payload.mime_type == "text/plain"
603            && let Some(text) = decode_body(payload.body.as_ref())
604        {
605            return text;
606        }
607
608        // Multipart — walk parts
609        if let Some(text) = find_text_in_parts(&payload.parts, "text/plain") {
610            return text;
611        }
612        if let Some(html) = find_text_in_parts(&payload.parts, "text/html") {
613            return strip_html(&html);
614        }
615    }
616
617    // Fallback to snippet
618    msg.snippet.clone()
619}
620
621/// Recursively search MIME parts for a given content type.
622fn find_text_in_parts(parts: &[MessagePart], mime_type: &str) -> Option<String> {
623    for part in parts {
624        if part.mime_type == mime_type
625            && let Some(text) = decode_body(part.body.as_ref())
626        {
627            return Some(text);
628        }
629        // Recurse into nested parts
630        if let Some(text) = find_text_in_parts(&part.parts, mime_type) {
631            return Some(text);
632        }
633    }
634    None
635}
636
637/// Decode a base64url-encoded Gmail message body.
638fn decode_body(body: Option<&MessageBody>) -> Option<String> {
639    body.and_then(|b| {
640        b.data.as_ref().and_then(|data| {
641            // Gmail API uses URL-safe base64 without padding
642            let standard = data.replace('-', "+").replace('_', "/");
643            // Restore padding stripped by Gmail API
644            let padded = match standard.len() % 4 {
645                2 => format!("{standard}=="),
646                3 => format!("{standard}="),
647                _ => standard,
648            };
649            BASE64
650                .decode(&padded)
651                .ok()
652                .and_then(|bytes| String::from_utf8(bytes).ok())
653        })
654    })
655}
656
657/// Basic HTML tag stripper (reuses the pattern from email_channel).
658fn strip_html(html: &str) -> String {
659    let mut result = String::new();
660    let mut in_tag = false;
661    for ch in html.chars() {
662        match ch {
663            '<' => in_tag = true,
664            '>' => in_tag = false,
665            _ if !in_tag => result.push(ch),
666            _ => {}
667        }
668    }
669    let mut normalized = String::with_capacity(result.len());
670    for word in result.split_whitespace() {
671        if !normalized.is_empty() {
672            normalized.push(' ');
673        }
674        normalized.push_str(word);
675    }
676    normalized
677}
