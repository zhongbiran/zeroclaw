23/// Lifecycle state of a phone call.
24#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
25#[serde(rename_all = "snake_case")]
26pub enum CallState {
27    /// Call is ringing (inbound or outbound).
28    Ringing,
29    /// Call is connected and audio is flowing.
30    InProgress,
31    /// Call has ended normally.
32    Completed,
33    /// Call failed to connect.
34    Failed,
35    /// Caller or callee hung up.
36    HungUp,
37    /// Call is queued (outbound, awaiting approval).
38    PendingApproval,
39}
40
41impl fmt::Display for CallState {
42    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
43        match self {
44            Self::Ringing => write!(f, "ringing"),
45            Self::InProgress => write!(f, "in_progress"),
46            Self::Completed => write!(f, "completed"),
47            Self::Failed => write!(f, "failed"),
48            Self::HungUp => write!(f, "hung_up"),
49            Self::PendingApproval => write!(f, "pending_approval"),
50        }
51    }
52}

62/// Tracks an active call's metadata and transcription.
63#[derive(Debug, Clone, Serialize, Deserialize)]
64pub struct CallRecord {
65    /// Unique call identifier (provider-specific SID/UUID).
66    pub call_id: String,
67    /// Direction: inbound or outbound.
68    pub direction: CallDirection,
69    /// Remote phone number (E.164).
70    pub remote_number: String,
71    /// Local phone number used.
72    pub local_number: String,
73    /// Current call state.
74    pub state: CallState,
75    /// When the call started (ISO-8601).
76    pub started_at: String,
77    /// When the call ended (ISO-8601), if applicable.
78    pub ended_at: Option<String>,
79    /// Duration in seconds (updated on completion).
80    pub duration_secs: u64,
81    /// Running transcript of the call.
82    pub transcript: Vec<TranscriptEntry>,
83}

85/// A single transcript entry from the call.
86#[derive(Debug, Clone, Serialize, Deserialize)]
87pub struct TranscriptEntry {
88    /// Who said it: `"caller"` or `"agent"`.
89    pub speaker: String,
90    /// The transcribed text.
91    pub text: String,
92    /// ISO-8601 timestamp.
93    pub timestamp: String,
94}

114    /// Get the provider-specific API base URL.
115    fn api_base_url(&self) -> &str {
116        match self.config.provider {
117            VoiceProvider::Twilio => "https://api.twilio.com/2010-04-01",
118            VoiceProvider::Telnyx => "https://api.telnyx.com/v2",
119            VoiceProvider::Plivo => "https://api.plivo.com/v1",
120        }

123    /// Place an outbound call via the configured provider.
124    pub async fn place_call(&self, to_number: &str) -> Result<String> {
125        if self.config.require_outbound_approval {
126            info!(to = to_number, "outbound call requires approval");
127            return Ok(format!("PENDING_APPROVAL:{to_number}"));
128        }
129        self.execute_outbound_call(to_number).await
130    }
131
132    async fn execute_outbound_call(&self, to_number: &str) -> Result<String> {
133        let webhook_url = self.webhook_url("/voice/status");
134
135        match self.config.provider {
136            VoiceProvider::Twilio => {
137                let url = format!(
138                    "{}/Accounts/{}/Calls.json",
139                    self.api_base_url(),
140                    self.config.account_id
141                );
142                let resp = self
143                    .client
144                    .post(&url)
145                    .basic_auth(&self.config.account_id, Some(&self.config.auth_token))
146                    .form(&[
147                        ("To", to_number),
148                        ("From", &self.config.from_number),
149                        ("StatusCallback", &webhook_url),
150                        ("Timeout", &self.config.max_call_duration_secs.to_string()),
151                    ])
152                    .send()
153                    .await?;
154
155                if !resp.status().is_success() {
156                    let body = resp.text().await.unwrap_or_default();
157                    bail!("Twilio call failed: {body}");
158                }
159
160                let json: serde_json::Value = serde_json::from_str(&resp.text().await?)?;
161                let call_sid = json["sid"].as_str().unwrap_or("unknown").to_string();
162                info!(call_sid = %call_sid, to = to_number, "outbound call placed via Twilio");
163                Ok(call_sid)
164            }
165            VoiceProvider::Telnyx => {
166                let url = format!("{}/calls", self.api_base_url());
167                let resp = self
168                    .client
169                    .post(&url)
170                    .bearer_auth(&self.config.auth_token)
171                    .json(&serde_json::json!({
172                        "connection_id": self.config.account_id,
173                        "to": to_number,
174                        "from": self.config.from_number,
175                        "webhook_url": webhook_url,
176                        "timeout_secs": self.config.max_call_duration_secs,
177                    }))
178                    .send()
179                    .await?;
180
181                if !resp.status().is_success() {
182                    let body = resp.text().await.unwrap_or_default();
183                    bail!("Telnyx call failed: {body}");
184                }
185
186                let json: serde_json::Value = serde_json::from_str(&resp.text().await?)?;
187                let call_id = json["data"]["call_control_id"]
188                    .as_str()
189                    .unwrap_or("unknown")
190                    .to_string();
191                info!(call_id = %call_id, to = to_number, "outbound call placed via Telnyx");
192                Ok(call_id)
193            }
194            VoiceProvider::Plivo => {
195                let url = format!(
196                    "{}/Account/{}/Call/",
197                    self.api_base_url(),
198                    self.config.account_id
199                );
200                let resp = self
201                    .client
202                    .post(&url)
203                    .basic_auth(&self.config.account_id, Some(&self.config.auth_token))
204                    .json(&serde_json::json!({
205                        "to": to_number,
206                        "from": self.config.from_number,
207                        "answer_url": self.webhook_url("/voice/answer"),
208                        "hangup_url": self.webhook_url("/voice/hangup"),
209                        "time_limit": self.config.max_call_duration_secs,
210                    }))
211                    .send()
212                    .await?;
213
214                if !resp.status().is_success() {
215                    let body = resp.text().await.unwrap_or_default();
216                    bail!("Plivo call failed: {body}");
217                }
218
219                let json: serde_json::Value = serde_json::from_str(&resp.text().await?)?;
220                let call_uuid = json["request_uuid"]
221                    .as_str()
222                    .unwrap_or("unknown")
223                    .to_string();
224                info!(call_uuid = %call_uuid, to = to_number, "outbound call placed via Plivo");
225                Ok(call_uuid)
226            }
227        }
228    }

239    /// Record a transcript entry for an active call.
240    pub async fn add_transcript_entry(&self, call_id: &str, speaker: &str, text: &str) {
241        let mut calls = self.active_calls.lock().await;
242        if let Some(record) = calls.get_mut(call_id) {
243            record.transcript.push(TranscriptEntry {
244                speaker: speaker.to_string(),
245                text: text.to_string(),
246                timestamp: chrono::Utc::now().to_rfc3339(),
247            });
248        }
249    }

263    /// Handle an incoming call webhook event.
264    pub async fn handle_inbound_call(
265        &self,
266        call_id: &str,
267        from_number: &str,
268        tx: &mpsc::Sender<ChannelMessage>,
269    ) -> Result<()> {
270        let record = CallRecord {
271            call_id: call_id.to_string(),
272            direction: CallDirection::Inbound,
273            remote_number: from_number.to_string(),
274            local_number: self.config.from_number.clone(),
275            state: CallState::Ringing,
276            started_at: chrono::Utc::now().to_rfc3339(),
277            ended_at: None,
278            duration_secs: 0,
279            transcript: Vec::new(),
280        };
281
282        {
283            let mut calls = self.active_calls.lock().await;
284            calls.insert(call_id.to_string(), record);
285        }
286
287        info!(
288            call_id = call_id,
289            from = from_number,
290            "inbound call received"
291        );
292
293        // Notify the agent about the incoming call
294        let msg = ChannelMessage {
295            id: call_id.to_string(),
296            sender: from_number.to_string(),
297            reply_target: from_number.to_string(),
298            content: format!("[Voice Call] Incoming call from {from_number} (call_id: {call_id})"),
299            channel: "voice_call".to_string(),
300            timestamp: chrono::Utc::now().timestamp().unsigned_abs(),
301            thread_ts: Some(call_id.to_string()),
302            interruption_scope_id: Some(call_id.to_string()),
303            attachments: vec![],
304        };
305        tx.send(msg)
306            .await
307            .map_err(|e| anyhow::anyhow!("Failed to send call event: {e}"))?;
308        Ok(())
309    }
310
311    /// Handle a call status update (state transition).
312    pub async fn handle_status_update(&self, call_id: &str, new_state: CallState) {
313        let mut calls = self.active_calls.lock().await;
314        if let Some(record) = calls.get_mut(call_id) {
315            let old_state = record.state;
316            record.state = new_state;
317
318            if matches!(
319                new_state,
320                CallState::Completed | CallState::Failed | CallState::HungUp
321            ) {
322                record.ended_at = Some(chrono::Utc::now().to_rfc3339());
323            }
324
325            debug!(
326                call_id = call_id,
327                old_state = %old_state,
328                new_state = %new_state,
329                "call state transition"
330            );
331        }
332    }
333
334    /// Save call transcript to workspace (if logging is enabled).
335    pub async fn save_transcript(
336        &self,
337        call_id: &str,
338        workspace_dir: &std::path::Path,
339    ) -> Result<()> {
340        if !self.config.transcription_logging {
341            return Ok(());
342        }
343
344        let calls = self.active_calls.lock().await;
345        let Some(record) = calls.get(call_id) else {
346            bail!("Call not found: {call_id}");
347        };
348
349        let logs_dir = workspace_dir.join("logs").join("calls");
350        std::fs::create_dir_all(&logs_dir)?;
351
352        let filename = format!("{}_{}.json", record.started_at.replace(':', "-"), call_id);
353        let path = logs_dir.join(filename);
354        let json = serde_json::to_string_pretty(record)?;
355        std::fs::write(&path, json)?;
356
357        info!(call_id = call_id, path = %path.display(), "call transcript saved");
358        Ok(())
359    }

366    fn name(&self) -> &str {
367        "voice_call"
368    }

370    async fn send(&self, message: &SendMessage) -> Result<()> {
371        // For active calls, TTS the message to the caller
372        if let Some(ref thread_ts) = message.thread_ts {
373            let calls = self.active_calls.lock().await;
374            if let Some(record) = calls.get(thread_ts)
375                && record.state == CallState::InProgress
376            {
377                debug!(
378                    call_id = thread_ts,
379                    "would TTS message to active call: {}", message.content
380                );
381                // TTS synthesis + streaming would be handled by the
382                // telephony provider's media stream API in production.
383                return Ok(());
384            }
385        }
386
387        debug!("voice_call send (no active call): {}", message.content);
388        Ok(())
389    }
390
391    async fn listen(&self, tx: mpsc::Sender<ChannelMessage>) -> Result<()> {
392        let port = self.config.webhook_port;
393        let active_calls = self.active_calls.clone();
394        let _tx = tx.clone();
395
396        info!(port = port, provider = %self.config.provider, "voice call webhook server starting");
397
398        // The webhook server runs as an axum HTTP server on the configured port.
399        // In production, this handles:
400        // - POST /voice/inbound — Twilio/Telnyx/Plivo call initiation webhook
401        // - POST /voice/status — Call status updates
402        // - POST /voice/transcription — Real-time transcription events
403        // - WebSocket /voice/media — Bidirectional audio streaming
404        //
405        // For now, we set up the server structure. Full endpoint
406        // implementation depends on provider-specific webhook payloads.
407
408        let app = axum::Router::new()
409            .route("/voice/health", axum::routing::get(|| async { "ok" }))
410            .with_state(active_calls);
411
412        let listener = tokio::net::TcpListener::bind(format!("0.0.0.0:{port}"))
413            .await
414            .map_err(|e| anyhow::anyhow!("Failed to bind voice webhook server: {e}"))?;
415
416        axum::serve(listener, app)
417            .await
418            .map_err(|e| anyhow::anyhow!("Voice webhook server error: {e}"))?;
419
420        Ok(())
421    }

423    async fn health_check(&self) -> bool {
424        // Check we can reach the provider API
425        let test_url = match self.config.provider {
426            VoiceProvider::Twilio => {
427                format!(
428                    "{}/Accounts/{}.json",
429                    self.api_base_url(),
430                    self.config.account_id
431                )
432            }
433            VoiceProvider::Telnyx => format!("{}/connections", self.api_base_url()),
434            VoiceProvider::Plivo => {
435                format!(
436                    "{}/Account/{}/",
437                    self.api_base_url(),
438                    self.config.account_id
439                )
440            }
441        };
442
443        match self.client.get(&test_url).send().await {
444            Ok(resp) => {
445                // 401 is expected without valid auth — it means the API is reachable
446                resp.status().is_success() || resp.status().as_u16() == 401
447            }
448            Err(e) => {
449                warn!(provider = %self.config.provider, "voice call health check failed: {e}");
450                false
451            }
452        }
453    }
