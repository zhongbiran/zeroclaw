55    /// Check if a sender email is in the allowlist
56    pub fn is_sender_allowed(&self, email: &str) -> bool {
57        if self.config.allowed_senders.is_empty() {
58            return false; // Empty = deny all
59        }
60        if self.config.allowed_senders.iter().any(|a| a == "*") {
61            return true; // Wildcard = allow all
62        }
63        let email_lower = email.to_lowercase();
64        self.config.allowed_senders.iter().any(|allowed| {
65            if allowed.starts_with('@') {
66                // Domain match with @ prefix: "@example.com"
67                email_lower.ends_with(&allowed.to_lowercase())
68            } else if allowed.contains('@') {
69                // Full email address match
70                allowed.eq_ignore_ascii_case(email)
71            } else {
72                // Domain match without @ prefix: "example.com"
73                email_lower.ends_with(&format!("@{}", allowed.to_lowercase()))
74            }
75        })
76    }

78    /// Strip HTML tags from content (basic)
79    pub fn strip_html(html: &str) -> String {
80        let mut result = String::new();
81        let mut in_tag = false;
82        for ch in html.chars() {
83            match ch {
84                '<' => in_tag = true,
85                '>' => in_tag = false,
86                _ if !in_tag => result.push(ch),
87                _ => {}
88            }
89        }
90        let mut normalized = String::with_capacity(result.len());
91        for word in result.split_whitespace() {
92            if !normalized.is_empty() {
93                normalized.push(' ');
94            }
95            normalized.push_str(word);
96        }
97        normalized
98    }

100    /// Extract the sender address from a parsed email
101    fn extract_sender(parsed: &mail_parser::Message) -> String {
102        parsed
103            .from()
104            .and_then(|addr| addr.first())
105            .and_then(|a| a.address())
106            .map(|s| s.to_string())
107            .unwrap_or_else(|| "unknown".into())
108    }

110    /// Extract readable text from a parsed email
111    fn extract_text(parsed: &mail_parser::Message) -> String {
112        if let Some(text) = parsed.body_text(0) {
113            return text.to_string();
114        }
115        if let Some(html) = parsed.body_html(0) {
116            return Self::strip_html(html.as_ref());
117        }
118        for part in parsed.attachments() {
119            let part: &mail_parser::MessagePart = part;
120            if let Some(ct) = MimeHeaders::content_type(part)
121                && ct.ctype() == "text"
122                && let Ok(text) = std::str::from_utf8(part.contents())
123            {
124                let name = MimeHeaders::attachment_name(part).unwrap_or("file");
125                return format!("[Attachment: {}]\n{}", name, text);
126            }
127        }
128        "(no readable content)".to_string()
129    }

131    /// Extract binary attachments from a parsed email as MediaAttachment entries.
132    fn extract_attachments(
133        &self,
134        parsed: &mail_parser::Message,
135    ) -> Vec<zeroclaw_api::media::MediaAttachment> {
136        let mut attachments = Vec::new();
137        let mut total_size = 0;
138
139        for part in parsed.attachments() {
140            let part: &mail_parser::MessagePart = part;
141            let ct = MimeHeaders::content_type(part);
142            let mime_str =
143                ct.map(|c| format!("{}/{}", c.ctype(), c.subtype().unwrap_or("octet-stream")));
144
145            // Skip text parts — already handled by extract_text()
146            if let Some(ref m) = mime_str
147                && m.starts_with("text/")
148            {
149                continue;
150            }
151
152            let data = part.contents().to_vec();
153            if data.is_empty() {
154                continue;
155            }
156
157            // Check size limit
158            total_size += data.len();
159            if total_size > self.config.max_attachment_bytes {
160                warn!(
161                    "Attachment size limit exceeded ({} bytes), dropping remaining attachments",
162                    self.config.max_attachment_bytes
163                );
164                break;
165            }
166
167            let file_name = MimeHeaders::attachment_name(part)
168                .unwrap_or("attachment")
169                .to_string();
170
171            attachments.push(zeroclaw_api::media::MediaAttachment {
172                file_name,
173                data,
174                mime_type: mime_str,
175            });
176        }
177        attachments
178    }

180    /// Connect to IMAP server with TLS and authenticate
181    async fn connect_imap(&self) -> Result<ImapSession> {
182        let addr = format!("{}:{}", self.config.imap_host, self.config.imap_port);
183        debug!("Connecting to IMAP server at {}", addr);
184
185        // Connect TCP
186        let tcp = TcpStream::connect(&addr).await?;
187
188        // Establish TLS using rustls
189        let certs = RootCertStore {
190            roots: webpki_roots::TLS_SERVER_ROOTS.into(),
191        };
192        let config = ClientConfig::builder()
193            .with_root_certificates(certs)
194            .with_no_client_auth();
195        let tls_stream: TlsConnector = Arc::new(config).into();
196        let sni: DnsName = self.config.imap_host.clone().try_into()?;
197        let stream = tls_stream.connect(sni.into(), tcp).await?;
198
199        // Create IMAP client
200        let client = async_imap::Client::new(stream);
201
202        // Login
203        let session = client
204            .login(&self.config.username, &self.config.password)
205            .await
206            .map_err(|(e, _)| anyhow!("IMAP login failed: {}", e))?;
207
208        debug!("IMAP login successful");
209        Ok(session)
210    }

212    /// Maximum number of messages fetched per IMAP round-trip.
213    /// Bounds peak memory when the mailbox has a large unseen backlog.
214    const MAX_FETCH_BATCH: usize = 10;
215
216    /// Fetch and process unseen messages from the selected mailbox.
217    ///
218    /// UIDs are fetched in chunks of [`Self::MAX_FETCH_BATCH`] to bound the
219    /// number of message bodies (and any audio attachments) held in memory at
220    /// once. Each chunk is marked `\Seen` immediately after fetch so that
221    /// successfully retrieved messages are not re-fetched if a later chunk fails.
222    async fn fetch_unseen(&self, session: &mut ImapSession) -> Result<Vec<ParsedEmail>> {
223        // Search for unseen messages
224        let uids = session.uid_search("UNSEEN").await?;
225        if uids.is_empty() {
226            return Ok(Vec::new());
227        }
228
229        debug!("Found {} unseen messages", uids.len());
230
231        let uid_list: Vec<u32> = uids.into_iter().collect();
232        let mut results = Vec::new();
233
234        for chunk in uid_list.chunks(Self::MAX_FETCH_BATCH) {
235            let uid_set: String = chunk
236                .iter()
237                .map(|u| u.to_string())
238                .collect::<Vec<_>>()
239                .join(",");
240
241            // Fetch message bodies for this chunk
242            let messages = session.uid_fetch(&uid_set, "RFC822").await?;
243            let messages: Vec<Fetch> = messages.try_collect().await?;
244
245            for msg in messages {
246                let uid = msg.uid.unwrap_or(0);
247                if let Some(body) = msg.body()
248                    && let Some(parsed) = MessageParser::default().parse(body)
249                {
250                    let sender = Self::extract_sender(&parsed);
251                    let subject = parsed.subject().unwrap_or("(no subject)").to_string();
252                    let body_text = Self::extract_text(&parsed);
253                    let content = format!("Subject: {}\n\n{}", subject, body_text);
254                    let msg_id = parsed
255                        .message_id()
256                        .map(|s| s.to_string())
257                        .unwrap_or_else(|| format!("gen-{}", Uuid::new_v4()));
258
259                    #[allow(clippy::cast_sign_loss)]
260                    let ts = parsed
261                        .date()
262                        .map(|d| {
263                            let naive = chrono::NaiveDate::from_ymd_opt(
264                                d.year as i32,
265                                u32::from(d.month),
266                                u32::from(d.day),
267                            )
268                            .and_then(|date| {
269                                date.and_hms_opt(
270                                    u32::from(d.hour),
271                                    u32::from(d.minute),
272                                    u32::from(d.second),
273                                )
274                            });
275                            naive.map_or(0, |n| n.and_utc().timestamp() as u64)
276                        })
277                        .unwrap_or_else(|| {
278                            SystemTime::now()
279                                .duration_since(UNIX_EPOCH)
280                                .map(|d| d.as_secs())
281                                .unwrap_or(0)
282                        });
283
284                    let attachments = self.extract_attachments(&parsed);
285
286                    results.push(ParsedEmail {
287                        _uid: uid,
288                        msg_id,
289                        sender,
290                        content,
291                        timestamp: ts,
292                        attachments,
293                    });
294                }
295            }
296
297            // Mark this chunk as seen before fetching the next
298            let _ = session
299                .uid_store(&uid_set, "+FLAGS (\\Seen)")
300                .await?
301                .try_collect::<Vec<_>>()
302                .await;
303        }
304
305        Ok(results)
306    }

308    /// Run the IDLE loop, returning when a new message arrives or timeout
309    /// Note: IDLE consumes the session and returns it via done()
310    async fn wait_for_changes(
311        &self,
312        session: ImapSession,
313    ) -> Result<(IdleWaitResult, ImapSession)> {
314        let idle_timeout = Duration::from_secs(self.config.idle_timeout_secs);
315
316        // Start IDLE mode - this consumes the session
317        let mut idle = session.idle();
318        idle.init().await?;
319
320        debug!("Entering IMAP IDLE mode");
321
322        // wait() returns (future, stop_source) - we only need the future
323        let (wait_future, _stop_source) = idle.wait();
324
325        // Wait for server notification or timeout
326        let result = timeout(idle_timeout, wait_future).await;
327
328        match result {
329            Ok(Ok(response)) => {
330                debug!("IDLE response: {:?}", response);
331                // Done with IDLE, return session to normal mode
332                let session = idle.done().await?;
333                let wait_result = match response {
334                    IdleResponse::NewData(_) => IdleWaitResult::NewMail,
335                    IdleResponse::Timeout => IdleWaitResult::Timeout,
336                    IdleResponse::ManualInterrupt => IdleWaitResult::Interrupted,
337                };
338                Ok((wait_result, session))
339            }
340            Ok(Err(e)) => {
341                // Try to clean up IDLE state
342                let _ = idle.done().await;
343                Err(anyhow!("IDLE error: {}", e))
344            }
345            Err(_) => {
346                // Timeout - RFC 2177 recommends restarting IDLE every 29 minutes
347                debug!("IDLE timeout reached, will re-establish");
348                let session = idle.done().await?;
349                Ok((IdleWaitResult::Timeout, session))
350            }
351        }
352    }

354    /// Main listen loop with automatic reconnection.
355    ///
356    /// Probes the server's CAPABILITY list after login and picks between:
357    /// - IMAP IDLE (RFC 2177) for instant push when the server advertises it.
358    /// - Periodic polling when the server does not support IDLE (e.g. seznam.cz).
359    async fn listen_with_reconnect(&self, tx: mpsc::Sender<ChannelMessage>) -> Result<()> {
360        let mut backoff = Duration::from_secs(1);
361        let max_backoff = Duration::from_secs(60);
362
363        loop {
364            match self.run_session(&tx).await {
365                Ok(()) => {
366                    // Clean exit (channel closed)
367                    return Ok(());
368                }
369                Err(e) => {
370                    error!(
371                        "IMAP session error: {}. Reconnecting in {:?}...",
372                        e, backoff
373                    );
374                    sleep(backoff).await;
375                    // Exponential backoff with cap
376                    backoff = std::cmp::min(backoff * 2, max_backoff);
377                }
378            }
379        }
380    }

382    /// Run a single IMAP session. Probes server capabilities and dispatches
383    /// to the IDLE or polling inner loop.
384    async fn run_session(&self, tx: &mpsc::Sender<ChannelMessage>) -> Result<()> {
385        // Connect and authenticate
386        let mut session = self.connect_imap().await?;
387
388        // Select the mailbox
389        session.select(&self.config.imap_folder).await?;
390
391        // Probe the server's post-auth capabilities to decide IDLE vs poll.
392        // RFC 3501 allows capabilities to change after authentication, so we
393        // probe after login rather than before.
394        let has_idle = {
395            let caps = session.capabilities().await?;
396            caps.has_str("IDLE")
397        };
398
399        // Drain any existing unseen messages first, regardless of mode
400        self.process_unseen(&mut session, tx).await?;
401
402        if has_idle {
403            info!(
404                "Email channel listening on {} (IMAP IDLE, instant push)",
405                self.config.imap_folder
406            );
407            self.run_idle_inner(session, tx).await
408        } else {
409            let poll_interval = Duration::from_secs(self.config.poll_interval_secs);
410            info!(
411                "Email channel listening on {} (IMAP polling, server lacks IDLE, interval: {:?})",
412                self.config.imap_folder, poll_interval
413            );
414            self.run_poll_inner(session, tx, poll_interval).await
415        }
416    }

418    /// IDLE-based wait loop. Consumes and returns the session across IDLE round trips.
419    async fn run_idle_inner(
420        &self,
421        mut session: ImapSession,
422        tx: &mpsc::Sender<ChannelMessage>,
423    ) -> Result<()> {
424        loop {
425            // Enter IDLE and wait for changes (consumes session, returns it via result)
426            match self.wait_for_changes(session).await {
427                Ok((IdleWaitResult::NewMail, returned_session)) => {
428                    debug!("New mail notification received");
429                    session = returned_session;
430                    self.process_unseen(&mut session, tx).await?;
431                }
432                Ok((IdleWaitResult::Timeout, returned_session)) => {
433                    // Re-check for mail after IDLE timeout (defensive)
434                    session = returned_session;
435                    self.process_unseen(&mut session, tx).await?;
436                }
437                Ok((IdleWaitResult::Interrupted, _)) => {
438                    info!("IDLE interrupted, exiting");
439                    return Ok(());
440                }
441                Err(e) => {
442                    // Connection likely broken, need to reconnect
443                    return Err(e);
444                }
445            }
446        }
447    }

449    /// Polling-based wait loop. Used when the server does not advertise IDLE.
450    /// Sleeps for `poll_interval` between UNSEEN checks and sends a NOOP each
451    /// cycle to keep the connection alive and detect drops early.
452    async fn run_poll_inner(
453        &self,
454        mut session: ImapSession,
455        tx: &mpsc::Sender<ChannelMessage>,
456        poll_interval: Duration,
457    ) -> Result<()> {
458        loop {
459            sleep(poll_interval).await;
460            // NOOP both keeps the connection alive and causes the server to
461            // flush any pending EXISTS/EXPUNGE updates before we search.
462            session.noop().await?;
463            self.process_unseen(&mut session, tx).await?;
464        }
465    }

467    /// Fetch unseen messages and send to channel
468    async fn process_unseen(
469        &self,
470        session: &mut ImapSession,
471        tx: &mpsc::Sender<ChannelMessage>,
472    ) -> Result<()> {
473        let messages = self.fetch_unseen(session).await?;
474
475        for email in messages {
476            // Check allowlist
477            if !self.is_sender_allowed(&email.sender) {
478                warn!("Blocked email from {}", email.sender);
479                continue;
480            }
481
482            let is_new = {
483                let mut seen = self.seen_messages.lock().await;
484                seen.insert(email.msg_id.clone())
485            };
486            if !is_new {
487                continue;
488            }
489
490            let msg = ChannelMessage {
491                id: email.msg_id,
492                reply_target: email.sender.clone(),
493                sender: email.sender,
494                content: email.content,
495                channel: "email".to_string(),
496                timestamp: email.timestamp,
497                thread_ts: None,
498                interruption_scope_id: None,
499                attachments: email.attachments,
500            };
501
502            if tx.send(msg).await.is_err() {
503                // Channel closed, exit cleanly
504                return Ok(());
505            }
506        }
507
508        Ok(())
509    }

511    fn create_smtp_transport(&self) -> Result<SmtpTransport> {
512        let creds = Credentials::new(self.config.username.clone(), self.config.password.clone());
513        let transport = if self.config.smtp_tls {
514            SmtpTransport::relay(&self.config.smtp_host)?
515                .port(self.config.smtp_port)
516                .credentials(creds)
517                .build()
518        } else {
519            SmtpTransport::builder_dangerous(&self.config.smtp_host)
520                .port(self.config.smtp_port)
521                .credentials(creds)
522                .build()
523        };
524        Ok(transport)
525    }

551    async fn send(&self, message: &SendMessage) -> Result<()> {
552        // Use explicit subject if provided, otherwise fall back to legacy parsing or default
553        let default_subject = self.config.default_subject.as_str();
554        let (subject, body) = if let Some(ref subj) = message.subject {
555            (subj.as_str(), message.content.as_str())
556        } else if message.content.starts_with("Subject: ") {
557            if let Some(pos) = message.content.find('\n') {
558                (&message.content[9..pos], message.content[pos + 1..].trim())
559            } else {
560                (default_subject, message.content.as_str())
561            }
562        } else {
563            (default_subject, message.content.as_str())
564        };
565
566        let email = if message.attachments.is_empty() {
567            // Existing plain-text path
568            Message::builder()
569                .from(self.config.from_address.parse()?)
570                .to(message.recipient.parse()?)
571                .subject(subject)
572                .singlepart(SinglePart::plain(body.to_string()))?
573        } else {
574            // Multipart with attachments
575            let mut multipart = MultiPart::mixed().singlepart(SinglePart::plain(body.to_string()));
576
577            for att in &message.attachments {
578                let content_type = att
579                    .mime_type
580                    .as_deref()
581                    .and_then(|m| ContentType::parse(m).ok())
582                    .unwrap_or_else(|| {
583                        ContentType::parse("application/octet-stream").expect("hardcoded MIME type")
584                    });
585
586                let attachment =
587                    Attachment::new(att.file_name.clone()).body(att.data.clone(), content_type);
588
589                multipart = multipart.singlepart(attachment);
590            }
591
592            Message::builder()
593                .from(self.config.from_address.parse()?)
594                .to(message.recipient.parse()?)
595                .subject(subject)
596                .multipart(multipart)?
597        };
598
599        let transport = self.create_smtp_transport()?;
600        transport.send(&email)?;
601        info!(
602            "Email sent to {} ({} attachments)",
603            message.recipient,
604            message.attachments.len()
605        );
606        Ok(())
607    }

609    async fn listen(&self, tx: mpsc::Sender<ChannelMessage>) -> Result<()> {
610        info!(
611            "Starting email channel on {} (IDLE preferred, polling fallback)",
612            self.config.imap_folder
613        );
614        self.listen_with_reconnect(tx).await
615    }

617    async fn health_check(&self) -> bool {
618        // Fully async health check - attempt IMAP connection
619        match timeout(Duration::from_secs(10), self.connect_imap()).await {
620            Ok(Ok(mut session)) => {
621                // Try to logout cleanly
622                let _ = session.logout().await;
623                true
624            }
625            Ok(Err(e)) => {
626                debug!("Health check failed: {}", e);
627                false
628            }
629            Err(_) => {
630                debug!("Health check timed out");
631                false
632            }
633        }
