11/// X/Twitter channel — uses the Twitter API v2 with OAuth 2.0 Bearer Token
12/// for sending tweets/DMs and filtered stream for receiving mentions.
13pub struct TwitterChannel {
14    bearer_token: String,
15    allowed_users: Vec<String>,
16    /// Message deduplication set.
17    dedup: Arc<RwLock<HashSet<String>>>,
18}
19
20/// Deduplication set capacity — evict half of entries when full.
21const DEDUP_CAPACITY: usize = 10_000;
22
23impl TwitterChannel {
24    pub fn new(bearer_token: String, allowed_users: Vec<String>) -> Self {
25        Self {
26            bearer_token,
27            allowed_users,
28            dedup: Arc::new(RwLock::new(HashSet::new())),
29        }
30    }

32    fn http_client(&self) -> reqwest::Client {
33        zeroclaw_config::schema::build_runtime_proxy_client("channel.twitter")
34    }

36    fn is_user_allowed(&self, user_id: &str) -> bool {
37        self.allowed_users.iter().any(|u| u == "*" || u == user_id)
38    }

40    /// Check and insert tweet ID for deduplication.
41    async fn is_duplicate(&self, tweet_id: &str) -> bool {
42        if tweet_id.is_empty() {
43            return false;
44        }
45
46        let mut dedup = self.dedup.write().await;
47
48        if dedup.contains(tweet_id) {
49            return true;
50        }
51
52        if dedup.len() >= DEDUP_CAPACITY {
53            let to_remove: Vec<String> = dedup.iter().take(DEDUP_CAPACITY / 2).cloned().collect();
54            for key in to_remove {
55                dedup.remove(&key);
56            }
57        }
58
59        dedup.insert(tweet_id.to_string());
60        false
61    }
62
63    /// Get the authenticated user's ID for filtered stream rules.
64    async fn get_authenticated_user_id(&self) -> anyhow::Result<String> {
65        let resp = self
66            .http_client()
67            .get(format!("{TWITTER_API_BASE}/users/me"))
68            .bearer_auth(&self.bearer_token)
69            .send()
70            .await?;
71
72        if !resp.status().is_success() {
73            let status = resp.status();
74            let err = resp.text().await.unwrap_or_default();
75            anyhow::bail!("Twitter users/me failed ({status}): {err}");
76        }
77
78        let data: serde_json::Value = resp.json().await?;
79        let user_id = data
80            .get("data")
81            .and_then(|d| d.get("id"))
82            .and_then(|id| id.as_str())
83            .ok_or_else(|| anyhow::anyhow!("Missing user id in Twitter response"))?
84            .to_string();
85
86        Ok(user_id)
87    }

89    /// Send a reply tweet.
90    async fn create_tweet(
91        &self,
92        text: &str,
93        reply_tweet_id: Option<&str>,
94    ) -> anyhow::Result<String> {
95        let mut body = json!({ "text": text });
96
97        if let Some(reply_id) = reply_tweet_id {
98            body["reply"] = json!({ "in_reply_to_tweet_id": reply_id });
99        }
100
101        let resp = self
102            .http_client()
103            .post(format!("{TWITTER_API_BASE}/tweets"))
104            .bearer_auth(&self.bearer_token)
105            .json(&body)
106            .send()
107            .await?;
108
109        if !resp.status().is_success() {
110            let status = resp.status();
111            let err = resp.text().await.unwrap_or_default();
112            anyhow::bail!("Twitter create tweet failed ({status}): {err}");
113        }
114
115        let data: serde_json::Value = resp.json().await?;
116        let tweet_id = data
117            .get("data")
118            .and_then(|d| d.get("id"))
119            .and_then(|id| id.as_str())
120            .unwrap_or("")
121            .to_string();
122
123        Ok(tweet_id)
124    }

126    /// Send a DM to a user.
127    async fn send_dm(&self, recipient_id: &str, text: &str) -> anyhow::Result<()> {
128        let body = json!({
129            "text": text,
130        });
131
132        let resp = self
133            .http_client()
134            .post(format!(
135                "{TWITTER_API_BASE}/dm_conversations/with/{recipient_id}/messages"
136            ))
137            .bearer_auth(&self.bearer_token)
138            .json(&body)
139            .send()
140            .await?;
141
142        if !resp.status().is_success() {
143            let status = resp.status();
144            let err = resp.text().await.unwrap_or_default();
145            anyhow::bail!("Twitter DM send failed ({status}): {err}");
146        }
147
148        Ok(())
149    }

152#[async_trait]
153impl Channel for TwitterChannel {
154    fn name(&self) -> &str {
155        "twitter"
156    }
157
158    async fn send(&self, message: &SendMessage) -> anyhow::Result<()> {
159        // recipient format: "dm:{user_id}" for DMs, "tweet:{tweet_id}" for replies
160        if let Some(user_id) = message.recipient.strip_prefix("dm:") {
161            // Twitter API enforces a 280 char limit on tweets but DMs can be up to 10000.
162            self.send_dm(user_id, &message.content).await
163        } else if let Some(tweet_id) = message.recipient.strip_prefix("tweet:") {
164            // Split long replies into tweet threads (280 char limit).
165            let chunks = split_tweet_text(&message.content, 280);
166            let mut reply_to = tweet_id.to_string();
167            for chunk in chunks {
168                reply_to = self.create_tweet(&chunk, Some(&reply_to)).await?;
169            }
170            Ok(())
171        } else {
172            // Default: treat as tweet reply
173            let chunks = split_tweet_text(&message.content, 280);
174            let mut reply_to = message.recipient.clone();
175            for chunk in chunks {
176                reply_to = self.create_tweet(&chunk, Some(&reply_to)).await?;
177            }
178            Ok(())
179        }
180    }
181
182    async fn listen(&self, tx: tokio::sync::mpsc::Sender<ChannelMessage>) -> anyhow::Result<()> {
183        tracing::info!("Twitter: authenticating...");
184        let bot_user_id = self.get_authenticated_user_id().await?;
185        tracing::info!("Twitter: authenticated as user {bot_user_id}");
186
187        // Poll mentions timeline (filtered stream requires elevated access).
188        // Using mentions timeline polling as a more accessible approach.
189        let mut since_id: Option<String> = None;
190        let poll_interval = std::time::Duration::from_secs(15);
191
192        loop {
193            let mut url = format!(
194                "{TWITTER_API_BASE}/users/{bot_user_id}/mentions?tweet.fields=author_id,conversation_id,created_at&expansions=author_id&max_results=20"
195            );
196
197            if let Some(ref id) = since_id {
198                use std::fmt::Write;
199                let _ = write!(url, "&since_id={id}");
200            }
201
202            match self
203                .http_client()
204                .get(&url)
205                .bearer_auth(&self.bearer_token)
206                .send()
207                .await
208            {
209                Ok(resp) if resp.status().is_success() => {
210                    let data: serde_json::Value = match resp.json().await {
211                        Ok(d) => d,
212                        Err(e) => {
213                            tracing::warn!("Twitter: failed to parse mentions response: {e}");
214                            tokio::time::sleep(poll_interval).await;
215                            continue;
216                        }
217                    };
218
219                    if let Some(tweets) = data.get("data").and_then(|d| d.as_array()) {
220                        // Build user lookup map from includes
221                        let user_map: std::collections::HashMap<String, String> = data
222                            .get("includes")
223                            .and_then(|i| i.get("users"))
224                            .and_then(|u| u.as_array())
225                            .map(|users| {
226                                users
227                                    .iter()
228                                    .filter_map(|u| {
229                                        let id = u.get("id")?.as_str()?.to_string();
230                                        let username = u.get("username")?.as_str()?.to_string();
231                                        Some((id, username))
232                                    })
233                                    .collect()
234                            })
235                            .unwrap_or_default();
236
237                        // Process tweets in chronological order (oldest first)
238                        for tweet in tweets.iter().rev() {
239                            let tweet_id = tweet.get("id").and_then(|i| i.as_str()).unwrap_or("");
240                            let author_id = tweet
241                                .get("author_id")
242                                .and_then(|a| a.as_str())
243                                .unwrap_or("");
244                            let text = tweet.get("text").and_then(|t| t.as_str()).unwrap_or("");
245
246                            // Skip own tweets
247                            if author_id == bot_user_id {
248                                continue;
249                            }
250
251                            if self.is_duplicate(tweet_id).await {
252                                continue;
253                            }
254
255                            let username = user_map
256                                .get(author_id)
257                                .cloned()
258                                .unwrap_or_else(|| author_id.to_string());
259
260                            if !self.is_user_allowed(&username) && !self.is_user_allowed(author_id)
261                            {
262                                tracing::debug!(
263                                    "Twitter: ignoring mention from unauthorized user: {username}"
264                                );
265                                continue;
266                            }
267
268                            // Strip the @mention from the text
269                            let clean_text = strip_at_mention(text, &bot_user_id);
270
271                            if clean_text.trim().is_empty() {
272                                continue;
273                            }
274
275                            let reply_target = format!("tweet:{tweet_id}");
276
277                            let channel_msg = ChannelMessage {
278                                id: Uuid::new_v4().to_string(),
279                                sender: username,
280                                reply_target,
281                                content: clean_text,
282                                channel: "twitter".to_string(),
283                                timestamp: std::time::SystemTime::now()
284                                    .duration_since(std::time::UNIX_EPOCH)
285                                    .unwrap_or_default()
286                                    .as_secs(),
287                                thread_ts: tweet
288                                    .get("conversation_id")
289                                    .and_then(|c| c.as_str())
290                                    .map(|s| s.to_string()),
291                                interruption_scope_id: None,
292                                attachments: vec![],
293                            };
294
295                            if tx.send(channel_msg).await.is_err() {
296                                tracing::warn!("Twitter: message channel closed");
297                                return Ok(());
298                            }
299
300                            // Track newest ID for pagination
301                            if since_id.as_deref().is_none_or(|s| tweet_id > s) {
302                                since_id = Some(tweet_id.to_string());
303                            }
304                        }
305                    }
306
307                    // Update newest_id from meta
308                    if let Some(newest) = data
309                        .get("meta")
310                        .and_then(|m| m.get("newest_id"))
311                        .and_then(|n| n.as_str())
312                    {
313                        since_id = Some(newest.to_string());
314                    }
315                }
316                Ok(resp) => {
317                    let status = resp.status();
318                    if status.as_u16() == 429 {
319                        // Rate limited — back off
320                        tracing::warn!("Twitter: rate limited, backing off 60s");
321                        tokio::time::sleep(std::time::Duration::from_secs(60)).await;
322                        continue;
323                    }
324                    let err = resp.text().await.unwrap_or_default();
325                    tracing::warn!("Twitter: mentions request failed ({status}): {err}");
326                }
327                Err(e) => {
328                    tracing::warn!("Twitter: mentions request error: {e}");
329                }
330            }
331
332            tokio::time::sleep(poll_interval).await;
333        }
334    }

336    async fn health_check(&self) -> bool {
337        self.get_authenticated_user_id().await.is_ok()
338    }

341/// Strip @mention from the beginning of a tweet text.
342fn strip_at_mention(text: &str, _bot_user_id: &str) -> String {
343    // Remove all leading @mentions (Twitter includes @bot_name at start of replies)
344    let mut result = text;
345    while let Some(rest) = result.strip_prefix('@') {
346        // Skip past the username (until whitespace or end)
347        match rest.find(char::is_whitespace) {
348            Some(idx) => result = rest[idx..].trim_start(),
349            None => return String::new(),
350        }
351    }
352    result.to_string()
353}

355/// Split text into tweet-sized chunks, breaking at word boundaries.
356fn split_tweet_text(text: &str, max_len: usize) -> Vec<String> {
357    if text.len() <= max_len {
358        return vec![text.to_string()];
359    }
360
361    let mut chunks = Vec::new();
362    let mut remaining = text;
363
364    while !remaining.is_empty() {
365        if remaining.len() <= max_len {
366            chunks.push(remaining.to_string());
367            break;
368        }
369
370        // Find last space within limit
371        let split_at = remaining[..max_len].rfind(' ').unwrap_or(max_len);
372
373        chunks.push(remaining[..split_at].to_string());
374        remaining = remaining[split_at..].trim_start();
375    }
376
377    chunks
378}
