1use ratatui::style::{Color, Modifier, Style};
2
3/// Icy-blue ZeroClaw palette. Only the colors actually referenced by a style
4/// helper below are kept — add more when a new widget needs them.
5const ICY_BLUE: Color = Color::Rgb(100, 200, 255);
6const ICY_CYAN: Color = Color::Rgb(140, 230, 255);
7const ICY_WHITE: Color = Color::Rgb(220, 240, 255);
8const FROST_DIM: Color = Color::Rgb(80, 130, 170);
9const CRAB_ACCENT: Color = Color::Rgb(255, 100, 80);
10const WARN_YELLOW: Color = Color::Rgb(255, 220, 80);
11const SELECTION_BG: Color = Color::Rgb(30, 60, 100);
12
13pub fn title_style() -> Style {
14    Style::default().fg(ICY_BLUE).add_modifier(Modifier::BOLD)
15}
16
17pub fn heading_style() -> Style {
18    Style::default().fg(ICY_CYAN).add_modifier(Modifier::BOLD)
19}
20
21pub fn body_style() -> Style {
22    Style::default().fg(ICY_WHITE)
23}
24
25pub fn dim_style() -> Style {
26    Style::default().fg(FROST_DIM)
27}
28
29pub fn accent_style() -> Style {
30    Style::default()
31        .fg(CRAB_ACCENT)
32        .add_modifier(Modifier::BOLD)
33}
34
35pub fn warn_style() -> Style {
36    Style::default().fg(WARN_YELLOW)
37}
38
39pub fn selected_style() -> Style {
40    Style::default()
41        .fg(ICY_BLUE)
42        .bg(SELECTION_BG)
43        .add_modifier(Modifier::BOLD)
44}
45
46pub fn input_style() -> Style {
47    Style::default().fg(ICY_WHITE)
48}
