12/// Runtime type classification for config property values.
13#[derive(Debug, Clone, Copy, PartialEq, Eq)]
14pub enum PropKind {
15    String,
16    Bool,
17    Integer,
18    Float,
19    /// An enum or other serde-serializable type (parsed as TOML string).
20    Enum,
21    /// A `Vec<String>` field; set via comma-separated input.
22    StringArray,
23    /// A `Vec<T>` field where `T` is a serializable struct (e.g. `Vec<McpServerConfig>`,
24    /// `Vec<PeripheralBoardConfig>`). Round-tripped on the wire as a JSON array of
25    /// objects; the dashboard renders a per-row sub-form using the JSON Schema
26    /// from `OPTIONS /api/config` to discover the element type's field shape.
27    /// Schema v3 / #5947 will migrate the load-bearing ones (mcp.servers etc.)
28    /// to `HashMap<String, T>` keyed tables; until then this kind covers them.
29    ObjectArray,
30    /// A struct-shaped scalar field (e.g. `Option<ModelPricing>`). Round-tripped
31    /// on the wire as a JSON object; the dashboard renders a sub-form for the
32    /// inner fields using the JSON Schema from `OPTIONS /api/config`. Distinct
33    /// from `String`, which inserts the raw value as a TOML string and breaks
34    /// the serde round-trip for typed structs (#6357 review).
35    Object,
36}

38/// Maps Rust types to PropKind at compile time.
39/// Scalars have explicit impls; the blanket impl catches everything
40/// else as `PropKind::Enum`.
41pub trait HasPropKind {
42    const PROP_KIND: PropKind;
43}
44
45macro_rules! impl_prop_kind {
46    ($kind:expr, $($ty:ty),+) => {
47        $(impl HasPropKind for $ty { const PROP_KIND: PropKind = $kind; })+
48    };
49}
50
51impl_prop_kind!(PropKind::Bool, bool);
52impl_prop_kind!(PropKind::String, String);
53impl_prop_kind!(PropKind::Float, f64, f32);
54impl_prop_kind!(
55    PropKind::Integer,
56    u8,
57    u16,
58    u32,
59    u64,
60    usize,
61    i8,
62    i16,
63    i32,
64    i64,
65    isize
66);
67impl HasPropKind for Vec<String> {
68    const PROP_KIND: PropKind = PropKind::StringArray;
69}

71/// Describes a single property field discovered via `#[derive(Configurable)]`.
72#[derive(Clone)]
73pub struct PropFieldInfo {
74    /// Full dotted name (e.g. `channels.telegram.draft-update-interval-ms`).
75    /// Owned so the `HashMap<String, T>` branch of the derive can inject the
76    /// runtime map key into the path (`providers.models.anthropic.api-key`)
77    /// — `&'static str` can't carry user-supplied keys.
78    pub name: String,
79    /// Category for grouping in property listings
80    pub category: &'static str,
81    /// Current value formatted for display (secrets show `"****"`)
82    pub display_value: String,
83    /// Raw Rust type string for display (e.g. `"bool"`, `"u64"`, `"Option<StreamMode>"`)
84    pub type_hint: &'static str,
85    /// Runtime type classification
86    pub kind: PropKind,
87    /// Whether this field is marked `#[secret]`
88    pub is_secret: bool,
89    /// Returns valid variant names for enum fields (None for non-enum fields)
90    pub enum_variants: Option<fn() -> Vec<String>>,
91    /// Field's `///` doc comment, flattened to a single line. Empty string
92    /// when the field has no doc comment. Onboard uses this as human-readable
93    /// prompt text instead of the raw kebab-case field name.
94    pub description: &'static str,
95    /// Whether this field's value is derived from a secret (`#[derived_from_secret]`).
96    /// Subject to the same write-only / no-readback rules as `#[secret]`.
97    /// Reserved for future schema additions; currently no fields are derived.
98    pub derived_from_secret: bool,
99}
100
101impl PropFieldInfo {
102    pub fn is_enum(&self) -> bool {
103        self.enum_variants.is_some()
104    }
105}
106
107impl std::fmt::Debug for PropFieldInfo {
108    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
109        f.debug_struct("PropFieldInfo")
110            .field("name", &self.name)
111            .field("kind", &self.kind)
112            .field("is_secret", &self.is_secret)
113            .finish_non_exhaustive()
114    }
115}

117/// Stable wire-form for an addable section — a `HashMap<String, T>` (Map) or
118/// `Vec<T>` (List) field whose value type implements `Configurable`. The
119/// dashboard / CLI use this to surface `+ Add` affordances without
120/// hardcoding the section list. Auto-discovered by the `Configurable` derive.
121#[derive(Debug, Clone, Copy, PartialEq, Eq)]
122#[cfg_attr(
123    feature = "schema-export",
124    derive(serde::Serialize, serde::Deserialize, schemars::JsonSchema)
125)]
126#[cfg_attr(feature = "schema-export", serde(rename_all = "snake_case"))]
127pub enum MapKeyKind {
128    /// `HashMap<String, T>` — key is user-supplied; new value is default.
129    Map,
130    /// `Vec<T>` — entries are appended; the user-supplied "key" is stored
131    /// in the value type's natural identifier field (e.g. `name`, `hint`).
132    List,
133}
134
135#[derive(Debug, Clone, Copy)]
136#[cfg_attr(
137    feature = "schema-export",
138    derive(serde::Serialize, schemars::JsonSchema)
139)]
140pub struct MapKeySection {
141    /// Dotted section path, e.g. `providers.models`, `mcp.servers`.
142    pub path: &'static str,
143    /// Whether the section is a map or a list.
144    pub kind: MapKeyKind,
145    /// Rust type name of the value, e.g. `ModelProviderConfig`. For display only.
146    pub value_type: &'static str,
147    /// Doc comment on the field (flattened to one line). What the user sees
148    /// when picking which kind of thing to add.
149    pub description: &'static str,
150}

152/// One row emitted by the `Configurable` derive's `nested_option_entries()`
153/// method — every `#[nested] Option<XConfig>` field on a struct shows up here
154/// with its `present` bit and the per-field `#[display_name = "..."]` /
155/// `#[description = "..."]` metadata. The integrations registry consumes
156/// this verbatim instead of carrying its own per-field hand-list.
157#[derive(Debug, Clone, Copy)]
158pub struct NestedOptionEntry {
159    /// snake_case field name on the parent struct (e.g. `"telegram"`,
160    /// `"voice_duplex"`).
161    pub field: &'static str,
162    /// `true` when the parent struct's field is `Some(_)`.
163    pub present: bool,
164    /// Display name from `#[display_name = "..."]`; falls back to a
165    /// title-cased rendering of the snake_case field name when the
166    /// attribute is absent.
167    pub display_name: &'static str,
168    /// One-line summary from `#[description = "..."]`. Empty when the
169    /// attribute is absent.
170    pub description: &'static str,
171}

173/// One row emitted by the `Configurable` derive's `integration_descriptor()`
174/// method on structs annotated with `#[integration(...)]`. Used for nested
175/// toggleable configs (e.g. `BrowserConfig`, `CronConfig`) where the
176/// integration is "active" iff a named bool field on the struct is `true`.
177#[derive(Debug, Clone, Copy)]
178pub struct IntegrationDescriptor {
179    pub display_name: &'static str,
180    pub description: &'static str,
181    /// Free-form category label (e.g. `"ToolsAutomation"`). The
182    /// integrations registry maps this string to its own
183    /// `IntegrationCategory` enum so the schema crate doesn't have to
184    /// depend on it.
185    pub category: &'static str,
186    /// Snapshot of the named status field at the moment this descriptor
187    /// was built (`status_field = "enabled"` ⇒ `self.enabled`).
188    pub active: bool,
189}

230/// Result of a single prompt — either the value the user chose, or a
231/// navigation signal. Backends return `Answer::Back` when the user presses
232/// the backend's back key (Esc on ratatui / dialoguer). Callers rewind.
233#[derive(Debug, Clone)]
234pub enum Answer<T> {
235    Value(T),
236    Back,
237}

239/// Prompt-surface the onboard orchestrator drives.
240///
241/// Async is deliberate: the orchestrator is already async (Config::load_or_init,
242/// Config::save), and a future gateway-backed onboarder (WebSocket → browser)
243/// needs to await network I/O per prompt. A sync trait would force that
244/// backend to bridge sync↔async via blocking threads and channels, which
245/// starves the tokio runtime under concurrent onboarding sessions. Blocking
246/// backends (dialoguer) wrap their calls in `tokio::task::spawn_blocking`.
247///
248/// Idempotency contract: prompts accept a `current` value and pre-populate it
249/// as the default. `secret(has_current=true)` returns `None` when the user
250/// declines to rotate; callers then skip the write. The orchestrator never
251/// calls `config.set_prop` unless the new value differs from `current`.
252#[async_trait::async_trait]
253pub trait OnboardUi: Send {
254    async fn confirm(&mut self, prompt: &str, default: bool) -> anyhow::Result<Answer<bool>>;
255
256    async fn string(
257        &mut self,
258        prompt: &str,
259        current: Option<&str>,
260    ) -> anyhow::Result<Answer<String>>;
261
262    /// `Answer::Value(Some(v))` = new secret entered. `Answer::Value(None)` =
263    /// user declined to update an existing secret (only when `has_current`).
264    /// `Answer::Back` = rewind.
265    async fn secret(
266        &mut self,
267        prompt: &str,
268        has_current: bool,
269    ) -> anyhow::Result<Answer<Option<String>>>;
270
271    async fn select(
272        &mut self,
273        prompt: &str,
274        items: &[SelectItem],
275        current: Option<usize>,
276    ) -> anyhow::Result<Answer<usize>>;
277
278    async fn editor(&mut self, hint: &str, initial: &str) -> anyhow::Result<Answer<String>>;
279
280    /// Announce a new section or subsection. `level == 1` = section
281    /// (Providers, Channels, …). `level == 2` = subsection within a section
282    /// (Hardware › Transport). Backends render these persistently so every
283    /// prompt remains anchored to its phase — rendered like Markdown
284    /// headings. `level == 1` resets any prior subsection.
285    fn heading(&mut self, level: u8, text: &str);
286    fn note(&mut self, msg: &str);
287    fn status(&mut self, msg: &str);
288    fn warn(&mut self, msg: &str);
289}
