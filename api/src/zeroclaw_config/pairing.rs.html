1// Gateway pairing mode — first-connect authentication.
2//
3// On startup the gateway generates a one-time pairing code printed to the
4// terminal. The first client must present this code via `X-Pairing-Code`
5// header on a `POST /pair` request. The server responds with a bearer token
6// that must be sent on all subsequent requests via `Authorization: Bearer <token>`.
7//
8// Already-paired tokens are persisted in config so restarts don't require
9// re-pairing.
10
11use parking_lot::Mutex;
12use sha2::{Digest, Sha256};
13use std::collections::{HashMap, HashSet};
14use std::sync::Arc;
15use std::time::Instant;
16
17/// Maximum failed pairing attempts before lockout.
18const MAX_PAIR_ATTEMPTS: u32 = 5;
19/// Lockout duration after too many failed pairing attempts.
20const PAIR_LOCKOUT_SECS: u64 = 300; // 5 minutes
21/// Maximum number of tracked client entries to bound memory usage.
22const MAX_TRACKED_CLIENTS: usize = 10_000;
23/// Retention period for failed-attempt entries with no activity.
24const FAILED_ATTEMPT_RETENTION_SECS: u64 = 900; // 15 min
25/// Minimum interval between full sweeps of the failed-attempt map.
26const FAILED_ATTEMPT_SWEEP_INTERVAL_SECS: u64 = 300; // 5 min
27
28/// Per-client failed attempt state with optional absolute lockout deadline.
29#[derive(Debug, Clone, Copy)]
30struct FailedAttemptState {
31    count: u32,
32    lockout_until: Option<Instant>,
33    last_attempt: Instant,
34}
35
36/// Manages pairing state for the gateway.
37///
38/// Bearer tokens are stored as SHA-256 hashes to prevent plaintext exposure
39/// in config files. When a new token is generated, the plaintext is returned
40/// to the client once, and only the hash is retained.
41// TODO: I've just made this work with parking_lot but it should use either flume or tokio's async mutexes
42#[derive(Debug, Clone)]
43pub struct PairingGuard {
44    /// Whether pairing is required at all.
45    require_pairing: bool,
46    /// One-time pairing code (generated on startup, consumed on first pair).
47    pairing_code: Arc<Mutex<Option<String>>>,
48    /// Set of SHA-256 hashed bearer tokens (persisted across restarts).
49    paired_tokens: Arc<Mutex<HashSet<String>>>,
50    /// Brute-force protection: per-client failed attempt state + last sweep timestamp.
51    failed_attempts: Arc<Mutex<(HashMap<String, FailedAttemptState>, Instant)>>,
52}
53
54impl PairingGuard {
55    /// Create a new pairing guard.
56    ///
57    /// If `require_pairing` is true and no tokens exist yet, a fresh
58    /// pairing code is generated and printed to the terminal. Once
59    /// paired, no code is generated on restart — operators can use
60    /// `generate_new_pairing_code()` or the CLI to create one on demand.
61    ///
62    /// Existing tokens are accepted in both forms:
63    /// - Plaintext (`zc_...`): hashed on load for backward compatibility
64    /// - Already hashed (64-char hex): stored as-is
65    pub fn new(require_pairing: bool, existing_tokens: &[String]) -> Self {
66        let tokens: HashSet<String> = existing_tokens
67            .iter()
68            .map(|t| {
69                if is_token_hash(t) {
70                    t.clone()
71                } else {
72                    hash_token(t)
73                }
74            })
75            .collect();
76        let code = if require_pairing && tokens.is_empty() {
77            Some(generate_code())
78        } else {
79            None
80        };
81        Self {
82            require_pairing,
83            pairing_code: Arc::new(Mutex::new(code)),
84            paired_tokens: Arc::new(Mutex::new(tokens)),
85            failed_attempts: Arc::new(Mutex::new((HashMap::new(), Instant::now()))),
86        }
87    }
88
89    /// The one-time pairing code (generated only on first startup when no tokens exist).
90    pub fn pairing_code(&self) -> Option<String> {
91        self.pairing_code.lock().clone()
92    }
93
94    /// Whether pairing is required at all.
95    pub fn require_pairing(&self) -> bool {
96        self.require_pairing
97    }

99    fn try_pair_blocking(&self, code: &str, client_id: &str) -> Result<Option<String>, u64> {
100        let client_id = normalize_client_key(client_id);
101        let now = Instant::now();
102
103        // Periodic sweep + lockout check
104        {
105            let mut guard = self.failed_attempts.lock();
106            let (ref mut map, ref mut last_sweep) = *guard;
107
108            // Sweep stale entries on interval
109            if now.duration_since(*last_sweep).as_secs() >= FAILED_ATTEMPT_SWEEP_INTERVAL_SECS {
110                prune_failed_attempts(map, now);
111                *last_sweep = now;
112            }
113
114            // Check brute force lockout for this specific client
115            if let Some(state) = map.get(&client_id)
116                && let Some(until) = state.lockout_until
117            {
118                if now < until {
119                    let remaining = (until - now).as_secs();
120                    return Err(remaining.max(1));
121                }
122                // Lockout expired — reset inline
123                map.remove(&client_id);
124            }
125        }
126
127        {
128            let mut pairing_code = self.pairing_code.lock();
129            if let Some(ref expected) = *pairing_code
130                && constant_time_eq(code.trim(), expected.trim())
131            {
132                // Reset failed attempts for this client on success
133                {
134                    let mut guard = self.failed_attempts.lock();
135                    guard.0.remove(&client_id);
136                }
137                let token = generate_token();
138                let mut tokens = self.paired_tokens.lock();
139                tokens.insert(hash_token(&token));
140
141                // Consume the pairing code so it cannot be reused
142                *pairing_code = None;
143
144                return Ok(Some(token));
145            }
146        }
147
148        // Increment failed attempts for this client
149        {
150            let mut guard = self.failed_attempts.lock();
151            let (ref mut map, _) = *guard;
152
153            // Enforce capacity bound: prune stale first, then LRU-evict if still full
154            if map.len() >= MAX_TRACKED_CLIENTS {
155                prune_failed_attempts(map, now);
156            }
157            if map.len() >= MAX_TRACKED_CLIENTS {
158                // Evict the least-recently-active entry
159                if let Some(lru_key) = map
160                    .iter()
161                    .min_by_key(|(_, s)| s.last_attempt)
162                    .map(|(k, _)| k.clone())
163                {
164                    map.remove(&lru_key);
165                }
166            }
167
168            let entry = map.entry(client_id).or_insert(FailedAttemptState {
169                count: 0,
170                lockout_until: None,
171                last_attempt: now,
172            });
173
174            entry.last_attempt = now;
175            entry.count += 1;
176
177            if entry.count >= MAX_PAIR_ATTEMPTS {
178                entry.lockout_until = Some(now + std::time::Duration::from_secs(PAIR_LOCKOUT_SECS));
179            }
180        }
181
182        Ok(None)
183    }

185    /// Attempt to pair with the given code. Returns a bearer token on success.
186    /// Returns `Err(lockout_seconds)` if locked out due to brute force.
187    /// `client_id` identifies the client for per-client lockout accounting.
188    pub async fn try_pair(&self, code: &str, client_id: &str) -> Result<Option<String>, u64> {
189        let this = self.clone();
190        let code = code.to_string();
191        let client_id = client_id.to_string();
192        // TODO: make this function the main one without spawning a task
193        let handle = tokio::task::spawn_blocking(move || this.try_pair_blocking(&code, &client_id));
194
195        handle
196            .await
197            .expect("failed to spawn blocking task this should not happen")
198    }

200    /// Check if a bearer token is valid (compares against stored hashes).
201    pub fn is_authenticated(&self, token: &str) -> bool {
202        if !self.require_pairing {
203            return true;
204        }
205        let hashed = hash_token(token);
206        let tokens = self.paired_tokens.lock();
207        tokens.contains(&hashed)
208    }
209
210    /// Returns true if the gateway is already paired (has at least one token).
211    pub fn is_paired(&self) -> bool {
212        let tokens = self.paired_tokens.lock();
213        !tokens.is_empty()
214    }
215
216    /// Get all paired token hashes (for persisting to config).
217    pub fn tokens(&self) -> Vec<String> {
218        let tokens = self.paired_tokens.lock();
219        tokens.iter().cloned().collect()
220    }

222    /// Generate a new pairing code, even if already paired.
223    ///
224    /// This allows adding additional clients without restarting the gateway.
225    /// The new code can be used exactly once to pair a new client.
226    pub fn generate_new_pairing_code(&self) -> Option<String> {
227        if !self.require_pairing {
228            return None;
229        }
230        let new_code = generate_code();
231        *self.pairing_code.lock() = Some(new_code.clone());
232        Some(new_code)
233    }

235    /// Get the token hash for a given plaintext token (for device registry lookup).
236    pub fn token_hash(token: &str) -> String {
237        use sha2::{Digest, Sha256};
238        hex::encode(Sha256::digest(token.as_bytes()))
239    }
240
241    /// Check if a token is paired and return its hash.
242    pub fn authenticate_and_hash(&self, token: &str) -> Option<String> {
243        if self.is_authenticated(token) {
244            Some(Self::token_hash(token))
245        } else {
246            None
247        }
248    }

251/// Normalize a client identifier: trim whitespace, map empty to `"unknown"`.
252fn normalize_client_key(key: &str) -> String {
253    let trimmed = key.trim();
254    if trimmed.is_empty() {
255        "unknown".to_string()
256    } else {
257        trimmed.to_string()
258    }

261/// Remove failed-attempt entries whose `last_attempt` is older than the retention window.
262fn prune_failed_attempts(map: &mut HashMap<String, FailedAttemptState>, now: Instant) {
263    map.retain(|_, state| {
264        now.duration_since(state.last_attempt).as_secs() < FAILED_ATTEMPT_RETENTION_SECS
265    });
266}

268/// Generate a 6-digit numeric pairing code using cryptographically secure randomness.
269fn generate_code() -> String {
270    // UUID v4 uses getrandom (backed by /dev/urandom on Linux, BCryptGenRandom
271    // on Windows) — a CSPRNG. We extract 4 bytes from it for a uniform random
272    // number in [0, 1_000_000).
273    //
274    // Rejection sampling eliminates modulo bias: values above the largest
275    // multiple of 1_000_000 that fits in u32 are discarded and re-drawn.
276    // The rejection probability is ~0.02%, so this loop almost always exits
277    // on the first iteration.
278    const UPPER_BOUND: u32 = 1_000_000;
279    const REJECT_THRESHOLD: u32 = (u32::MAX / UPPER_BOUND) * UPPER_BOUND;
280
281    loop {
282        let uuid = uuid::Uuid::new_v4();
283        let bytes = uuid.as_bytes();
284        let raw = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
285
286        if raw < REJECT_THRESHOLD {
287            return format!("{:06}", raw % UPPER_BOUND);
288        }
289    }
290}

292/// Generate a cryptographically-adequate bearer token with 256-bit entropy.
293///
294/// Uses `rand::rng()` which is backed by the OS CSPRNG
295/// (/dev/urandom on Linux, BCryptGenRandom on Windows, SecRandomCopyBytes
296/// on macOS). The 32 random bytes (256 bits) are hex-encoded for a
297/// 64-character token, providing 256 bits of entropy.
298fn generate_token() -> String {
299    let bytes: [u8; 32] = rand::random();
300    format!("zc_{}", hex::encode(bytes))
301}
302
303/// SHA-256 hash a bearer token for storage. Returns lowercase hex.
304fn hash_token(token: &str) -> String {
305    format!("{:x}", Sha256::digest(token.as_bytes()))
306}

308/// Check if a stored value looks like a SHA-256 hash (64 hex chars)
309/// rather than a plaintext token.
310fn is_token_hash(value: &str) -> bool {
311    value.len() == 64 && value.chars().all(|c| c.is_ascii_hexdigit())
312}

314/// Constant-time string comparison to prevent timing attacks.
315///
316/// This function is critical to the security of the pairing mechanism:
317/// when verifying the one-time pairing code, timing side-channels could
318/// allow an attacker to deduce the correct code character-by-character.
319///
320/// Implementation details that ensure constant-time execution:
321/// 1. Does not short-circuit on length mismatch — always iterates over
322///    the longer input to avoid leaking length information via timing.
323/// 2. Uses bitwise AND (&) instead of logical AND (&&) to ensure both
324///    comparisons always execute, preventing timing variations that could
325///    reveal whether the length check or byte comparison failed first.
326///
327/// SECURITY NOTE: The use of `&` instead of `&&` is intentional and
328/// required for constant-time behavior. Do not change to `&&` or clippy
329/// suggestions that would reintroduce short-circuit evaluation.
330#[allow(clippy::needless_bitwise_bool)]
331pub fn constant_time_eq(a: &str, b: &str) -> bool {
332    let a = a.as_bytes();
333    let b = b.as_bytes();
334
335    // Track length mismatch as a usize (non-zero = different lengths)
336    let len_diff = a.len() ^ b.len();
337
338    // XOR each byte, padding the shorter input with zeros.
339    // Iterates over max(a.len(), b.len()) to avoid timing differences.
340    let max_len = a.len().max(b.len());
341    let mut byte_diff = 0u8;
342    for i in 0..max_len {
343        let x = *a.get(i).unwrap_or(&0);
344        let y = *b.get(i).unwrap_or(&0);
345        byte_diff |= x ^ y;
346    }
347    // Intentional use of bitwise & (not &&) to ensure constant-time execution
348    // and prevent timing side-channel attacks. Both comparisons must execute.
349    (len_diff == 0) & (byte_diff == 0)
350}
